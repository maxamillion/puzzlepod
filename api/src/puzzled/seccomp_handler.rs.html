2// DC: Single-threaded seccomp notification handling is sufficient for typical
3// workloads (<100 execve/s). For higher throughput, consider per-branch handler threads.
4//
5// WS9 LIMITATION: The single-threaded poll loop means one slow notification
6// (e.g., slow OPA evaluation, disk I/O) blocks ALL agents' gated syscalls.
7// The kernel seccomp USER_NOTIF has no built-in timeout — agents block until
8// puzzled responds. Per-notification timing is logged so operators can identify
9// slow handlers. A thread pool would fix this but requires careful fd lifecycle
10// management — tracked as future work.
11//
12// Known limitation: single-threaded poll loop means one slow notification can
13// delay all agents' gated syscalls. The systemd watchdog timeout (default 30s)
14// is the worst-case freeze duration before SIGABRT. Per-notification deadline
15// of 2s mitigates but does not eliminate head-of-line blocking.
16
17//! Centralized seccomp USER_NOTIF polling thread.
18//!
19//! Manages an epoll set of seccomp notification file descriptors (one per
20//! active branch). When a notification arrives, reads the syscall arguments
21//! via `/proc/<pid>/mem` and evaluates against the agent's profile.
22//!
23//! Runs in `tokio::task::spawn_blocking` since epoll_wait is a blocking call.
24

38/// M7: epoll_wait timeout in milliseconds for seccomp notification polling.

42/// WS9: Wall-clock budget from when the poll loop begins handling a notification
43/// (including `receive`) until the response is sent. If the handler would allow
44/// the syscall but this deadline has passed, the response is forced to EPERM.
45#[cfg(target_os = "linux")]
46pub const SECCOMP_NOTIF_HANDLER_WALL_DEADLINE: Duration = Duration::from_secs(2);

48/// WS9: Notifications denied with EPERM because the wall-clock deadline was exceeded
49/// while the policy outcome would have been allow (metrics / observability).
50#[cfg(target_os = "linux")]
51pub static SECCOMP_NOTIF_HANDLER_WALL_DEADLINE_DENIES: AtomicU64 = AtomicU64::new(0);

53/// §3.4 G23: Credential proxy context for seccomp-aware gateway blocking.
54///
55/// When a credential proxy is active for a branch, the seccomp handler
56/// blocks direct connections to the gateway IP on the proxy port range
57/// to prevent agents from bypassing the transparent proxy.
58#[derive(Debug, Clone)]
59pub struct CredentialProxyContext {
60    /// Whether the credential proxy is active for this branch.
61    pub enabled: bool,
62    /// Gateway IP address (container → host).
63    pub proxy_gateway_ip: std::net::IpAddr,
64    /// Proxy port for this specific branch.
65    pub proxy_port: u16,
66    /// Ports being intercepted by DNAT.
67    pub proxied_ports: Vec<u16>,
68    /// Global proxy port range (all branches).
69    pub global_port_range: std::ops::RangeInclusive<u16>,
70}

72/// Commands sent to the polling thread via channel.
73pub enum SeccompCommand {
74    /// Register a new seccomp notify fd for a branch.
75    Register {
76        notify_fd: i32,
77        branch_id: BranchId,
78        profile: Box<AgentProfile>,
79        /// §3.4 G23: Optional credential proxy context for gateway blocking.
80        credential_proxy: Option<CredentialProxyContext>,
81    },
82    /// Unregister a branch's notify fd (branch terminated).
83    Unregister { branch_id: BranchId },
84    /// H9: Register a cgroup memory.events file for OOM kill monitoring.
85    #[cfg(target_os = "linux")]
86    RegisterOomMonitor {
87        branch_id: BranchId,
88        /// Path to the cgroup's memory.events file (e.g., /sys/fs/cgroup/puzzled/<branch>/memory.events)
89        memory_events_path: std::path::PathBuf,
90    },
91    /// Shut down the polling thread.
92    Shutdown,
93}

95/// Handle returned when the polling thread is spawned.
96// Q2: wake_fd is a raw eventfd shared across clones (via #[derive(Clone)]).
97// It is closed in poll_loop when a Shutdown command is received. Known limitation:
98// if the handler is dropped without sending Shutdown, the fd is leaked.
99#[derive(Clone)]
100pub struct SeccompNotifHandler {
101    sender: tokio::sync::mpsc::Sender<SeccompCommand>,
102    /// File descriptor for the eventfd used to wake the epoll loop immediately
103    /// after sending a command, rather than waiting for the 100ms timeout.
104    #[cfg(target_os = "linux")]
105    wake_fd: std::os::unix::io::RawFd,
106}
107
108impl SeccompNotifHandler {
109    /// Create and spawn the seccomp notification handler thread.
110    ///
111    /// Returns a handle for sending register/unregister commands.
112    pub fn spawn() -> Self {
113        let (tx, rx) = tokio::sync::mpsc::channel(64);
114
115        // H4: Create eventfd before spawning the thread so the handler can
116        // write to it after each send to wake the epoll loop immediately.
117        // S1: Check eventfd() return — -1 indicates failure (e.g., fd exhaustion)
118        #[cfg(target_os = "linux")]
119        let wake_fd = unsafe { libc::eventfd(0, libc::EFD_NONBLOCK | libc::EFD_CLOEXEC) };
120        #[cfg(target_os = "linux")]
121        if wake_fd < 0 {
122            tracing::error!(
123                error = %std::io::Error::last_os_error(),
124                "S1: failed to create eventfd for seccomp wake — poll loop will rely on timeout"
125            );
126        }
127        #[cfg(not(target_os = "linux"))]
128        let _wake_fd: i32 = -1;
129
130        #[cfg(target_os = "linux")]
131        let wake_fd_for_thread = wake_fd;
132
133        // Use tokio::task::spawn_blocking if a runtime is available,
134        // otherwise fall back to std::thread::spawn (for sync tests).
135        let poll_fn = move || {
136            #[cfg(target_os = "linux")]
137            {
138                if let Err(e) = poll_loop(rx, wake_fd_for_thread) {
139                    tracing::error!(error = %e, "seccomp polling thread exited with error");
140                }
141            }
142            #[cfg(not(target_os = "linux"))]
143            {
144                if let Err(e) = poll_loop(rx) {
145                    tracing::error!(error = %e, "seccomp polling thread exited with error");
146                }
147            }
148        };
149
150        if tokio::runtime::Handle::try_current().is_ok() {
151            tokio::task::spawn_blocking(poll_fn);
152        } else {
153            std::thread::spawn(poll_fn);
154        }
155
156        Self {
157            sender: tx,
158            #[cfg(target_os = "linux")]
159            wake_fd,
160        }
161    }
162
163    /// H4: Write to the wake_fd eventfd to wake the epoll loop immediately
164    /// after sending a command via the channel.
165    #[cfg(target_os = "linux")]
166    fn wake_poll_loop(&self) {
167        // S7: Check write() return to detect wake failures
168        let val: u64 = 1;
169        let ret = unsafe {
170            libc::write(
171                self.wake_fd,
172                &val as *const u64 as *const libc::c_void,
173                std::mem::size_of::<u64>(),
174            )
175        };
176        if ret < 0 {
177            tracing::warn!(
178                error = %std::io::Error::last_os_error(),
179                "S7: failed to write to wake eventfd — poll loop will wake on next timeout"
180            );
181        }
182    }
183
184    /// No-op on non-Linux platforms.
185    #[cfg(not(target_os = "linux"))]
186    fn wake_poll_loop(&self) {}
187
188    /// Register a seccomp notify fd for a branch (async).
189    pub async fn register_async(
190        &self,
191        notify_fd: i32,
192        branch_id: BranchId,
193        profile: AgentProfile,
194        credential_proxy: Option<CredentialProxyContext>,
195    ) -> Result<()> {
196        self.sender
197            .send(SeccompCommand::Register {
198                notify_fd,
199                branch_id,
200                profile: Box::new(profile),
201                credential_proxy,
202            })
203            .await
204            .map_err(|_| PuzzledError::SeccompNotif("handler channel closed".to_string()))?;
205        // H4: Wake the epoll loop immediately so the new fd is registered without delay
206        self.wake_poll_loop();
207        Ok(())
208    }
209
210    /// Register a seccomp notify fd for a branch (non-blocking).
211    ///
212    /// H4: Returns an error if the channel is full, so the caller can fail
213    /// branch creation instead of silently running without seccomp mediation.
214    pub fn register(
215        &self,
216        notify_fd: i32,
217        branch_id: BranchId,
218        profile: AgentProfile,
219        credential_proxy: Option<CredentialProxyContext>,
220    ) -> Result<()> {
221        self.sender
222            .try_send(SeccompCommand::Register {
223                notify_fd,
224                branch_id,
225                profile: Box::new(profile),
226                credential_proxy,
227            })
228            .map_err(|e| {
229                PuzzledError::SeccompNotif(format!("registering seccomp notify fd: {}", e))
230            })?;
231        // H4: Wake the epoll loop immediately so the new fd is registered without delay
232        self.wake_poll_loop();
233        Ok(())
234    }
235
236    /// Unregister a branch's seccomp notify fd (non-blocking, fire-and-forget).
237    pub fn unregister(&self, branch_id: BranchId) {
238        if self
239            .sender
240            .try_send(SeccompCommand::Unregister { branch_id })
241            .is_ok()
242        {
243            // H4: Wake the epoll loop immediately to process the unregister
244            self.wake_poll_loop();
245        }
246    }
247
248    /// Unregister by notify fd — convenience for rollback paths.
249    pub fn unregister_by_fd(&self, _notify_fd: i32) {
250        // The polling thread tracks fd->branch mapping; sending Unregister by BranchId
251        // is the canonical path. This is a no-op placeholder.
252    }
253
254    /// H9: Register OOM kill monitoring for a branch's cgroup.
255    /// Watches the cgroup's memory.events file via inotify and logs a warning
256    /// when an OOM kill is detected.
257    #[cfg(target_os = "linux")]
258    pub fn register_oom_monitor(
259        &self,
260        branch_id: BranchId,
261        memory_events_path: std::path::PathBuf,
262    ) -> Result<()> {
263        self.sender
264            .try_send(SeccompCommand::RegisterOomMonitor {
265                branch_id,
266                memory_events_path,
267            })
268            .map_err(|e| PuzzledError::SeccompNotif(format!("registering OOM monitor: {}", e)))?;
269        // H4: Wake the epoll loop immediately to register the OOM monitor
270        self.wake_poll_loop();
271        Ok(())
272    }
273
274    /// Shut down the polling thread.
275    pub async fn shutdown(&self) {
276        if self.sender.send(SeccompCommand::Shutdown).await.is_ok() {
277            // H4: Wake the epoll loop immediately to process shutdown
278            self.wake_poll_loop();
279        }
280    }
281}

283/// Entry for a registered branch in the polling thread.
284#[cfg_attr(not(target_os = "linux"), allow(dead_code))]
285struct BranchEntry {
286    notify_fd: i32,
287    profile: AgentProfile,
288    /// Per-branch exec counter — tracks total execve/execveat calls.
289    /// Compared against the profile's exec budget to enforce limits.
290    exec_count: std::sync::atomic::AtomicU64,
291    /// §3.4 G23: Credential proxy context for gateway blocking.
292    credential_proxy: Option<CredentialProxyContext>,
293}
294
295/// Main polling loop — runs in a blocking thread.
296///
297/// Uses epoll to wait on multiple seccomp notification fds simultaneously.
298/// Processes commands from the channel between epoll waits.
299#[cfg(target_os = "linux")]
300fn poll_loop(
301    mut rx: tokio::sync::mpsc::Receiver<SeccompCommand>,
302    wake_fd: std::os::unix::io::RawFd,
303) -> Result<()> {
304    // Create epoll instance
305    let epoll_fd = unsafe { libc::epoll_create1(libc::EPOLL_CLOEXEC) };
306    if epoll_fd < 0 {
307        return Err(PuzzledError::SeccompNotif(format!(
308            "epoll_create1 failed: {}",
309            std::io::Error::last_os_error()
310        )));
311    }
312
313    // H4: wake_fd is created by SeccompNotifHandler::spawn() and shared with
314    // the handler struct, so that callers can write to it after sending commands
315    // to wake the epoll loop immediately instead of waiting for the 100ms timeout.
316    if wake_fd < 0 {
317        unsafe { libc::close(epoll_fd) };
318        return Err(PuzzledError::SeccompNotif(
319            "invalid wake_fd passed to poll_loop".to_string(),
320        ));
321    }
322
323    // Add wake_fd to epoll
324    let mut wake_event = libc::epoll_event {
325        events: libc::EPOLLIN as u32,
326        u64: wake_fd as u64,
327    };
328    unsafe {
329        libc::epoll_ctl(epoll_fd, libc::EPOLL_CTL_ADD, wake_fd, &mut wake_event);
330    }
331
332    let mut active_fds: HashMap<i32, (BranchId, usize)> = HashMap::new(); // notify_fd -> (branch_id, index)
333    let mut branches: HashMap<BranchId, BranchEntry> = HashMap::new();
334    let mut events = vec![libc::epoll_event { events: 0, u64: 0 }; 64];
335    // M-sc2: Track fds that are being closed to avoid processing in-flight
336    // notifications for fds that are in the process of being unregistered.
337    // The fd is added here before epoll_del to prevent a race where a
338    // notification arrives between the epoll_del and the fd being removed
339    // from active_fds.
340    let mut closing_fds: HashSet<i32> = HashSet::new();
341
342    // H9: inotify instance for monitoring cgroup memory.events files for OOM kills
343    let inotify_fd = unsafe { libc::inotify_init1(libc::IN_NONBLOCK | libc::IN_CLOEXEC) };
344    if inotify_fd < 0 {
345        tracing::warn!(
346            error = %std::io::Error::last_os_error(),
347            "failed to create inotify instance for OOM monitoring, continuing without it"
348        );
349    } else {
350        // Add inotify_fd to epoll
351        let mut ino_event = libc::epoll_event {
352            events: libc::EPOLLIN as u32,
353            u64: inotify_fd as u64,
354        };
355        unsafe {
356            libc::epoll_ctl(epoll_fd, libc::EPOLL_CTL_ADD, inotify_fd, &mut ino_event);
357        }
358    }
359    // Maps inotify watch descriptor -> branch_id for OOM monitoring
360    let mut oom_watches: HashMap<i32, BranchId> = HashMap::new();
361    // Maps branch_id -> inotify watch descriptor for cleanup
362    let mut oom_branch_to_wd: HashMap<BranchId, i32> = HashMap::new();
363
364    tracing::info!("seccomp polling thread started");
365
366    loop {
367        // Process any pending commands (non-blocking)
368        while let Ok(cmd) = rx.try_recv() {
369            match cmd {
370                SeccompCommand::Register {
371                    notify_fd,
372                    branch_id,
373                    profile,
374                    credential_proxy,
375                } => {
376                    tracing::debug!(
377                        branch = %branch_id,
378                        notify_fd,
379                        "registering seccomp notify fd"
380                    );
381
382                    // Add to epoll
383                    let mut event = libc::epoll_event {
384                        events: libc::EPOLLIN as u32,
385                        u64: notify_fd as u64,
386                    };
387                    let ret = unsafe {
388                        libc::epoll_ctl(epoll_fd, libc::EPOLL_CTL_ADD, notify_fd, &mut event)
389                    };
390                    if ret < 0 {
391                        tracing::error!(
392                            notify_fd,
393                            error = %std::io::Error::last_os_error(),
394                            "failed to add notify fd to epoll"
395                        );
396                        continue;
397                    }
398
399                    active_fds.insert(notify_fd, (branch_id.clone(), 0));
400                    branches.insert(
401                        branch_id,
402                        BranchEntry {
403                            notify_fd,
404                            profile: *profile,
405                            exec_count: std::sync::atomic::AtomicU64::new(0),
406                            credential_proxy,
407                        },
408                    );
409                }
410                SeccompCommand::Unregister { branch_id } => {
411                    tracing::debug!(branch = %branch_id, "unregistering seccomp notify fd");
412
413                    if let Some(entry) = branches.remove(&branch_id) {
414                        // M-sc2: Mark fd as closing BEFORE epoll_del to prevent
415                        // processing in-flight notifications for this fd.
416                        closing_fds.insert(entry.notify_fd);
417                        unsafe {
418                            libc::epoll_ctl(
419                                epoll_fd,
420                                libc::EPOLL_CTL_DEL,
421                                entry.notify_fd,
422                                std::ptr::null_mut(),
423                            );
424                        }
425                        active_fds.remove(&entry.notify_fd);
426                        closing_fds.remove(&entry.notify_fd);
427                    }
428
429                    // H9: Remove OOM watch for this branch if present
430                    if let Some(wd) = oom_branch_to_wd.remove(&branch_id) {
431                        if inotify_fd >= 0 {
432                            unsafe {
433                                libc::inotify_rm_watch(inotify_fd, wd);
434                            }
435                        }
436                        oom_watches.remove(&wd);
437                    }
438                }
439                // H9: Register OOM monitoring for a branch's cgroup memory.events
440                SeccompCommand::RegisterOomMonitor {
441                    branch_id,
442                    memory_events_path,
443                } => {
444                    if inotify_fd < 0 {
445                        tracing::warn!(
446                            branch = %branch_id,
447                            "cannot register OOM monitor: inotify not available"
448                        );
449                    } else {
450                        use std::ffi::CString;
451                        use std::os::unix::ffi::OsStrExt;
452
453                        if let Ok(c_path) = CString::new(memory_events_path.as_os_str().as_bytes())
454                        {
455                            let wd = unsafe {
456                                libc::inotify_add_watch(
457                                    inotify_fd,
458                                    c_path.as_ptr(),
459                                    libc::IN_MODIFY,
460                                )
461                            };
462                            if wd < 0 {
463                                tracing::warn!(
464                                    branch = %branch_id,
465                                    path = %memory_events_path.display(),
466                                    error = %std::io::Error::last_os_error(),
467                                    "failed to add inotify watch for OOM monitoring"
468                                );
469                            } else {
470                                tracing::debug!(
471                                    branch = %branch_id,
472                                    path = %memory_events_path.display(),
473                                    "registered OOM kill monitor"
474                                );
475                                oom_watches.insert(wd, branch_id.clone());
476                                oom_branch_to_wd.insert(branch_id, wd);
477                            }
478                        }
479                    }
480                }
481                SeccompCommand::Shutdown => {
482                    tracing::info!("seccomp polling thread shutting down");
483                    unsafe {
484                        libc::close(wake_fd);
485                        if inotify_fd >= 0 {
486                            libc::close(inotify_fd);
487                        }
488                        libc::close(epoll_fd);
489                    }
490                    return Ok(());
491                }
492            }
493        }
494
495        // M7: Wait for events (EPOLL_TIMEOUT_MS timeout to check for commands)
496        let n = unsafe {
497            libc::epoll_wait(
498                epoll_fd,
499                events.as_mut_ptr(),
500                events.len() as i32,
501                EPOLL_TIMEOUT_MS,
502            )
503        };
504
505        if n < 0 {
506            let err = std::io::Error::last_os_error();
507            if err.kind() == std::io::ErrorKind::Interrupted {
508                continue;
509            }
510            tracing::error!(error = %err, "epoll_wait failed");
511            continue;
512        }
513
514        // T5: Use try_from for consistency with fanotify epoll loops
515        for event in events.iter().take(usize::try_from(n).unwrap_or(0)) {
516            // T6: Use try_from for consistency with fanotify J6 pattern
517            // Copy from packed struct field to avoid misaligned reference (UB)
518            let event_data = event.u64;
519            let fd = match i32::try_from(event_data) {
520                Ok(v) => v,
521                Err(_) => {
522                    tracing::warn!(raw = event_data, "T6: epoll event data exceeds i32 range");
523                    continue;
524                }
525            };
526
527            // Skip wake_fd events — just drain the eventfd
528            if fd == wake_fd {
529                let mut buf = [0u8; 8];
530                unsafe {
531                    libc::read(wake_fd, buf.as_mut_ptr() as *mut libc::c_void, 8);
532                }
533                continue;
534            }
535
536            // H9: Handle inotify events for OOM kill monitoring
537            if fd == inotify_fd && inotify_fd >= 0 {
538                let mut ino_buf = [0u8; 4096];
539                let n_read = unsafe {
540                    libc::read(
541                        inotify_fd,
542                        ino_buf.as_mut_ptr() as *mut libc::c_void,
543                        ino_buf.len(),
544                    )
545                };
546                if n_read > 0 {
547                    // K3: Clamp n_read to buffer size to prevent out-of-bounds access
548                    let n_read_clamped = std::cmp::min(n_read as usize, ino_buf.len());
549                    // Parse inotify events to find which watch descriptor fired
550                    let mut offset = 0usize;
551                    while offset + std::mem::size_of::<libc::inotify_event>() <= n_read_clamped {
552                        let event = unsafe {
553                            &*(ino_buf.as_ptr().add(offset) as *const libc::inotify_event)
554                        };
555                        if let Some(branch_id) = oom_watches.get(&event.wd) {
556                            tracing::warn!(
557                                branch = %branch_id,
558                                "cgroup memory.events modified — possible OOM kill detected for branch"
559                            );
560                        }
561                        // Advance to next event: sizeof(inotify_event) + name_len
562                        offset += std::mem::size_of::<libc::inotify_event>() + event.len as usize;
563                    }
564                }
565                continue;
566            }
567
568            // M-sc2: Skip in-flight handling for fds that are being closed
569            if closing_fds.contains(&fd) {
570                tracing::debug!(fd, "skipping notification for closing fd (M-sc2)");
571                continue;
572            }
573
574            // Find the branch for this notify fd
575            if let Some((branch_id, _)) = active_fds.get(&fd) {
576                if let Some(entry) = branches.get(branch_id) {
577                    // C1: Budget enforcement is now inside handle_notification_counted
578                    // using atomic increment-then-check (no TOCTOU race).
579                    let exec_budget =
580                        entry
581                            .profile
582                            .resource_limits
583                            .max_exec_calls
584                            .unwrap_or_else(|| {
585                                entry.profile.resource_limits.max_pids.saturating_mul(10)
586                            }) as u64; // S4: safe widening u32 → u64
587                                       // WS9: Wall-clock start for this notification (receive + policy + respond).
588                    let notif_start = std::time::Instant::now();
589                    if let Err(e) = SeccompBuilder::handle_notification_counted(
590                        fd,
591                        &entry.profile,
592                        &entry.exec_count,
593                        exec_budget,
594                        entry.credential_proxy.as_ref(),
595                        notif_start,
596                    ) {
597                        tracing::debug!(
598                            branch = %branch_id,
599                            error = %e,
600                            "seccomp notification handling error"
601                        );
602                    }
603                    let elapsed = notif_start.elapsed();
604                    if elapsed.as_millis() > 500 {
605                        tracing::warn!(
606                            branch = %branch_id,
607                            elapsed_ms = elapsed.as_millis() as u64,
608                            "WS9: slow seccomp notification handler (>500ms) — \
609                             all other agents' gated syscalls blocked during this time"
610                        );
611                    }
612                }
613            }
614        }
615    }
616}

618#[cfg(not(target_os = "linux"))]
619fn poll_loop(mut rx: tokio::sync::mpsc::Receiver<SeccompCommand>) -> Result<()> {
620    // On non-Linux, just drain commands until shutdown
621    loop {
622        match rx.blocking_recv() {
623            Some(SeccompCommand::Shutdown) | None => return Ok(()),
624            _ => {}
625        }
626    }

633/// ioctl number for SECCOMP_IOCTL_NOTIF_ADDFD (kernel 5.9+).
634///
635/// This ioctl injects a file descriptor into the notifying (agent) process's
636/// fd table, eliminating the TOCTOU window in execve handling:
637///
638/// Without ADDFD (current approach):
639///   1. Agent calls execve("/usr/bin/python3")
640///   2. seccomp USER_NOTIF fires; puzzled reads path from /proc/<pid>/mem
641///   3. Agent could modify the memory between steps 2 and 4 (TOCTOU window)
642///   4. puzzled validates path and responds with ALLOW
643///   5. Kernel resumes agent's execve with potentially modified path
644///
645/// With ADDFD (TOCTOU-safe approach):
646///   1. Agent calls execve("/usr/bin/python3")
647///   2. seccomp USER_NOTIF fires; puzzled reads path from /proc/<pid>/mem
648///   3. puzzled opens the binary file ITSELF (O_PATH)
649///   4. puzzled uses SECCOMP_IOCTL_NOTIF_ADDFD to inject the fd into agent
650///   5. puzzled responds with SECCOMP_USER_NOTIF_FLAG_CONTINUE, pointing
651///      the execve at the injected fd via /proc/<pid>/fd/<N>
652///   6. Kernel resumes agent's execve using the fd puzzled verified
653///
654/// The TOCTOU window is eliminated because puzzled opens the file itself
655/// and injects a reference to the exact file it validated.
656///
657/// Requires kernel 5.9+ (SECCOMP_IOCTL_NOTIF_ADDFD was added in v5.9).
658#[cfg_attr(not(target_os = "linux"), allow(dead_code))]
659const SECCOMP_IOCTL_NOTIF_ADDFD: u64 = 0x40182103;
660
661/// Kernel struct for SECCOMP_IOCTL_NOTIF_ADDFD (kernel 5.9+).
662///
663/// Mirrors `struct seccomp_notif_addfd` from `<linux/seccomp.h>`.
664/// Used to inject a file descriptor from puzzled's fd table into the
665/// notifying (agent) process's fd table.
666#[cfg(target_os = "linux")]
667#[repr(C)]
668#[derive(Debug)]
669struct SeccompNotifAddfd {
670    /// The notification ID (must match the pending notification).
671    id: u64,
672    /// Flags: 0 or SECCOMP_ADDFD_FLAG_SETFD.
673    flags: u32,
674    /// File descriptor in puzzled's fd table to inject.
675    srcfd: u32,
676    /// Desired fd number in agent (0 = kernel picks the next available).
677    newfd: u32,
678    /// Flags for the new fd (e.g., O_CLOEXEC).
679    newfd_flags: u32,
680}
681
682/// Inject a validated file descriptor into the agent process for TOCTOU-safe execve.
683///
684/// Reads the execve path from `/proc/<pid>/mem`, validates it against the
685/// agent's exec_allowlist, opens the binary with `O_PATH`, and uses
686/// `SECCOMP_IOCTL_NOTIF_ADDFD` to inject the fd into the agent process.
687///
688/// Returns:
689/// - `Ok(Some(fd))` — the fd number in the agent's process (ADDFD succeeded)
690/// - `Ok(None)` — kernel too old for ADDFD; caller should fall back to standard approach
691/// - `Err(msg)` — validation failed or unrecoverable error
692///
693/// Kernel version requirement: 5.9+ (the ioctl was added in commit 7cf97b12).
694/// On older kernels, fall back to the standard TOCTOU-mitigated approach
695/// H6: TOCTOU-safe variant that accepts a pre-read path string.
696///
697/// This function uses a path that has already been read from `/proc/<pid>/mem`
698/// by the caller, instead of re-reading it. This eliminates the TOCTOU window
699/// where an attacker could swap the file between the two reads.
700///
701/// The caller (handle_notification) reads the path ONCE and passes the same
702/// string to both `validate_execve_with_path()` and this function.
703#[cfg(target_os = "linux")]
704#[allow(dead_code)]
705pub fn inject_fd_for_execve_with_path(
706    notify_fd: i32,
707    notify_id: u64,
708    _pid: u32,
709    path_str: &str,
710    profile: &AgentProfile,
711) -> std::result::Result<Option<i32>, String> {
712    use std::ffi::CString;
713
714    if path_str.is_empty() {
715        return Err("execve path is empty".to_string());
716    }
717
718    // Step 1: Validate the path against exec_allowlist (using the pre-read path)
719    // H-5: Use the same glob-style matching logic as validate_execve_with_path
720    // to prevent allowlist mismatch between validation and fd injection.
721    let allowed = profile.exec_allowlist.iter().any(|pattern| {
722        if pattern.ends_with('*') {
723            // Prefix glob: /usr/bin/* matches /usr/bin/anything
724            let prefix = &pattern[..pattern.len() - 1];
725            path_str.starts_with(prefix)
726        } else {
727            // Exact match
728            path_str == *pattern
729        }
730    });
731
732    if !allowed {
733        return Err(format!(
734            "execve path '{}' not in exec_allowlist (pid {})",
735            path_str, _pid
736        ));
737    }
738
739    // P2-N1: Check exec_denylist — deny overrides allow (mirrors validate.rs logic)
740    if !profile.exec_denylist.is_empty() {
741        let matches_denylist = |p: &str| -> bool {
742            profile.exec_denylist.iter().any(|pattern| {
743                if pattern.ends_with('*') {
744                    let prefix = &pattern[..pattern.len() - 1];
745                    p.starts_with(prefix)
746                } else {
747                    p == *pattern
748                }
749            })
750        };
751        if matches_denylist(path_str) {
752            return Err(format!(
753                "execve path '{}' denied by exec_denylist (pid {})",
754                path_str, _pid
755            ));
756        }
757    }
758
759    // Step 2: Canonicalize and open the binary with O_PATH
760    let canonical = std::fs::canonicalize(path_str)
761        .map_err(|e| format!("failed to canonicalize '{}': {}", path_str, e))?;
762
763    let c_path = CString::new(canonical.as_os_str().as_encoded_bytes())
764        .map_err(|e| format!("path contains interior null byte: {}", e))?;
765
766    let binary_fd = unsafe { libc::open(c_path.as_ptr(), libc::O_PATH | libc::O_CLOEXEC) };
767    if binary_fd < 0 {
768        return Err(format!(
769            "failed to open '{}' with O_PATH: {}",
770            canonical.display(),
771            std::io::Error::last_os_error()
772        ));
773    }
774
775    // Step 3: Use SECCOMP_IOCTL_NOTIF_ADDFD to inject fd into the agent process
776    let addfd_req = SeccompNotifAddfd {
777        id: notify_id,
778        flags: 0,
779        srcfd: binary_fd as u32, // S3: safe — binary_fd >= 0 guaranteed by check above
780        newfd: 0,
781        newfd_flags: 0,
782    };
783
784    let ret = unsafe {
785        libc::ioctl(
786            notify_fd,
787            SECCOMP_IOCTL_NOTIF_ADDFD as libc::c_ulong,
788            &addfd_req as *const SeccompNotifAddfd,
789        )
790    };
791
792    // Close our copy of the O_PATH fd
793    unsafe { libc::close(binary_fd) };
794
795    if ret < 0 {
796        let err = std::io::Error::last_os_error();
797        let errno = err.raw_os_error().unwrap_or(0);
798
799        if errno == libc::ENOTTY || errno == libc::EINVAL {
800            return Ok(None);
801        }
802
803        return Err(format!(
804            "SECCOMP_IOCTL_NOTIF_ADDFD failed for pid {}: {}",
805            _pid, err
806        ));
807    }
808
809    Ok(Some(ret))
810}
