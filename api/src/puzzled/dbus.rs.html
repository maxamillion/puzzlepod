20/// A-M2: Read the SELinux context for a process from `/proc/<pid>/attr/current`.
21///
22/// Returns `None` if the file doesn't exist (SELinux not enabled), if the process
23/// has exited, or if the context is the default "unconfined" placeholder.
24#[cfg_attr(not(target_os = "linux"), allow(dead_code))]
25pub(crate) fn read_selinux_context(pid: u32) -> Option<String> {
26    let path = format!("/proc/{}/attr/current", pid);
27    match std::fs::read_to_string(&path) {
28        Ok(ctx) => {
29            let ctx = ctx.trim_end_matches('\0').trim().to_string();
30            if ctx.is_empty() || ctx == "unconfined" {
31                None
32            } else {
33                Some(ctx)
34            }
35        }
36        Err(_) => None,
37    }
38}

40/// K21/K22: Sanitize user-supplied reason strings before logging.
41/// Replaces control characters (0x00-0x1F, 0x7F) with underscores to prevent
42/// log injection attacks (e.g., fake log lines via embedded newlines).
43fn sanitize_log_reason(s: &str) -> String {
44    s.chars()
45        .map(|c| if c.is_control() { '_' } else { c })
46        .collect()

49/// M10: Simple per-UID rate limiter for branch creation.
50/// Tracks timestamps of recent requests per UID; rejects if > MAX_PER_MINUTE.
51/// A3: Bounded to MAX_TRACKED_UIDS to prevent memory exhaustion from UID cycling.
52struct RateLimiter {
53    /// Recent request timestamps per caller UID.
54    requests: HashMap<u32, Vec<std::time::Instant>>,
55}
56
57impl RateLimiter {
58    const MAX_PER_MINUTE: usize = 10;
59    /// A3: Maximum number of UIDs tracked. Prevents memory exhaustion if an attacker
60    /// cycles through many UIDs. When the limit is reached, stale UIDs (no requests
61    /// in the last 60s) are evicted first; if still full, the oldest UID is evicted.
62    const MAX_TRACKED_UIDS: usize = 4096;
63
64    fn new() -> Self {
65        Self {
66            requests: HashMap::new(),
67        }
68    }
69
70    /// Check if a request from this UID is allowed. Returns false if rate-limited.
71    fn check(&mut self, uid: u32) -> bool {
72        let now = std::time::Instant::now();
73        let cutoff = now - std::time::Duration::from_secs(60);
74
75        // A3: Evict stale UIDs before checking, to keep map bounded
76        self.requests.retain(|_, times| {
77            times.retain(|t| *t > cutoff);
78            !times.is_empty()
79        });
80
81        // A3: If still at capacity after stale eviction, evict the UID
82        // with the oldest most-recent request to make room
83        if self.requests.len() >= Self::MAX_TRACKED_UIDS && !self.requests.contains_key(&uid) {
84            if let Some((&oldest_uid, _)) = self
85                .requests
86                .iter()
87                .min_by_key(|(_, times)| times.last().copied().unwrap_or(now))
88            {
89                self.requests.remove(&oldest_uid);
90            }
91        }
92
93        let times = self.requests.entry(uid).or_default();
94        if times.len() >= Self::MAX_PER_MINUTE {
95            return false;
96        }
97        times.push(now);
98        true
99    }
100}

152    /// No sandboxed process is spawned. Call `activate_branch` to spawn one.
153    async fn create_branch(

156        #[zbus(connection)] connection: &zbus::Connection,
157        profile: &str,
158        base_path: &str,

162        // potentially sensitive command arguments into log output.
163        tracing::info!(
164            profile,

167            "CreateBranch requested"

170        // L15: Reject agent registration if daemon is not fully initialized
171        if !self.initialized.load(std::sync::atomic::Ordering::Acquire) {
172            tracing::warn!("CreateBranch rejected: daemon not fully initialized");
173            return Err(zbus::fdo::Error::Failed(
174                "daemon is not fully initialized; try again shortly".into(),
175            ));
176        }
177
178        let uid = get_caller_uid(&header, connection).await?;
179
180        // Validate D-Bus inputs before rate limiting — invalid requests
181        // should not consume rate limit quota.
182        validate_dbus_inputs(profile, base_path, command_json)?;
183
184        // M10: Rate-limit branch creation per caller UID
185        {
186            let mut limiter = self.rate_limiter.lock().unwrap_or_else(|e| e.into_inner());
187            if !limiter.check(uid) {
188                tracing::warn!(uid, "branch creation rate-limited");
189                return Err(zbus::fdo::Error::Failed(format!(
190                    "rate limited: max {} branch creates per minute per UID",
191                    RateLimiter::MAX_PER_MINUTE
192                )));
193            }
194        }
195
196        // DC2: Idempotency check
197        let idempotency_key = {
198            use sha2::{Digest, Sha256};
199            let mut hasher = Sha256::new();
200            hasher.update(uid.to_le_bytes());
201            hasher.update(b"\x00");
202            hasher.update(profile.as_bytes());
203            hasher.update(b"\x00");
204            hasher.update(base_path.as_bytes());
205            hasher.update(b"\x00");
206            hasher.update(command_json.as_bytes());
207            format!("{:x}", hasher.finalize())
208        };
209        {
210            let mut cache = self
211                .idempotency_cache
212                .lock()
213                .unwrap_or_else(|e| e.into_inner());
214            cache.retain(|_, entry| entry.created_at.elapsed() < IDEMPOTENCY_TTL);
215            if let Some(entry) = cache.get(&idempotency_key) {
216                // Verify the cached branch still exists. If it was committed
217                // or rolled back, the entry is stale and should be evicted
218                // so a new branch can be created with the same parameters.
219                let still_valid = serde_json::from_str::<serde_json::Value>(&entry.result_json)
220                    .ok()
221                    .and_then(|v| v.get("id").and_then(|id| id.as_str().map(String::from)))
222                    .and_then(|id_str| puzzled_types::BranchId::validated(id_str).ok())
223                    .map(|bid| self.manager.inspect(&bid).is_some())
224                    .unwrap_or(false);
225
226                if still_valid {
227                    tracing::info!(
228                        profile,
229                        base_path,
230                        "CreateBranch returning cached result (idempotent retry)"
231                    );
232                    return Ok(entry.result_json.clone());
233                }
234                cache.remove(&idempotency_key);
235            }
236        }
237
238        let manager = &*self.manager;
239
240        let agent_uid = if uid == 0 { 65534 } else { uid };
241
242        let info = manager
243            .create_branch(profile, std::path::Path::new(base_path), agent_uid)
244            .map_err(|e| zbus::fdo::Error::Failed(e.to_string()))?;
245
246        // Y1: Register the agent UID with the trust manager using profile-specific
247        // initial scores. PRD §4.1.9: "Callers should call this when a branch is
248        // created with a known profile to ensure the correct initial score is used."
249        // Existing UIDs are not overwritten (or_insert_with semantics).
250        {
251            let mut trust = self.trust_manager.lock().unwrap_or_else(|e| e.into_inner());
252            trust.register_uid(agent_uid, profile);
253        }
254
255        let json =
256            serde_json::to_string(&info).map_err(|e| zbus::fdo::Error::Failed(e.to_string()))?;
257
258        // L-br1: Audit event for branch creation is emitted in BranchManager::create()
259        // (branch.rs). Removed the duplicate here to avoid double-logging.
260
261        // ── Attestation bridge (§3.1) ──────────────────────────────────────
262        // Write branch_created event to persistent NDJSON audit store for
263        // Ed25519 signature and Merkle tree leaf.  BranchManager::create()
264        // logs to AuditLogger (syslog/netlink) but not AuditStore.
265        //
266        // ⚠ PODMAN MERGE CONFLICT RISK: This block touches dbus.rs which is
267        // also modified by the Podman-native architecture refactor.  Keep
268        // clearly delimited so it can be re-applied after a Podman merge.
269        {
270            let create_event = AuditEvent::BranchCreated {
271                branch_id: info.id.clone(),
272                profile: profile.to_string(),
273                uid: agent_uid,
274            };
275            let identity = puzzled_types::AgentIdentity {
276                uid: agent_uid,
277                profile: profile.to_string(),
278                selinux_context: info.selinux_context.clone(),
279                framework: None,
280            };
281            if let Err(e) = self.audit_store.lock().await.store_with_context(
282                &create_event,
283                Some(identity),
284                None,
285            ) {
286                tracing::warn!(branch_id = %info.id, error = %e, "failed to store branch_created event in audit store");
287            }
288        }
289        // ── End attestation bridge ──────────────────────────────────────────
290
291        // DC2: Store result in idempotency cache
292        {
293            let mut cache = self
294                .idempotency_cache
295                .lock()
296                .unwrap_or_else(|e| e.into_inner());
297
298            if cache.len() >= MAX_IDEMPOTENCY_ENTRIES {
299                let oldest_key = cache
300                    .iter()
301                    .min_by_key(|(_, entry)| entry.created_at)
302                    .map(|(k, _)| k.clone());
303                if let Some(key) = oldest_key {
304                    cache.remove(&key);
305                }
306            }
307
308            cache.insert(
309                idempotency_key,
310                IdempotencyCacheEntry {
311                    result_json: json.clone(),
312                    created_at: std::time::Instant::now(),
313                },
314            );
315        }
316
317        // H6: Emit D-Bus signal for branch creation
318        let branch_id_str = info.id.to_string();
319        if let Some(iface_ref) = get_signal_emitter(connection).await {
320            if let Err(e) = ManagerInterface::branch_created(
321                iface_ref.signal_emitter(),
322                &branch_id_str,
323                profile,
324            )
325            .await
326            {
327                tracing::debug!("F11: D-Bus signal emission failed: {e}");
328            }
329        }
330
331        Ok(json)
332    }

334    /// Spawn a sandboxed process inside an existing branch.
335    ///
336    /// `command_json` is a JSON array of strings (e.g., `["/usr/bin/python3", "agent.py"]`).
337    /// If empty string or `"[]"`, the child falls back to a pause() loop.

339        &self,
340        #[zbus(header)] header: zbus::message::Header<'_>,

345        validate_branch_id(branch_id)?;
346        // J21: Log length instead of verbatim command_json to avoid leaking sensitive data
347        tracing::info!(
348            branch_id,
349            command_json_len = command_json.len(),
350            "ActivateBranch requested"
351        );
352
353        if !self.initialized.load(std::sync::atomic::Ordering::Acquire) {
354            return Err(zbus::fdo::Error::Failed(
355                "daemon is not fully initialized; try again shortly".into(),
356            ));
357        }
358
359        let (uid, id) =
360            validate_and_authorize(&header, connection, branch_id, &self.manager).await?;
361

382
383        let info = self
384            .manager
385            .inspect(&id)
386            .ok_or_else(|| zbus::fdo::Error::Failed("branch not found after activation".into()))?;
387
388        let json =
389            serde_json::to_string(&info).map_err(|e| zbus::fdo::Error::Failed(e.to_string()))?;
390
391        Ok(json)
392    }

394    /// Commit an active branch, applying governance policy.
395    ///
396    /// L-db3: Returns a JSON string (serialized `CommitResult`) rather than structured
397    /// D-Bus types. This is intentional for flexibility — the JSON format allows adding
398    /// fields without breaking the D-Bus interface contract. The PRD specifies structured
399    /// D-Bus types, but JSON serialization was chosen during implementation for forward
400    /// compatibility.
401    async fn commit_branch(
402        &self,
403        #[zbus(header)] header: zbus::message::Header<'_>,
404        #[zbus(connection)] connection: &zbus::Connection,
405        branch_id: &str,
406    ) -> zbus::fdo::Result<String> {
407        validate_branch_id(branch_id)?;
408        tracing::info!(branch_id, "CommitBranch requested");
409
410        #[allow(unused_variables)]
411        let (uid, id) =
412            validate_and_authorize(&header, connection, branch_id, &self.manager).await?;
413
414        // M14: Idempotency — if branch is already committed, return cached result
415        // m4 TODO: Cache the actual CommitResult from the first successful commit
416        // so that repeated calls return the real files_committed/bytes_committed
417        // values instead of zeros. Consider storing CommitResult in BranchInfo or
418        // a separate DashMap<BranchId, CommitResult>.
419        if let Some(info) = self.manager.inspect(&id) {
420            if info.state == puzzled_types::BranchState::Committed {
421                tracing::info!(branch_id, "CommitBranch idempotent — already committed");
422                let result = puzzled_types::CommitResult {
423                    branch_id: id.clone(),
424                    policy_result: puzzled_types::PolicyDecision::Approved,
425                    files_committed: 0,
426                    bytes_committed: 0,
427                };
428                let json = serde_json::to_string(&result)
429                    .map_err(|e| zbus::fdo::Error::Failed(e.to_string()))?;
430                return Ok(json);
431            }
432        }
433
434        // M20: Look up the branch profile and owner UID for enriched signal emission
435        // (before commit changes state — inspect may return None after commit).
436        let (branch_profile, branch_uid) = self
437            .manager
438            .inspect(&id)
439            .map(|info| (info.profile.clone(), info.uid))
440            .unwrap_or_else(|| (String::new(), 0));
441
442        // M24: Compute a SHA-256 changeset hash BEFORE commit, because commit
443        // merges the OverlayFS upper layer into the base and destroys the diff.
444        // Try to load the IMA commit manifest for this branch; if available,
445        // hash the manifest JSON. Otherwise, fall back to hashing the diff changeset.
446        let changeset_hash = {
447            use sha2::{Digest, Sha256};
448            let manifest_dir = self.manager.config().branch_root.join("manifests");
449            let manifest_path = manifest_dir.join(format!("{}.manifest.yaml", branch_id));
450            let manifest_json = manifest_path
451                .exists()
452                .then(|| std::fs::read_to_string(&manifest_path).ok())
453                .flatten()
454                .and_then(|yaml_str| {
455                    let val: serde_json::Value = serde_yaml::from_str(&yaml_str).ok()?;
456                    serde_json::to_string(&val).ok()
457                });
458            // R6: Use sentinel value with branch_id on failure to prevent constant-hash collision
459            let hash_input = manifest_json.unwrap_or_else(|| {
460                self.manager
461                    .diff(&id)
462                    .ok()
463                    .and_then(|d| serde_json::to_string(&d).ok())
464                    .unwrap_or_else(|| {
465                        tracing::error!(branch_id = %id, "R6: changeset hash fallback — both manifest and diff unavailable");
466                        format!("UNAVAILABLE:{}", id)
467                    })
468            });
469            let mut hasher = Sha256::new();
470            hasher.update(hash_input.as_bytes());
471            format!("{:x}", hasher.finalize())
472        };
473
474        let result = self
475            .manager
476            .commit(&id)
477            .map_err(|e| zbus::fdo::Error::Failed(e.to_string()))?;
478
479        let json =
480            serde_json::to_string(&result).map_err(|e| zbus::fdo::Error::Failed(e.to_string()))?;
481
482        // PXH4: Emit audit event for branch commit
483        self.audit_logger.log(AuditEvent::BranchCommitted {
484            branch_id: id.clone(),
485            files: result.files_committed,
486            bytes: result.bytes_committed,
487        });
488
489        // ── Attestation bridge (§3.1) ──────────────────────────────────────
490        // Write governance event to the persistent NDJSON audit store so it
491        // gets an Ed25519 signature and Merkle tree leaf when attestation is
492        // enabled.  The AuditLogger above goes to syslog/netlink only.
493        //
494        // ⚠ PODMAN MERGE CONFLICT RISK: This block touches dbus.rs which is
495        // also modified by the Podman-native architecture refactor
496        // (podman_puzzled_architecture_prd.md).  Keep this block clearly
497        // delimited so it can be re-applied after a Podman merge.
498        {
499            let commit_event = AuditEvent::BranchCommitted {
500                branch_id: id.clone(),
501                files: result.files_committed,
502                bytes: result.bytes_committed,
503            };
504            let commit_identity = self.manager.inspect(&id).map(|info| {
505                puzzled_types::AgentIdentity {
506                    uid: info.uid,
507                    profile: info.profile.clone(),
508                    // A-M2: Read SELinux context from /proc/<pid>/attr/current.
509                    selinux_context: info.selinux_context.clone(),
510                    framework: None, // TODO: populate from agent framework metadata when available
511                }
512            });
513            if let Err(e) = self.audit_store.lock().await.store_with_context(
514                &commit_event,
515                commit_identity,
516                Some(changeset_hash.clone()),
517            ) {
518                tracing::warn!(branch_id, error = %e, "failed to store commit event in audit store");
519            }
520        }
521        // ── End attestation bridge ──────────────────────────────────────────
522
523        // ── Gap 4: Store CommitRejected/PolicyViolation in audit store ────
524        if let puzzled_types::PolicyDecision::Rejected(ref violations) = result.policy_result {
525            let reject_reason = format!("{} policy violation(s)", violations.len());
526            let reject_identity = self.manager.inspect(&id).map(|info| {
527                puzzled_types::AgentIdentity {
528                    uid: info.uid,
529                    profile: info.profile.clone(),
530                    // A-M2: Read SELinux context from /proc/<pid>/attr/current.
531                    selinux_context: info.selinux_context.clone(),
532                    framework: None, // TODO: populate from agent framework metadata when available
533                }
534            });
535            // Store CommitRejected event
536            let reject_event = AuditEvent::CommitRejected {
537                branch_id: id.clone(),
538                reason: reject_reason.clone(),
539            };
540            self.audit_logger.log(AuditEvent::CommitRejected {
541                branch_id: id.clone(),
542                reason: reject_reason.clone(),
543            });
544            if let Err(e) = self.audit_store.lock().await.store_with_context(
545                &reject_event,
546                reject_identity.clone(),
547                Some(changeset_hash.clone()),
548            ) {
549                tracing::warn!(branch_id, error = %e, "failed to store commit_rejected event in audit store");
550            }
551            // Store individual PolicyViolation events
552            for v in violations {
553                let pv_event = AuditEvent::PolicyViolation {
554                    branch_id: id.clone(),
555                    rule: v.rule.clone(),
556                    message: v.message.clone(),
557                };
558                self.audit_logger.log(AuditEvent::PolicyViolation {
559                    branch_id: id.clone(),
560                    rule: v.rule.clone(),
561                    message: v.message.clone(),
562                });
563                if let Err(e) = self.audit_store.lock().await.store_with_context(
564                    &pv_event,
565                    reject_identity.clone(),
566                    None,
567                ) {
568                    tracing::warn!(branch_id, error = %e, "failed to store policy_violation event in audit store");
569                }
570            }
571        }
572        // ── End Gap 4 ──────────────────────────────────────────────────────
573
574        // V1: Update trust score based on governance outcome.
575        // W1: Use branch_uid captured BEFORE commit — inspect may return None after.
576        // Y2: Exhaustive match — no wildcard. PolicyDecision::Error is a policy
577        // evaluation failure, NOT an approval. Map it to "commit_rejected" so the
578        // trust score reflects the failure rather than silently rewarding it.
579        let trust_event_type = match &result.policy_result {
580            puzzled_types::PolicyDecision::Approved => "commit_approved",
581            puzzled_types::PolicyDecision::Rejected(_) => "policy_violation",
582            puzzled_types::PolicyDecision::Error(_) => "commit_rejected",
583        };
584        // W9: Capture score in the same lock scope to avoid TOCTOU.
585        let (trust_transition_result, trust_score_after) = {
586            let mut trust = self.trust_manager.lock().unwrap_or_else(|e| e.into_inner());
587            let transition = trust.on_audit_event(trust_event_type, branch_uid, Some(branch_id));
588            let score = trust.get_score(branch_uid).map(|s| s.score).unwrap_or(0);
589            (transition, score)
590        };
591
592        // V2: Record governance provenance for this commit.
593        // W2: policy_version from audit_store; changeset_hash as manifest_hash.
594        {
595            let policy_version_str = {
596                let store = self.audit_store.lock().await;
597                store.policy_hash().unwrap_or("unknown").to_string()
598            };
599            let violations: Vec<String> = match &result.policy_result {
600                puzzled_types::PolicyDecision::Rejected(vs) => {
601                    vs.iter().map(|v| v.rule.clone()).collect()
602                }
603                _ => vec![],
604            };
605            let gov_result = match &result.policy_result {
606                puzzled_types::PolicyDecision::Approved => "approved",
607                puzzled_types::PolicyDecision::Rejected(_) => "rejected",
608                puzzled_types::PolicyDecision::Error(_) => "error",
609            };
610            if let Err(e) = crate::provenance::record_governance(
611                &self.provenance_store,
612                branch_id,
613                &policy_version_str,
614                gov_result,
615                &violations,
616                Some(changeset_hash.clone()),
617                &[], // change_ids — populated by file-level provenance if available
618            ) {
619                tracing::warn!(branch_id, error = %e, "V2: failed to record governance provenance");
620            }
621        }
622
623        // V3: Emit trust_transition signal if a tier boundary was crossed.
624        // W9: Use trust_score_after captured in same lock scope as on_audit_event.
625        if let Some((old_level, new_level)) = &trust_transition_result {
626            if old_level != new_level {
627                if let Some(iface_ref) = get_signal_emitter(connection).await {
628                    let ctx = iface_ref.signal_emitter();
629                    if let Err(e) = ManagerInterface::trust_transition(
630                        ctx,
631                        branch_uid,
632                        old_level.as_str(),
633                        new_level.as_str(),
634                        trust_score_after,
635                        trust_event_type,
636                    )
637                    .await
638                    {
639                        tracing::debug!("V3: trust_transition signal emission failed: {e}");
640                    }
641                }
642            }
643        }
644
645        // B2: Clean up behavioral trigger throttle entry for this branch.
646        // Prevents unbounded growth of the throttle map over time.
647        {
648            let mut throttle = self
649                .behavioral_trigger_last_emitted
650                .lock()
651                .unwrap_or_else(|e| e.into_inner());
652            throttle.remove(branch_id);
653        }
654
655        // H6: Emit D-Bus signals for commit result
656        // H-10: Detect governance review pending (Approved + files_committed==0
657        // + branch still exists in GovernanceReview state)
658        let is_governance_review = matches!(
659            &result.policy_result,
660            puzzled_types::PolicyDecision::Approved
661        ) && result.files_committed == 0
662            && self
663                .manager
664                .inspect(&id)
665                .map(|info| info.state == puzzled_types::BranchState::GovernanceReview)
666                .unwrap_or(false);
667
668        if let Some(iface_ref) = get_signal_emitter(connection).await {
669            let ctx = iface_ref.signal_emitter();
670            if is_governance_review {
671                // H-10: Emit governance_review_pending signal with diff summary
672                // Use pending_review_summary() for actual changeset counts
673                let (file_count, total_bytes) =
674                    self.manager.pending_review_summary(&id).unwrap_or((0, 0));
675                let diff_summary = serde_json::json!({
676                    "files": file_count,
677                    "bytes": total_bytes,
678                    "status": "awaiting_review"
679                })
680                .to_string();
681                if let Err(e) =
682                    ManagerInterface::governance_review_pending(ctx, branch_id, &diff_summary).await
683                {
684                    tracing::debug!("F11: D-Bus signal emission failed: {e}");
685                }
686            } else {
687                match &result.policy_result {
688                    puzzled_types::PolicyDecision::Approved => {
689                        if let Err(e) = ManagerInterface::branch_committed(
690                            ctx,
691                            branch_id,
692                            &changeset_hash,
693                            &branch_profile,
694                        )
695                        .await
696                        {
697                            tracing::debug!("F11: D-Bus signal emission failed: {e}");
698                        }
699                    }
700                    puzzled_types::PolicyDecision::Rejected(violations) => {
701                        let violations_json = serde_json::to_string(violations).unwrap_or_else(|e| {
702                            tracing::error!(error = %e, "R26: failed to serialize policy violations for signal");
703                            String::new()
704                        });
705                        let reason = format!("{} policy violation(s)", violations.len());
706                        if let Err(e) = ManagerInterface::policy_violation(
707                            ctx,
708                            branch_id,
709                            &violations_json,
710                            &changeset_hash,
711                            &reason,
712                            &branch_profile,
713                        )
714                        .await
715                        {
716                            tracing::debug!("F11: D-Bus signal emission failed: {e}");
717                        }
718                    }
719                    // Z11: Explicit handling for PolicyDecision::Error — emit
720                    // policy_violation signal with the error message so D-Bus
721                    // subscribers are aware of governance evaluation failures.
722                    puzzled_types::PolicyDecision::Error(ref err_msg) => {
723                        let reason = format!("policy evaluation error: {}", err_msg);
724                        if let Err(e) = ManagerInterface::policy_violation(
725                            ctx,
726                            branch_id,
727                            "[]",
728                            &changeset_hash,
729                            &reason,
730                            &branch_profile,
731                        )
732                        .await
733                        {
734                            tracing::debug!("Z11: D-Bus signal emission failed: {e}");
735                        }
736                    }
737                }
738            }
739        }
740
741        Ok(json)
742    }

744    /// Roll back a branch, discarding all changes.
745    ///
746    /// M27: The `reason` parameter is logged and included in the audit trail.
747    async fn rollback_branch(
748        &self,
749        #[zbus(header)] header: zbus::message::Header<'_>,
750        #[zbus(connection)] connection: &zbus::Connection,
751        branch_id: &str,
752        reason: &str,
753    ) -> zbus::fdo::Result<bool> {
754        validate_branch_id(branch_id)?;
755        // K21: Sanitize reason before logging to prevent log injection
756        let sanitized_reason = sanitize_log_reason(reason);
757        tracing::info!(branch_id, reason = %sanitized_reason, "RollbackBranch requested");
758
759        let (_uid, id) =
760            validate_and_authorize(&header, connection, branch_id, &self.manager).await?;
761
762        // M14: Idempotency — if branch is already rolled back, return success
763        if let Some(info) = self.manager.inspect(&id) {
764            if info.state == puzzled_types::BranchState::RolledBack {
765                tracing::info!(branch_id, "RollbackBranch idempotent — already rolled back");
766                return Ok(true);
767            }
768        }
769
770        let rollback_reason = if reason.is_empty() {
771            "D-Bus rollback request".to_string()
772        } else {
773            format!("D-Bus rollback request: {}", reason)
774        };
775        self.manager
776            .rollback(&rollback_reason, &id)
777            .map_err(|e| zbus::fdo::Error::Failed(e.to_string()))?;
778
779        // PXH4: Emit audit event for branch rollback
780        self.audit_logger.log(AuditEvent::BranchRolledBack {
781            branch_id: id.clone(),
782            reason: rollback_reason.clone(),
783        });
784
785        // ── Attestation bridge (§3.1) ──────────────────────────────────────
786        // Write rollback event to persistent NDJSON audit store for Ed25519
787        // signature and Merkle tree leaf.  See commit_branch for rationale.
788        //
789        // ⚠ PODMAN MERGE CONFLICT RISK: This block touches dbus.rs which is
790        // also modified by the Podman-native architecture refactor.  Keep
791        // clearly delimited so it can be re-applied after a Podman merge.
792        {
793            let rollback_event = AuditEvent::BranchRolledBack {
794                branch_id: id.clone(),
795                reason: rollback_reason.clone(),
796            };
797            let rollback_identity = self.manager.inspect(&id).map(|info| {
798                puzzled_types::AgentIdentity {
799                    uid: info.uid,
800                    profile: info.profile.clone(),
801                    // A-M2: Read SELinux context from /proc/<pid>/attr/current.
802                    selinux_context: info.selinux_context.clone(),
803                    framework: None, // TODO: populate from agent framework metadata when available
804                }
805            });
806            if let Err(e) = self.audit_store.lock().await.store_with_context(
807                &rollback_event,
808                rollback_identity,
809                None,
810            ) {
811                tracing::warn!(branch_id, error = %e, "failed to store rollback event in audit store");
812            }
813        }
814        // ── End attestation bridge ──────────────────────────────────────────
815
816        // W11: Record governance provenance for rollback decision.
817        {
818            let policy_version_str = {
819                let store = self.audit_store.lock().await;
820                store.policy_hash().unwrap_or("unknown").to_string()
821            };
822            if let Err(e) = crate::provenance::record_governance(
823                &self.provenance_store,
824                branch_id,
825                &policy_version_str,
826                "rollback",
827                std::slice::from_ref(&rollback_reason),
828                None,
829                &[],
830            ) {
831                tracing::warn!(branch_id, error = %e, "W11: failed to record rollback provenance");
832            }
833        }
834
835        // X1: Clean up provenance data for the rolled-back branch.
836        // PRD §4.3.8: "Branch rollback/cleanup removes the provenance directory."
837        // Must happen AFTER W11 recording so the rollback decision is persisted
838        // before the directory is removed.
839        if let Err(e) = self.provenance_store.cleanup_branch(branch_id) {
840            tracing::warn!(branch_id, error = %e, "X1: failed to clean up provenance data");
841        }
842
843        // B2: Clean up behavioral trigger throttle entry for this branch.
844        {
845            let mut throttle = self
846                .behavioral_trigger_last_emitted
847                .lock()
848                .unwrap_or_else(|e| e.into_inner());
849            throttle.remove(branch_id);
850        }
851
852        // H6: Emit D-Bus signal for rollback
853        // L-db1: Include reason in BranchRolledBack signal
854        if let Some(iface_ref) = get_signal_emitter(connection).await {
855            if let Err(e) = ManagerInterface::branch_rolled_back(
856                iface_ref.signal_emitter(),
857                branch_id,
858                &rollback_reason,
859            )
860            .await
861            {
862                tracing::debug!("F11: D-Bus signal emission failed: {e}");
863            }
864        }
865
866        Ok(true)
867    }

1580    /// §3.1: Verify the attestation chain for a branch.
1581    /// Returns JSON with verification results (chain length, signature validity, etc.).
1582    /// R1: Root-only — exports full audit chain data.
1583    async fn verify_attestation_chain(
1584        &self,
1585        #[zbus(header)] header: zbus::message::Header<'_>,
1586        #[zbus(connection)] connection: &zbus::Connection,
1587        branch_id: &str,
1588    ) -> zbus::fdo::Result<String> {
1589        validate_branch_id(branch_id)?;
1590        let uid = get_caller_uid(&header, connection).await?;
1591        if uid != 0 {
1592            return Err(zbus::fdo::Error::AccessDenied(
1593                "R1: only root may verify attestation chains".into(),
1594            ));
1595        }
1596        tracing::info!(branch_id = %branch_id, "§3.1: verify_attestation_chain called");
1597
1598        // Read the public key for Ed25519 signature verification
1599        let attestation_dir = &self.manager.config().attestation.attestation_dir;
1600        let pubkey_path = attestation_dir.join("public_key.hex");
1601        let verifying_key: Option<ed25519_dalek::VerifyingKey> =
1602            std::fs::read_to_string(&pubkey_path)
1603                .ok()
1604                .and_then(|hex_str| {
1605                    let hex_str = hex_str.trim();
1606                    // H44: Guard against odd-length hex strings to prevent
1607                    // panic on out-of-bounds slice in the parsing loop.
1608                    if hex_str.len() % 2 != 0 {
1609                        return None;
1610                    }
1611                    let bytes: Vec<u8> = (0..hex_str.len())
1612                        .step_by(2)
1613                        .filter_map(|i| u8::from_str_radix(&hex_str[i..i + 2], 16).ok())
1614                        .collect();
1615                    if bytes.len() != 32 {
1616                        return None;
1617                    }
1618                    let mut arr = [0u8; 32];
1619                    arr.copy_from_slice(&bytes);
1620                    ed25519_dalek::VerifyingKey::from_bytes(&arr).ok()
1621                });
1622
1623        let store = self.audit_store.lock().await;
1624        // A-I1: Query all events for this branch to verify chain continuity.
1625        // No limit is applied because verification must cover the entire chain —
1626        // a truncated query would silently miss chain breaks or invalid signatures.
1627        // This is an admin operation, not a high-frequency query.
1628        let events = store
1629            .query(Some(branch_id), None, None, None)
1630            .map_err(|e| zbus::fdo::Error::Failed(format!("querying audit events: {}", e)))?;
1631
1632        let mut chain_length: u64 = 0;
1633        let mut signatures_valid: u64 = 0;
1634        let mut signatures_invalid: u64 = 0;
1635        let mut signatures_missing: u64 = 0;
1636        let mut signatures_not_verified: u64 = 0;
1637        let mut chain_breaks: u64 = 0;
1638        let mut timestamp_violations: u64 = 0;
1639        let mut prev_record_id: Option<String> = None;
1640        let mut prev_timestamp: Option<String> = None;
1641
1642        for event in &events {
1643            if let Some(ref record_id) = event.record_id {
1644                chain_length += 1;
1645
1646                // Check parent_record_id chain continuity: each record's
1647                // parent_record_id should match the previous record's record_id
1648                if chain_length > 1 {
1649                    let expected_parent = prev_record_id.as_deref();
1650                    let actual_parent = event.parent_record_id.as_deref();
1651                    if expected_parent != actual_parent {
1652                        chain_breaks += 1;
1653                    }
1654                }
1655
1656                // J22: Use parsed DateTime comparison instead of lexicographic string
1657                // comparison, which can give wrong results for non-uniform formats.
1658                if let Some(ref prev_ts) = prev_timestamp {
1659                    match (
1660                        chrono::DateTime::parse_from_rfc3339(&event.timestamp),
1661                        chrono::DateTime::parse_from_rfc3339(prev_ts),
1662                    ) {
1663                        (Ok(curr), Ok(prev)) => {
1664                            if curr < prev {
1665                                timestamp_violations += 1;
1666                            }
1667                        }
1668                        _ => {
1669                            // If either timestamp fails to parse, count as violation
1670                            timestamp_violations += 1;
1671                        }
1672                    }
1673                }
1674                prev_timestamp = Some(event.timestamp.clone());
1675                prev_record_id = Some(record_id.clone());
1676
1677                // Actually verify Ed25519 signature
1678                if let Some(ref sig_hex) = event.signature {
1679                    if let Some(ref vk) = verifying_key {
1680                        // Reconstruct canonical attestation string
1681                        let canonical =
1682                            crate::audit_store::AuditStore::build_canonical_attestation(event);
1683                        // Decode signature hex (reject invalid hex rather than silently dropping)
1684                        let sig_bytes: Option<Vec<u8>> = if sig_hex.len() % 2 != 0 {
1685                            None
1686                        } else {
1687                            (0..sig_hex.len())
1688                                .step_by(2)
1689                                .map(|i| {
1690                                    sig_hex
1691                                        .get(i..i + 2)
1692                                        .and_then(|s| u8::from_str_radix(s, 16).ok())
1693                                })
1694                                .collect()
1695                        };
1696                        if let Some(sig_bytes) = sig_bytes.filter(|b| b.len() == 64) {
1697                            let mut sig_arr = [0u8; 64];
1698                            sig_arr.copy_from_slice(&sig_bytes);
1699                            let signature = ed25519_dalek::Signature::from_bytes(&sig_arr);
1700                            if ed25519_dalek::Verifier::verify(vk, canonical.as_bytes(), &signature)
1701                                .is_ok()
1702                            {
1703                                signatures_valid += 1;
1704                            } else {
1705                                signatures_invalid += 1;
1706                            }
1707                        } else {
1708                            // Malformed hex or wrong length
1709                            signatures_invalid += 1;
1710                        }
1711                    }
1712                    if verifying_key.is_none() {
1713                        signatures_not_verified += 1;
1714                    }
1715                } else {
1716                    signatures_missing += 1;
1717                }
1718            }
1719        }
1720
1721        Ok(serde_json::json!({
1722            "branch_id": branch_id,
1723            "chain_length": chain_length,
1724            "signatures_valid": signatures_valid,
1725            "signatures_invalid": signatures_invalid,
1726            "signatures_missing": signatures_missing,
1727            "signatures_not_verified": signatures_not_verified,
1728            "chain_breaks": chain_breaks,
1729            "timestamp_violations": timestamp_violations,
1730            "chain_intact": signatures_missing == 0 && signatures_not_verified == 0 && signatures_invalid == 0 && chain_breaks == 0 && timestamp_violations == 0 && chain_length > 0,
1731        })
1732        .to_string())
1733    }

2641    /// §3.4 G19: Unlock a passphrase-encrypted credential.
2642    ///
2643    /// Derives AES key via Argon2id from the passphrase, decrypts the credential,
2644    /// and stores the plaintext in the secure credential store. The passphrase
2645    /// is zeroized immediately after key derivation.
2646    async fn unlock_credential(
2647        &self,
2648        #[zbus(header)] header: zbus::message::Header<'_>,
2649        #[zbus(connection)] connection: &zbus::Connection,
2650        credential_name: &str,
2651        // SECURITY NOTE: The passphrase is borrowed from the D-Bus message buffer.
2652        // It cannot be explicitly zeroized because zbus does not expose the underlying
2653        // allocation. The passphrase exists in memory for the duration of the Argon2id
2654        // KDF call (~300ms at 64MiB memory cost). This is an accepted limitation of
2655        // the zbus framework. Mitigated by: puzzled process isolation (PR_SET_DUMPABLE=0,
2656        // SELinux puzzled_t, RLIMIT_CORE=0).
2657        passphrase: &str,
2658    ) -> zbus::fdo::Result<bool> {
2659        let uid = get_caller_uid(&header, connection).await?;
2660        // H-1: Access control — only root can unlock credentials, matching
2661        // store_credential and rotate_credential patterns.
2662        if uid != 0 {
2663            return Err(zbus::fdo::Error::AccessDenied(
2664                "Only root can unlock credentials".into(),
2665            ));
2666        }
2667        tracing::info!(
2668            credential_name,
2669            uid,
2670            "§3.4 G19: UnlockCredential requested (passphrase redacted)"
2671        );
2672
2673        // Verify credential store is available
2674        let _store = self.manager.credential_store().ok_or_else(|| {
2675            zbus::fdo::Error::Failed(
2676                "credential store not initialized (credentials.enabled=false)".into(),
2677            )
2678        })?;
2679
2680        // Look for the encrypted credential file
2681        let secrets_dir = std::env::var("XDG_CONFIG_HOME")
2682            .map(std::path::PathBuf::from)
2683            .or_else(|_| std::env::var("HOME").map(|h| std::path::PathBuf::from(h).join(".config")))
2684            .unwrap_or_else(|_| std::path::PathBuf::from("/etc"))
2685            .join("puzzled/secrets");
2686        let enc_path = secrets_dir.join(format!("{}.enc", credential_name));
2687
2688        if !enc_path.exists() {
2689            // H46: Generic error message — do not expose filesystem paths in D-Bus responses.
2690            // The full path is logged server-side for admin debugging.
2691            tracing::warn!(
2692                credential = %credential_name,
2693                path = %enc_path.display(),
2694                "§3.4 H46: encrypted credential file not found"
2695            );
2696            return Err(zbus::fdo::Error::Failed(format!(
2697                "encrypted credential '{}' not found",
2698                credential_name,
2699            )));
2700        }
2701
2702        // Read encrypted file
2703        let encrypted_data = std::fs::read(&enc_path).map_err(|e| {
2704            // H46: Generic error — log the actual path and error server-side only.
2705            tracing::warn!(
2706                credential = %credential_name,
2707                path = %enc_path.display(),
2708                error = %e,
2709                "§3.4 H46: failed to read encrypted credential file"
2710            );
2711            zbus::fdo::Error::Failed(format!(
2712                "failed to read encrypted credential '{}'",
2713                credential_name
2714            ))
2715        })?;
2716
2717        // Decrypt with passphrase via Argon2id
2718        let plaintext = puzzle_proxy::credential_backends::decrypt_with_passphrase(
2719            credential_name,
2720            &encrypted_data,
2721            passphrase.as_bytes(),
2722        )
2723        .map_err(|e| {
2724            zbus::fdo::Error::Failed(format!(
2725                "failed to decrypt credential '{}': {} (wrong passphrase?)",
2726                credential_name, e
2727            ))
2728        })?;
2729
2730        // §3.4 G19: Store decrypted credential in the CredentialStore so phantom
2731        // tokens can resolve against it.
2732        //
2733        // M5: If the credential already exists in the store (e.g., was previously
2734        // provisioned), preserve its scoped allowed_profiles and target_domains
2735        // rather than defaulting to wildcards. This prevents an unlock from
2736        // widening a credential's scope beyond what its CredentialSpec permits.
2737        let credential_value = String::from_utf8(plaintext.to_vec()).map_err(|_| {
2738            zbus::fdo::Error::Failed(format!(
2739                "credential '{}' contains invalid UTF-8",
2740                credential_name
2741            ))
2742        })?;
2743
2744        // Look up existing credential metadata to preserve scoping
2745        let store_read = self.manager.credential_store().ok_or_else(|| {
2746            zbus::fdo::Error::Failed(
2747                "credential store not initialized (credentials.enabled=false)".into(),
2748            )
2749        })?;
2750        let (existing_profiles, existing_domains, existing_injection) = {
2751            let guard = store_read.read().await;
2752            match guard.get(credential_name) {
2753                Some(existing) => (
2754                    existing.allowed_profiles.clone(),
2755                    existing.target_domains.clone(),
2756                    existing.injection.clone(),
2757                ),
2758                None => (
2759                    // No existing credential — use empty vecs (credential will only
2760                    // be usable after a profile provisions it with proper scoping)
2761                    vec![],
2762                    vec![],
2763                    puzzle_proxy::credentials::InjectionMethod::BearerHeader,
2764                ),
2765            }
2766        };
2767
2768        let cred = puzzle_proxy::credentials::StoredCredential {
2769            name: credential_name.to_string(),
2770            credential_type: puzzle_proxy::credentials::CredentialType::ApiKey,
2771            value: credential_value.into(),
2772            allowed_profiles: existing_profiles,
2773            target_domains: existing_domains,
2774            injection: existing_injection,
2775            expires_at: None,
2776            created_at: chrono::Utc::now().to_rfc3339(),
2777            rotated_at: None,
2778        };
2779
2780        let store = self.manager.credential_store().ok_or_else(|| {
2781            zbus::fdo::Error::Failed(
2782                "credential store not initialized (credentials.enabled=false)".into(),
2783            )
2784        })?;
2785        let mut guard = store.write().await;
2786        guard
2787            .store(cred)
2788            .map_err(|e| zbus::fdo::Error::Failed(format!("storing unlocked credential: {}", e)))?;
2789        drop(guard);
2790
2791        // Emit audit event for credential unlock
2792        self.audit_logger.log(AuditEvent::CredentialStored {
2793            credential_name: credential_name.to_string(),
2794            caller_uid: uid,
2795        });
2796
2797        tracing::info!(
2798            credential_name,
2799            "§3.4 G19: credential unlocked and stored successfully"
2800        );
2801
2802        Ok(true)
2803    }

3657/// L4: Validate that a branch_id is within length limits and contains only
3658/// allowed characters (alphanumeric, hyphens, underscores).
3659fn validate_branch_id(branch_id: &str) -> zbus::fdo::Result<()> {
3660    if branch_id.is_empty() {
3661        return Err(zbus::fdo::Error::Failed(
3662            "branch_id must not be empty".into(),
3663        ));
3664    }
3665    if branch_id.len() > MAX_BRANCH_ID_LEN {
3666        return Err(zbus::fdo::Error::Failed(format!(
3667            "branch_id exceeds maximum length of {} characters",
3668            MAX_BRANCH_ID_LEN
3669        )));
3670    }
3671    if !branch_id
3672        .chars()
3673        .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
3674    {
3675        return Err(zbus::fdo::Error::Failed(
3676            "branch_id contains invalid characters (allowed: alphanumeric, hyphen, underscore)"
3677                .into(),
3678        ));
3679    }
3680    Ok(())
3681}

3683/// Validate D-Bus inputs for path traversal, shell metacharacters, null bytes,
3684/// and length limits (L4).
3685fn validate_dbus_inputs(
3686    profile: &str,
3687    base_path: &str,
3688    command_json: &str,
3689) -> zbus::fdo::Result<()> {
3690    // L4: profile: length limit + alphanumeric + hyphens/underscores only
3691    if profile.is_empty() {
3692        return Err(zbus::fdo::Error::Failed(
3693            "profile name must not be empty".into(),
3694        ));
3695    }
3696    if profile.len() > MAX_PROFILE_NAME_LEN {
3697        return Err(zbus::fdo::Error::Failed(format!(
3698            "profile name exceeds maximum length of {} characters",
3699            MAX_PROFILE_NAME_LEN
3700        )));
3701    }
3702    if !profile
3703        .chars()
3704        .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
3705    {
3706        return Err(zbus::fdo::Error::Failed(
3707            "invalid profile name (allowed: alphanumeric, hyphen, underscore)".into(),
3708        ));
3709    }
3710    // L4: base_path: length limit, must be absolute, no null bytes, no ".." components
3711    if base_path.len() > MAX_BASE_PATH_LEN {
3712        return Err(zbus::fdo::Error::Failed(format!(
3713            "base_path exceeds maximum length of {} characters",
3714            MAX_BASE_PATH_LEN
3715        )));
3716    }
3717    if !base_path.starts_with('/') {
3718        return Err(zbus::fdo::Error::Failed(
3719            "base_path must be absolute".into(),
3720        ));
3721    }
3722    if base_path.contains('\0') || base_path.split('/').any(|c| c == "..") {
3723        return Err(zbus::fdo::Error::Failed(
3724            "base_path contains illegal components".into(),
3725        ));
3726    }
3727    // L4: command_json: length limit
3728    if command_json.len() > MAX_COMMAND_JSON_LEN {
3729        return Err(zbus::fdo::Error::Failed(format!(
3730            "command_json exceeds maximum length of {} characters",
3731            MAX_COMMAND_JSON_LEN
3732        )));
3733    }
3734    // L4: command_json: must be valid JSON array of strings (or empty)
3735    if !command_json.is_empty() {
3736        let parsed: Result<Vec<String>, _> = serde_json::from_str(command_json);
3737        if parsed.is_err() {
3738            return Err(zbus::fdo::Error::Failed(
3739                "command_json must be a valid JSON array of strings".into(),
3740            ));
3741        }
3742    }
3743    Ok(())
3744}

3822/// PM11: Minimum interval between BehavioralTrigger signals for the same branch.
3823const BEHAVIORAL_TRIGGER_THROTTLE: std::time::Duration = std::time::Duration::from_secs(10);
3824
3825/// PM11: Check whether a BehavioralTrigger signal should be emitted for this branch.
3826///
3827/// Returns `true` if at least `BEHAVIORAL_TRIGGER_THROTTLE` (10 seconds) has elapsed
3828/// since the last emission for this branch. Updates the timestamp on emission.
3829/// Callers should gate `ManagerInterface::behavioral_trigger()` behind this check
3830/// to avoid flooding D-Bus subscribers during sustained anomalous behavior.
3831///
3832/// M-db4: Also evicts entries older than 300s and caps at MAX_BEHAVIORAL_TRIGGER_ENTRIES.
3833pub fn should_emit_behavioral_trigger(
3834    last_emitted: &std::sync::Mutex<HashMap<String, std::time::Instant>>,
3835    branch_id: &str,
3836) -> bool {
3837    let now = std::time::Instant::now();
3838    let mut map = last_emitted.lock().unwrap_or_else(|e| e.into_inner());
3839
3840    // M-db4: Evict entries older than 300s to prevent unbounded growth
3841    map.retain(|_, ts| now.duration_since(*ts) < BEHAVIORAL_TRIGGER_MAX_AGE);
3842
3843    // M-db4: Cap at MAX_BEHAVIORAL_TRIGGER_ENTRIES
3844    if map.len() >= MAX_BEHAVIORAL_TRIGGER_ENTRIES {
3845        // Evict the oldest entry
3846        let oldest_key = map.iter().min_by_key(|(_, ts)| *ts).map(|(k, _)| k.clone());
3847        if let Some(key) = oldest_key {
3848            map.remove(&key);
3849        }
3850    }
3851
3852    if let Some(last) = map.get(branch_id) {
3853        if now.duration_since(*last) < BEHAVIORAL_TRIGGER_THROTTLE {
3854            tracing::debug!(
3855                branch_id,
3856                "PM11: BehavioralTrigger signal throttled (< 10s since last emission)"
3857            );
3858            return false;
3859        }
3860    }
3861    map.insert(branch_id.to_string(), now);
3862    true
3863}
