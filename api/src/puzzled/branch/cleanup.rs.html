11    /// Cleanly shut down all active branches during daemon shutdown.
12    ///
13    /// Iterates all active/frozen branches and cleans up their resources:
14    /// kills cgroup processes, unmounts OverlayFS, removes cgroup scopes,
15    /// and cleans up network resources. Transitions branches to Terminated
16    /// and emits audit events. Does NOT remove upper directories (state is
17    /// preserved for recovery on restart).
18    pub fn shutdown_all(&self) {
19        let branch_ids: Vec<BranchId> = self.branches.iter().map(|r| r.key().clone()).collect();
20        if branch_ids.is_empty() {
21            return;
22        }
23
24        tracing::info!(count = branch_ids.len(), "shutting down active branches");
25
26        for id in &branch_ids {
27            let state = match self.branches.get(id) {
28                Some(info) => info.state,
29                None => continue,
30            };
31
32            // Only shut down branches that are still alive (including Degraded/Ready)
33            if !matches!(
34                state,
35                BranchState::Active
36                    | BranchState::Ready
37                    | BranchState::Frozen
38                    | BranchState::Creating
39                    | BranchState::Degraded
40            ) {
41                continue;
42            }
43
44            // H-26: Thaw frozen or degraded-frozen branches before cleanup so cgroup.kill works
45            // S6: Log thaw failures — if thaw fails, subsequent cgroup.kill will also fail
46            #[cfg(target_os = "linux")]
47            if state == BranchState::Frozen || state == BranchState::Degraded {
48                self.thaw_cgroup(id);
49            }
50
51            self.cleanup_branch_resources(id);
52
53            // Transition state to Terminated
54            if let Some(mut info) = self.branches.get_mut(id) {
55                info.state = BranchState::Terminated;
56            }
57
58            self.audit.log(crate::audit::AuditEvent::BranchRolledBack {
59                branch_id: id.clone(),
60                reason: "daemon shutdown".to_string(),
61            });
62
63            tracing::info!(branch = %id, "branch terminated during shutdown");
64        }
65    }

67    /// Clean up all branch resources: sandbox (cgroup, overlay, pidfd, seccomp, BPF LSM,
68    /// fanotify), network setup, network journal, and conflict tracking.
69    ///
70    /// This is the single cleanup path used by FailSilent, FailOperational, and rollback
71    /// to ensure no resource leaks. Does NOT remove the upper directory or transition
72    /// branch state — those are caller responsibilities.
73    pub(super) fn cleanup_branch_resources(&self, id: &BranchId) {
74        // §3.4: Revoke phantom tokens for this branch
75        if let Some(ref ptm) = self.phantom_token_manager {
76            // Must guarantee revocation — use blocking write to avoid silently skipping
77            let mut ptm_guard = tokio::task::block_in_place(|| {
78                tokio::runtime::Handle::current().block_on(ptm.write())
79            });
80            ptm_guard.revoke_branch(id);
81            tracing::debug!(branch = %id, "§3.4: phantom tokens revoked for branch");
82        }
83
84        // Clean up sandbox resources (cgroup, overlay, pidfd, seccomp, BPF LSM, fanotify, network)
85        #[cfg(target_os = "linux")]
86        self.cleanup_sandbox_resources(id);
87
88        // Abort the proxy server task if running
89        if let Some((_, task)) = self.proxy_tasks.remove(id) {
90            task.abort();
91            tracing::debug!(branch = %id, "HTTP proxy task aborted");
92        }
93
94        // Discard network journal if present
95        if let Some((_, journal)) = self.network_journals.remove(id) {
96            journal.discard();
97        }
98
99        // C8: Cancel any active path reservation for this branch
100        // Clean up conflict tracking
101        {
102            let mut detector = unlock_poisoned(self.conflict_detector.lock());
103            detector.cancel_reservation(id);
104            detector.unregister_branch(id);
105        }
106    }

108    /// Clean up sandbox resources (seccomp, network, fanotify, BPF LSM, cgroup, overlay, pidfd).
109    ///
110    /// Called from both commit (after success) and rollback. Does NOT remove the
111    /// upper directory or transition branch state — those are caller responsibilities.
112    #[cfg(target_os = "linux")]
113    pub(super) fn cleanup_sandbox_resources(&self, id: &BranchId) {
114        if let Some((_, mut handle)) = self.sandboxes.remove(id) {
115            // Unregister seccomp notify fd
116            if handle.seccomp_notify_fd.is_some() {
117                if let Some(ref handler) = self.seccomp_handler {
118                    handler.unregister(id.clone());
119                }
120            }
121
122            // Clean up network resources
123            if let Some((_, net_setup)) = self.network_setups.remove(id) {
124                net_setup.cleanup();
125            }
126
127            // Remove fanotify trigger receiver
128            self.fanotify_triggers.remove(id);
129
130            // S17: Clean up BPF LSM rate limits — log error to detect map entry leaks
131            if let Some(ref bpf_lsm) = self.bpf_lsm {
132                if let Ok(meta) = std::fs::metadata(&handle.cgroup_path) {
133                    use std::os::unix::fs::MetadataExt;
134                    if let Err(e) = bpf_lsm.remove_cgroup(meta.ino()) {
135                        tracing::warn!(
136                            branch = %id,
137                            cgroup_ino = meta.ino(),
138                            error = %e,
139                            "S17: BPF LSM cgroup removal failed — map entry may leak"
140                        );
141                    }
142                }
143            }
144
145            // S4: Kill processes via cgroup — log error instead of silently discarding
146            if let Err(e) = crate::sandbox::cgroup::CgroupManager::remove_scope(&handle.cgroup_path)
147            {
148                tracing::error!(
149                    branch = %id,
150                    cgroup = %handle.cgroup_path.display(),
151                    error = %e,
152                    "S4: cgroup remove_scope failed — orphaned agent processes may persist"
153                );
154            }
155
156            // S5: Unmount OverlayFS — log error instead of silently discarding.
157            // A failed unmount leaves the branch filesystem visible and accessible.
158            if let Err(e) = nix::mount::umount2(&handle.merged_dir, MntFlags::MNT_DETACH) {
159                tracing::error!(
160                    branch = %id,
161                    merged = %handle.merged_dir.display(),
162                    error = %e,
163                    "S5: OverlayFS unmount failed — branch filesystem may remain accessible"
164                );
165            } else {
166                tracing::debug!(merged = %handle.merged_dir.display(), "OverlayFS unmounted");
167            }
168
169            // Close pidfd and mark it closed so Drop doesn't double-close.
170            // Double-close causes fd reuse races in parallel tests (SIGABRT).
171            if handle.pidfd >= 0 {
172                unsafe { libc::close(handle.pidfd) };
173                handle.pidfd = -1;
174            }
175
176            tracing::info!(branch = %id, "sandbox resources cleaned up");
177        }
178    }

180    /// Thaw the cgroup for a branch (best-effort, logs on failure).
181    pub(super) fn thaw_cgroup(&self, id: &BranchId) {
182        #[cfg(target_os = "linux")]
183        {
184            if let Some(handle) = self.sandboxes.get(id) {
185                if let Err(e) = crate::sandbox::cgroup::CgroupManager::thaw(&handle.cgroup_path) {
186                    tracing::warn!(branch = %id, error = %e, "failed to thaw cgroup");
187                }
188            } else if let Some(pid) = self.branches.get(id).and_then(|r| r.pid) {
189                match crate::sandbox::cgroup::cgroup_v2_fs_path_for_pid(pid) {
190                    Ok(path) => {
191                        if let Err(e) = crate::sandbox::cgroup::CgroupManager::thaw(&path) {
192                            tracing::warn!(
193                                branch = %id,
194                                pid,
195                                error = %e,
196                                "failed to thaw cgroup (resolved via PID; no sandbox handle)"
197                            );
198                        }
199                    }
200                    Err(e) => tracing::warn!(
201                        branch = %id,
202                        pid,
203                        error = %e,
204                        "failed to resolve cgroup path for thaw via PID"
205                    ),
206                }
207            }
208        }
209        let _ = id; // suppress unused warning on non-Linux
210    }

212    /// Replay the network journal for a committed branch (async, best-effort).
213    ///
214    /// Aborts the proxy server first so no new entries are written during replay.
215    pub(super) fn replay_network_journal(&self, id: &BranchId) {
216        // Stop the proxy before replaying — no new entries should arrive during replay
217        if let Some((_, task)) = self.proxy_tasks.remove(id) {
218            task.abort();
219            tracing::debug!(branch = %id, "HTTP proxy task aborted before journal replay");
220        }
221
222        if let Some((_, journal)) = self.network_journals.remove(id) {
223            let branch_id_clone = id.clone();
224            // Guard against missing Tokio runtime (e.g., in synchronous tests).
225            if let Ok(handle) = tokio::runtime::Handle::try_current() {
226                handle.spawn(async move {
227                    match journal.replay(&[]).await {
228                        Ok(count) if count > 0 => {
229                            tracing::info!(
230                                branch = %branch_id_clone,
231                                replayed = count,
232                                "network journal replayed"
233                            );
234                        }
235                        Ok(_) => {}
236                        Err(e) => {
237                            tracing::warn!(
238                                branch = %branch_id_clone,
239                                error = %e,
240                                "network journal replay failed (continuing)"
241                            );
242                        }
243                    }
244                });
245            } else {
246                tracing::debug!(
247                    branch = %branch_id_clone,
248                    "skipping network journal replay (no Tokio runtime)"
249                );
250            }
251        }
252    }
