31/// Manages the lifecycle of all active branches.

96/// BC2: RAII guard that removes a branch from `committing_branches` on drop,
97/// ensuring cleanup on all exit paths (normal return, early return, panic).

103impl Drop for CommitGuard {
104    fn drop(&mut self) {
105        let mut committing = unlock_poisoned(self.committing_branches.lock());
106        committing.remove(&self.branch_id);
107    }

270    ///
271    /// Used by EnsureBranch for idempotent branch creation.
272    ///
273    /// M5: This performs a linear scan over the DashMap. O(n) is acceptable
274    /// for Phase 1: max 64 branches (configurable), and EnsureBranch is a

292    /// Attach governance to a running container (Podman-native mode).
293    ///
294    /// Called by the OCI hook at createRuntime stage. Registers the container
295    /// PID with the branch and optionally starts BPF LSM and fanotify monitoring.
296    pub fn attach_governance(
297        &self,
298        id: &BranchId,
299        container_pid: u32,
300        _container_id: &str,
301    ) -> Result<()> {
302        // M4: Validate PID before attaching governance
303        if container_pid == 0 {
304            return Err(crate::error::PuzzledError::Branch(
305                "container_pid must be > 0".to_string(),
306            ));
307        }
308
309        // M4: Verify the PID exists via pidfd_open (race-free check).
310        // The fd is closed immediately — we only need to confirm the PID is valid.
311        #[cfg(target_os = "linux")]
312        {
313            let fd = unsafe { libc::syscall(libc::SYS_pidfd_open, container_pid, 0) };
314            if fd < 0 {
315                return Err(crate::error::PuzzledError::Branch(format!(
316                    "container_pid {} does not exist or is not accessible: {}",
317                    container_pid,
318                    std::io::Error::last_os_error()
319                )));
320            }
321            // H6: Safe cast — use i32::try_from to prevent truncation of large fd values.
322            let fd_i32 = i32::try_from(fd).map_err(|_| {
323                crate::error::PuzzledError::Branch(format!(
324                    "pidfd_open returned fd {} which overflows i32",
325                    fd
326                ))
327            })?;
328            unsafe { libc::close(fd_i32) };
329        }
330
331        let mut branch = self.branches.get_mut(id).ok_or_else(|| {
332            crate::error::PuzzledError::Branch(format!("branch {} not found", id))
333        })?;
334
335        branch.pid = Some(container_pid);
336        tracing::info!(
337            branch_id = %id,
338            container_pid,
339            "governance attached to container process"
340        );
341
342        Ok(())
343    }

355    /// Create a branch workspace: set up overlay directories and store metadata.
356    /// No process is spawned and no sandbox primitives (cgroups, namespaces,
357    /// seccomp) are applied. The branch enters `Ready` state and is writable
358    /// via `upper_dir`. Call `activate_branch()` to spawn a sandboxed process
359    /// and transition to `Active`.
360    pub fn create_branch(
361        &self,
362        profile_name: &str,
363        base_path: &std::path::Path,
364        uid: u32,
365    ) -> Result<BranchInfo> {
366        if !base_path.is_absolute() {
367            return Err(PuzzledError::Branch(format!(
368                "base_path must be absolute, got: {}",
369                base_path.display()
370            )));
371        }
372        if !base_path.exists() {
373            return Err(PuzzledError::Branch(format!(
374                "base_path does not exist: {}",
375                base_path.display()
376            )));
377        }
378        if !base_path.is_dir() {
379            return Err(PuzzledError::Branch(format!(
380                "base_path is not a directory: {}",
381                base_path.display()
382            )));
383        }
384
385        if self.branches.len() >= self.config.max_branches as usize {
386            return Err(PuzzledError::Branch(format!(
387                "maximum branches ({}) reached",
388                self.config.max_branches
389            )));
390        }
391
392        let _profile = self
393            .profile_loader
394            .get(profile_name)
395            .ok_or_else(|| PuzzledError::Profile(format!("profile '{}' not found", profile_name)))?
396            .clone();
397
398        let branch_id = BranchId::new();
399        let branch_dir = self.config.branch_root.join(branch_id.as_str());
400        let (upper_dir, work_dir, _merged_dir) =
401            crate::sandbox::overlay::OverlayMount::create_dirs(&branch_dir)?;
402
403        let info = BranchInfo {
404            id: branch_id.clone(),
405            profile: profile_name.to_string(),
406            base_path: base_path.to_path_buf(),
407            upper_dir,
408            work_dir,
409            state: BranchState::Ready,
410            created_at: Utc::now(),
411            expires_at: None,
412            pid: None,
413            uid,
414            selinux_context: None,
415        };
416
417        self.branches.insert(branch_id.clone(), info.clone());
418
419        self.audit.log(AuditEvent::BranchCreated {
420            branch_id: branch_id.clone(),
421            profile: profile_name.to_string(),
422            uid,
423        });
424
425        if let Some(m) = self.get_metrics() {
426            m.record_create(profile_name, 0.0);
427        }
428
429        tracing::info!(
430            branch = %branch_id,
431            profile = profile_name,
432            uid,
433            "branch created (workspace ready)"
434        );
435
436        Ok(info)
437    }

463    /// Kill an active agent and roll back its branch.
464    ///
465    /// On Linux, kills agent processes via cgroup before rollback.
466    pub fn kill_agent(&self, id: &BranchId) -> Result<()> {
467        let state = self
468            .branches
469            .get(id)
470            .map(|r| r.state)
471            .ok_or_else(|| PuzzledError::NotFound(format!("branch {}", id)))?;
472        if state != BranchState::Active {
473            return Err(PuzzledError::Branch(format!(
474                "branch {} is in state {}, expected Active",
475                id, state
476            )));
477        }
478        // S3: Kill via cgroup before rollback — log error instead of silently discarding
479        #[cfg(target_os = "linux")]
480        if let Some(handle) = self.sandboxes.get(id) {
481            if let Err(e) = crate::sandbox::cgroup::CgroupManager::kill(&handle.cgroup_path) {
482                tracing::error!(
483                    branch = %id,
484                    cgroup = %handle.cgroup_path.display(),
485                    error = %e,
486                    "S3: cgroup kill failed during agent termination — \
487                     agent processes may still be running"
488                );
489            }
490        }
491        self.rollback("agent killed by operator", id)
492    }

494    /// §3.3: Quarantine a branch by freezing its cgroup.
495    ///
496    /// Called when the DLP engine detects a Quarantine-level violation.
497    /// The branch processes are frozen (not killed) so the changeset can be
498    /// inspected by an operator before deciding to commit or rollback.
499    pub fn quarantine_branch(&self, id: &BranchId) -> Result<()> {
500        let state = self
501            .branches
502            .get(id)
503            .map(|r| r.state)
504            .ok_or_else(|| PuzzledError::NotFound(format!("branch {}", id)))?;
505        if state != BranchState::Active {
506            return Err(PuzzledError::Branch(format!(
507                "branch {} is in state {}, cannot quarantine",
508                id, state
509            )));
510        }
511
512        #[cfg(target_os = "linux")]
513        if let Some(handle) = self.sandboxes.get(id) {
514            crate::sandbox::cgroup::CgroupManager::freeze(&handle.cgroup_path)?;
515            tracing::warn!(
516                branch = %id,
517                "§3.3: branch quarantined via cgroup.freeze (DLP violation)"
518            );
519        }
520        #[cfg(not(target_os = "linux"))]
521        {
522            let _ = id;
523            Err(PuzzledError::Sandbox(
524                "quarantine (cgroup.freeze) is only supported on Linux".into(),
525            ))
526        }
527        #[cfg(target_os = "linux")]
528        Ok(())
529    }

542    pub fn rollback(&self, reason: &str, id: &BranchId) -> Result<()> {
543        // BC2: Prevent rollback/unregister while a commit is in progress for this branch.
544        // The commit flow uses rollback_internal() for its own rollback needs.
545        {
546            let committing = unlock_poisoned(self.committing_branches.lock());
547            if committing.contains(id) {
548                return Err(PuzzledError::Branch(format!(
549                    "branch {} is currently mid-commit, cannot rollback",
550                    id
551                )));
552            }
553        }
554
555        self.rollback_internal(reason, id)
556    }

558    /// Internal rollback that bypasses the BC2 committing_branches check.
559    /// Used by commit() when it needs to rollback on error (e.g., conflict, cgroup freeze failure).
560    pub(super) fn rollback_internal(&self, reason: &str, id: &BranchId) -> Result<()> {
561        let (upper_dir, rollback_profile, created_at) = {
562            let info = self
563                .branches
564                .get(id)
565                .ok_or_else(|| PuzzledError::NotFound(format!("branch {}", id)))?;
566            (
567                info.upper_dir.clone(),
568                info.profile.clone(),
569                info.created_at,
570            )
571        };
572
573        // Clean up all branch resources (sandbox, network, conflict tracking)
574        self.cleanup_branch_resources(id);
575
576        // Remove the upper directory (zero residue)
577        // Validate that branch_dir is under branch_root to prevent accidental deletion
578        // of unrelated directories if upper_dir is pathological.
579        // Canonicalize both paths to resolve symlinks and ".." components before comparison.
580        if upper_dir.exists() {
581            let branch_dir = upper_dir.parent().unwrap_or(&upper_dir);
582            let branch_dir_canon =
583                std::fs::canonicalize(branch_dir).unwrap_or(branch_dir.to_path_buf());
584            let root_canon = std::fs::canonicalize(&self.config.branch_root)
585                .unwrap_or(self.config.branch_root.clone());
586            if branch_dir_canon.starts_with(&root_canon) && branch_dir.exists() {
587                // L1: TOCTOU mitigation — re-verify canonicalized path still starts_with
588                // the expected parent directory immediately before deletion.
589                match std::fs::canonicalize(branch_dir) {
590                    Ok(recheck_canon) if recheck_canon.starts_with(&root_canon) => {
591                        // H1: Propagate remove_dir_all errors instead of silently ignoring them.
592                        if let Err(e) = std::fs::remove_dir_all(&recheck_canon) {
593                            tracing::error!(
594                                branch_dir = %recheck_canon.display(),
595                                error = %e,
596                                "failed to remove branch directory during rollback"
597                            );
598                            return Err(PuzzledError::Branch(format!(
599                                "failed to remove branch directory {}: {}",
600                                recheck_canon.display(),
601                                e
602                            )));
603                        }
604                    }
605                    Ok(recheck_canon) => {
606                        tracing::error!(
607                            branch_dir = %recheck_canon.display(),
608                            branch_root = %root_canon.display(),
609                            "TOCTOU: branch dir changed between checks, refusing to delete"
610                        );
611                        return Err(PuzzledError::Branch(
612                            "TOCTOU: branch directory path changed between canonicalization checks"
613                                .to_string(),
614                        ));
615                    }
616                    Err(e) => {
617                        tracing::error!(
618                            branch_dir = %branch_dir.display(),
619                            error = %e,
620                            "failed to re-canonicalize branch dir for TOCTOU check"
621                        );
622                        return Err(PuzzledError::Branch(format!(
623                            "failed to re-canonicalize branch dir: {}",
624                            e
625                        )));
626                    }
627                }
628            } else {
629                tracing::error!(
630                    branch_dir = %branch_dir.display(),
631                    branch_root = %self.config.branch_root.display(),
632                    "refusing to delete branch dir outside branch_root (path traversal blocked)"
633                );
634            }
635        }
636
637        // S18: Clean up XFS quota — log error to detect quota starvation over time
638        if let Err(e) = crate::sandbox::quota::QuotaManager::remove(&upper_dir) {
639            tracing::warn!(
640                branch = %id,
641                upper_dir = %upper_dir.display(),
642                error = %e,
643                "S18: XFS quota cleanup failed — quota reserves may leak"
644            );
645        }
646
647        self.transition(id, BranchState::RolledBack)?;
648
649        // C4: Remove terminal branch from self.branches to free the slot
650        self.branches.remove(id);
651
652        self.audit.log(AuditEvent::BranchRolledBack {
653            branch_id: id.clone(),
654            reason: reason.to_string(),
655        });
656
657        // Wire metrics: rollback and branch lifetime duration
658        if let Some(m) = self.get_metrics() {
659            m.record_rollback(&rollback_profile);
660            let lifetime =
661                // S9: precision loss irrelevant for metrics (>285M years)
662                (chrono::Utc::now() - created_at).num_milliseconds().max(0) as f64 / 1000.0;
663            m.branch_duration_seconds.observe(lifetime);
664            m.commit_outcomes
665                .get_or_create(&crate::metrics::OutcomeLabels {
666                    outcome: "rolled_back".to_string(),
667                })
668                .inc();
669        }
670
671        tracing::info!(branch = %id, reason = reason, "branch rolled back");
672        Ok(())
673    }

675    /// Transition a branch to a new state, validating the state machine.
676    pub(super) fn transition(&self, id: &BranchId, new_state: BranchState) -> Result<()> {
677        let mut info = self
678            .branches
679            .get_mut(id)
680            .ok_or_else(|| PuzzledError::NotFound(format!("branch {}", id)))?;
681
682        let valid = match (info.state, new_state) {
683            (BranchState::Creating, BranchState::Ready) => true,
684            (BranchState::Ready, BranchState::Active) => true,
685            (BranchState::Ready, BranchState::Frozen) => true, // direct-mode commit (no process to freeze)
686            (BranchState::Ready, BranchState::RolledBack) => true, // workspace cancelled before activation
687            (BranchState::Active, BranchState::Frozen) => true,
688            (BranchState::Active, BranchState::RolledBack) => true,
689            // H4: Active -> Exited (agent process exited cleanly with exit code 0)
690            (BranchState::Active, BranchState::Exited) => true,
691            // H4: Active -> Terminated (agent killed by signal or non-zero exit)
692            (BranchState::Active, BranchState::Terminated) => true,
693            (BranchState::Frozen, BranchState::Active) => true, // Recovery from FailSilent/FailOperational
694            (BranchState::Frozen, BranchState::Committing) => true,
695            (BranchState::Frozen, BranchState::RolledBack) => true,
696            (BranchState::Frozen, BranchState::Committed) => true, // empty changeset
697            // H4: Frozen -> Terminated (agent killed while frozen, e.g., OOM during commit)
698            (BranchState::Frozen, BranchState::Terminated) => true,
699            (BranchState::Committing, BranchState::Committed) => true,
700            (BranchState::Committing, BranchState::Active) => true, // M7: recovery from wal_commit failure
701            (BranchState::Committing, BranchState::Failed) => true,
702            // H4: Exited -> Frozen (freeze for commit after clean exit — agent has stopped,
703            // freeze the cgroup to safely read the diff)
704            (BranchState::Exited, BranchState::Frozen) => true,
705            // H4: Terminated -> RolledBack (cleanup after termination — discard changes)
706            (BranchState::Terminated, BranchState::RolledBack) => true,
707            // H-10: GovernanceReview -> Committed (approved by human reviewer)
708            (BranchState::GovernanceReview, BranchState::Committed) => true,
709            // H-10: GovernanceReview -> RolledBack (rejected by human reviewer)
710            (BranchState::GovernanceReview, BranchState::RolledBack) => true,
711            // H-9: Committing -> GovernanceReview (policy approved, awaiting human review)
712            (BranchState::Committing, BranchState::GovernanceReview) => true,
713            // H-26: Any -> Degraded (FailOperational/FailSilent modes)
714            (_, BranchState::Degraded) => true,
715            (_, BranchState::Failed) => true, // Any state can fail
716            _ => false,
717        };
718
719        if !valid {
720            return Err(PuzzledError::Branch(format!(
721                "invalid state transition: {} -> {}",
722                info.state, new_state
723            )));
724        }
725
726        tracing::debug!(
727            branch = %id,
728            from = %info.state,
729            to = %new_state,
730            "state transition"
731        );
732
733        info.state = new_state;
734        Ok(())
735    }

737    /// Execute a WAL-protected commit: log intent, copy files, mark complete.
738    ///
739    /// Delegates to `CommitExecutor` which handles the file operation mechanics
740    /// (symlink safety, atomic rename, metadata, crash-safe rollback).
741    pub(super) fn wal_commit(
742        &self,
743        branch_id: &BranchId,
744        changes: &[FileChange],
745        base_path: &std::path::Path,
746    ) -> Result<()> {
747        let upper_dir = self
748            .branches
749            .get(branch_id)
750            .map(|r| r.upper_dir.clone())
751            .ok_or_else(|| PuzzledError::NotFound(format!("branch {}", branch_id)))?;
752
753        let executor = crate::commit::CommitExecutor::new(&self.wal);
754        executor.execute(branch_id, changes, base_path, &upper_dir)
755    }

757    /// Recover branches from disk after daemon restart.
758    ///
759    /// Scans the branch_root for existing branch directories and the WAL
760    /// for incomplete commits that need rollback.
761    pub fn recover(&self) -> Result<()> {
762        // Recover incomplete WAL entries
763        let incomplete = self.wal.recover()?;
764
765        // L1/M6: Collect base filesystem paths from WAL operations BEFORE
766        // reverse_operations() removes the WAL files. These paths are where
767        // `.puzzled_old` orphan files are actually created during commit
768        // (via renameat2 RENAME_EXCHANGE), not in branch_root.
769        let mut all_base_paths = HashSet::new();
770
771        for branch_id in &incomplete {
772            tracing::warn!(
773                branch = %branch_id,
774                "rolling back incomplete commit from previous run"
775            );
776
777            // Read the WAL to find which operations completed and need reversal
778            let (operations, completed) = self.wal.read_operations_with_status(branch_id)?;
779
780            // L1/M6: Extract base filesystem paths from CopyFile targets before
781            // the WAL file is removed by reverse_operations().
782            let branch_base_paths = WriteAheadLog::collect_base_paths_from_wal(&operations);
783            all_base_paths.extend(branch_base_paths);
784
785            if !completed.is_empty() {
786                tracing::info!(
787                    branch = %branch_id,
788                    total_ops = operations.len(),
789                    completed_ops = completed.len(),
790                    "reversing partially completed operations using WAL backups"
791                );
792
793                // Reverse completed operations using backups
794                // H2: If recovery fails, abort branch deletion and return an error
795                // instead of continuing with a partially-reversed state.
796                if let Err(e) = self
797                    .wal
798                    .reverse_operations(branch_id, &operations, &completed)
799                {
800                    tracing::error!(
801                        branch = %branch_id,
802                        error = %e,
803                        "failed to fully reverse partial commit — aborting branch cleanup (manual intervention required)"
804                    );
805                    return Err(PuzzledError::Wal(format!(
806                        "WAL recovery failed for branch {}: {} — manual intervention required",
807                        branch_id, e
808                    )));
809                }
810            }
811
812            // Clean up the branch directory (with path traversal protection)
813            let branch_dir = self.config.branch_root.join(branch_id.as_str());
814            let mut drop_branch_entry = !branch_dir.exists();
815            if branch_dir.exists() {
816                let canonical_branch = match branch_dir.canonicalize() {
817                    Ok(p) => p,
818                    Err(e) => {
819                        tracing::error!(
820                            branch = %branch_id,
821                            error = %e,
822                            "failed to canonicalize branch directory during WAL recovery — skipping"
823                        );
824                        continue;
825                    }
826                };
827                let canonical_root = match self.config.branch_root.canonicalize() {
828                    Ok(p) => p,
829                    Err(e) => {
830                        tracing::error!(
831                            error = %e,
832                            "failed to canonicalize branch root during WAL recovery — skipping"
833                        );
834                        continue;
835                    }
836                };
837                if !canonical_branch.starts_with(&canonical_root) {
838                    tracing::error!(
839                        branch = %branch_id,
840                        branch_dir = %canonical_branch.display(),
841                        branch_root = %canonical_root.display(),
842                        "WAL recovery: branch directory escapes branch root — skipping (possible path traversal)"
843                    );
844                    continue;
845                }
846                if let Err(e) = std::fs::remove_dir_all(&branch_dir) {
847                    tracing::error!(
848                        branch = %branch_id,
849                        error = %e,
850                        "failed to remove branch directory during WAL recovery"
851                    );
852                    return Err(PuzzledError::Branch(format!(
853                        "failed to remove branch directory during WAL recovery: {}",
854                        e
855                    )));
856                }
857                drop_branch_entry = true;
858            }
859            // M15: Drop in-memory branch after WAL cleanup when state was loaded first.
860            if drop_branch_entry {
861                self.branches.remove(branch_id);
862            }
863        }
864
865        // L1: Scan branch_root for orphaned .puzzled_old files (original behavior).
866        WriteAheadLog::cleanup_orphan_puzzled_old(&self.config.branch_root);
867
868        // L1/M6: Scan base filesystem paths extracted from WAL CopyFile targets.
869        // This fixes the audit finding: previously only branch_root was scanned,
870        // but .puzzled_old files are created in base filesystem commit target
871        // directories (via renameat2 RENAME_EXCHANGE), not in branch_root.
872        if !all_base_paths.is_empty() {
873            tracing::info!(
874                base_path_count = all_base_paths.len(),
875                "L1/M6: scanning WAL-derived base filesystem paths for orphaned .puzzled_old files"
876            );
877            for base_path in &all_base_paths {
878                tracing::debug!(
879                    path = %base_path.display(),
880                    "L1/M6: cleaning orphan .puzzled_old files in base path"
881                );
882                WriteAheadLog::cleanup_orphan_puzzled_old(base_path);
883            }
884        }
885
886        // M-br4: Scan branch_root for directories not in loaded state.
887        // Orphaned branch directories (from crashes) are registered as Degraded
888        // so operators can inspect and clean them up.
889        if self.config.branch_root.exists() {
890            match std::fs::read_dir(&self.config.branch_root) {
891                Ok(entries) => {
892                    for entry in entries.flatten() {
893                        let path = entry.path();
894                        if !path.is_dir() {
895                            continue;
896                        }
897                        if let Some(name) = path.file_name().and_then(|n| n.to_str()) {
898                            // Skip puzzled's own internal subdirectories
899                            const INTERNAL_DIRS: &[&str] =
900                                &["wal", "audit", "manifests", "staging", "journal"];
901                            if INTERNAL_DIRS.contains(&name) {
902                                continue;
903                            }
904                            // S10: Use validated() to skip dirs with invalid chars
905                            let branch_id = match BranchId::validated(name.to_string()) {
906                                Ok(id) => id,
907                                Err(_) => {
908                                    tracing::warn!(
909                                        dir = name,
910                                        "S10: skipping orphaned dir with invalid branch name"
911                                    );
912                                    continue;
913                                }
914                            };
915                            if self.branches.contains_key(&branch_id) {
916                                continue; // Already loaded from state
917                            }
918                            // Skip WAL-recovered branches (already cleaned up above)
919                            if incomplete.iter().any(|b| b.as_str() == name) {
920                                continue;
921                            }
922                            // Create a Degraded BranchInfo for operator inspection
923                            let upper_dir = path.join("upper");
924                            let work_dir = path.join("work");
925                            let info = BranchInfo {
926                                id: branch_id.clone(),
927                                profile: "unknown".to_string(),
928                                base_path: PathBuf::from("/"),
929                                upper_dir,
930                                work_dir,
931                                state: BranchState::Degraded,
932                                created_at: Utc::now(),
933                                pid: None,
934                                uid: 0,
935                                expires_at: None,
936                                selinux_context: None,
937                            };
938                            self.branches.insert(branch_id.clone(), info);
939                            tracing::warn!(
940                                branch = %branch_id,
941                                path = %path.display(),
942                                "M-br4: orphaned branch directory found during recovery — registered as Degraded"
943                            );
944                        }
945                    }
946                }
947                Err(e) => {
948                    tracing::warn!(
949                        branch_root = %self.config.branch_root.display(),
950                        error = %e,
951                        "M-br4: failed to scan branch_root for orphaned branches"
952                    );
953                }
954            }
955        }
956
957        // M15: Terminate surviving agents after restore — seccomp USER_NOTIF fds are not
958        // recoverable; gated syscalls return ENOSYS permanently for pre-crash agents.
959        #[cfg(target_os = "linux")]
960        kill_surviving_agents_after_recovery(&self.branches);
961
962        self.audit.log(AuditEvent::WalRecovery {
963            // H4: Safe cast — prevent truncation if recovery count exceeds u32::MAX.
964            branches_recovered: u32::try_from(incomplete.len()).unwrap_or(u32::MAX),
965        });
966
967        if !incomplete.is_empty() {
968            tracing::info!(count = incomplete.len(), "recovered incomplete commits");
969        }
970
971        Ok(())
972    }

999    /// H-10: Approve a branch in GovernanceReview state.
1000    ///
1001    /// Retrieves the stored changeset from `pending_reviews`, calls
1002    /// `finalize_approved_commit()` to perform the WAL commit, and returns
1003    /// the `CommitResult` with actual file/byte counts.
1004    pub fn approve_branch(&self, id: &BranchId) -> Result<puzzled_types::CommitResult> {
1005        // H-1/H-2: Use committing_branches guard to prevent concurrent
1006        // approve+reject race. Only one thread can hold this for a given branch.
1007        {
1008            let mut committing = unlock_poisoned(self.committing_branches.lock());
1009            if !committing.insert(id.clone()) {
1010                return Err(PuzzledError::Branch(format!(
1011                    "concurrent operation already in progress for branch {}",
1012                    id
1013                )));
1014            }
1015        }
1016        let _commit_guard = CommitGuard {
1017            committing_branches: Arc::clone(&self.committing_branches),
1018            branch_id: id.clone(),
1019        };
1020
1021        // H-2: Atomically check state via get_mut (hold ref across check + transition)
1022        {
1023            let info = self
1024                .branches
1025                .get(id)
1026                .ok_or_else(|| PuzzledError::NotFound(format!("branch {}", id)))?;
1027
1028            if info.state != BranchState::GovernanceReview {
1029                return Err(PuzzledError::Branch(format!(
1030                    "branch {} is in state {}, expected GovernanceReview",
1031                    id, info.state
1032                )));
1033            }
1034        }
1035
1036        // Retrieve the stored changeset — must exist if we're in GovernanceReview
1037        let (_, (changes, base_path)) = self.pending_reviews.remove(id).ok_or_else(|| {
1038            PuzzledError::Branch(format!(
1039                "branch {} is in GovernanceReview but has no pending review data",
1040                id
1041            ))
1042        })?;
1043
1044        tracing::info!(
1045            branch = %id,
1046            files = changes.len(),
1047            "H-10: governance review approved, finalizing commit"
1048        );
1049
1050        // H10: Re-verify cgroup freeze before finalize — the agent may have been
1051        // thawed between the original freeze (at commit time) and this approval.
1052        // If the sandbox handle exists, re-freeze to ensure TOCTOU-safe commit.
1053        #[cfg(target_os = "linux")]
1054        if let Some(handle) = self.sandboxes.get(id) {
1055            if let Err(e) = crate::sandbox::cgroup::CgroupManager::freeze(&handle.cgroup_path) {
1056                tracing::warn!(
1057                    branch = %id,
1058                    error = %e,
1059                    "H10: failed to re-freeze cgroup before approved commit — \
1060                     proceeding with stored changeset (TOCTOU risk if agent modified files)"
1061                );
1062            }
1063        }
1064
1065        self.audit.log(AuditEvent::BranchFrozen {
1066            branch_id: id.clone(),
1067        });
1068
1069        // Finalize the commit using the stored changeset
1070        self.finalize_approved_commit(id, &changes, &base_path, PolicyDecision::Approved)
1071    }

1073    /// H-10: Reject a branch in GovernanceReview state.
1074    ///
1075    /// Removes the stored changeset from `pending_reviews`, thaws the cgroup,
1076    /// performs a full rollback (cleanup + upper dir removal), and emits an
1077    /// audit event.
1078    pub fn reject_branch(&self, id: &BranchId, reason: &str) -> Result<()> {
1079        // H-1/H-2: Use committing_branches guard to prevent concurrent
1080        // approve+reject race. Only one thread can hold this for a given branch.
1081        {
1082            let mut committing = unlock_poisoned(self.committing_branches.lock());
1083            if !committing.insert(id.clone()) {
1084                return Err(PuzzledError::Branch(format!(
1085                    "concurrent operation already in progress for branch {}",
1086                    id
1087                )));
1088            }
1089        }
1090        let _commit_guard = CommitGuard {
1091            committing_branches: Arc::clone(&self.committing_branches),
1092            branch_id: id.clone(),
1093        };
1094
1095        // H-2: Check state atomically
1096        {
1097            let info = self
1098                .branches
1099                .get(id)
1100                .ok_or_else(|| PuzzledError::NotFound(format!("branch {}", id)))?;
1101
1102            if info.state != BranchState::GovernanceReview {
1103                return Err(PuzzledError::Branch(format!(
1104                    "branch {} is in state {}, expected GovernanceReview",
1105                    id, info.state
1106                )));
1107            }
1108        }
1109
1110        // Remove pending review data (if any — may already be cleaned up by timeout)
1111        self.pending_reviews.remove(id);
1112
1113        // Cancel any path reservations held during the review period
1114        unlock_poisoned(self.conflict_detector.lock()).cancel_reservation(id);
1115
1116        // Thaw the cgroup before rollback so processes can be properly cleaned up
1117        self.thaw_cgroup(id);
1118
1119        let reject_reason = format!("governance review rejected: {}", reason);
1120
1121        tracing::info!(
1122            branch = %id,
1123            reason = %reject_reason,
1124            "H-10: governance review rejected, rolling back"
1125        );
1126
1127        self.audit.log(AuditEvent::BranchRolledBack {
1128            branch_id: id.clone(),
1129            reason: reject_reason.clone(),
1130        });
1131
1132        self.rollback_internal(&reject_reason, id)
1133    }

1135    /// Serialize branch state to disk for zero-downtime restart.
1136    ///
1137    /// Writes `state.json` atomically via temp+rename. Only serializes branch
1138    /// metadata (BranchInfo). Sandbox handles (pidfds, seccomp notify fds) cannot
1139    /// be serialized and must be re-established on restart.
1140    pub fn save_state(&self) -> Result<()> {
1141        // M-br2: Write state file to runtime_dir (ephemeral storage, e.g. /run/puzzled)
1142        // instead of branch_root (persistent storage).
1143        let runtime_dir = &self.config.runtime_dir;
1144        if !runtime_dir.exists() {
1145            std::fs::create_dir_all(runtime_dir).map_err(|e| {
1146                PuzzledError::Branch(format!(
1147                    "creating runtime_dir {}: {}",
1148                    runtime_dir.display(),
1149                    e
1150                ))
1151            })?;
1152        }
1153        let state_path = runtime_dir.join("state.json");
1154
1155        let branches: Vec<BranchInfo> = self.branches.iter().map(|r| r.value().clone()).collect();
1156        let json = serde_json::to_string_pretty(&branches)
1157            .map_err(|e| PuzzledError::Branch(format!("serializing state: {}", e)))?;
1158
1159        // L2: Use tempfile::NamedTempFile for unpredictable temp file names,
1160        // then atomically rename to the final destination.
1161        // Write to temp file and fsync to ensure data is durable before rename.
1162        // Without fsync, a crash after rename could leave a partial state file.
1163        {
1164            use std::io::Write;
1165            let named_tmp = tempfile::NamedTempFile::new_in(runtime_dir)
1166                .map_err(|e| PuzzledError::Branch(format!("creating temp state file: {}", e)))?;
1167            // Write and flush via BufWriter, then drop the writer to release the borrow.
1168            {
1169                let mut writer = std::io::BufWriter::new(named_tmp.as_file());
1170                writer
1171                    .write_all(json.as_bytes())
1172                    .map_err(|e| PuzzledError::Branch(format!("writing state: {}", e)))?;
1173                writer
1174                    .flush()
1175                    .map_err(|e| PuzzledError::Branch(format!("flushing state: {}", e)))?;
1176            }
1177            // fsync the underlying file to ensure durability before rename.
1178            named_tmp
1179                .as_file()
1180                .sync_all()
1181                .map_err(|e| PuzzledError::Branch(format!("fsync state file: {}", e)))?;
1182            named_tmp
1183                .persist(&state_path)
1184                .map_err(|e| PuzzledError::Branch(format!("persisting state file: {}", e)))?;
1185        }
1186
1187        tracing::debug!(branches = branches.len(), "state saved");
1188        Ok(())
1189    }
1190
1191    /// Load branch state from disk after daemon restart.
1192    ///
1193    /// Restores Ready/Active branches whose upper_dir still exists on disk as
1194    /// `Degraded`: seccomp notify, cgroup, and overlay enforcement are not
1195    /// re-established after restart. Does NOT restore sandbox handles.
1196    pub fn load_state(&self) -> Result<()> {
1197        // M-br2: Read state file from runtime_dir
1198        let state_path = self.config.runtime_dir.join("state.json");
1199        if !state_path.exists() {
1200            tracing::debug!("no state.json found, starting fresh");
1201            return Ok(());
1202        }
1203
1204        let json = std::fs::read_to_string(&state_path)?;
1205        let branches: Vec<BranchInfo> = serde_json::from_str(&json)
1206            .map_err(|e| PuzzledError::Branch(format!("deserializing state: {}", e)))?;
1207
1208        let mut restored = 0;
1209        for mut info in branches {
1210            if matches!(info.state, BranchState::Active | BranchState::Ready)
1211                && info.upper_dir.exists()
1212            {
1213                let previous_state = info.state;
1214                tracing::warn!(
1215                    branch = %info.id,
1216                    previous_state = %previous_state,
1217                    "restored branch was Ready/Active — kernel enforcement cannot survive daemon restart; marking Degraded"
1218                );
1219                info.state = BranchState::Degraded;
1220                self.branches.insert(info.id.clone(), info);
1221                restored += 1;
1222            }
1223        }
1224
1225        // M-br3: Delete the state file after successful load to prevent
1226        // stale state from being loaded on subsequent restarts.
1227        if let Err(e) = std::fs::remove_file(&state_path) {
1228            tracing::warn!(
1229                path = %state_path.display(),
1230                error = %e,
1231                "M-br3: failed to delete state file after load"
1232            );
1233        }
1234
1235        tracing::info!(restored, "loaded branch state from disk");
1236        Ok(())
1237    }

1239    /// Re-attach fanotify monitors to branches restored from state on daemon restart.
1240    ///
1241    /// Called after `load_state()` to restore behavioral monitoring for branches
1242    /// that survived a daemon restart. If a monitor can't be attached (e.g., the
1243    /// OverlayFS mount is gone), the branch is marked as needing a full diff.
1244    #[cfg(target_os = "linux")]
1245    pub fn reattach_monitors(&self) {
1246        let active_branches: Vec<(BranchId, std::path::PathBuf, String)> = self
1247            .branches
1248            .iter()
1249            .filter(|r| matches!(r.state, BranchState::Active | BranchState::Ready))
1250            .map(|r| {
1251                let id = r.key().clone();
1252                let merged = r
1253                    .base_path
1254                    .parent()
1255                    .unwrap_or(&r.base_path)
1256                    .join(id.as_str())
1257                    .join("merged");
1258                (id, merged, r.profile.clone())
1259            })
1260            .collect();
1261
1262        let mut attached = 0u32;
1263        for (id, merged_dir, profile_name) in &active_branches {
1264            // Skip if merged dir doesn't exist (mount is gone)
1265            if !merged_dir.exists() {
1266                tracing::warn!(
1267                    branch = %id,
1268                    merged_dir = %merged_dir.display(),
1269                    "merged directory missing — skipping fanotify reattachment"
1270                );
1271                continue;
1272            }
1273
1274            let behavioral_config = self
1275                .profile_loader
1276                .get(profile_name)
1277                .map(|p| p.behavioral.clone())
1278                .unwrap_or_default();
1279
1280            match crate::sandbox::fanotify::FanotifyMonitor::init(
1281                id.clone(),
1282                merged_dir.clone(),
1283                behavioral_config,
1284            ) {
1285                Ok(monitor) => {
1286                    let (trigger_rx, _counters, _touched, _needs_full_diff, _shutdown) =
1287                        monitor.start();
1288                    self.fanotify_triggers.insert(id.clone(), trigger_rx);
1289                    attached += 1;
1290                    tracing::info!(branch = %id, "fanotify monitor reattached after restart");
1291                }
1292                Err(e) => {
1293                    tracing::warn!(
1294                        branch = %id,
1295                        error = %e,
1296                        "failed to reattach fanotify monitor — behavioral monitoring degraded"
1297                    );
1298                }
1299            }
1300        }
1301
1302        if !active_branches.is_empty() {
1303            tracing::info!(
1304                total = active_branches.len(),
1305                attached,
1306                "fanotify monitor reattachment complete"
1307            );
1308        }
1309    }

1311    /// Enforce branch lifetime limits — roll back any Active branches that have
1312    /// exceeded the configured watchdog timeout.
1313    ///
1314    /// Called periodically from the watchdog task in main.rs.
1315    ///
1316    /// Clean up old committed conflict records to prevent unbounded memory growth.
1317    /// Delegates to ConflictDetector::cleanup_old_committed().
1318    pub fn cleanup_committed_conflicts(&self) {
1319        if let Ok(mut detector) = self.conflict_detector.lock() {
1320            detector.cleanup_old_committed(chrono::Duration::hours(1));
1321        }
1322    }
1323
1324    pub fn enforce_timeouts(&self) {
1325        let timeout_secs = self.config.watchdog_timeout_secs;
1326        if timeout_secs == 0 {
1327            return;
1328        }
1329
1330        let now = chrono::Utc::now();
1331        // H4: Check both Active and Frozen branches for timeouts. Frozen branches
1332        // can be stuck (e.g., after a FailSilent that never resumed), and should also
1333        // be subject to watchdog timeout enforcement.
1334        let expired: Vec<BranchId> = self
1335            .branches
1336            .iter()
1337            .filter(|r| {
1338                matches!(
1339                    r.state,
1340                    BranchState::Active | BranchState::Ready | BranchState::Frozen
1341                )
1342            })
1343            .filter(|r| {
1344                let age = now.signed_duration_since(r.created_at);
1345                // H3: Safe cast — prevent truncation of large u64 timeout values.
1346                age.num_seconds() > i64::try_from(timeout_secs).unwrap_or(i64::MAX)
1347            })
1348            .map(|r| r.key().clone())
1349            .collect();
1350
1351        for id in expired {
1352            // Recheck state — a commit may have started since we collected the expired list
1353            let current_state = self.branches.get(&id).map(|r| r.state);
1354            if !matches!(
1355                current_state,
1356                Some(BranchState::Active) | Some(BranchState::Ready) | Some(BranchState::Frozen)
1357            ) {
1358                continue;
1359            }
1360
1361            // If frozen, thaw before rollback so processes can be properly cleaned up
1362            #[cfg(target_os = "linux")]
1363            if current_state == Some(BranchState::Frozen) {
1364                self.thaw_cgroup(&id);
1365            }
1366
1367            tracing::warn!(
1368                branch = %id,
1369                timeout_secs,
1370                "branch exceeded watchdog timeout, rolling back"
1371            );
1372
1373            // Wire metrics: watchdog timeout
1374            if let Some(m) = self.get_metrics() {
1375                m.watchdog_timeouts_total.inc();
1376            }
1377
1378            // M8: rollback() itself emits the BranchRolledBack audit event
1379            // on success, so no separate audit log is needed here.
1380            if let Err(e) = self.rollback("watchdog timeout exceeded", &id) {
1381                tracing::error!(
1382                    branch = %id,
1383                    error = %e,
1384                    "failed to rollback timed-out branch"
1385                );
1386            }
1387        }
1388
1389        // H-10: Check GovernanceReview branches for review timeout expiry.
1390        // Uses a separate timeout (governance_review_timeout_seconds) from the
1391        // watchdog timeout, since governance reviews may legitimately take longer
1392        // than normal branch lifetimes.
1393        let review_timeout_secs = self.config.governance_review_timeout_seconds;
1394        if review_timeout_secs == 0 {
1395            return;
1396        }
1397
1398        let review_expired: Vec<BranchId> = self
1399            .branches
1400            .iter()
1401            .filter(|r| r.state == BranchState::GovernanceReview)
1402            .filter(|r| {
1403                let age = now.signed_duration_since(r.created_at);
1404                // H3: Safe cast — prevent truncation of large u64 timeout values.
1405                age.num_seconds() > i64::try_from(review_timeout_secs).unwrap_or(i64::MAX)
1406            })
1407            .map(|r| r.key().clone())
1408            .collect();
1409
1410        for id in review_expired {
1411            // Recheck state — approval may have occurred since we collected the list
1412            let current_state = self.branches.get(&id).map(|r| r.state);
1413            if current_state != Some(BranchState::GovernanceReview) {
1414                continue;
1415            }
1416
1417            tracing::warn!(
1418                branch = %id,
1419                review_timeout_secs,
1420                "H-10: governance review timeout expired, auto-rejecting"
1421            );
1422
1423            // Wire metrics: watchdog timeout (governance review variant)
1424            if let Some(m) = self.get_metrics() {
1425                m.watchdog_timeouts_total.inc();
1426            }
1427
1428            // Use reject_branch which handles pending_reviews cleanup, cgroup thaw,
1429            // conflict reservation cancellation, and rollback_internal.
1430            if let Err(e) = self.reject_branch(&id, "governance review timeout expired") {
1431                tracing::error!(
1432                    branch = %id,
1433                    error = %e,
1434                    "H-10: failed to auto-reject timed-out governance review"
1435                );
1436            }
1437        }
1438    }
1439}

1441/// Enumerate `agent-*.scope` cgroup directories under `puzzle.slice` (and `user-*.slice` children).
1442#[cfg(target_os = "linux")]
1443fn collect_puzzle_agent_scope_dirs(slice: &std::path::Path) -> Vec<std::path::PathBuf> {
1444    let mut out = Vec::new();
1445    let Ok(entries) = std::fs::read_dir(slice) else {
1446        return out;
1447    };
1448    for e in entries.flatten() {
1449        let p = e.path();
1450        if !p.is_dir() {
1451            continue;
1452        }
1453        let name = e.file_name().to_string_lossy().into_owned();
1454        if name.starts_with("user-") && name.ends_with(".slice") {
1455            if let Ok(inner) = std::fs::read_dir(&p) {
1456                for e2 in inner.flatten() {
1457                    let p2 = e2.path();
1458                    if !p2.is_dir() {
1459                        continue;
1460                    }
1461                    let n2 = e2.file_name();
1462                    let n2s = n2.to_string_lossy();
1463                    if n2s.starts_with("agent-") && n2s.ends_with(".scope") {
1464                        out.push(p2);
1465                    }
1466                }
1467            }
1468        } else if name.starts_with("agent-") && name.ends_with(".scope") {
1469            out.push(p);
1470        }
1471    }
1472    out
1473}

1475/// M15: SIGKILL surviving agent PIDs after daemon restart; scan `puzzle.slice` scopes for stragglers.
1476#[cfg(target_os = "linux")]
1477fn kill_surviving_agents_after_recovery(branches: &DashMap<BranchId, BranchInfo>) {
1478    use nix::errno::Errno;
1479    use nix::sys::signal::{kill, Signal};
1480    use nix::unistd::Pid;
1481
1482    for entry in branches.iter() {
1483        let id = entry.key().clone();
1484        let info = entry.value();
1485        if info.state != BranchState::Degraded {
1486            continue;
1487        }
1488        let Some(pid) = info.pid else {
1489            continue;
1490        };
1491        if pid == 0 {
1492            continue;
1493        }
1494        let Ok(pid_i32) = i32::try_from(pid) else {
1495            continue;
1496        };
1497        match kill(Pid::from_raw(pid_i32), Signal::SIGKILL) {
1498            Ok(()) => {
1499                tracing::warn!(
1500                    pid,
1501                    branch = %id,
1502                    "killing surviving agent process {} from degraded branch {} — seccomp notif fd lost, agent cannot be re-governed",
1503                    pid,
1504                    id
1505                );
1506            }
1507            Err(Errno::ESRCH) => {}
1508            Err(e) => {
1509                tracing::warn!(
1510                    pid,
1511                    branch = %id,
1512                    error = ?e,
1513                    "could not SIGKILL surviving agent process (may have exited)"
1514                );
1515            }
1516        }
1517    }
1518
1519    let slice = std::path::Path::new("/sys/fs/cgroup/puzzle.slice");
1520    if !slice.is_dir() {
1521        return;
1522    }
1523    for scope_path in collect_puzzle_agent_scope_dirs(slice) {
1524        let Some(scope_name) = scope_path.file_name().and_then(|n| n.to_str()) else {
1525            continue;
1526        };
1527        let Some(rest) = scope_name
1528            .strip_prefix("agent-")
1529            .and_then(|s| s.strip_suffix(".scope"))
1530        else {
1531            continue;
1532        };
1533        let bid = match BranchId::validated(rest.to_string()) {
1534            Ok(id) => id,
1535            Err(_) => continue,
1536        };
1537        let orphan_scope = match branches.get(&bid) {
1538            None => true,
1539            Some(r) => r.state == BranchState::Degraded,
1540        };
1541        if !orphan_scope {
1542            continue;
1543        }
1544        let procs_path = scope_path.join("cgroup.procs");
1545        let contents = match std::fs::read_to_string(&procs_path) {
1546            Ok(c) => c,
1547            Err(e) => {
1548                tracing::debug!(
1549                    path = %procs_path.display(),
1550                    error = %e,
1551                    "M15: could not read cgroup.procs"
1552                );
1553                continue;
1554            }
1555        };
1556        for token in contents.split_whitespace() {
1557            let Ok(pid_u) = token.parse::<u32>() else {
1558                continue;
1559            };
1560            if pid_u == 0 {
1561                continue;
1562            }
1563            let Ok(pid_i32) = i32::try_from(pid_u) else {
1564                continue;
1565            };
1566            match kill(Pid::from_raw(pid_i32), Signal::SIGKILL) {
1567                Ok(()) => {
1568                    tracing::warn!(
1569                        pid = pid_u,
1570                        cgroup = %scope_path.display(),
1571                        branch = %bid,
1572                        "killing surviving agent process {} from orphaned puzzle.slice scope — seccomp notif fd lost, agent cannot be re-governed",
1573                        pid_u
1574                    );
1575                }
1576                Err(Errno::ESRCH) => {}
1577                Err(e) => {
1578                    tracing::debug!(
1579                        pid = pid_u,
1580                        error = ?e,
1581                        "M15: SIGKILL for cgroup.procs pid failed"
1582                    );
1583                }
1584            }
1585        }
1586    }
