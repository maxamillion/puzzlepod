2//! Landlock rules JSON generation for the puzzle-init shim.
3//!
4//! Generates a JSON file consumed by the `puzzle-init` binary inside
5//! the container. The shim reads the rules, builds a Landlock ruleset,
6//! calls `landlock_restrict_self()` (irrevocable), then execs the real command.
7
8use std::path::{Path, PathBuf};
9
10use puzzled_types::AgentProfile;
11use serde::{Deserialize, Serialize};
12
13use crate::error::{PuzzledError, Result};
14
15/// Landlock rules structure consumed by the puzzle-init shim.
16#[derive(Debug, Clone, Serialize, Deserialize)]
17pub struct LandlockRules {
18    /// Landlock ABI version to request (e.g., "V4").
19    pub abi: String,
20    /// Paths allowed for read access.
21    pub read: Vec<String>,
22    /// Paths allowed for write access (implies read).
23    pub write: Vec<String>,
24    /// Paths allowed for execution.
25    pub exec: Vec<String>,
26    /// Whether to allow LANDLOCK_ACCESS_FS_REFER (cross-directory renames).
27    #[serde(default)]
28    pub allow_refer: bool,
29}
30
31/// Generate Landlock rules JSON from an agent profile.
32///
33/// The rules translate the profile's filesystem allowlists into a format
34/// that the `puzzle-init` shim can parse and apply via `landlock_restrict_self()`.
35///
36/// # Arguments
37/// * `profile` - The agent profile containing filesystem rules
38/// * `workspace_path` - The branch workspace directory (always writable)
39pub fn generate_landlock_rules(
40    profile: &AgentProfile,
41    workspace_path: &Path,
42) -> Result<LandlockRules> {
43    // J81: Filter read paths against both the general denylist AND the read_denylist.
44    let mut read_paths: Vec<String> = profile
45        .filesystem
46        .read_allowlist
47        .iter()
48        .filter(|p| !is_denylisted(p, &profile.filesystem.denylist))
49        .filter(|p| !is_denylisted(p, &profile.filesystem.read_denylist))
50        .map(|p| p.to_string_lossy().to_string())
51        .collect();
52
53    // J81: Filter write paths against both the general denylist AND the write_denylist.
54    let mut write_paths: Vec<String> = profile
55        .filesystem
56        .write_allowlist
57        .iter()
58        .filter(|p| !is_denylisted(p, &profile.filesystem.denylist))
59        .filter(|p| !is_denylisted(p, &profile.filesystem.write_denylist))
60        .map(|p| p.to_string_lossy().to_string())
61        .collect();
62
63    // Always include the workspace directory as writable
64    let workspace_str = workspace_path.to_string_lossy().to_string();
65    if !write_paths.contains(&workspace_str) {
66        write_paths.push(workspace_str);
67    }
68
69    let exec_paths: Vec<String> = profile.exec_allowlist.to_vec();
70
71    // Standard read paths that agents commonly need
72    let standard_read_paths = [
73        "/usr",
74        "/lib",
75        "/lib64",
76        "/etc/ld.so.cache",
77        "/etc/ld.so.conf",
78    ];
79    // J81: Standard read paths must also be checked against read_denylist.
80    for path in &standard_read_paths {
81        let path_str = path.to_string();
82        if !read_paths.contains(&path_str)
83            && !is_denylisted_str(path, &profile.filesystem.denylist)
84            && !is_denylisted_str(path, &profile.filesystem.read_denylist)
85        {
86            read_paths.push(path_str);
87        }
88    }
89
90    Ok(LandlockRules {
91        abi: "V4".to_string(),
92        read: read_paths,
93        write: write_paths,
94        exec: exec_paths,
95        allow_refer: profile.allow_symlinks,
96    })
97}

99/// Write Landlock rules to a JSON file.
100pub fn write_landlock_rules(rules: &LandlockRules, output_path: &Path) -> Result<PathBuf> {
101    if let Some(parent) = output_path.parent() {
102        std::fs::create_dir_all(parent).map_err(|e| {
103            PuzzledError::Sandbox(format!(
104                "creating landlock rules directory {}: {}",
105                parent.display(),
106                e
107            ))
108        })?;
109    }
110
111    let json = serde_json::to_string_pretty(rules)
112        .map_err(|e| PuzzledError::Sandbox(format!("serializing landlock rules: {}", e)))?;
113
114    std::fs::write(output_path, &json).map_err(|e| {
115        PuzzledError::Sandbox(format!(
116            "writing landlock rules to {}: {}",
117            output_path.display(),
118            e
119        ))
120    })?;
121
122    Ok(output_path.to_path_buf())
123}

125/// Check if a path is in the denylist.
126///
127/// K1: Canonicalize both the path and denylist entries before comparison
128/// to prevent symlink-based bypass (matching pattern in landlock.rs).
129/// Falls back to raw path if canonicalize fails.
130fn is_denylisted(path: &Path, denylist: &[PathBuf]) -> bool {
131    // K1: Canonicalize the path, fall back to raw path on failure
132    let canonical_path = std::fs::canonicalize(path).unwrap_or_else(|_| path.to_path_buf());
133    denylist.iter().any(|d| {
134        // K1: Canonicalize each denylist entry, fall back to raw path on failure
135        let canonical_d = std::fs::canonicalize(d).unwrap_or_else(|_| d.to_path_buf());
136        canonical_path.starts_with(&canonical_d)
137    })
138}
139
140/// Check if a string path is in the denylist.
141///
142/// K1: Canonicalize both the path and denylist entries before comparison
143/// to prevent symlink-based bypass (matching pattern in landlock.rs).
144/// Falls back to raw path if canonicalize fails.
145fn is_denylisted_str(path: &str, denylist: &[PathBuf]) -> bool {
146    let p = Path::new(path);
147    // K1: Canonicalize the path, fall back to raw path on failure
148    let canonical_p = std::fs::canonicalize(p).unwrap_or_else(|_| p.to_path_buf());
149    denylist.iter().any(|d| {
150        // K1: Canonicalize each denylist entry, fall back to raw path on failure
151        let canonical_d = std::fs::canonicalize(d).unwrap_or_else(|_| d.to_path_buf());
152        canonical_p.starts_with(&canonical_d)
153    })
154}

161    #[test]
162    fn test_generate_rules_from_restricted_profile() {
163        let profile = create_restricted_profile();
164        let rules = generate_landlock_rules(&profile, Path::new("/workspace")).unwrap();
165
166        assert_eq!(rules.abi, "V4");
167        // Workspace is always included as writable
168        assert!(rules.write.contains(&"/workspace".to_string()));
169        assert!(!rules.allow_refer);
170    }

172    #[test]
173    fn test_generate_rules_from_standard_profile() {
174        let profile = create_test_profile("standard");
175        let rules = generate_landlock_rules(&profile, Path::new("/workspace")).unwrap();
176
177        // read_allowlist paths should be included (minus denylisted)
178        assert!(rules.read.iter().any(|p| p.contains("/usr")));
179        assert!(rules.read.iter().any(|p| p.contains("/tmp")));
180        // /etc/shadow is in denylist, /root is in denylist
181        // write_allowlist should include /tmp
182        assert!(rules.write.iter().any(|p| p.contains("/tmp")));
183        // Workspace always included
184        assert!(rules.write.contains(&"/workspace".to_string()));
185    }
