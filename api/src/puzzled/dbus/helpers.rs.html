63/// A-M2: Read the SELinux context for a process from `/proc/<pid>/attr/current`.
64///
65/// Returns `None` if the file doesn't exist (SELinux not enabled), if the process
66/// has exited, or if the context is the default "unconfined" placeholder.
67#[cfg_attr(not(target_os = "linux"), allow(dead_code))]
68pub(crate) fn read_selinux_context(pid: u32) -> Option<String> {
69    let path = format!("/proc/{}/attr/current", pid);
70    match std::fs::read_to_string(&path) {
71        Ok(ctx) => {
72            let ctx = ctx.trim_end_matches('\0').trim().to_string();
73            if ctx.is_empty() || ctx == "unconfined" {
74                None
75            } else {
76                Some(ctx)
77            }
78        }
79        Err(_) => None,
80    }
81}

83/// K21/K22: Sanitize user-supplied reason strings before logging.
84/// Replaces control characters (0x00-0x1F, 0x7F) with underscores to prevent
85/// log injection attacks (e.g., fake log lines via embedded newlines).
86pub(crate) fn sanitize_log_reason(s: &str) -> String {
87    s.chars()
88        .map(|c| if c.is_control() { '_' } else { c })
89        .collect()
90}

92/// Reject the D-Bus call if the caller is not root (UID 0).
93pub(crate) fn require_root(uid: u32, operation: &str) -> zbus::fdo::Result<()> {
94    if uid != 0 {
95        tracing::warn!(uid, "{operation} rejected: caller is not root");
96        Err(zbus::fdo::Error::AccessDenied(format!(
97            "only root (UID 0) may {operation}"
98        )))
99    } else {
100        Ok(())
101    }
102}

104/// L4: Validate that a branch_id is within length limits and contains only
105/// allowed characters (alphanumeric, hyphens, underscores).
106pub(crate) fn validate_branch_id(branch_id: &str) -> zbus::fdo::Result<()> {
107    if branch_id.is_empty() {
108        return Err(zbus::fdo::Error::Failed(
109            "branch_id must not be empty".into(),
110        ));
111    }
112    if branch_id.len() > MAX_BRANCH_ID_LEN {
113        return Err(zbus::fdo::Error::Failed(format!(
114            "branch_id exceeds maximum length of {} characters",
115            MAX_BRANCH_ID_LEN
116        )));
117    }
118    if !branch_id
119        .chars()
120        .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
121    {
122        return Err(zbus::fdo::Error::Failed(
123            "branch_id contains invalid characters (allowed: alphanumeric, hyphen, underscore)"
124                .into(),
125        ));
126    }
127    Ok(())
128}

130/// Validate D-Bus inputs for path traversal, shell metacharacters, null bytes,
131/// and length limits (L4).
132pub(crate) fn validate_dbus_inputs(
133    profile: &str,
134    base_path: &str,
135    command_json: &str,
136) -> zbus::fdo::Result<()> {
137    // L4: profile: length limit + alphanumeric + hyphens/underscores only
138    if profile.is_empty() {
139        return Err(zbus::fdo::Error::Failed(
140            "profile name must not be empty".into(),
141        ));
142    }
143    if profile.len() > MAX_PROFILE_NAME_LEN {
144        return Err(zbus::fdo::Error::Failed(format!(
145            "profile name exceeds maximum length of {} characters",
146            MAX_PROFILE_NAME_LEN
147        )));
148    }
149    if !profile
150        .chars()
151        .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
152    {
153        return Err(zbus::fdo::Error::Failed(
154            "invalid profile name (allowed: alphanumeric, hyphen, underscore)".into(),
155        ));
156    }
157    // L4: base_path: length limit, must be absolute, no null bytes, no ".." components
158    if base_path.len() > MAX_BASE_PATH_LEN {
159        return Err(zbus::fdo::Error::Failed(format!(
160            "base_path exceeds maximum length of {} characters",
161            MAX_BASE_PATH_LEN
162        )));
163    }
164    if !base_path.starts_with('/') {
165        return Err(zbus::fdo::Error::Failed(
166            "base_path must be absolute".into(),
167        ));
168    }
169    if base_path.contains('\0') || base_path.split('/').any(|c| c == "..") {
170        return Err(zbus::fdo::Error::Failed(
171            "base_path contains illegal components".into(),
172        ));
173    }
174    // L4: command_json: length limit
175    if command_json.len() > MAX_COMMAND_JSON_LEN {
176        return Err(zbus::fdo::Error::Failed(format!(
177            "command_json exceeds maximum length of {} characters",
178            MAX_COMMAND_JSON_LEN
179        )));
180    }
181    // L4: command_json: must be valid JSON array of strings (or empty)
182    if !command_json.is_empty() {
183        let parsed: Result<Vec<String>, _> = serde_json::from_str(command_json);
184        if parsed.is_err() {
185            return Err(zbus::fdo::Error::Failed(
186                "command_json must be a valid JSON array of strings".into(),
187            ));
188        }
189    }
190    Ok(())
191}

193/// Extract the caller UID from D-Bus peer credentials.
194pub(crate) async fn get_caller_uid(
195    header: &zbus::message::Header<'_>,
196    connection: &zbus::Connection,
197) -> zbus::fdo::Result<u32> {
198    if let Some(sender) = header.sender() {
199        let dbus_proxy = zbus::fdo::DBusProxy::new(connection)
200            .await
201            .map_err(|e| zbus::fdo::Error::Failed(format!("D-Bus proxy: {}", e)))?;
202        let bus_name: zbus::names::BusName = sender.clone().into();
203        dbus_proxy
204            .get_connection_unix_user(bus_name)
205            .await
206            .map_err(|e| zbus::fdo::Error::Failed(format!("getting caller UID: {}", e)))
207    } else {
208        Err(zbus::fdo::Error::Failed(
209            "cannot determine caller identity".into(),
210        ))
211    }
212}

214/// Check that a caller has access to a branch (must be owner or root).
215/// H-30: Returns error when branch is not found instead of silently allowing access.
216pub(crate) fn check_branch_access(
217    caller_uid: u32,
218    manager: &BranchManager,
219    id: &BranchId,
220) -> zbus::fdo::Result<()> {
221    if caller_uid == 0 {
222        return Ok(());
223    }
224    if let Some(info) = manager.inspect(id) {
225        if info.uid == caller_uid {
226            return Ok(());
227        }
228        Err(zbus::fdo::Error::AccessDenied(
229            "you do not own this branch".into(),
230        ))
231    } else {
232        // H-30: Branch not found — fail closed instead of allowing access
233        Err(zbus::fdo::Error::Failed("branch not found".into()))
234    }
235}

237/// Validate a branch ID and authorize the caller.
238#[allow(dead_code)] // Used by zbus interface methods; may appear unused on some targets
239pub(crate) async fn validate_and_authorize(
240    header: &zbus::message::Header<'_>,
241    connection: &zbus::Connection,
242    branch_id: &str,
243    manager: &BranchManager,
244) -> zbus::fdo::Result<(u32, BranchId)> {
245    let uid = get_caller_uid(header, connection).await?;
246    let id = BranchId::validated(branch_id.to_string())
247        .map_err(|e| zbus::fdo::Error::Failed(format!("invalid branch_id: {}", e)))?;
248    check_branch_access(uid, manager, &id)?;
249    Ok((uid, id))
250}

252/// PM11: Check whether a BehavioralTrigger signal should be emitted for this branch.
253///
254/// Returns `true` if at least `BEHAVIORAL_TRIGGER_THROTTLE` (10 seconds) has elapsed
255/// since the last emission for this branch. Updates the timestamp on emission.
256/// Callers should gate `ManagerInterface::behavioral_trigger()` behind this check
257/// to avoid flooding D-Bus subscribers during sustained anomalous behavior.
258///
259/// M-db4: Also evicts entries older than 300s and caps at MAX_BEHAVIORAL_TRIGGER_ENTRIES.
260#[allow(dead_code)] // Public API for behavioral trigger throttling; used by tests in dbus.rs
261pub fn should_emit_behavioral_trigger(
262    last_emitted: &std::sync::Mutex<HashMap<String, std::time::Instant>>,
263    branch_id: &str,
264) -> bool {
265    let now = std::time::Instant::now();
266    let mut map = unlock_poisoned(last_emitted.lock());
267
268    // M-db4: Evict entries older than 300s to prevent unbounded growth
269    map.retain(|_, ts| now.duration_since(*ts) < BEHAVIORAL_TRIGGER_MAX_AGE);
270
271    // M-db4: Cap at MAX_BEHAVIORAL_TRIGGER_ENTRIES
272    if map.len() >= MAX_BEHAVIORAL_TRIGGER_ENTRIES {
273        // Evict the oldest entry
274        let oldest_key = map.iter().min_by_key(|(_, ts)| *ts).map(|(k, _)| k.clone());
275        if let Some(key) = oldest_key {
276            map.remove(&key);
277        }
278    }
279
280    if let Some(last) = map.get(branch_id) {
281        if now.duration_since(*last) < BEHAVIORAL_TRIGGER_THROTTLE {
282            tracing::debug!(
283                branch_id,
284                "PM11: BehavioralTrigger signal throttled (< 10s since last emission)"
285            );
286            return false;
287        }
288    }
289    map.insert(branch_id.to_string(), now);
290    true
291}

297/// M10: Simple per-UID rate limiter for branch creation.
298/// Tracks timestamps of recent requests per UID; rejects if > MAX_PER_MINUTE.
299/// A3: Bounded to MAX_TRACKED_UIDS to prevent memory exhaustion from UID cycling.
300pub(crate) struct RateLimiter {
301    /// Recent request timestamps per caller UID.
302    pub(crate) requests: HashMap<u32, Vec<std::time::Instant>>,
303}
304
305impl RateLimiter {
306    pub(crate) const MAX_PER_MINUTE: usize = 10;
307    /// A3: Maximum number of UIDs tracked. Prevents memory exhaustion if an attacker
308    /// cycles through many UIDs. When the limit is reached, stale UIDs (no requests
309    /// in the last 60s) are evicted first; if still full, the oldest UID is evicted.
310    pub(crate) const MAX_TRACKED_UIDS: usize = 4096;
311
312    pub(crate) fn new() -> Self {
313        Self {
314            requests: HashMap::new(),
315        }
316    }
317
318    /// Check if a request from this UID is allowed. Returns false if rate-limited.
319    pub(crate) fn check(&mut self, uid: u32) -> bool {
320        let now = std::time::Instant::now();
321        let cutoff = now - std::time::Duration::from_secs(60);
322
323        // A3: Evict stale UIDs before checking, to keep map bounded
324        self.requests.retain(|_, times| {
325            times.retain(|t| *t > cutoff);
326            !times.is_empty()
327        });
328
329        // A3: If still at capacity after stale eviction, evict the UID
330        // with the oldest most-recent request to make room
331        if self.requests.len() >= Self::MAX_TRACKED_UIDS && !self.requests.contains_key(&uid) {
332            if let Some((&oldest_uid, _)) = self
333                .requests
334                .iter()
335                .min_by_key(|(_, times)| times.last().copied().unwrap_or(now))
336            {
337                self.requests.remove(&oldest_uid);
338            }
339        }
340
341        let times = self.requests.entry(uid).or_default();
342        if times.len() >= Self::MAX_PER_MINUTE {
343            return false;
344        }
345        times.push(now);
346        true
347    }
348}

357/// Shared daemon state grouped into a single context struct.
358/// All fields are `Arc`-wrapped, so `Clone` is cheap (reference-count bumps only).
359#[derive(Clone)]
360pub(crate) struct DaemonServices {
361    pub(crate) manager: Arc<BranchManager>,
362    pub(crate) audit_store: Arc<tokio::sync::Mutex<AuditStore>>,
363    pub(crate) rate_limiter: Arc<std::sync::Mutex<RateLimiter>>,
364    /// L15: Tracks whether the daemon is fully initialized.
365    /// Set to true after all subsystems are ready.
366    pub(crate) initialized: Arc<std::sync::atomic::AtomicBool>,
367    /// DC2: Idempotency cache for CreateBranch. Key is "{profile}:{base_path}:{command_json}"
368    /// from the same caller UID. Prevents duplicate branch creation on D-Bus retries.
369    pub(crate) idempotency_cache: Arc<std::sync::Mutex<HashMap<String, IdempotencyCacheEntry>>>,
370    /// PXH4: Audit logger for emitting audit events on D-Bus operations.
371    pub(crate) audit_logger: Arc<AuditLogger>,
372    /// PM11: Tracks last BehavioralTrigger signal emission time per branch.
373    /// Rate-limited to at most 1 signal per branch per 10 seconds to prevent
374    /// flooding D-Bus subscribers during sustained anomalous behavior.
375    pub(crate) behavioral_trigger_last_emitted:
376        Arc<std::sync::Mutex<HashMap<String, std::time::Instant>>>,
377    /// §4.1: Trust manager for per-UID trust scoring and behavioral baselines.
378    pub(crate) trust_manager: Arc<std::sync::Mutex<TrustManager>>,
379    /// §4.3: Provenance store for per-branch causal chain records.
380    pub(crate) provenance_store: Arc<ProvenanceStore>,
381    /// §4.5: Identity manager for JWT-SVID issuance (requires ima feature).
382    #[cfg(feature = "ima")]
383    pub(crate) identity_manager: Arc<crate::identity::IdentityManager>,
384}

386impl DaemonServices {
387    /// Store an audit event in the persistent NDJSON audit store (attestation bridge §3.1).
388    ///
389    /// Resolves agent identity from the branch manager and writes the event with
390    /// optional changeset hash for Ed25519 signature and Merkle tree leaf.
391    pub(crate) async fn store_audit_event(
392        &self,
393        event: &AuditEvent,
394        branch_id: &BranchId,
395        changeset_hash: Option<String>,
396    ) {
397        let identity = self
398            .manager
399            .inspect(branch_id)
400            .map(|info| info.agent_identity());
401        if let Err(e) =
402            self.audit_store
403                .lock()
404                .await
405                .store_with_context(event, identity, changeset_hash)
406        {
407            tracing::warn!(
408                branch_id = %branch_id,
409                error = %e,
410                "failed to store event in audit store"
411            );
412        }
413    }
414
415    /// Record a governance provenance entry, fetching policy_hash from the audit store.
416    pub(crate) async fn record_governance_provenance(
417        &self,
418        branch_id: &str,
419        result: &str,
420        violations: &[String],
421        manifest_hash: Option<String>,
422    ) {
423        let policy_version_str = {
424            let store = self.audit_store.lock().await;
425            store.policy_hash().unwrap_or("unknown").to_string()
426        };
427        if let Err(e) = crate::provenance::record_governance(
428            &self.provenance_store,
429            branch_id,
430            &policy_version_str,
431            result,
432            violations,
433            manifest_hash,
434            &[],
435        ) {
436            tracing::warn!(
437                branch_id,
438                error = %e,
439                "failed to record governance provenance"
440            );
441        }
442    }
443
444    /// Remove the behavioral trigger throttle entry for a terminated branch.
445    pub(crate) fn cleanup_behavioral_throttle(&self, branch_id: &str) {
446        unlock_poisoned(self.behavioral_trigger_last_emitted.lock()).remove(branch_id);
447    }
448
449    /// Update trust score and return transition info for optional signal emission.
450    pub(crate) fn update_trust_score(
451        &self,
452        event_type: &str,
453        branch_uid: u32,
454        branch_id: &str,
455    ) -> (Option<(TrustLevel, TrustLevel)>, u32) {
456        let mut trust = unlock_poisoned(self.trust_manager.lock());
457        let transition = trust.on_audit_event(event_type, branch_uid, Some(branch_id));
458        let score = trust.get_score(branch_uid).map(|s| s.score).unwrap_or(0);
459        (transition, score)
460    }
