6/// Landlock ruleset builder — creates an irrevocable filesystem ACL
7/// from an agent profile.
8///
9/// Once applied via `landlock_restrict_self()`, the ruleset:
10/// - Survives puzzled crash (kernel-enforced)
11/// - Cannot be removed by the agent process
12/// - Is inherited by all child processes
13pub struct LandlockBuilder;
14
15/// Log conflicts where a read allowlist path prefixes a read denylist path.
16///
17/// Landlock is allowlist-based: granting read on a directory allows the whole
18/// subtree. A `read_denylist` entry under that tree is ineffective until the
19/// path exists and canonicalizes; operators need an explicit ERROR when the
20/// profile pairs a broad allow with a narrower deny.
21#[cfg(target_os = "linux")]
22fn log_read_allowlist_parent_of_read_denylist_conflicts(
23    read_allowlist: &[std::path::PathBuf],
24    read_denylist: &[std::path::PathBuf],
25) {
26    for allow in read_allowlist {
27        let allow = allow.as_path();
28        for deny in read_denylist {
29            let deny = deny.as_path();
30            if deny.starts_with(allow) && deny != allow {
31                tracing::error!(
32                    read_allowlist = %allow.display(),
33                    read_denylist = %deny.display(),
34                    "read_allowlist path is a parent of read_denylist path; Landlock allows the \
35                     entire parent subtree, so read_denylist cannot exclude a descendant that is \
36                     not yet present or not canonicalized"
37                );
38            }
39        }
40    }
41}
42
43impl LandlockBuilder {
44    /// Build and apply a Landlock ruleset from an agent profile.
45    ///
46    /// Translates profile filesystem rules into Landlock access rights:
47    /// - read_allowlist -> ReadFile, ReadDir access rules
48    /// - write_allowlist + branch merged dir -> full write access rules
49    /// - denylist -> excluded from ruleset (Landlock is allowlist-based)
50    /// - read_denylist -> paths excluded from read rules (M-ll2)
51    /// - write_denylist -> paths excluded from write rules (M-ll2)
52    ///
53    /// `extra_write_paths` should include the branch merged directory.
54    /// `proxy_port` is the TCP port of the governance proxy that agents are
55    /// allowed to connect to. Defaults to 3128 if `None`.
56    #[cfg(target_os = "linux")]
57    pub fn apply(
58        profile: &AgentProfile,
59        extra_write_paths: &[std::path::PathBuf],
60        proxy_port: Option<u16>,
61    ) -> Result<()> {
62        use landlock::{
63            Access, AccessFs, LandlockStatus, PathBeneath, PathFd, Ruleset, RulesetAttr,
64            RulesetCreatedAttr, RulesetStatus, ABI,
65        };
66
67        // m18: Request ABI V5 (includes IOCTL_DEV), falling back to V4 (6.7+),
68        // V3 (6.2+), V2 (5.19+), V1 (5.13+). The landlock crate's best-effort
69        // negotiation clamps to the highest kernel-supported ABI automatically.
70        let abi = ABI::V5;
71        tracing::info!(
72            requested_abi = "V5",
73            "Landlock ABI negotiation (kernel will clamp to highest supported)"
74        );
75
76        let read_access = AccessFs::from_read(abi);
77        let write_access = AccessFs::from_all(abi);
78
79        let mut ruleset = Ruleset::default()
80            .handle_access(write_access)
81            .map_err(|e| {
82                crate::error::PuzzledError::Sandbox(format!("creating Landlock ruleset: {}", e))
83            })?
84            .create()
85            .map_err(|e| {
86                crate::error::PuzzledError::Sandbox(format!("creating Landlock ruleset: {}", e))
87            })?;
88
89        // §3.4/M-cred1: Credential store paths are ALWAYS denied to agent processes.
90        // Agents must never read the credential store directly — they can only
91        // access credentials through the proxy's phantom token resolution.
92        // These paths are prepended to the profile denylist unconditionally.
93        let credential_store_paths: Vec<std::path::PathBuf> = {
94            let mut paths = vec![
95                std::path::PathBuf::from("/var/lib/puzzled/credentials"),
96                // §3.4: PRD default store_path is /etc/puzzled/credentials/store.enc
97                std::path::PathBuf::from("/etc/puzzled/credentials"),
98            ];
99            // Also deny the user-level credential store path if HOME is set
100            if let Ok(home) = std::env::var("HOME") {
101                paths.push(std::path::PathBuf::from(home).join(".local/share/puzzled/credentials"));
102            }
103            paths
104        };
105
106        // H1: Filter allowlist paths against denylist before adding rules.
107        // Landlock is default-deny — the deny is achieved by NOT granting access.
108        // Any path that matches or is under a denylist entry is excluded.
109        //
110        // M7: Canonicalize denylist paths before checking starts_with().
111        // Without canonicalization, an agent could bypass the denylist by using
112        // symlinks or relative paths (e.g., /var/data/../etc/shadow).
113        //
114        // M-ll1: When canonicalize() fails for a denylist entry, skip the entry
115        // with a warning instead of using the raw (uncanonicalized) path. The raw
116        // path could miss the target due to symlinks.
117        let canonical_denylist: Vec<std::path::PathBuf> = profile
118            .filesystem
119            .denylist
120            .iter()
121            .chain(credential_store_paths.iter())
122            .filter_map(|d| match std::fs::canonicalize(d) {
123                Ok(canonical) => Some(canonical),
124                Err(e) => {
125                    // M-ll1: Skip uncanonicalized denylist entries — using a raw path
126                    // could miss the actual target if symlinks are involved, allowing
127                    // an agent to bypass the denylist.
128                    tracing::error!(
129                        path = %d.display(),
130                        error = %e,
131                        "denylist path canonicalization failed, skipping entry \
132                         (raw path could miss target due to symlinks); skipped denylist \
133                         provides no protection until the path exists"
134                    );
135                    None
136                }
137            })
138            .collect();
139
140        // M-ll2: Canonicalize read_denylist and write_denylist paths for filtering.
141        // These provide fine-grained per-access-type deny lists beyond the general denylist.
142        let canonical_read_denylist: Vec<std::path::PathBuf> = profile
143            .filesystem
144            .read_denylist
145            .iter()
146            .filter_map(|d| match std::fs::canonicalize(d) {
147                Ok(canonical) => Some(canonical),
148                Err(e) => {
149                    // M-ll1: Same canonicalization fallback as general denylist.
150                    tracing::error!(
151                        path = %d.display(),
152                        error = %e,
153                        "read_denylist path canonicalization failed, skipping entry \
154                         (raw path could miss target due to symlinks); skipped entry \
155                         provides no read exclusion until the path exists"
156                    );
157                    None
158                }
159            })
160            .collect();
161
162        let canonical_write_denylist: Vec<std::path::PathBuf> = profile
163            .filesystem
164            .write_denylist
165            .iter()
166            .filter_map(|d| match std::fs::canonicalize(d) {
167                Ok(canonical) => Some(canonical),
168                Err(e) => {
169                    // M-ll1: Same canonicalization fallback as general denylist.
170                    tracing::error!(
171                        path = %d.display(),
172                        error = %e,
173                        "write_denylist path canonicalization failed, skipping entry \
174                         (raw path could miss target due to symlinks); skipped entry \
175                         provides no write exclusion until the path exists"
176                    );
177                    None
178                }
179            })
180            .collect();
181
182        log_read_allowlist_parent_of_read_denylist_conflicts(
183            &profile.filesystem.read_allowlist,
184            &profile.filesystem.read_denylist,
185        );
186
187        // S2: Helper to canonicalize a path for denylist comparison.
188        // If canonicalize fails, treat the path as DENYLISTED (fail-closed)
189        // to prevent bypass via broken symlinks or permission-denied paths.
190        let canonicalize_or_deny = |p: &std::path::Path| -> (std::path::PathBuf, bool) {
191            match std::fs::canonicalize(p) {
192                Ok(canonical) => (canonical, false),
193                Err(e) => {
194                    tracing::warn!(
195                        path = %p.display(),
196                        error = %e,
197                        "S2: canonicalize failed for denylist check — treating path as \
198                         DENYLISTED (fail-closed) to prevent symlink bypass"
199                    );
200                    (p.to_path_buf(), true)
201                }
202            }
203        };
204
205        let is_denylisted = |p: &std::path::Path| -> bool {
206            // S2: If canonicalize fails, deny the path (fail-closed)
207            let (canonical_p, failed) = canonicalize_or_deny(p);
208            if failed {
209                return true;
210            }
211            canonical_denylist
212                .iter()
213                .any(|d| canonical_p.starts_with(d))
214        };
215
216        // M-ll2: Check if a path is in the read_denylist (general denylist OR read-specific denylist)
217        let is_read_denylisted = |p: &std::path::Path| -> bool {
218            let (canonical_p, failed) = canonicalize_or_deny(p);
219            if failed {
220                return true;
221            }
222            canonical_denylist
223                .iter()
224                .any(|d| canonical_p.starts_with(d))
225                || canonical_read_denylist
226                    .iter()
227                    .any(|d| canonical_p.starts_with(d))
228        };
229
230        // M-ll2: Check if a path is in the write_denylist (general denylist OR write-specific denylist)
231        let is_write_denylisted = |p: &std::path::Path| -> bool {
232            let (canonical_p, failed) = canonicalize_or_deny(p);
233            if failed {
234                return true;
235            }
236            canonical_denylist
237                .iter()
238                .any(|d| canonical_p.starts_with(d))
239                || canonical_write_denylist
240                    .iter()
241                    .any(|d| canonical_p.starts_with(d))
242        };
243
244        // Add read-only rules for read_allowlist paths
245        for path in &profile.filesystem.read_allowlist {
246            // M-ll2: Check both general denylist and read-specific denylist
247            if is_read_denylisted(path) {
248                tracing::warn!(
249                    path = %path.display(),
250                    "skipping denylisted path from read allowlist"
251                );
252                continue;
253            }
254            // H-25: Non-existent paths handling based on fail_mode.
255            // In FailClosed mode, a missing allowlist path is an error —
256            // the profile may be misconfigured. In other modes, skip with a warning.
257            if !path.exists() {
258                if profile.fail_mode == puzzled_types::FailMode::FailClosed {
259                    return Err(crate::error::PuzzledError::Sandbox(format!(
260                        "read allowlist path does not exist and fail_mode is FailClosed: {}",
261                        path.display()
262                    )));
263                }
264                tracing::warn!(
265                    path = %path.display(),
266                    "skipping non-existent read allowlist path"
267                );
268                continue;
269            }
270            match PathFd::new(path) {
271                Ok(fd) => {
272                    ruleset = ruleset
273                        .add_rule(PathBeneath::new(fd, read_access))
274                        .map_err(|e| {
275                            crate::error::PuzzledError::Sandbox(format!(
276                                "adding Landlock read rule for {}: {}",
277                                path.display(),
278                                e
279                            ))
280                        })?;
281                }
282                Err(e) => {
283                    tracing::warn!(
284                        path = %path.display(),
285                        error = %e,
286                        "skipping Landlock read rule (path not accessible)"
287                    );
288                }
289            }
290        }
291
292        // Add read-write rules for write_allowlist paths
293        for path in &profile.filesystem.write_allowlist {
294            // M-ll2: Check both general denylist and write-specific denylist
295            if is_write_denylisted(path) {
296                tracing::warn!(
297                    path = %path.display(),
298                    "skipping denylisted path from write allowlist"
299                );
300                continue;
301            }
302            // H-25: Non-existent paths handling based on fail_mode.
303            if !path.exists() {
304                if profile.fail_mode == puzzled_types::FailMode::FailClosed {
305                    return Err(crate::error::PuzzledError::Sandbox(format!(
306                        "write allowlist path does not exist and fail_mode is FailClosed: {}",
307                        path.display()
308                    )));
309                }
310                tracing::warn!(
311                    path = %path.display(),
312                    "skipping non-existent write allowlist path"
313                );
314                continue;
315            }
316            match PathFd::new(path) {
317                Ok(fd) => {
318                    ruleset = ruleset
319                        .add_rule(PathBeneath::new(fd, write_access))
320                        .map_err(|e| {
321                            crate::error::PuzzledError::Sandbox(format!(
322                                "adding Landlock write rule for {}: {}",
323                                path.display(),
324                                e
325                            ))
326                        })?;
327                }
328                Err(e) => {
329                    tracing::warn!(
330                        path = %path.display(),
331                        error = %e,
332                        "skipping Landlock write rule (path not accessible)"
333                    );
334                }
335            }
336        }
337
338        // Add write rules for extra paths (branch merged dir)
339        for path in extra_write_paths {
340            if is_denylisted(path) {
341                tracing::warn!(
342                    path = %path.display(),
343                    "skipping denylisted extra write path"
344                );
345                continue;
346            }
347            if path.exists() {
348                match PathFd::new(path) {
349                    Ok(fd) => {
350                        ruleset = ruleset
351                            .add_rule(PathBeneath::new(fd, write_access))
352                            .map_err(|e| {
353                                crate::error::PuzzledError::Sandbox(format!(
354                                    "adding Landlock write rule for {}: {}",
355                                    path.display(),
356                                    e
357                                ))
358                            })?;
359                    }
360                    Err(e) => {
361                        tracing::warn!(
362                            path = %path.display(),
363                            error = %e,
364                            "skipping Landlock write rule (path not accessible)"
365                        );
366                    }
367                }
368            }
369        }
370
371        // DC: Gap between net and fs ruleset application is pre-exec setup phase only —
372        // no untrusted agent code is running. Rulesets applied in narrowing order:
373        // network first (broader surface), then filesystem (more granular).
374
375        // C10: Landlock network rules (ABI v4+, kernel 6.7+)
376        //
377        // Landlock ABI v4 introduced AccessNet::{ConnectTcp, BindTcp} for
378        // network access control. These are enforced in-kernel, irrevocable,
379        // and survive puzzled crash — just like filesystem rules.
380        //
381        // M10: Port-only restriction is a Landlock ABI v4 limitation —
382        // IP-based filtering requires nftables defense-in-depth.
383        //
384        // Landlock's ConnectTcp/BindTcp rules operate on TCP port numbers only.
385        // They cannot restrict connections by destination IP address, CIDR range,
386        // or DNS domain name. For IP-level filtering, the architecture relies on:
387        //   1. Network namespace isolation (CLONE_NEWNET)
388        //   2. nftables rules inside the agent namespace (setup_gated/setup_monitored)
389        //   3. seccomp USER_NOTIF for connect/bind argument inspection
390        //
391        // The Landlock network layer provides an additional irrevocable port
392        // restriction that survives puzzled crash — even if nftables rules are
393        // somehow circumvented, the agent can only connect to the proxy port.
394        //
395        // NOTE: As of landlock crate v0.4.x, AccessNet support may not be
396        // available in the Rust crate. If `landlock::AccessNet` is not present,
397        // we fall back to raw syscall-based Landlock network rule addition.
398        // For now, we use the crate's AccessNet if available, otherwise log a
399        // warning. The seccomp USER_NOTIF layer provides defense-in-depth for
400        // connect/bind even if Landlock network rules cannot be applied.
401        {
402            use landlock::{AccessNet, NetPort};
403
404            // PM5: Use the proxy port from config instead of hardcoding.
405            // Falls back to 3128 (the default_proxy_port from config.rs) if not provided.
406            let proxy_port_val: u16 = proxy_port.unwrap_or(3128);
407
408            // Create a new ruleset handling network access
409            let mut net_ruleset = Ruleset::default()
410                .handle_access(AccessNet::ConnectTcp | AccessNet::BindTcp)
411                .map_err(|e| {
412                    crate::error::PuzzledError::Sandbox(format!(
413                        "creating Landlock network ruleset: {}",
414                        e
415                    ))
416                })?
417                .create()
418                .map_err(|e| {
419                    crate::error::PuzzledError::Sandbox(format!(
420                        "creating Landlock network ruleset: {}",
421                        e
422                    ))
423                })?;
424
425            // ConnectTcp: only allow connections to the HTTP proxy port.
426            // All agent network traffic must go through the governance proxy.
427            net_ruleset = net_ruleset
428                .add_rule(NetPort::new(proxy_port_val, AccessNet::ConnectTcp))
429                .map_err(|e| {
430                    crate::error::PuzzledError::Sandbox(format!(
431                        "adding Landlock ConnectTcp rule for proxy port {}: {}",
432                        proxy_port_val, e
433                    ))
434                })?;
435
436            // BindTcp: deny all bind operations by not adding any BindTcp rules.
437            // Landlock is default-deny — since we declared we handle BindTcp but
438            // added no rules for it, all bind() calls will be blocked.
439
440            // H-24: Apply network ruleset FIRST (broader surface), before filesystem
441            // ruleset (more granular). Both are applied during pre-exec setup phase
442            // only — no untrusted agent code is running during this gap.
443            let net_status = net_ruleset.restrict_self().map_err(|e| {
444                crate::error::PuzzledError::Sandbox(format!(
445                    "applying Landlock network ruleset: {}",
446                    e
447                ))
448            })?;
449
450            if let LandlockStatus::Available { effective_abi, .. } = net_status.landlock {
451                if effective_abi < ABI::V5 {
452                    tracing::info!(
453                        negotiated_abi = %effective_abi,
454                        "Landlock negotiated ABI v{} (requested v5) — IOCTL_DEV restrictions unavailable on this kernel",
455                        effective_abi,
456                    );
457                }
458                // FailClosed + Gated/Blocked requires kernel Landlock network ABI (v4+). Refuse
459                // when the negotiated ABI is below v4, not only when ruleset status is partial.
460                if profile.fail_mode == puzzled_types::FailMode::FailClosed
461                    && matches!(
462                        profile.network.mode,
463                        puzzled_types::NetworkMode::Gated | puzzled_types::NetworkMode::Blocked
464                    )
465                    && effective_abi < ABI::V4
466                {
467                    return Err(crate::error::PuzzledError::Sandbox(format!(
468                        "Landlock effective ABI v{} is below v4 — network restrictions unavailable \
469                         on this kernel; fail_mode is FailClosed with network mode {:?}. \
470                         Refusing to create sandbox.",
471                        effective_abi,
472                        profile.network.mode,
473                    )));
474                }
475            }
476
477            match net_status.ruleset {
478                RulesetStatus::FullyEnforced => {
479                    tracing::info!(
480                        proxy_port = proxy_port_val,
481                        "Landlock network rules fully enforced (ConnectTcp: proxy only, BindTcp: denied)"
482                    );
483                }
484                RulesetStatus::PartiallyEnforced => {
485                    // SC3: If profile requires FailClosed and has network rules
486                    // (Gated or Blocked mode), ABI < v4 is a security violation.
487                    // Return an error instead of silently degrading.
488                    if profile.fail_mode == puzzled_types::FailMode::FailClosed
489                        && matches!(
490                            profile.network.mode,
491                            puzzled_types::NetworkMode::Gated | puzzled_types::NetworkMode::Blocked
492                        )
493                    {
494                        return Err(crate::error::PuzzledError::Sandbox(
495                            "Landlock network rules only partially enforced (kernel ABI < v4, requires 6.7+) \
496                             but profile requires FailClosed with network restrictions (Gated/Blocked). \
497                             Cannot guarantee kernel-enforced network ACL."
498                                .to_string(),
499                        ));
500                    }
501                    tracing::warn!(
502                        "Landlock network rules partially enforced — kernel may not support ABI v4 network restrictions. \
503                         Falling back to seccomp USER_NOTIF for connect/bind gating."
504                    );
505                }
506                RulesetStatus::NotEnforced => {
507                    // SC3: Same FailClosed check for NotEnforced.
508                    if profile.fail_mode == puzzled_types::FailMode::FailClosed
509                        && matches!(
510                            profile.network.mode,
511                            puzzled_types::NetworkMode::Gated | puzzled_types::NetworkMode::Blocked
512                        )
513                    {
514                        return Err(crate::error::PuzzledError::Sandbox(
515                            "Landlock network rules not enforced (kernel does not support ABI v4, requires 6.7+) \
516                             but profile requires FailClosed with network restrictions (Gated/Blocked). \
517                             Cannot guarantee kernel-enforced network ACL."
518                                .to_string(),
519                        ));
520                    }
521                    tracing::warn!(
522                        "Landlock network rules not enforced — kernel does not support Landlock ABI v4 (requires 6.7+). \
523                         Agent network access will be gated by seccomp USER_NOTIF and network namespace only."
524                    );
525                }
526            }
527        }
528
529        // H-24: Apply the filesystem ruleset AFTER network ruleset — irrevocable.
530        // Network ruleset was applied first (broader surface area), filesystem
531        // ruleset applied second (more granular). Both applied pre-exec, so no
532        // untrusted code runs during the gap between applications.
533        let status = ruleset.restrict_self().map_err(|e| {
534            crate::error::PuzzledError::Sandbox(format!("applying Landlock ruleset: {}", e))
535        })?;
536
537        match status.ruleset {
538            RulesetStatus::FullyEnforced => {
539                tracing::info!("Landlock ruleset fully enforced");
540            }
541            RulesetStatus::PartiallyEnforced => {
542                // PartiallyEnforced is expected when read/write rules include
543                // directory-only flags (e.g. ReadDir) for file paths like
544                // /etc/localtime or /dev/null. The landlock crate automatically
545                // strips incompatible flags, so security is not reduced.
546                // This is NOT the same as a kernel ABI limitation.
547                tracing::info!(
548                    "Landlock filesystem ruleset partially enforced \
549                     (expected: directory-only flags stripped from file paths)"
550                );
551            }
552            RulesetStatus::NotEnforced => {
553                return Err(crate::error::PuzzledError::Sandbox(
554                    "Landlock ruleset not enforced: kernel does not support Landlock. \
555                     Cannot launch agent without filesystem ACL."
556                        .to_string(),
557                ));
558            }
559        }
560
561        Ok(())
562    }
