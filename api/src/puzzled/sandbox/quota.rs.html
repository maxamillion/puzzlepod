6/// Status of an enforcement mechanism setup attempt.
7#[derive(Debug, Clone, PartialEq, Eq)]
8pub enum EnforcementStatus {
9    /// Mechanism was successfully configured.
10    Active,
11    /// Mechanism was not available but setup continued (best-effort).
12    Unavailable(String),
13}
14
15/// XFS project quota manager — enforces per-branch storage and inode limits
16/// on the OverlayFS upper layer directory.
17///
18/// Returns `EnforcementStatus` to let callers decide whether a failure
19/// is fatal based on the profile's `EnforcementRequirements`.
20pub struct QuotaManager;
21
22impl QuotaManager {
23    /// Set up XFS project quotas for a branch upper directory.
24    ///
25    /// 1. Assign a unique project ID to the upper dir via FS_IOC_FSSETXATTR
26    /// 2. Set block limit (storage_quota_mb) via quotactl(Q_XSETPQLIM)
27    /// 3. Set inode limit (inode_quota) via quotactl(Q_XSETPQLIM)
28    ///
29    /// Returns `EnforcementStatus::Active` on success, or
30    /// `EnforcementStatus::Unavailable` with a reason if quotas are not available.
31    /// Callers should check the profile's `require_quota` to decide if
32    /// `Unavailable` is fatal.
33    #[cfg(target_os = "linux")]
34    pub fn setup(
35        upper_dir: &Path,
36        storage_quota_mb: u64,
37        inode_quota: u64,
38    ) -> Result<EnforcementStatus> {
39        use std::ffi::CString;
40
41        // Try to detect if we're on XFS with project quotas enabled
42        let dir_str = upper_dir.to_string_lossy().to_string();
43        let _c_path = match CString::new(dir_str.as_str()) {
44            Ok(p) => p,
45            Err(e) => {
46                tracing::warn!(
47                    path = %upper_dir.display(),
48                    error = %e,
49                    "XFS quota setup skipped: invalid path"
50                );
51                return Ok(EnforcementStatus::Unavailable(format!("invalid path: {e}")));
52            }
53        };
54
55        // Generate a deterministic project ID from the directory path (SHA-256).
56        // SHA-256 is used instead of DefaultHasher because DefaultHasher (SipHash)
57        // is seeded with per-process random bytes — hash values differ across runs
58        // and Rust versions, breaking crash recovery.
59        let project_id = {
60            use sha2::{Digest, Sha256};
61            let hash = Sha256::digest(upper_dir.to_string_lossy().as_bytes());
62            u32::from_le_bytes([hash[0], hash[1], hash[2], hash[3]])
63        };
64
65        // Try to set the project ID via FS_IOC_FSSETXATTR ioctl
66        // This requires the directory to be on XFS with prjquota mount option
67        let fd = match std::fs::File::open(upper_dir) {
68            Ok(f) => f,
69            Err(e) => {
70                tracing::warn!(
71                    path = %upper_dir.display(),
72                    error = %e,
73                    "XFS quota setup skipped: cannot open directory"
74                );
75                return Ok(EnforcementStatus::Unavailable(format!(
76                    "cannot open directory: {e}"
77                )));
78            }
79        };
80
81        // FS_IOC_FSSETXATTR = 0x40100020 (set extended attributes)
82        const FS_IOC_FSSETXATTR: libc::c_ulong = 0x40100020;
83        const FSX_PROJID: u32 = 0x04;
84
85        #[repr(C)]
86        struct FsxAttr {
87            fsx_xflags: u32,
88            fsx_extsize: u32,
89            fsx_nextents: u32,
90            fsx_projid: u32,
91            fsx_cowextsize: u32,
92            fsx_pad: [u8; 8],
93        }
94
95        let attr = FsxAttr {
96            fsx_xflags: FSX_PROJID,
97            fsx_extsize: 0,
98            fsx_nextents: 0,
99            fsx_projid: project_id,
100            fsx_cowextsize: 0,
101            fsx_pad: [0; 8],
102        };
103
104        use std::os::unix::io::AsRawFd;
105        let ret =
106            unsafe { libc::ioctl(fd.as_raw_fd(), FS_IOC_FSSETXATTR, &attr as *const FsxAttr) };
107
108        if ret < 0 {
109            let err = std::io::Error::last_os_error();
110            tracing::warn!(
111                path = %upper_dir.display(),
112                project_id,
113                error = %err,
114                "XFS project ID assignment failed (best-effort, continuing)"
115            );
116            return Ok(EnforcementStatus::Unavailable(format!(
117                "FS_IOC_FSSETXATTR failed: {err}"
118            )));
119        }
120
121        // Set project quota limits via quotactl(Q_XSETPQLIM).
122        // If this fails after FS_IOC_FSSETXATTR succeeded, the project ID is
123        // assigned but limits are not enforced — report as Unavailable so the
124        // caller can decide based on the profile's require_quota setting.
125        if let Err(reason) =
126            Self::set_project_quota(upper_dir, project_id, storage_quota_mb, inode_quota)
127        {
128            tracing::warn!(
129                path = %upper_dir.display(),
130                project_id,
131                error = %reason,
132                "XFS project ID assigned but quotactl failed"
133            );
134            return Ok(EnforcementStatus::Unavailable(format!(
135                "project ID set but quota limits failed: {reason}"
136            )));
137        }
138
139        tracing::info!(
140            path = %upper_dir.display(),
141            project_id,
142            storage_quota_mb,
143            inode_quota,
144            "XFS project quota configured"
145        );
146
147        Ok(EnforcementStatus::Active)
148    }

163    /// Set XFS project quota limits via quotactl(Q_XSETPQLIM).
164    ///
165    /// Returns `Ok(())` on success, or `Err(reason)` if quotactl fails
166    /// (e.g., not on XFS, quotas not enabled, insufficient privileges).
167    #[cfg(target_os = "linux")]
168    fn set_project_quota(
169        upper_dir: &Path,
170        project_id: u32,
171        storage_quota_mb: u64,
172        inode_quota: u64,
173    ) -> std::result::Result<(), String> {
174        use std::ffi::CString;
175
176        // Q_XSETPQLIM = QCMD(Q_XSETQLIM, PRJQUOTA) = 0x00800005
177        const Q_XSETPQLIM: libc::c_int = 0x00800005_u32 as i32;
178
179        #[repr(C)]
180        struct FsDiskQuota {
181            d_version: i8,
182            d_flags: i8,
183            d_fieldmask: u16,
184            d_id: u32,
185            d_blk_hardlimit: u64,
186            d_blk_softlimit: u64,
187            d_ino_hardlimit: u64,
188            d_ino_softlimit: u64,
189            d_bcount: u64,
190            d_icount: u64,
191            d_itimer: i32,
192            d_btimer: i32,
193            d_iwarns: u16,
194            d_bwarns: u16,
195            d_padding2: i32,
196            d_rtb_hardlimit: u64,
197            d_rtb_softlimit: u64,
198            d_rtbcount: u64,
199            d_rtbtimer: i32,
200            d_rtbwarns: u16,
201            d_padding3: i16,
202            d_padding4: [i8; 8],
203        }
204
205        let dq = FsDiskQuota {
206            d_version: 1,             // FS_DQUOT_VERSION
207            d_flags: 2,               // XFS_PROJ_QUOTA
208            d_fieldmask: 0x08 | 0x80, // FS_DQ_BHARD | FS_DQ_IHARD
209            d_id: project_id,
210            // A4: XFS quota d_blk_hardlimit is always in "basic blocks" (512 bytes),
211            // regardless of the filesystem's actual block size. This is the XFS quota
212            // interface convention (see fs/xfs/xfs_dquot.h: XFS_FSB_TO_BB). Conversion:
213            // MB * 1024 (KB/MB) * 2 (512-byte blocks per KB) = MB * 2048 basic blocks.
214            d_blk_hardlimit: storage_quota_mb * 1024 * 2,
215            d_blk_softlimit: 0,
216            d_ino_hardlimit: inode_quota,
217            d_ino_softlimit: 0,
218            d_bcount: 0,
219            d_icount: 0,
220            d_itimer: 0,
221            d_btimer: 0,
222            d_iwarns: 0,
223            d_bwarns: 0,
224            d_padding2: 0,
225            d_rtb_hardlimit: 0,
226            d_rtb_softlimit: 0,
227            d_rtbcount: 0,
228            d_rtbtimer: 0,
229            d_rtbwarns: 0,
230            d_padding3: 0,
231            d_padding4: [0; 8],
232        };
233
234        let dev = Self::find_block_device(upper_dir)
235            .ok_or_else(|| "could not determine block device for quota enforcement".to_string())?;
236
237        let c_dev = CString::new(dev.as_str())
238            .map_err(|_| "invalid block device path for quotactl".to_string())?;
239
240        let ret = unsafe {
241            libc::quotactl(
242                Q_XSETPQLIM,
243                c_dev.as_ptr(),
244                // R1: Mask high bit to prevent sign wrap when casting u32 -> i32.
245                // XFS project IDs above 0x7FFF_FFFF would become negative i32,
246                // causing quotactl to fail or apply to the wrong project.
247                (project_id & 0x7FFF_FFFF) as i32,
248                &dq as *const FsDiskQuota as *mut libc::c_char,
249            )
250        };
251        if ret < 0 {
252            let err = std::io::Error::last_os_error();
253            return Err(format!("quotactl(Q_XSETPQLIM) failed: {err}"));
254        }
255
256        Ok(())
257    }

259    /// Find the block device for the filesystem containing the given path
260    /// by reading /proc/mounts.
261    #[cfg(target_os = "linux")]
262    fn find_block_device(path: &Path) -> Option<String> {
263        let path_str = path.to_string_lossy();
264        let mounts = std::fs::read_to_string("/proc/mounts").ok()?;
265
266        let mut best_match: Option<(&str, &str)> = None;
267        for line in mounts.lines() {
268            let parts: Vec<&str> = line.split_whitespace().collect();
269            if parts.len() < 2 {
270                continue;
271            }
272            let mount_point = parts[1];
273            if path_str.starts_with(mount_point)
274                && best_match
275                    .map(|(_, mp)| mount_point.len() > mp.len())
276                    .unwrap_or(true)
277            {
278                best_match = Some((parts[0], mount_point));
279            }
280        }
281
282        best_match.map(|(dev, _)| dev.to_string())
283    }

285    /// Remove project quota for a branch (best-effort cleanup).
286    ///
287    /// 1. Zero out quota limits via quotactl(Q_XSETPQLIM) with b_hard=0, i_hard=0
288    /// 2. Reset project ID to 0 via FS_IOC_FSSETXATTR
289    ///
290    /// Errors are logged but never propagated — rollback must not fail due to
291    /// quota cleanup issues.
292    #[cfg(target_os = "linux")]
293    pub fn remove(upper_dir: &Path) -> Result<()> {
294        // Recompute the same deterministic project ID used during setup
295        let project_id = {
296            use sha2::{Digest, Sha256};
297            let hash = Sha256::digest(upper_dir.to_string_lossy().as_bytes());
298            u32::from_le_bytes([hash[0], hash[1], hash[2], hash[3]])
299        };
300
301        // Step 1: Zero out quota limits via quotactl(Q_XSETPQLIM)
302        if let Err(e) = Self::set_project_quota(upper_dir, project_id, 0, 0) {
303            tracing::warn!(
304                path = %upper_dir.display(),
305                project_id,
306                error = %e,
307                "best-effort quota limit cleanup failed"
308            );
309        }
310
311        // Step 2: Reset project ID to 0 via FS_IOC_FSSETXATTR
312        const FS_IOC_FSSETXATTR: libc::c_ulong = 0x40100020;
313
314        #[repr(C)]
315        struct FsxAttr {
316            fsx_xflags: u32,
317            fsx_extsize: u32,
318            fsx_nextents: u32,
319            fsx_projid: u32,
320            fsx_cowextsize: u32,
321            fsx_pad: [u8; 8],
322        }
323
324        let attr = FsxAttr {
325            fsx_xflags: 0,
326            fsx_extsize: 0,
327            fsx_nextents: 0,
328            fsx_projid: 0,
329            fsx_cowextsize: 0,
330            fsx_pad: [0; 8],
331        };
332
333        match std::fs::File::open(upper_dir) {
334            Ok(fd) => {
335                use std::os::unix::io::AsRawFd;
336                let ret = unsafe {
337                    libc::ioctl(fd.as_raw_fd(), FS_IOC_FSSETXATTR, &attr as *const FsxAttr)
338                };
339                if ret < 0 {
340                    let err = std::io::Error::last_os_error();
341                    tracing::warn!(
342                        path = %upper_dir.display(),
343                        error = %err,
344                        "best-effort project ID reset failed"
345                    );
346                }
347            }
348            Err(e) => {
349                tracing::warn!(
350                    path = %upper_dir.display(),
351                    error = %e,
352                    "best-effort quota cleanup skipped: cannot open directory"
353                );
354            }
355        }
356
357        tracing::debug!(
358            path = %upper_dir.display(),
359            project_id,
360            "XFS quota cleanup completed (best-effort)"
361        );
362        Ok(())
363    }

365    /// Remove project quota for a branch (no-op on non-Linux).
366    #[cfg(not(target_os = "linux"))]
367    pub fn remove(upper_dir: &Path) -> Result<()> {
368        tracing::debug!(
369            path = %upper_dir.display(),
370            "XFS quota cleanup skipped (not on Linux)"
371        );
