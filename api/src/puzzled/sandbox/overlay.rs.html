6/// OverlayFS mount management for branch isolation.
7///
8/// Each branch gets:
9/// - `upper/` — copy-on-write layer (agent writes here)
10/// - `work/`  — OverlayFS internal work directory
11/// - `merged/` — union mount point visible to the agent
12///
13/// The lower layer is the base filesystem (read-only from the agent's perspective).
14#[allow(dead_code)] // Fields used by mount/unmount on Linux only
15pub struct OverlayMount {
16    upper_dir: PathBuf,
17    work_dir: PathBuf,
18    merged_dir: PathBuf,
19    lower_dir: PathBuf,
20    /// Tracks the fuse-overlayfs child process in rootless mode.
21    /// `None` for kernel OverlayFS (rootful) mounts.
22    fuse_child: Option<std::process::Child>,
23}
24
25impl OverlayMount {
26    /// Create directory structure for a new branch.
27    ///
28    /// M-ov1: Before creating directories, checks that `branch_dir` and its
29    /// subdirectories are not symlinks. A symlink in the branch path could be
30    /// used to redirect writes to arbitrary locations outside the branch.
31    // V29: Intermediate path components are not checked for symlinks — the branch base path
32    // is admin-configured in puzzled.conf, not agent-controlled. Agent-created symlinks are
33    // caught by the leaf directory checks and Landlock enforcement.
34    pub fn create_dirs(branch_dir: &Path) -> Result<(PathBuf, PathBuf, PathBuf)> {
35        // M-ov1: Check branch_dir itself for symlinks before creating subdirs.
36        if branch_dir.exists()
37            && branch_dir
38                .symlink_metadata()
39                .map(|m| m.file_type().is_symlink())
40                .unwrap_or(false)
41        {
42            return Err(crate::error::PuzzledError::Sandbox(format!(
43                "symlink detected in branch directory path: {}",
44                branch_dir.display()
45            )));
46        }
47
48        let upper = branch_dir.join("upper");
49        let work = branch_dir.join("work");
50        let merged = branch_dir.join("merged");
51
52        // M-ov1: Check each subdirectory for symlinks before create_dir_all.
53        for subdir in &[&upper, &work, &merged] {
54            if subdir.exists()
55                && subdir
56                    .symlink_metadata()
57                    .map(|m| m.file_type().is_symlink())
58                    .unwrap_or(false)
59            {
60                return Err(crate::error::PuzzledError::Sandbox(format!(
61                    "symlink detected in branch directory path: {}",
62                    subdir.display()
63                )));
64            }
65        }
66
67        std::fs::create_dir_all(&upper)?;
68        std::fs::create_dir_all(&work)?;
69        std::fs::create_dir_all(&merged)?;
70
71        // H28: Post-creation symlink verification to close the TOCTOU window
72        // between the pre-creation check and create_dir_all(). An attacker could
73        // race to replace a path component with a symlink after the check but
74        // before directory creation.
75        for subdir in &[&upper, &work, &merged] {
76            if subdir
77                .symlink_metadata()
78                .map(|m| m.file_type().is_symlink())
79                .unwrap_or(false)
80            {
81                return Err(crate::error::PuzzledError::Sandbox(format!(
82                    "H28: symlink detected after creation in branch directory path: {}",
83                    subdir.display()
84                )));
85            }
86        }
87
88        Ok((upper, work, merged))
89    }

91    /// Validate that a mount path does not contain characters that could inject
92    /// additional mount options (commas) or other control characters.
93    #[cfg_attr(not(target_os = "linux"), allow(dead_code))]
94    fn validate_mount_path(path: &Path) -> Result<()> {
95        // H1: Use to_str() instead of to_string_lossy() to reject non-UTF-8 paths
96        // explicitly. to_string_lossy() silently replaces non-UTF-8 bytes with U+FFFD,
97        // which could mask malicious byte sequences in path names.
98        let s = path.to_str().ok_or_else(|| {
99            crate::error::PuzzledError::Sandbox(format!(
100                "mount path contains non-UTF-8 bytes: {}",
101                path.to_string_lossy()
102            ))
103        })?;
104
105        // V34: Shell metacharacters are safe in mount() context — no shell invocation
106        // Check for characters that could inject mount options or escape shell contexts:
107        // - comma: separates mount options, enabling option injection
108        // - newline: breaks line-based parsing
109        // - null: C string terminator, truncates path
110        // - backslash: escape character that could alter path interpretation
111        // - control characters (ASCII < 0x20): could interfere with terminal output,
112        //   log parsing, or shell interpretation
113        if let Some(bad_char) = s
114            .chars()
115            .find(|&c| c == ',' || c == '\\' || c == '\0' || (c as u32) < 0x20)
116        {
117            let description = match bad_char {
118                ',' => "comma".to_string(),
119                '\\' => "backslash".to_string(),
120                '\n' => "newline".to_string(),
121                '\0' => "null".to_string(),
122                c => format!("control character 0x{:02x}", c as u32),
123            };
124            return Err(crate::error::PuzzledError::Sandbox(format!(
125                "mount path contains illegal character ({}): {}",
126                description, s
127            )));
128        }
129        Ok(())
130    }

132    /// Mount the OverlayFS filesystem.
133    ///
134    /// mount -t overlay overlay -o lowerdir=<lower>,upperdir=<upper>,workdir=<work> <merged>
135    ///
136    /// # Arguments
137    /// * `allow_exec` — if false, MS_NOEXEC is added to mount flags (default behavior).
138    ///   Set to true only for profiles that legitimately need to execute binaries from
139    ///   the overlay (e.g., compiler workloads).
140    #[cfg(target_os = "linux")]
141    pub fn mount(
142        lower: &Path,
143        upper: &Path,
144        work: &Path,
145        merged: &Path,
146        allow_exec: bool,
147    ) -> Result<Self> {
148        use nix::mount::MsFlags;
149
150        // H5: Validate paths before constructing mount options to prevent injection
151        Self::validate_mount_path(lower)?;
152        Self::validate_mount_path(upper)?;
153        Self::validate_mount_path(work)?;
154        Self::validate_mount_path(merged)?;
155
156        // Rootless mode: use fuse-overlayfs instead of kernel overlayfs
157        if nix::unistd::geteuid().as_raw() != 0 {
158            return Self::mount_fuse_overlayfs(lower, upper, work, merged, allow_exec);
159        }
160
161        let options = Self::build_mount_options(lower, upper, work);
162
163        // H17: Prevent setuid binaries and device nodes in the overlay.
164        // MS_NOSUID ignores set-user-ID and set-group-ID bits on execution.
165        // MS_NODEV prevents interpretation of block/character special devices.
166        // MS_NOEXEC prevents execution of binaries from the overlay unless
167        // the profile explicitly allows it via allow_exec_overlay.
168        let mut flags = MsFlags::MS_NOSUID | MsFlags::MS_NODEV;
169        if !allow_exec {
170            flags |= MsFlags::MS_NOEXEC;
171        }
172
173        nix::mount::mount(
174            Some("overlay"),
175            merged,
176            Some("overlay"),
177            flags,
178            Some(options.as_str()),
179        )
180        .map_err(|e| {
181            crate::error::PuzzledError::Sandbox(format!(
182                "mounting OverlayFS at {}: {}",
183                merged.display(),
184                e
185            ))
186        })?;
187
188        tracing::info!(
189            merged = %merged.display(),
190            lower = %lower.display(),
191            "OverlayFS mounted"
192        );
193
194        Ok(Self {
195            upper_dir: upper.to_path_buf(),
196            work_dir: work.to_path_buf(),
197            merged_dir: merged.to_path_buf(),
198            lower_dir: lower.to_path_buf(),
199            fuse_child: None,
200        })
201    }

217    /// M-ov2: Validate a path received over the socketpair from the parent process.
218    ///
219    /// Checks:
220    /// - Length does not exceed 4096 bytes (PATH_MAX on Linux)
221    /// - Path is absolute (starts with `/`)
222    /// - Path contains no null bytes
223    pub fn validate_received_path(path_bytes: &[u8], label: &str) -> Result<()> {
224        // M-ov2: Length bounds check
225        if path_bytes.len() > 4096 {
226            return Err(crate::error::PuzzledError::Sandbox(format!(
227                "mount param '{}' exceeds max path length (got {}, max 4096)",
228                label,
229                path_bytes.len()
230            )));
231        }
232
233        // M-ov2: Check for null bytes
234        if path_bytes.contains(&0) {
235            return Err(crate::error::PuzzledError::Sandbox(format!(
236                "mount param '{}' contains null byte",
237                label
238            )));
239        }
240
241        // M-ov2: Validate path is absolute
242        if path_bytes.first() != Some(&b'/') {
243            return Err(crate::error::PuzzledError::Sandbox(format!(
244                "mount param '{}' is not an absolute path",
245                label
246            )));
247        }
248
249        Ok(())
250    }

252    /// Build the OverlayFS mount options string.
253    ///
254    /// Extracted for testability — the actual `mount()` call is Linux-only,
255    /// but the options string can be validated cross-platform.
256    #[cfg_attr(not(test), allow(dead_code))]
257    fn build_mount_options(lower: &Path, upper: &Path, work: &Path) -> String {
258        // metacopy=off: ensures full data copy-up (not just metadata).
259        // redirect_dir=off: prevents rename-based bypass of upper-layer
260        // tracking — without this, agents could use rename operations to
261        // evade the diff engine (Issue #10).
262        format!(
263            "lowerdir={},upperdir={},workdir={},metacopy=off,redirect_dir=off",
264            lower.display(),
265            upper.display(),
266            work.display()
267        )
268    }

270    /// Mount using fuse-overlayfs for rootless (non-root) operation.
271    ///
272    /// Spawns fuse-overlayfs in foreground mode (`-f`) so the process handle
273    /// can be tracked for cleanup. Verifies the mount established by checking
274    /// that the process is still running and the merged path is a mountpoint.
275    #[cfg(target_os = "linux")]
276    fn mount_fuse_overlayfs(
277        lower: &Path,
278        upper: &Path,
279        work: &Path,
280        merged: &Path,
281        allow_exec: bool,
282    ) -> Result<Self> {
283        let fuse_bin = Self::find_fuse_overlayfs()?;
284
285        // H17: nosuid/nodev always set; noexec unless profile allows execution
286        let mut opts = format!(
287            "lowerdir={},upperdir={},workdir={},nosuid,nodev",
288            lower.display(),
289            upper.display(),
290            work.display()
291        );
292        if !allow_exec {
293            opts.push_str(",noexec");
294        }
295
296        let mut child = std::process::Command::new(&fuse_bin)
297            .arg("-f")
298            .arg("-o")
299            .arg(&opts)
300            .arg(merged)
301            .stdin(std::process::Stdio::null())
302            .stdout(std::process::Stdio::null())
303            .spawn()
304            .map_err(|e| {
305                crate::error::PuzzledError::Sandbox(format!("spawning fuse-overlayfs: {}", e))
306            })?;
307
308        std::thread::sleep(std::time::Duration::from_millis(100));
309
310        // If fuse-overlayfs exited within the settle window, the mount failed
311        match child.try_wait() {
312            Ok(Some(status)) => {
313                return Err(crate::error::PuzzledError::Sandbox(format!(
314                    "fuse-overlayfs exited immediately with {} — \
315                     verify fuse-overlayfs is installed correctly and {} is a valid mountpoint",
316                    status,
317                    merged.display()
318                )));
319            }
320            Ok(None) => { /* still running — mount is likely up */ }
321            Err(e) => {
322                return Err(crate::error::PuzzledError::Sandbox(format!(
323                    "checking fuse-overlayfs process status: {}",
324                    e
325                )));
326            }
327        }
328
329        // Verify mount via device ID comparison with parent directory
330        if !Self::is_fuse_mountpoint(merged) {
331            let _ = child.kill();
332            let _ = child.wait();
333            return Err(crate::error::PuzzledError::Sandbox(format!(
334                "fuse-overlayfs process is running but {} does not appear to be a mountpoint",
335                merged.display()
336            )));
337        }
338
339        tracing::info!(
340            merged = %merged.display(),
341            lower = %lower.display(),
342            "fuse-overlayfs mounted (rootless)"
343        );
344
345        Ok(Self {
346            upper_dir: upper.to_path_buf(),
347            work_dir: work.to_path_buf(),
348            merged_dir: merged.to_path_buf(),
349            lower_dir: lower.to_path_buf(),
350            fuse_child: Some(child),
351        })
352    }

354    /// Locate the `fuse-overlayfs` binary in well-known paths or `$PATH`.
355    #[cfg(target_os = "linux")]
356    fn find_fuse_overlayfs() -> Result<PathBuf> {
357        for candidate in &["/usr/bin/fuse-overlayfs", "/usr/local/bin/fuse-overlayfs"] {
358            let p = Path::new(candidate);
359            if p.is_file() {
360                return Ok(p.to_path_buf());
361            }
362        }
363        if let Some(found) = Self::find_in_path("fuse-overlayfs") {
364            return Ok(found);
365        }
366        Err(crate::error::PuzzledError::Sandbox(
367            "rootless mode requires fuse-overlayfs but it was not found in PATH".to_string(),
368        ))
369    }

371    /// Search `$PATH` for a binary by name.
372    #[cfg(target_os = "linux")]
373    fn find_in_path(binary: &str) -> Option<PathBuf> {
374        std::env::var_os("PATH").and_then(|paths| {
375            std::env::split_paths(&paths)
376                .map(|dir| dir.join(binary))
377                .find(|p| p.is_file())
378        })
379    }

381    /// Check whether `path` is a mountpoint by comparing its device ID
382    /// with that of its parent directory.
383    #[cfg(target_os = "linux")]
384    fn is_fuse_mountpoint(path: &Path) -> bool {
385        use std::os::unix::fs::MetadataExt;
386        let Ok(path_meta) = std::fs::metadata(path) else {
387            return false;
388        };
389        let Some(parent) = path.parent() else {
390            return false;
391        };
392        let Ok(parent_meta) = std::fs::metadata(parent) else {
393            return false;
394        };
395        path_meta.dev() != parent_meta.dev()
396    }

398    /// Unmount a FUSE filesystem using fusermount3 (preferred) or fusermount.
399    #[cfg(target_os = "linux")]
400    fn fusermount_unmount(merged: &Path) -> Result<()> {
401        for cmd in &["fusermount3", "fusermount"] {
402            if Self::find_in_path(cmd).is_some() {
403                let output = std::process::Command::new(cmd)
404                    .arg("-u")
405                    .arg(merged)
406                    .output()
407                    .map_err(|e| {
408                        crate::error::PuzzledError::Sandbox(format!(
409                            "running {} -u {}: {}",
410                            cmd,
411                            merged.display(),
412                            e
413                        ))
414                    })?;
415                if output.status.success() {
416                    return Ok(());
417                }
418                let stderr = String::from_utf8_lossy(&output.stderr);
419                tracing::warn!(
420                    cmd,
421                    stderr = %stderr.trim(),
422                    "fusermount failed, trying fallback"
423                );
424            }
425        }
426        Err(crate::error::PuzzledError::Sandbox(format!(
427            "failed to unmount fuse-overlayfs at {}: \
428             neither fusermount3 nor fusermount succeeded",
429            merged.display()
430        )))
431    }

433    /// Unmount the OverlayFS filesystem.
434    #[cfg(target_os = "linux")]
435    pub fn unmount(&mut self) -> Result<()> {
436        if let Some(ref mut child) = self.fuse_child {
437            // Rootless: unmount via fusermount3/fusermount, then clean up child
438            Self::fusermount_unmount(&self.merged_dir)?;
439
440            match child.try_wait() {
441                Ok(None) => {
442                    let _ = child.kill();
443                    let _ = child.wait();
444                }
445                Ok(Some(_)) => {}
446                Err(e) => {
447                    tracing::warn!(
448                        error = %e,
449                        "failed to check fuse-overlayfs status during unmount"
450                    );
451                    let _ = child.kill();
452                    let _ = child.wait();
453                }
454            }
455
456            tracing::info!(
457                merged = %self.merged_dir.display(),
458                "fuse-overlayfs unmounted (rootless)"
459            );
460            return Ok(());
461        }
462
463        // Rootful: kernel unmount (unchanged)
464        nix::mount::umount2(&self.merged_dir, nix::mount::MntFlags::MNT_DETACH).map_err(|e| {
465            crate::error::PuzzledError::Sandbox(format!(
466                "unmounting OverlayFS at {}: {}",
467                self.merged_dir.display(),
468                e
469            ))
470        })?;
471
472        tracing::info!(merged = %self.merged_dir.display(), "OverlayFS unmounted");
473        Ok(())
474    }
