17/// Check if a path matches an exec allowlist.
18///
19/// Supports exact matches and trailing glob patterns (e.g., "/usr/bin/*").
20///
21/// C3: Callers must canonicalize paths before calling this function.
22/// The debug_assert catches path traversal attempts in development/testing.
23/// In production, `validate_execve_with_path` canonicalizes via `/proc/<pid>/root`.
24pub(crate) fn path_matches_exec_allowlist(path: &str, allowlist: &[String]) -> bool {
25    debug_assert!(
26        !path.contains("/../") && !path.ends_with("/.."),
27        "C3: path_matches_exec_allowlist called with non-canonical path containing '..': {path}"
28    );
29    if allowlist.is_empty() {
30        return false;
31    }
32    allowlist.iter().any(|pattern| {
33        if pattern.ends_with('*') {
34            let prefix = &pattern[..pattern.len() - 1];
35            path.starts_with(prefix)
36        } else {
37            path == *pattern
38        }
39    })
40}

42/// Check if a path matches an exec denylist.
43///
44/// Deny overrides allow — if a path matches the denylist, it is blocked
45/// regardless of the allowlist.
46///
47/// C3: Callers must canonicalize paths before calling this function.
48pub(crate) fn path_matches_exec_denylist(path: &str, denylist: &[String]) -> bool {
49    debug_assert!(
50        !path.contains("/../") && !path.ends_with("/.."),
51        "C3: path_matches_exec_denylist called with non-canonical path containing '..': {path}"
52    );
53    if denylist.is_empty() {
54        return false;
55    }
56    denylist.iter().any(|pattern| {
57        if pattern.ends_with('*') {
58            let prefix = &pattern[..pattern.len() - 1];
59            path.starts_with(prefix)
60        } else {
61            path == *pattern
62        }
63    })
64}

66/// Evaluate exec policy: allowed by allowlist AND not denied by denylist.
67pub(crate) fn is_exec_allowed(path: &str, profile: &AgentProfile) -> bool {
68    if !path_matches_exec_allowlist(path, &profile.exec_allowlist) {
69        return false;
70    }
71    !path_matches_exec_denylist(path, &profile.exec_denylist)
72}

74/// Check if a connect destination should be allowed under the given network config.
75///
76/// `ip` is the destination IP address string, `is_loopback` indicates localhost.
77pub(crate) fn is_connect_allowed_by_policy(
78    ip: &str,
79    is_loopback: bool,
80    profile: &AgentProfile,
81) -> bool {
82    use puzzled_types::NetworkMode;
83
84    match profile.network.mode {
85        NetworkMode::Blocked => false,
86        NetworkMode::Unrestricted | NetworkMode::Monitored => true,
87        NetworkMode::Gated => {
88            if is_loopback {
89                return true;
90            }
91            profile
92                .network
93                .allowed_domains
94                .iter()
95                .any(|domain| *domain == ip || domain == "*")
96        }
97    }
98}

117/// Validate an execve syscall against the profile's exec_allowlist.
118///
119/// Reads the binary path from the agent's memory, canonicalizes it to
120/// resolve path traversal (e.g., `/usr/bin/../../etc/shadow`), and checks
121/// against allowed patterns. Landlock remains the backstop for filesystem
122/// access, but this prevents obvious seccomp bypasses.
123#[cfg(target_os = "linux")]
124#[allow(dead_code)] // H6: Kept for fallback; validate_execve_with_path is preferred
125pub(super) fn validate_execve(pid: u32, path_addr: usize, profile: &AgentProfile) -> bool {
126    let raw_path = match read_string_from_proc_mem(pid, path_addr) {
127        Ok(p) => p,
128        Err(e) => {
129            tracing::warn!(pid, error = %e, "failed to read execve path, denying");
130            return false;
131        }
132    };
133    validate_execve_with_path(pid, &raw_path, profile)
134}
135
136/// H6: Validate an execve syscall using a pre-read path string.
137///
138/// This variant accepts a path that has already been read from
139/// `/proc/<pid>/mem`, eliminating the TOCTOU window that exists when the
140/// path is read separately by `validate_execve()` and `inject_fd_for_execve()`.
141/// The caller reads the path once and passes the same string to both functions.
142///
143/// Canonicalizes the path to resolve path traversal, then checks against
144/// the profile's exec_allowlist and exec_denylist.
145#[cfg(target_os = "linux")]
146pub(super) fn validate_execve_with_path(pid: u32, raw_path: &str, profile: &AgentProfile) -> bool {
147    // If exec_allowlist is empty, deny all execs
148    if profile.exec_allowlist.is_empty() {
149        tracing::info!(pid, path = %raw_path, "execve denied: empty allowlist");
150        return false;
151    }
152
153    // Canonicalize the path to resolve traversal (.. components).
154    // Use /proc/<pid>/root to resolve relative to the agent's filesystem view.
155    let path = match std::fs::canonicalize(format!("/proc/{}/root/{}", pid, raw_path)) {
156        Ok(canonical) => {
157            // Strip the /proc/<pid>/root prefix to get the agent-relative path
158            let root_prefix = format!("/proc/{}/root", pid);
159            canonical
160                .to_string_lossy()
161                .strip_prefix(&root_prefix)
162                .unwrap_or(&canonical.to_string_lossy())
163                .to_string()
164        }
165        Err(_) => {
166            // M2: Normalize path traversal components even when canonicalize fails.
167            // Landlock is the backstop defense, but seccomp should not be weakened.
168            normalize_path_components(raw_path)
169        }
170    };
171
172    if path != raw_path {
173        tracing::debug!(
174            pid,
175            raw = %raw_path,
176            canonical = %path,
177            "execve path canonicalized (traversal resolved)"
178        );
179    }
180
181    // Check if the path matches any allowed pattern.
182    // Check BOTH the raw path and the canonical path. Allowlist entries
183    // like "/usr/bin/python3" should match even when the binary is a
184    // symlink to "/usr/bin/python3.13". The raw path is what the profile
185    // author specified; the canonical path catches traversal bypasses.
186    let matches_allowlist = |p: &str| -> bool {
187        profile.exec_allowlist.iter().any(|pattern| {
188            if pattern.ends_with('*') {
189                let prefix = &pattern[..pattern.len() - 1];
190                p.starts_with(prefix)
191            } else {
192                p == *pattern
193            }
194        })
195    };
196    let allowed = matches_allowlist(&path) || matches_allowlist(raw_path);
197
198    // Check exec_denylist — deny overrides allow
199    // P2-N4: Check both canonical path AND raw_path against denylist,
200    // mirroring the allowlist pattern to prevent bypass via symlinks.
201    if allowed && !profile.exec_denylist.is_empty() {
202        let matches_denylist = |p: &str| -> bool {
203            profile.exec_denylist.iter().any(|pattern| {
204                if pattern.ends_with('*') {
205                    let prefix = &pattern[..pattern.len() - 1];
206                    p.starts_with(prefix)
207                } else {
208                    p == *pattern
209                }
210            })
211        };
212        let denied = matches_denylist(&path) || matches_denylist(raw_path);
213        if denied {
214            tracing::info!(
215                pid,
216                path = %path,
217                raw_path = %raw_path,
218                "execve denied: matched exec_denylist (deny overrides allow)"
219            );
220            return false;
221        }
222    }
223
224    if !allowed {
225        tracing::info!(
226            pid,
227            path = %path,
228            "execve denied: not in allowlist"
229        );
230    } else {
231        tracing::debug!(
232            pid,
233            path = %path,
234            "execve allowed"
235        );
236    }
237
238    allowed
239}

241/// Validate a connect syscall against the profile's network config.
242///
243/// Reads the sockaddr from agent memory and checks against allowed domains/IPs.
244#[cfg(target_os = "linux")]
245pub(super) fn validate_connect(
246    pid: u32,
247    addr: usize,
248    len: usize,
249    profile: &AgentProfile,
250    credential_proxy: Option<&crate::seccomp_handler::CredentialProxyContext>,
251) -> bool {
252    use puzzled_types::NetworkMode;
253
254    match profile.network.mode {
255        NetworkMode::Blocked => {
256            tracing::info!(pid, "connect denied: network mode is Blocked");
257            return false;
258        }
259        NetworkMode::Unrestricted | NetworkMode::Monitored => {
260            // Still check gateway blocking for credential proxy even in unrestricted mode
261        }
262        NetworkMode::Gated => {
263            // Fall through to domain checking
264        }
265    }
266
267    let (family, ip, port) = match read_sockaddr_from_proc_mem(pid, addr, len) {
268        Ok(sa) => sa,
269        Err(e) => {
270            tracing::warn!(pid, error = %e, "failed to read connect sockaddr, denying");
271            return false;
272        }
273    };
274
275    // §3.4 G23: Block direct connections to the credential proxy gateway.
276    // Agents must use the transparent proxy (DNAT) — direct connections to the
277    // gateway IP on the proxy port range bypass credential isolation.
278    if let Some(ctx) = credential_proxy {
279        if ctx.enabled {
280            // Parse the IP from sockaddr for type-safe comparison. This handles
281            // IPv6 normalization (e.g., ::ffff:10.0.2.2 vs 10.0.2.2) correctly.
282            let is_gateway = ip
283                .parse::<std::net::IpAddr>()
284                .is_ok_and(|parsed| parsed == ctx.proxy_gateway_ip);
285            if is_gateway {
286                // Block connections to any port in the global proxy port range
287                // (cross-branch defense: prevents connecting to another branch's proxy)
288                if ctx.global_port_range.contains(&port) {
289                    tracing::error!(
290                        pid,
291                        ip = %ip,
292                        port,
293                        "§3.4 G23: BLOCKED — direct connect to credential proxy gateway \
294                         on proxy port range (bypass attempt)"
295                    );
296                    return false;
297                }
298                // Block connections to gateway on non-proxied ports too — only DNAT'd
299                // traffic (on proxied_ports like 80/443) should reach the gateway
300                if !ctx.proxied_ports.contains(&port) {
301                    tracing::error!(
302                        pid,
303                        ip = %ip,
304                        port,
305                        "§3.4 G23: BLOCKED — direct connect to credential proxy gateway \
306                         on non-proxied port"
307                    );
308                    return false;
309                }
310            }
311        }
312    }
313
314    // For Unrestricted/Monitored, allow after gateway check
315    if matches!(
316        profile.network.mode,
317        NetworkMode::Unrestricted | NetworkMode::Monitored
318    ) {
319        return true;
320    }
321
322    // H7: AF_UNIX sockets are no longer unconditionally allowed.
323    // Gate through policy — only allow if the socket path matches the
324    // profile's allowed_domains list (which may include Unix socket paths).
325    // If no Unix socket paths are configured, block AF_UNIX by default
326    // (fail-closed). This prevents agents from connecting to arbitrary
327    // Unix domain sockets (e.g., Docker socket, D-Bus system bus).
328    if family as i32 == libc::AF_UNIX {
329        // Check if any allowed_domains entry looks like a Unix socket path
330        let unix_allowed = profile.network.allowed_domains.iter().any(|entry| {
331            // Wildcard allows everything including Unix sockets
332            entry == "*"
333            // Entries starting with "/" are treated as Unix socket path patterns
334            || (entry.starts_with('/') && {
335                if entry.ends_with('*') {
336                    ip.starts_with(&entry[..entry.len() - 1])
337                } else {
338                    ip == *entry
339                }
340            })
341        });
342        if !unix_allowed {
343            tracing::warn!(
344                pid,
345                socket_path = %ip,
346                "connect denied: AF_UNIX socket not in allowed_domains (H7: no longer unconditionally allowed)"
347            );
348        } else {
349            tracing::debug!(pid, socket_path = %ip, "AF_UNIX connect allowed by policy");
350        }
351        return unix_allowed;
352    }
353
354    // U8: Block connect to 0.0.0.0 (INADDR_ANY) — should not be a connect target
355    if ip == "0.0.0.0" {
356        tracing::warn!(
357            pid,
358            "connect denied: 0.0.0.0 (INADDR_ANY) is not a valid connect target"
359        );
360        return false;
361    }
362
363    // U13: Loopback connections are always allowed — agents need localhost for proxy communication
364    // T8: Include IPv4-mapped IPv6 loopback; T10: remove dead "::1" (formatter uses expanded form)
365    if ip == "127.0.0.1" || ip == "0:0:0:0:0:0:0:1" || ip == "0:0:0:0:0:ffff:7f00:1" {
366        return true;
367    }
368
369    // For Gated mode, check against allowed domains list.
370    // Since we have an IP address, not a hostname, we check if the IP matches
371    // any allowed domain. In practice, the HTTP proxy handles domain-level
372    // filtering; seccomp connect gating catches direct IP connections.
373    let allowed = profile.network.allowed_domains.iter().any(|domain| {
374        // Check if the domain entry is an IP address or CIDR
375        *domain == ip || domain == "*"
376    });
377
378    if !allowed {
379        tracing::info!(
380            pid,
381            ip = %ip,
382            port,
383            "connect denied: not in allowed domains"
384        );
385    } else {
386        tracing::debug!(pid, ip = %ip, port, "connect allowed");
387    }
388
389    allowed
390}

392/// Validate a bind syscall against the profile's network config.
393///
394/// Checks if the agent is allowed to bind to the requested port.
395#[cfg(target_os = "linux")]
396pub(super) fn validate_bind(pid: u32, addr: usize, len: usize, profile: &AgentProfile) -> bool {
397    use puzzled_types::NetworkMode;
398
399    match profile.network.mode {
400        NetworkMode::Blocked => {
401            tracing::info!(pid, "bind denied: network mode is Blocked");
402            return false;
403        }
404        NetworkMode::Unrestricted | NetworkMode::Monitored => {
405            return true;
406        }
407        NetworkMode::Gated => {
408            // Fall through to validation
409        }
410    }
411
412    let (_family, ip, port) = match read_sockaddr_from_proc_mem(pid, addr, len) {
413        Ok(sa) => sa,
414        Err(e) => {
415            tracing::warn!(pid, error = %e, "failed to read bind sockaddr, denying");
416            return false;
417        }
418    };
419
420    // V31: Port 0 binding allowed — ephemeral port in namespaced network is bounded
421    // T8: Include IPv4-mapped IPv6 loopback; T10: remove dead "::1"
422    if ip == "127.0.0.1" || ip == "0:0:0:0:0:0:0:1" || ip == "0:0:0:0:0:ffff:7f00:1" {
423        return true;
424    }
425
426    // In Gated mode, deny binding to non-loopback addresses
427    // Agents should only listen on loopback; external access goes through proxy
428    tracing::info!(
429        pid,
430        ip = %ip,
431        port,
432        "bind denied: non-loopback bind in Gated mode"
433    );
434    false
435}

437/// M2: Normalize path by resolving `.` and `..` components without touching the filesystem.
438/// This prevents path traversal attacks when `canonicalize()` fails (e.g., file doesn't exist yet).
439fn normalize_path_components(path: &str) -> String {
440    let mut parts: Vec<&str> = Vec::new();
441    for component in path.split('/') {
442        match component {
443            "" | "." => {}
444            ".." => {
445                parts.pop();
446            }
447            other => parts.push(other),
448        }
449    }
450    format!("/{}", parts.join("/"))
451}
