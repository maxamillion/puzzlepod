24/// H-23: Handler-side deadline for expensive steps within one notification.
25/// If elapsed time exceeds this threshold before an expensive operation
26/// (reading process memory, policy check, ADDFD injection), the syscall
27/// is denied and the handler returns early.
28///
29/// This is **only** a userspace handler budget: it does **not** unblock the
30/// agent's syscall from the kernel's perspective. The notifying thread stays
31/// blocked in the kernel until puzzled sends an allow/deny response.
32#[cfg_attr(not(target_os = "linux"), allow(dead_code))]
33const NOTIFICATION_TIMEOUT: Duration = Duration::from_secs(5);

36    /// C2: Deny a pending seccomp notification (exec budget exceeded).
37    /// Receives the notification and responds with EPERM without evaluating policy.
38    #[cfg(target_os = "linux")]
39    pub fn deny_notification(notify_fd: i32) -> Result<()> {
40        use libseccomp::*;
41
42        let req = ScmpNotifReq::receive(notify_fd).map_err(|e| {
43            crate::error::PuzzledError::SeccompNotif(format!(
44                "receiving seccomp notification for denial: {}",
45                e
46            ))
47        })?;
48
49        let resp = ScmpNotifResp::new_error(req.id, -(libc::EPERM), ScmpNotifRespFlags::empty());
50
51        if super::notify_id_valid(notify_fd, req.id).is_ok() {
52            resp.respond(notify_fd).map_err(|e| {
53                crate::error::PuzzledError::SeccompNotif(format!(
54                    "responding with denial to seccomp notification: {}",
55                    e
56                ))
57            })?;
58        }
59
60        Ok(())
61    }

70    /// C7: Handle a seccomp notification, incrementing exec_count only for
71    /// execve/execveat syscalls (not connect/bind/clone).
72    ///
73    /// Reads the notification, inspects the syscall arguments (execve path,
74    /// connect address, bind port) via /proc/<pid>/mem, evaluates against the
75    /// agent profile, and responds with ALLOW or DENY.
76    ///
77    /// H-23: Enforces a 5-second timeout. Before each expensive operation,
78    /// checks elapsed time and denies the syscall if the timeout is exceeded.
79    ///
80    /// TOCTOU protection: uses SECCOMP_IOCTL_NOTIF_ID_VALID to check the
81    /// notification is still valid before responding.
82    ///
83    /// DC: H-6 connect/bind sockaddr TOCTOU — fundamental USER_NOTIF limitation.
84    /// Agent threads could modify sockaddr between our /proc/pid/mem read and the
85    /// kernel's response. Mitigated by seccomp blocking `process_vm_writev`.
86    ///
87    /// DC: clone_args memory TOCTOU — clone_args lives in agent memory and could
88    /// be modified between seccomp read and kernel processing. Mitigated by
89    /// `notify_id_valid` check before responding.
90    ///
91    /// DC: H-4 execve canonicalize→open TOCTOU — fundamental race between path
92    /// canonicalization and fd injection. ADDFD with O_PATH is the mitigation.
93    /// Opening the canonicalized path is unavoidable.
94    ///
95    /// C1: Budget enforcement uses increment-then-check (atomic fetch_add followed
96    /// by comparison) to eliminate the TOCTOU race in the previous check-then-increment
97    /// pattern. Ordering::Relaxed is sufficient because exec_count is a monotonically
98    /// increasing counter on a single atomic — no other memory operations depend on
99    /// its ordering relative to other variables.
100    ///
101    /// WS9: `handler_wall_start` must be `Instant::now()` from the poll loop immediately
102    /// before calling this function so the wall-clock deadline includes `receive`.
103    #[cfg(target_os = "linux")]
104    pub fn handle_notification_counted(
105        notify_fd: i32,
106        profile: &AgentProfile,
107        exec_count: &std::sync::atomic::AtomicU64,
108        exec_budget: u64,
109        credential_proxy: Option<&seccomp_handler::CredentialProxyContext>,
110        handler_wall_start: Instant,
111    ) -> Result<()> {
112        use libseccomp::*;
113        use std::sync::atomic::Ordering;
114
115        // H-23: Record entry time for timeout enforcement
116        let entry_time = Instant::now();
117        let req = ScmpNotifReq::receive(notify_fd).map_err(|e| {
118            crate::error::PuzzledError::SeccompNotif(format!(
119                "receiving seccomp notification: {}",
120                e
121            ))
122        })?;
123
124        let syscall = req.data.syscall;
125        let pid = req.pid;
126
127        // Issue #5: Resolve syscall names with explicit error handling rather
128        // than .ok() which silently returns None. If resolution fails, the
129        // handler logs a clear error and denies the syscall (fail-closed).
130        let resolve = |name: &str| -> Option<ScmpSyscall> {
131            match ScmpSyscall::from_name(name) {
132                Ok(nr) => Some(nr),
133                Err(e) => {
134                    tracing::error!(
135                        syscall = name,
136                        error = %e,
137                        "failed to resolve syscall name for USER_NOTIF handler — \
138                         notifications for this syscall will be denied (fail-closed)"
139                    );
140                    None
141                }
142            }
143        };
144
145        // C7: Only increment exec counter for execve/execveat
146        let execve_nr = resolve("execve");
147        let execveat_nr = resolve("execveat");
148        if execve_nr == Some(syscall) || execveat_nr == Some(syscall) {
149            // C1: Increment-then-check eliminates TOCTOU race. fetch_add returns
150            // the previous value atomically, so concurrent notifications cannot
151            // both pass the budget check.
152            let prev = exec_count.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
153            if exec_budget > 0 && prev >= exec_budget {
154                tracing::error!(
155                    pid,
156                    exec_count = prev + 1,
157                    budget = exec_budget,
158                    "exec budget exceeded — denying notification (C1)"
159                );
160                return Self::respond_deny(notify_fd, req.id);
161            }
162        }
163
164        tracing::debug!(
165            pid,
166            syscall = ?syscall,
167            notify_id = req.id,
168            "seccomp USER_NOTIF received (counted)"
169        );
170
171        // Match syscall number to determine validation logic
172        let connect_nr = resolve("connect");
173        let bind_nr = resolve("bind");
174        let clone3_nr = resolve("clone3");
175        let clone_nr = resolve("clone");
176
177        let allow = if execve_nr == Some(syscall) {
178            // execve: arg0 = pointer to path string
179            let path_addr = req.data.args[0] as usize;
180            // H-23: Check timeout before reading process memory
181            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
182                tracing::warn!(
183                    pid,
184                    "seccomp notification timeout before reading execve path (H-23), denying"
185                );
186                false
187            } else {
188                // H6: Read the path ONCE from /proc/<pid>/mem and pass the same string
189                // to both validate_execve and inject_fd_for_execve. This eliminates the
190                // TOCTOU window where an attacker could swap the file between two reads.
191                match read_string_from_proc_mem(pid, path_addr) {
192                    Ok(execve_path) => {
193                        // H-23: Check timeout before policy check
194                        if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
195                            tracing::warn!(pid, "seccomp notification timeout before execve policy check (H-23), denying");
196                            false
197                        } else if validate_execve_with_path(pid, &execve_path, profile) {
198                            // H-23: Check timeout before ADDFD injection
199                            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
200                                tracing::warn!(pid, "seccomp notification timeout before ADDFD injection (H-23), denying");
201                                false
202                            } else {
203                                // SC1: Use SECCOMP_ADDFD for TOCTOU-safe execve when available.
204                                // inject_fd_for_execve_with_path uses the already-read path
205                                // instead of re-reading from /proc/<pid>/mem.
206                                match seccomp_handler::inject_fd_for_execve_with_path(
207                                    notify_fd,
208                                    req.id,
209                                    pid,
210                                    &execve_path,
211                                    profile,
212                                ) {
213                                    Ok(Some(_fd)) => true,
214                                    Ok(None) => {
215                                        tracing::warn!(pid, "ADDFD unavailable (kernel too old), allowing execve via CONTINUE (path validated)");
216                                        true
217                                    }
218                                    Err(e) => {
219                                        // ADDFD is a TOCTOU mitigation (H-4), not the primary
220                                        // control. The path was validated by validate_execve_with_path
221                                        // and Landlock enforces filesystem access independently.
222                                        tracing::warn!(pid, error = %e, "SECCOMP_ADDFD inject failed for execve, allowing via CONTINUE (path validated)");
223                                        true
224                                    }
225                                }
226                            }
227                        } else {
228                            false
229                        }
230                    }
231                    Err(e) => {
232                        tracing::warn!(pid, error = %e, "failed to read execve path, denying");
233                        false
234                    }
235                }
236            }
237        } else if execveat_nr == Some(syscall) {
238            // execveat: arg1 = pointer to path string (arg0 = dirfd)
239            let path_addr = req.data.args[1] as usize;
240            // H-23: Check timeout before reading process memory
241            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
242                tracing::warn!(
243                    pid,
244                    "seccomp notification timeout before reading execveat path (H-23), denying"
245                );
246                false
247            } else {
248                // H6: Read the path ONCE — same TOCTOU fix as execve above.
249                match read_string_from_proc_mem(pid, path_addr) {
250                    Ok(execve_path) => {
251                        // H-23: Check timeout before policy check
252                        if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
253                            tracing::warn!(pid, "seccomp notification timeout before execveat policy check (H-23), denying");
254                            false
255                        } else if validate_execve_with_path(pid, &execve_path, profile) {
256                            // H-23: Check timeout before ADDFD injection
257                            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
258                                tracing::warn!(pid, "seccomp notification timeout before execveat ADDFD injection (H-23), denying");
259                                false
260                            } else {
261                                match seccomp_handler::inject_fd_for_execve_with_path(
262                                    notify_fd,
263                                    req.id,
264                                    pid,
265                                    &execve_path,
266                                    profile,
267                                ) {
268                                    Ok(Some(_fd)) => true,
269                                    Ok(None) => {
270                                        tracing::warn!(pid, "ADDFD unavailable (kernel too old), allowing execveat via CONTINUE (path validated)");
271                                        true
272                                    }
273                                    Err(e) => {
274                                        // ADDFD is a TOCTOU mitigation (H-4), not the primary
275                                        // control. The path was validated by validate_execve_with_path
276                                        // and Landlock enforces filesystem access independently.
277                                        tracing::warn!(
278                                            pid,
279                                            error = %e,
280                                            "SECCOMP_ADDFD inject failed for execveat, allowing via CONTINUE (path validated)"
281                                        );
282                                        true
283                                    }
284                                }
285                            }
286                        } else {
287                            false
288                        }
289                    }
290                    Err(e) => {
291                        tracing::warn!(pid, error = %e, "failed to read execveat path, denying");
292                        false
293                    }
294                }
295            }
296        } else if connect_nr == Some(syscall) {
297            // H-23: Check timeout before reading connect sockaddr
298            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
299                tracing::warn!(
300                    pid,
301                    "seccomp notification timeout before connect validation (H-23), denying"
302                );
303                false
304            } else {
305                validate_connect(
306                    pid,
307                    req.data.args[1] as usize,
308                    req.data.args[2] as usize,
309                    profile,
310                    credential_proxy,
311                )
312            }
313        } else if clone3_nr == Some(syscall) {
314            // H-23: Check timeout before reading clone3 flags from process memory
315            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
316                tracing::warn!(
317                    pid,
318                    "seccomp notification timeout before reading clone3 flags (H-23), denying"
319                );
320                false
321            } else {
322                // SC2: clone3 USER_NOTIF handler — check clone flags for namespace escape.
323                // When BPF clone guard is not active, clone3 is gated through USER_NOTIF.
324                //
325                // CRITICAL: For clone3, args[0] is a POINTER to struct clone_args, NOT
326                // the flags directly. The flags field is at offset 0 of the struct.
327                // V33: clone_args.flags is at offset 0 — stable kernel ABI since 5.3
328                // We must read the flags from the agent's memory via /proc/<pid>/mem.
329                let clone_flags = match read_u64_from_proc_mem(pid, req.data.args[0] as usize) {
330                    Ok(flags) => flags,
331                    Err(e) => {
332                        tracing::warn!(
333                            pid,
334                            error = %e,
335                            "failed to read clone3 flags from process memory, denying (fail-closed)"
336                        );
337                        return Self::respond_deny(notify_fd, req.id);
338                    }
339                };
340                // H8: Block all namespace isolation flags — agents must not create
341                // new IPC, UTS, or cgroup namespaces in addition to the original set.
342                let forbidden_flags: u64 = libc::CLONE_NEWNS as u64
343                    | libc::CLONE_NEWPID as u64
344                    | libc::CLONE_NEWNET as u64
345                    | libc::CLONE_NEWUSER as u64
346                    | libc::CLONE_NEWIPC as u64
347                    | libc::CLONE_NEWUTS as u64
348                    | libc::CLONE_NEWCGROUP as u64;
349                if clone_flags & forbidden_flags != 0 {
350                    tracing::warn!(
351                        pid,
352                        clone_flags = format!("0x{:x}", clone_flags),
353                        "clone3 denied: namespace escape flags detected (no BPF clone guard)"
354                    );
355                    false
356                } else {
357                    true
358                }
359            }
360        } else if clone_nr == Some(syscall) {
361            // H-23: Check timeout before validating clone flags
362            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
363                tracing::warn!(
364                    pid,
365                    "seccomp notification timeout before clone flag validation (H-23), denying"
366                );
367                false
368            } else {
369                // Legacy clone: args[0] IS the flags directly (unlike clone3 which uses a pointer).
370                let clone_flags = req.data.args[0];
371                // H8: Block all namespace isolation flags — same set as clone3 above.
372                let forbidden_flags: u64 = libc::CLONE_NEWNS as u64
373                    | libc::CLONE_NEWPID as u64
374                    | libc::CLONE_NEWNET as u64
375                    | libc::CLONE_NEWUSER as u64
376                    | libc::CLONE_NEWIPC as u64
377                    | libc::CLONE_NEWUTS as u64
378                    | libc::CLONE_NEWCGROUP as u64;
379                if clone_flags & forbidden_flags != 0 {
380                    tracing::warn!(
381                        pid,
382                        clone_flags = format!("0x{:x}", clone_flags),
383                        "legacy clone denied: namespace escape flags detected (no BPF clone guard)"
384                    );
385                    false
386                } else {
387                    true
388                }
389            }
390        } else if bind_nr == Some(syscall) {
391            // H-23: Check timeout before reading bind sockaddr
392            if entry_time.elapsed() > NOTIFICATION_TIMEOUT {
393                tracing::warn!(
394                    pid,
395                    "seccomp notification timeout before bind validation (H-23), denying"
396                );
397                false
398            } else {
399                validate_bind(
400                    pid,
401                    req.data.args[1] as usize,
402                    req.data.args[2] as usize,
403                    profile,
404                )
405            }
406        } else {
407            tracing::error!(pid, syscall = ?syscall, "unknown seccomp notify syscall, denying (fail-closed)");
408            false
409        };
410
411        // TOCTOU mitigation: After reading from /proc/<pid>/mem and before
412        // acting on the result, re-check that the notification is still valid.
413        // This narrows the TOCTOU window where the agent process could have
414        // been replaced or the memory contents changed between our read and
415        // the response. The kernel's own notify_id_valid check in respond()
416        // provides the final guarantee, but this belt-and-suspenders check
417        // catches stale data earlier.
418        if super::notify_id_valid(notify_fd, req.id).is_err() {
419            tracing::debug!(
420                pid,
421                notify_id = req.id,
422                "seccomp notification expired after validation (TOCTOU window)"
423            );
424            return Ok(());
425        }
426
427        // WS9: Hard wall-clock deadline from poll-loop start. If we would allow but took
428        // too long, deny with EPERM so the poll loop can eventually serve other agents.
429        let allow = if allow
430            && handler_wall_start.elapsed() > seccomp_handler::SECCOMP_NOTIF_HANDLER_WALL_DEADLINE
431        {
432            seccomp_handler::SECCOMP_NOTIF_HANDLER_WALL_DEADLINE_DENIES
433                .fetch_add(1, Ordering::Relaxed);
434            tracing::warn!(
435                pid,
436                notify_id = req.id,
437                elapsed_ms = handler_wall_start.elapsed().as_millis() as u64,
438                "WS9: seccomp notification wall-clock deadline exceeded — denying with EPERM \
439                 (single-threaded handler; other agents' gated syscalls were blocked)"
440            );
441            false
442        } else {
443            allow
444        };
445
446        // Build the response
447        // For allowed syscalls, use CONTINUE to let the kernel execute the original
448        // syscall (critical for execve — new_val(0) would fake-succeed without loading
449        // the binary). For denied syscalls, return EPERM.
450        let resp = if allow {
451            ScmpNotifResp::new_continue(req.id, ScmpNotifRespFlags::empty())
452        } else {
453            ScmpNotifResp::new_error(req.id, -(libc::EPERM), ScmpNotifRespFlags::empty())
454        };
455
456        // Verify the notification is still valid (final TOCTOU protection)
457        if super::notify_id_valid(notify_fd, req.id).is_ok() {
458            resp.respond(notify_fd).map_err(|e| {
459                crate::error::PuzzledError::SeccompNotif(format!(
460                    "responding to seccomp notification: {}",
461                    e
462                ))
463            })?;
464        }
465
466        Ok(())
467    }

469    /// Helper to respond with denial.
470    #[cfg(target_os = "linux")]
471    fn respond_deny(notify_fd: i32, notify_id: u64) -> Result<()> {
472        use libseccomp::*;
473        let resp = ScmpNotifResp::new_error(notify_id, -(libc::EPERM), ScmpNotifRespFlags::empty());
474        if super::notify_id_valid(notify_fd, notify_id).is_ok() {
475            resp.respond(notify_fd).map_err(|e| {
476                crate::error::PuzzledError::SeccompNotif(format!("responding with denial: {}", e))
477            })?;
478        }
479        Ok(())
480    }
