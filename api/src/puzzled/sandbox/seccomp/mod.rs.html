9/// seccomp-BPF filter builder with dual-mode support:
10///
11/// **Permissive** (default-allow):
12///   - Default action: **ALLOW**
13///   - **KILL_PROCESS**: Known escape vectors (ptrace, mount, bpf, io_uring, ...)
14///   - **USER_NOTIF**: High-impact syscalls gated through puzzled
15///   - Unknown syscalls pass through — avoids silent workload failures
16///
17/// **Strict** (default-deny):
18///   - Default action: **EPERM**
19///   - **ALLOW**: Safe syscalls from a curated allowlist (~120 entries)
20///   - **USER_NOTIF**: High-impact syscalls gated through puzzled
21///   - **KILL_PROCESS**: Known escape vectors
22///   - Novel/unknown syscalls return EPERM
23///
24/// Both modes share the same KillProcess deny list (escape vectors) and
25/// USER_NOTIF list (execve, connect, bind). The difference is what happens
26/// to syscalls not in any list: Permissive allows them, Strict returns EPERM.
27///
28/// Once loaded via `seccomp()`, the filter is irrevocable.
29pub struct SeccompBuilder {
30    /// SC2: Whether the BPF LSM clone guard is active.
31    /// When false, clone3 is added to the seccomp USER_NOTIF list so
32    /// namespace escape flags can be checked by the notification handler.
33    pub bpf_clone_guard_active: bool,
34    /// Filter strategy: Permissive (default-allow) or Strict (default-deny).
35    pub seccomp_mode: puzzled_types::SeccompMode,
36}

154/// Apply a minimal seccomp hardening filter to the puzzled daemon itself.
155///
156/// Blocks syscalls that puzzled should never need (see [`DAEMON_DENY_SYSCALLS`]).
157/// Uses KillProcess action — a denied escape-vector syscall indicates compromise.
158/// This is a defense-in-depth measure — even if puzzled is compromised, it cannot
159/// load kernel modules, reboot the system, or escape its namespace.
160#[cfg(target_os = "linux")]
161pub fn apply_daemon_hardening() -> Result<()> {
162    use libseccomp::*;
163
164    let mut filter = ScmpFilterContext::new_filter(ScmpAction::Allow).map_err(|e| {
165        crate::error::PuzzledError::Sandbox(format!("creating daemon seccomp filter: {}", e))
166    })?;
167
168    // Issue #3: Use KillProcess instead of EPERM. A denied escape-vector
169    // syscall in the daemon indicates compromise — the daemon should crash
170    // immediately (triggering systemd restart and audit logging) rather than
171    // continue running in a potentially compromised state.
172    //
173    // Issue #4: Fail hard on add_rule errors. Partial hardening is dangerous
174    // and undetectable. If a deny rule cannot be added, startup must fail.
175    for name in DAEMON_DENY_SYSCALLS {
176        match ScmpSyscall::from_name(name) {
177            Ok(syscall) => {
178                filter
179                    .add_rule(ScmpAction::KillProcess, syscall)
180                    .map_err(|e| {
181                        crate::error::PuzzledError::Sandbox(format!(
182                            "adding daemon hardening rule for {}: {}",
183                            name, e
184                        ))
185                    })?;
186            }
187            Err(e) => {
188                return Err(crate::error::PuzzledError::Sandbox(format!(
189                    "daemon hardening: syscall '{}' not found: {} \
190                     (libseccomp may be too old or syscall name invalid)",
191                    name, e
192                )));
193            }
194        }
195    }
196
197    // x86_64-only: block modify_ldt (segmentation-based sandbox escapes).
198    #[cfg(target_arch = "x86_64")]
199    {
200        let name = "modify_ldt";
201        match ScmpSyscall::from_name(name) {
202            Ok(syscall) => {
203                filter
204                    .add_rule(ScmpAction::KillProcess, syscall)
205                    .map_err(|e| {
206                        crate::error::PuzzledError::Sandbox(format!(
207                            "adding daemon hardening rule for {}: {}",
208                            name, e
209                        ))
210                    })?;
211            }
212            Err(e) => {
213                return Err(crate::error::PuzzledError::Sandbox(format!(
214                    "daemon hardening: syscall '{}' not found: {}",
215                    name, e
216                )));
217            }
218        }
219    }
220
221    filter.load().map_err(|e| {
222        crate::error::PuzzledError::Sandbox(format!("loading daemon seccomp filter: {}", e))
223    })?;
224
225    tracing::info!(
226        blocked = DAEMON_DENY_SYSCALLS.len(),
227        "puzzled self-hardening seccomp filter loaded (KillProcess)"
228    );
229    Ok(())
230}

242/// Validate that a seccomp notification ID is still valid (TOCTOU protection).
243///
244/// Uses `ioctl(SECCOMP_IOCTL_NOTIF_ID_VALID)` directly rather than relying on
245/// libseccomp bindings, which may not expose this ioctl. The kernel checks that
246/// the notification identified by `id` has not been invalidated (e.g., because
247/// the notifying process exited or was killed between our /proc/pid/mem read
248/// and our response).
249///
250/// Returns Ok(()) if the notification is still valid, Err otherwise.
251#[cfg(target_os = "linux")]
252#[allow(dead_code)] // Will be called from seccomp notification handler once wired up
253fn notify_id_valid(notify_fd: i32, id: u64) -> std::result::Result<(), String> {
254    // SECCOMP_IOCTL_NOTIF_ID_VALID is defined as SECCOMP_IOW(2, __u64)
255    // which expands to _IOW('!', 2, u64) = 0x40082102 on most architectures.
256    //
257    // The ioctl takes a pointer to the notification ID (u64) and returns 0
258    // if valid, or -1 with errno=ENOENT if the notification has expired.
259    const SECCOMP_IOCTL_NOTIF_ID_VALID: libc::c_ulong = 0x40082102;
260
261    let mut check_id = id;
262    // SAFETY: We pass a valid fd (the seccomp notify fd) and a pointer to a
263    // stack-allocated u64 with the correct ioctl number. The kernel will read
264    // the u64 value and validate it against its internal notification table.
265    let ret = unsafe {
266        libc::ioctl(
267            notify_fd,
268            SECCOMP_IOCTL_NOTIF_ID_VALID as libc::c_ulong,
269            &mut check_id as *mut u64,
270        )
271    };
272
273    if ret == 0 {
274        Ok(())
275    } else {
276        let errno = std::io::Error::last_os_error();
277        Err(format!("notification id {} no longer valid: {}", id, errno))
278    }
279}

367    /// Helper to check if a path matches an allowlist (mirrors validate_execve logic).
368    ///
369    /// Note: In the real validate_execve, paths are canonicalized via
370    /// /proc/<pid>/root before matching. This helper tests the post-
371    /// canonicalization matching logic only.
372    fn path_matches_allowlist(path: &str, allowlist: &[String]) -> bool {
373        if allowlist.is_empty() {
374            return false;
375        }
376        allowlist.iter().any(|pattern| {
377            if pattern.ends_with('*') {
378                let prefix = &pattern[..pattern.len() - 1];
379                path.starts_with(prefix)
380            } else {
381                path == *pattern
382            }
383        })
384    }
