306impl super::SeccompBuilder {
307    /// Build and load a seccomp-BPF filter.
308    ///
309    /// The filter strategy depends on `self.seccomp_mode`:
310    ///
311    /// **Permissive** (default-allow):
312    ///   - Default: ALLOW
313    ///   - KILL_PROCESS for escape vectors ([`DENY_SYSCALLS`])
314    ///   - USER_NOTIF for execve/connect/bind (+ clone when BPF guard inactive)
315    ///   - Raw socket blocking via argument filtering
316    ///
317    /// **Strict** (default-deny):
318    ///   - Default: EPERM
319    ///   - ALLOW for curated allowlist ([`ALLOW_SYSCALLS`])
320    ///   - KILL_PROCESS for escape vectors ([`DENY_SYSCALLS`])
321    ///   - USER_NOTIF for execve/connect/bind (+ clone when BPF guard inactive)
322    ///   - Raw socket blocking via argument filtering
323    ///
324    /// Returns the USER_NOTIF file descriptor for puzzled to poll.
325    /// Requires libseccomp >= 2.5.0 for USER_NOTIF support (get_notify_fd).
326    #[cfg(target_os = "linux")]
327    pub fn apply(&self, _profile: &AgentProfile) -> Result<Option<i32>> {
328        use libseccomp::*;
329
330        let default_action = match self.seccomp_mode {
331            SeccompMode::Permissive => ScmpAction::Allow,
332            SeccompMode::Strict => ScmpAction::Errno(libc::EPERM),
333        };
334
335        let mut filter = ScmpFilterContext::new_filter(default_action).map_err(|e| {
336            crate::error::PuzzledError::Sandbox(format!("creating seccomp filter: {}", e))
337        })?;
338
339        // ── Tier 1: Allowlist (Strict mode only) ──────────────────────────
340        // In Permissive mode the default action is ALLOW, so we skip this.
341        let mut allow_count = 0usize;
342        if self.seccomp_mode == SeccompMode::Strict {
343            for syscall_name in ALLOW_SYSCALLS {
344                match ScmpSyscall::from_name(syscall_name) {
345                    Ok(syscall) => {
346                        filter.add_rule(ScmpAction::Allow, syscall).map_err(|e| {
347                            crate::error::PuzzledError::Sandbox(format!(
348                                "adding seccomp allow rule for {}: {}",
349                                syscall_name, e
350                            ))
351                        })?;
352                        allow_count += 1;
353                    }
354                    Err(e) => {
355                        tracing::debug!(
356                            syscall = syscall_name,
357                            error = %e,
358                            "skipping syscall not available on this architecture"
359                        );
360                    }
361                }
362            }
363        }
364
365        // clone/clone3: when BPF clone guard is active, allow directly (BPF
366        // LSM filters flags). In Permissive mode they're already allowed by
367        // default; in Strict mode we need an explicit rule.
368        // NOTE: clone and clone3 are intentionally NOT in the deny list.
369        // Agents need clone/clone3 for thread creation (CLONE_VM|CLONE_FS).
370        // Namespace escape via clone flags (CLONE_NEWNS, CLONE_NEWPID, etc.)
371        // is prevented by the seccomp USER_NOTIF handler or BPF LSM argument
372        // filtering — blocking clone entirely would prevent any multi-threaded
373        // workload from functioning. The unshare and setns syscalls remain
374        // blocked to cover the other namespace escape vectors.
375        if self.bpf_clone_guard_active && self.seccomp_mode == SeccompMode::Strict {
376            for name in &["clone", "clone3"] {
377                if let Ok(syscall) = ScmpSyscall::from_name(name) {
378                    filter.add_rule(ScmpAction::Allow, syscall).map_err(|e| {
379                        crate::error::PuzzledError::Sandbox(format!(
380                            "adding seccomp allow rule for {}: {}",
381                            name, e
382                        ))
383                    })?;
384                    allow_count += 1;
385                }
386            }
387        }
388
389        // ── Tier 3: KillProcess deny list (both modes) ───────────────────
390        for syscall_name in DENY_SYSCALLS {
391            match ScmpSyscall::from_name(syscall_name) {
392                Ok(syscall) => {
393                    filter
394                        .add_rule(ScmpAction::KillProcess, syscall)
395                        .map_err(|e| {
396                            crate::error::PuzzledError::Sandbox(format!(
397                                "adding seccomp deny rule for {}: {}",
398                                syscall_name, e
399                            ))
400                        })?;
401                }
402                Err(e) => {
403                    return Err(crate::error::PuzzledError::Sandbox(format!(
404                        "seccomp deny rule for '{}' failed: {} \
405                         (libseccomp may be too old or syscall name invalid)",
406                        syscall_name, e
407                    )));
408                }
409            }
410        }
411
412        // ── Raw socket blocking (both modes) ──────────────────────────────
413        // socket(domain, type, protocol) — arg1 (type) is checked for
414        // SOCK_RAW and SOCK_PACKET. MaskedEqual ignores SOCK_NONBLOCK and
415        // SOCK_CLOEXEC flags.
416        // Issue #6: Fail hard if socket syscall cannot be resolved. SOCK_RAW
417        // allows packet injection, ARP spoofing, and kernel exploit delivery.
418        // The socket syscall exists on every supported Linux system.
419        let socket_syscall = ScmpSyscall::from_name("socket").map_err(|e| {
420            crate::error::PuzzledError::Sandbox(format!(
421                "socket syscall not found — cannot block SOCK_RAW/SOCK_PACKET: {}",
422                e
423            ))
424        })?;
425        {
426            use libseccomp::{ScmpArgCompare, ScmpCompareOp};
427            const SOCK_TYPE_MASK: u64 = 0x0F;
428            const SOCK_RAW: u64 = 3;
429            const SOCK_PACKET: u64 = 10;
430
431            let raw_cmp =
432                ScmpArgCompare::new(1, ScmpCompareOp::MaskedEqual(SOCK_TYPE_MASK), SOCK_RAW);
433            filter
434                .add_rule_conditional(ScmpAction::KillProcess, socket_syscall, &[raw_cmp])
435                .map_err(|e| {
436                    crate::error::PuzzledError::Sandbox(format!(
437                        "adding seccomp rule to block SOCK_RAW: {}",
438                        e
439                    ))
440                })?;
441
442            let packet_cmp =
443                ScmpArgCompare::new(1, ScmpCompareOp::MaskedEqual(SOCK_TYPE_MASK), SOCK_PACKET);
444            filter
445                .add_rule_conditional(ScmpAction::KillProcess, socket_syscall, &[packet_cmp])
446                .map_err(|e| {
447                    crate::error::PuzzledError::Sandbox(format!(
448                        "adding seccomp rule to block SOCK_PACKET: {}",
449                        e
450                    ))
451                })?;
452
453            tracing::debug!("seccomp: SOCK_RAW and SOCK_PACKET blocked via argument filtering");
454        }
455
456        // ── Tier 2: USER_NOTIF (both modes) ──────────────────────────────
457        let mut notify_syscalls: Vec<&str> = BASE_NOTIFY_SYSCALLS.to_vec();
458        if !self.bpf_clone_guard_active {
459            tracing::warn!(
460                "BPF clone guard not active — adding clone3/clone to seccomp USER_NOTIF \
461                 for namespace escape prevention"
462            );
463            notify_syscalls.push("clone3");
464            notify_syscalls.push("clone");
465        }
466
467        for syscall_name in &notify_syscalls {
468            match ScmpSyscall::from_name(syscall_name) {
469                Ok(syscall) => {
470                    filter.add_rule(ScmpAction::Notify, syscall).map_err(|e| {
471                        crate::error::PuzzledError::Sandbox(format!(
472                            "adding seccomp notify rule for {}: {}",
473                            syscall_name, e
474                        ))
475                    })?;
476                }
477                Err(e) => {
478                    tracing::warn!(
479                        syscall = syscall_name,
480                        error = %e,
481                        "skipping unknown syscall in seccomp notify list"
482                    );
483                }
484            }
485        }
486
487        // Load the filter — irrevocable once loaded
488        filter.load().map_err(|e| {
489            crate::error::PuzzledError::Sandbox(format!("loading seccomp filter: {}", e))
490        })?;
491
492        let notify_fd = filter.get_notify_fd().map_err(|e| {
493            crate::error::PuzzledError::Sandbox(format!("getting seccomp notify fd: {}", e))
494        })?;
495
496        let mode_label = match self.seccomp_mode {
497            SeccompMode::Permissive => "permissive (default-allow)",
498            SeccompMode::Strict => "strict (default-deny)",
499        };
500
501        tracing::info!(
502            notify_fd,
503            mode = mode_label,
504            allow_count,
505            deny_count = DENY_SYSCALLS.len(),
506            notify_count = notify_syscalls.len(),
507            "seccomp-BPF filter loaded [{}]: {} allowed, {} kill-on-use, \
508             {} USER_NOTIF-gated; Landlock enforces filesystem ACL",
509            mode_label,
510            allow_count,
511            DENY_SYSCALLS.len(),
512            notify_syscalls.len()
513        );
514
515        Ok(if notify_fd >= 0 {
516            Some(notify_fd)
517        } else {
518            None
519        })
520    }
521
522    #[cfg(not(target_os = "linux"))]
523    pub fn apply(&self, _profile: &AgentProfile) -> Result<Option<i32>> {
524        Err(crate::error::PuzzledError::Sandbox(
525            "seccomp requires Linux".to_string(),
526        ))
527    }
528}
