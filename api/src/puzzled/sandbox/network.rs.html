2//! Network isolation and gating for agent sandboxes.
3//!
4//! Implements three network modes:
5//! - **Blocked**: Empty network namespace with no interfaces (existing behavior)
6//! - **Gated**: veth pair (agent NS ↔ host NS) with nftables restricting agent
7//!   to HTTP proxy only; proxy does domain-level filtering
8//! - **Monitored**: veth pair with full routing, nftables logging
9//!
10//! The agent's environment gets `HTTP_PROXY` / `HTTPS_PROXY` pointing to the
11//! in-namespace proxy endpoint for transparent domain filtering.
12
13#[cfg(target_os = "linux")]
14use crate::error::PuzzledError;
15use crate::error::Result;
16use puzzled_types::NetworkMode;
17use std::net::SocketAddr;
18
19/// Network configuration for a running sandbox.
20pub struct NetworkSetup {
21    /// Network mode in use.
22    pub mode: NetworkMode,
23    /// veth interface name on the host side (if Gated/Monitored).
24    pub host_veth: Option<String>,
25    /// veth interface name inside the agent namespace (if Gated/Monitored).
26    pub agent_veth: Option<String>,
27    /// Proxy address the agent should use (if Gated).
28    pub proxy_addr: Option<SocketAddr>,
29    /// nftables ruleset name (for cleanup).
30    pub nft_table: Option<String>,
31    /// Named network namespace (for cleanup).
32    pub netns_name: Option<String>,
33}
34
35impl NetworkSetup {
36    /// Set up network isolation for a branch.
37    ///
38    /// For Blocked mode, the named network namespace (created by the parent
39    /// before clone3 and joined by the child via setns) provides an empty
40    /// network namespace — no additional setup needed.
41    ///
42    /// For Gated/Monitored modes, creates a veth pair and configures nftables.
43    /// `netns_name` is the named network namespace created by
44    /// `create_named_netns()` before clone3.
45    #[cfg(target_os = "linux")]
46    pub fn configure(
47        branch_id: &str,
48        mode: NetworkMode,
49        netns_name: &str,
50        proxy_port: u16,
51    ) -> Result<Self> {
52        match mode {
53            NetworkMode::Blocked => {
54                tracing::info!(
55                    branch_id,
56                    "network mode: Blocked (empty namespace, no interfaces)"
57                );
58                Ok(Self {
59                    mode,
60                    host_veth: None,
61                    agent_veth: None,
62                    proxy_addr: None,
63                    nft_table: None,
64                    netns_name: None,
65                })
66            }
67            NetworkMode::Gated => Self::setup_gated(branch_id, netns_name, proxy_port),
68            NetworkMode::Monitored => Self::setup_monitored(branch_id, netns_name),
69            NetworkMode::Unrestricted => {
70                tracing::warn!(branch_id, "network mode: Unrestricted (no isolation)");
71                Ok(Self {
72                    mode,
73                    host_veth: None,
74                    agent_veth: None,
75                    proxy_addr: None,
76                    nft_table: None,
77                    netns_name: None,
78                })
79            }
80        }
81    }

83    /// Set up Gated network mode.
84    ///
85    /// Creates a veth pair, assigns IP addresses, and configures nftables
86    /// to restrict the agent to only communicate with the HTTP proxy.
87    #[cfg(target_os = "linux")]
88    fn setup_gated(branch_id: &str, netns_name: &str, proxy_port: u16) -> Result<Self> {
89        // M9: Generate deterministic, collision-resistant veth names using CRC32
90        // of the full branch_id. This avoids collisions that occur when using
91        // branch_id[..8] for branches with shared prefixes (e.g., UUIDs).
92        // The CRC32 hash produces an 8-char hex string, fitting within IFNAMSIZ (15).
93        let short_id = format!("{:08x}", crc32fast::hash(branch_id.as_bytes()));
94        let host_veth = format!("vh{}", &short_id);
95        let agent_veth = format!("va{}", &short_id);
96
97        // L7: Randomized /30 subnet allocation to avoid IP collision when
98        // multiple branches are active. Uses SHA256 of branch_id plus a random
99        // salt for entropy. If a collision is detected (ip addr add fails),
100        // retries with additional randomness up to MAX_SUBNET_RETRIES times.
101        const MAX_SUBNET_RETRIES: u32 = 5;
102        let mut retry = 0u32;
103
104        // Clean up stale veth pair from a previous failed run (if any).
105        // This prevents "RTNETLINK answers: File exists" errors when re-creating
106        // veth pairs after a crash or test failure that left interfaces behind.
107        // Q3: Log stale veth cleanup failures instead of silently discarding
108        if let Err(e) = run_ip_cmd(&["link", "del", &host_veth]) {
109            tracing::trace!(error = %e, veth = %host_veth, "Q3: stale veth cleanup failed (may not exist)");
110        }
111
112        // Create veth pair (before subnet assignment — subnet retries only re-do IP config)
113        run_ip_cmd(&[
114            "link",
115            "add",
116            &host_veth,
117            "type",
118            "veth",
119            "peer",
120            "name",
121            &agent_veth,
122        ])?;
123
124        // Move agent-side veth into the agent's network namespace.
125        // Uses the named netns (created from pidfd via setns) instead of PID-based
126        // /proc/<pid>/ns/net, which may not be visible on some VM configurations.
127        if let Err(e) = run_ip_cmd(&["link", "set", &agent_veth, "netns", netns_name]) {
128            let _ = run_ip_cmd(&["link", "del", &host_veth]);
129            return Err(PuzzledError::Network(format!(
130                "moving veth to netns '{}': {}",
131                netns_name, e
132            )));
133        }
134
135        let (_host_ip, agent_ip, proxy_ip) = loop {
136            let (octet3, octet4_base) = {
137                use sha2::{Digest, Sha256};
138                let mut hasher = Sha256::new();
139                hasher.update(branch_id.as_bytes());
140                // Mix in retry counter for collision recovery
141                hasher.update(retry.to_le_bytes());
142                // Mix in PID and timestamp for additional entropy across restarts
143                hasher.update(std::process::id().to_le_bytes());
144                hasher.update(
145                    std::time::SystemTime::now()
146                        .duration_since(std::time::UNIX_EPOCH)
147                        .unwrap_or_default()
148                        .as_nanos()
149                        .to_le_bytes(),
150                );
151                let hash = hasher.finalize();
152                (hash[0], hash[1] & 0xFC) // /30 subnet: 4 addresses per subnet
153            };
154            let h_ip = format!("10.42.{}.{}/30", octet3, octet4_base | 1);
155            let a_ip = format!("10.42.{}.{}/30", octet3, octet4_base | 2);
156            let p_ip = format!("10.42.{}.{}", octet3, octet4_base | 1);
157
158            // Try to assign the IP — failure likely means subnet collision
159            match run_ip_cmd(&["addr", "add", &h_ip, "dev", &host_veth]) {
160                Ok(()) => break (h_ip, a_ip, p_ip),
161                Err(e) => {
162                    retry += 1;
163                    if retry >= MAX_SUBNET_RETRIES {
164                        return Err(PuzzledError::Network(format!(
165                            "failed to allocate non-colliding /30 subnet after {} retries: {}",
166                            MAX_SUBNET_RETRIES, e
167                        )));
168                    }
169                    tracing::warn!(
170                        branch_id,
171                        retry,
172                        subnet = format!("10.42.{}.{}/30", octet3, octet4_base),
173                        "subnet collision detected, retrying with different hash"
174                    );
175                }
176            }
177        };
178        run_ip_cmd(&["link", "set", &host_veth, "up"])?;
179
180        // Configure agent side (via named netns)
181        run_netns_ip(netns_name, &["addr", "add", &agent_ip, "dev", &agent_veth])?;
182        run_netns_ip(netns_name, &["link", "set", &agent_veth, "up"])?;
183        run_netns_ip(netns_name, &["link", "set", "lo", "up"])?;
184        run_netns_ip(netns_name, &["route", "add", "default", "via", &proxy_ip])?;
185
186        // G15: Validate proxy_ip matches expected 10.42.x.y pattern before
187        // interpolation into nftables rules to prevent injection attacks.
188        debug_assert!(
189            proxy_ip.starts_with("10.42."),
190            "G15: proxy_ip must be in 10.42.0.0/16 subnet"
191        );
192        if !proxy_ip.starts_with("10.42.") {
193            return Err(PuzzledError::Network(format!(
194                "G15: unexpected proxy_ip '{}' — must be in 10.42.0.0/16",
195                proxy_ip
196            )));
197        }
198
199        // H30: Validate short_id contains only hex chars before interpolation
200        // into nftables rules to prevent injection attacks.
201        if !short_id.chars().all(|c| c.is_ascii_hexdigit()) {
202            return Err(PuzzledError::Network(format!(
203                "H30: short_id '{}' contains non-hex characters",
204                short_id
205            )));
206        }
207
208        // Configure nftables: agent can only reach proxy on host side
209        let nft_table = format!("agent_{}", short_id);
210        let nft_rules = format!(
211            r#"table inet {table} {{
212    chain agent_output {{
213        type filter hook output priority 0; policy drop;
214        # Allow loopback
215        oifname "lo" accept
216        # Allow traffic to proxy
217        ip daddr {proxy_ip} tcp dport {proxy_port} accept
218        # U11: UDP 53 is intentional — DNS resolution for Gated mode. DNS tunneling is mitigated by the proxy's domain allowlist.
219        ip daddr {proxy_ip} udp dport 53 accept
220        # Allow established/related
221        ct state established,related accept
222        # Log and drop everything else
223        log prefix "agent-drop: " drop
224    }}
225}}"#,
226            table = nft_table,
227            proxy_ip = proxy_ip,
228            proxy_port = proxy_port,
229        );
230
231        // Apply nftables rules inside agent namespace
232        run_netns_nft(netns_name, &nft_rules)?;
233
234        let proxy_addr: SocketAddr = format!("{}:{}", proxy_ip, proxy_port)
235            .parse()
236            .map_err(|e| PuzzledError::Network(format!("invalid proxy addr: {}", e)))?;
237
238        tracing::info!(
239            branch_id,
240            host_veth = %host_veth,
241            agent_veth = %agent_veth,
242            proxy_addr = %proxy_addr,
243            "network mode: Gated (veth + nftables + proxy)"
244        );
245
246        Ok(Self {
247            mode: NetworkMode::Gated,
248            host_veth: Some(host_veth),
249            agent_veth: Some(agent_veth),
250            proxy_addr: Some(proxy_addr),
251            nft_table: Some(nft_table),
252            netns_name: Some(netns_name.to_string()),
253        })
254    }

256    /// Set up Monitored network mode.
257    ///
258    /// Creates a veth pair with full routing but adds nftables logging
259    /// for all network activity.
260    #[cfg(target_os = "linux")]
261    fn setup_monitored(branch_id: &str, netns_name: &str) -> Result<Self> {
262        // M9: Use CRC32 hash for deterministic, collision-resistant veth names
263        let short_id = format!("{:08x}", crc32fast::hash(branch_id.as_bytes()));
264        let host_veth = format!("vh{}", &short_id);
265        let agent_veth = format!("va{}", &short_id);
266
267        // L7: Randomized /30 subnet allocation (see setup_gated for full rationale)
268        const MAX_SUBNET_RETRIES: u32 = 5;
269        let mut retry = 0u32;
270
271        // Clean up stale veth pair from a previous failed run (if any).
272        // Q3: Log stale veth cleanup failures instead of silently discarding
273        if let Err(e) = run_ip_cmd(&["link", "del", &host_veth]) {
274            tracing::trace!(error = %e, veth = %host_veth, "Q3: stale veth cleanup failed (may not exist)");
275        }
276
277        // Create veth pair
278        run_ip_cmd(&[
279            "link",
280            "add",
281            &host_veth,
282            "type",
283            "veth",
284            "peer",
285            "name",
286            &agent_veth,
287        ])?;
288
289        // Move agent-side veth into the agent's network namespace.
290        if let Err(e) = run_ip_cmd(&["link", "set", &agent_veth, "netns", netns_name]) {
291            let _ = run_ip_cmd(&["link", "del", &host_veth]);
292            return Err(PuzzledError::Network(format!(
293                "moving veth to netns '{}': {}",
294                netns_name, e
295            )));
296        }
297
298        let (_host_ip, agent_ip, proxy_ip) = loop {
299            let (octet3, octet4_base) = {
300                use sha2::{Digest, Sha256};
301                let mut hasher = Sha256::new();
302                hasher.update(branch_id.as_bytes());
303                hasher.update(retry.to_le_bytes());
304                hasher.update(std::process::id().to_le_bytes());
305                hasher.update(
306                    std::time::SystemTime::now()
307                        .duration_since(std::time::UNIX_EPOCH)
308                        .unwrap_or_default()
309                        .as_nanos()
310                        .to_le_bytes(),
311                );
312                let hash = hasher.finalize();
313                (hash[0], hash[1] & 0xFC)
314            };
315            let h_ip = format!("10.42.{}.{}/30", octet3, octet4_base | 1);
316            let a_ip = format!("10.42.{}.{}/30", octet3, octet4_base | 2);
317            let p_ip = format!("10.42.{}.{}", octet3, octet4_base | 1);
318
319            match run_ip_cmd(&["addr", "add", &h_ip, "dev", &host_veth]) {
320                Ok(()) => break (h_ip, a_ip, p_ip),
321                Err(e) => {
322                    retry += 1;
323                    if retry >= MAX_SUBNET_RETRIES {
324                        return Err(PuzzledError::Network(format!(
325                            "failed to allocate non-colliding /30 subnet after {} retries: {}",
326                            MAX_SUBNET_RETRIES, e
327                        )));
328                    }
329                    tracing::warn!(
330                        branch_id,
331                        retry,
332                        subnet = format!("10.42.{}.{}/30", octet3, octet4_base),
333                        "subnet collision detected, retrying with different hash"
334                    );
335                }
336            }
337        };
338        run_ip_cmd(&["link", "set", &host_veth, "up"])?;
339
340        run_netns_ip(netns_name, &["addr", "add", &agent_ip, "dev", &agent_veth])?;
341        run_netns_ip(netns_name, &["link", "set", &agent_veth, "up"])?;
342        run_netns_ip(netns_name, &["link", "set", "lo", "up"])?;
343        run_netns_ip(netns_name, &["route", "add", "default", "via", &proxy_ip])?;
344
345        // T13: Enable IP forwarding in the agent's network namespace only (not globally).
346        // Writing to the namespace-specific sysctl avoids modifying host-wide routing state.
347        let ns_forward_path = format!("/proc/sys/net/ipv4/conf/{}/forwarding", host_veth);
348        if let Err(e) = std::fs::write(&ns_forward_path, "1") {
349            // U9: Removed global ip_forward fallback — too dangerous to leak forwarding to all interfaces
350            tracing::error!(
351                error = %e,
352                path = %ns_forward_path,
353                "failed to enable forwarding on veth — monitored network mode may not route traffic \
354                 (global ip_forward fallback removed for safety)"
355            );
356        }
357
358        // H30: Validate short_id contains only hex chars before interpolation
359        // into nftables rules to prevent injection attacks.
360        if !short_id.chars().all(|c| c.is_ascii_hexdigit()) {
361            return Err(PuzzledError::Network(format!(
362                "H30: short_id '{}' contains non-hex characters",
363                short_id
364            )));
365        }
366
367        // Nftables: log all agent network activity
368        // V30: No nftables log rate limit — syslog rotation/rate-limiting should be configured
369        // at the system level (journald RateLimitBurst, rsyslog rate limiting) for Monitored mode.
370        let nft_table = format!("agent_{}", short_id);
371        let nft_rules = format!(
372            r#"table inet {table} {{
373    chain agent_output {{
374        type filter hook output priority 0; policy accept;
375        log prefix "agent-monitor: " accept
376    }}
377}}"#,
378            table = nft_table,
379        );
380
381        run_netns_nft(netns_name, &nft_rules)?;
382
383        tracing::info!(
384            branch_id,
385            host_veth = %host_veth,
386            "network mode: Monitored (veth + logging)"
387        );
388
389        Ok(Self {
390            mode: NetworkMode::Monitored,
391            host_veth: Some(host_veth),
392            agent_veth: Some(agent_veth),
393            proxy_addr: None,
394            nft_table: Some(nft_table),
395            netns_name: Some(netns_name.to_string()),
396        })
397    }

411    /// Clean up network resources for a branch.
412    pub fn cleanup(&self) {
413        #[cfg(target_os = "linux")]
414        {
415            // Removing the host veth automatically removes the peer
416            if let Some(ref veth) = self.host_veth {
417                let _ = run_ip_cmd(&["link", "del", veth]);
418            }
419            // Clean up named network namespace
420            if let Some(ref name) = self.netns_name {
421                delete_named_netns(name);
422            }
423            // nftables table is inside the namespace and dies with it
424        }
425    }

427    /// Get environment variables the agent process should have set.
428    pub fn agent_env(&self) -> Vec<(String, String)> {
429        let mut env = Vec::new();
430        if let Some(addr) = self.proxy_addr {
431            let proxy_url = format!("http://{}", addr);
432            env.push(("HTTP_PROXY".to_string(), proxy_url.clone()));
433            env.push(("HTTPS_PROXY".to_string(), proxy_url.clone()));
434            env.push(("http_proxy".to_string(), proxy_url.clone()));
435            env.push(("https_proxy".to_string(), proxy_url));
436        }
437        env
438    }

445/// Run an `ip` command.
446#[cfg(target_os = "linux")]
447fn run_ip_cmd(args: &[&str]) -> Result<()> {
448    let output = std::process::Command::new("ip")
449        .args(args)
450        .output()
451        .map_err(|e| PuzzledError::Network(format!("running ip {}: {}", args.join(" "), e)))?;
452
453    if !output.status.success() {
454        let stderr = String::from_utf8_lossy(&output.stderr);
455        return Err(PuzzledError::Network(format!(
456            "ip {} failed: {}",
457            args.join(" "),
458            stderr.trim()
459        )));
460    }
461    Ok(())
462}

464/// Run an `ip` command inside a named network namespace via nsenter.
465#[cfg(target_os = "linux")]
466fn run_netns_ip(netns_name: &str, args: &[&str]) -> Result<()> {
467    let netns_arg = format!("--net=/var/run/netns/{}", netns_name);
468    let mut cmd = std::process::Command::new("nsenter");
469    cmd.args([netns_arg.as_str(), "ip"]);
470    cmd.args(args);
471
472    let output = cmd
473        .output()
474        .map_err(|e| PuzzledError::Network(format!("nsenter ip {}: {}", args.join(" "), e)))?;
475
476    if !output.status.success() {
477        let stderr = String::from_utf8_lossy(&output.stderr);
478        return Err(PuzzledError::Network(format!(
479            "nsenter ip {} failed: {}",
480            args.join(" "),
481            stderr.trim()
482        )));
483    }
484    Ok(())
485}

487/// Apply nftables rules inside a named network namespace.
488#[cfg(target_os = "linux")]
489fn run_netns_nft(netns_name: &str, rules: &str) -> Result<()> {
490    let netns_arg = format!("--net=/var/run/netns/{}", netns_name);
491    let output = std::process::Command::new("nsenter")
492        .args([netns_arg.as_str(), "nft", "-f", "-"])
493        .stdin(std::process::Stdio::piped())
494        .stdout(std::process::Stdio::piped())
495        .stderr(std::process::Stdio::piped())
496        .spawn()
497        .and_then(|mut child| {
498            use std::io::Write;
499            if let Some(ref mut stdin) = child.stdin {
500                stdin.write_all(rules.as_bytes())?;
501            }
502            child.wait_with_output()
503        })
504        .map_err(|e| PuzzledError::Network(format!("nsenter nft: {}", e)))?;
505
506    if !output.status.success() {
507        let stderr = String::from_utf8_lossy(&output.stderr);
508        return Err(PuzzledError::Network(format!(
509            "nft rules failed: {}",
510            stderr.trim()
511        )));
512    }
513    Ok(())
514}

520/// Create a named network namespace using `ip netns add`.
521///
522/// This creates an empty network namespace with a persistent reference at
523/// `/var/run/netns/<name>`. The parent calls this BEFORE clone3 so the
524/// named netns is visible in the parent's mount namespace. The child then
525/// joins it via `setns()`.
526///
527/// This approach avoids the problems with:
528/// R4: Validate network namespace name to prevent path traversal.
529/// Names must not contain '/', '\0', or '..' components.
530#[allow(dead_code)] // called from #[cfg(target_os = "linux")] code only
531fn validate_netns_name(name: &str) -> crate::error::Result<()> {
532    use crate::error::PuzzledError;
533    if name.is_empty() {
534        return Err(PuzzledError::Network("netns name must not be empty".into()));
535    }
536    if name.contains('/') || name.contains('\0') || name.contains("..") {
537        return Err(PuzzledError::Network(format!(
538            "R4: netns name '{}' contains path traversal characters",
539            name
540        )));
541    }
542    if name.len() > 64 {
543        return Err(PuzzledError::Network(format!(
544            "R4: netns name too long ({} > 64)",
545            name.len()
546        )));
547    }
548    Ok(())
549}
550
551#[cfg(target_os = "linux")]
552pub fn create_named_netns(name: &str) -> Result<()> {
553    // R4: Validate name before using in path construction
554    validate_netns_name(name)?;
555
556    // Clean up any stale netns from a previous failed run
557    delete_named_netns(name);
558
559    let output = std::process::Command::new("ip")
560        .args(["netns", "add", name])
561        .output()
562        .map_err(|e| PuzzledError::Network(format!("running ip netns add {}: {}", name, e)))?;
563
564    if !output.status.success() {
565        let stderr = String::from_utf8_lossy(&output.stderr);
566        return Err(PuzzledError::Network(format!(
567            "ip netns add {} failed: {}",
568            name,
569            stderr.trim()
570        )));
571    }
572
573    // Verify the netns file exists
574    let netns_path = format!("/var/run/netns/{}", name);
575    if !std::path::Path::new(&netns_path).exists() {
576        return Err(PuzzledError::Network(format!(
577            "ip netns add {} succeeded but {} does not exist",
578            name, netns_path
579        )));
580    }
581
582    tracing::debug!(name, path = %netns_path, "created named network namespace");
583    Ok(())
584}

592/// Delete a named network namespace.
593#[cfg(target_os = "linux")]
594pub fn delete_named_netns(name: &str) {
595    // R4: Validate name to prevent path traversal in umount2/remove_file
596    if name.contains('/') || name.contains('\0') || name.contains("..") || name.is_empty() {
597        tracing::error!(name, "R4: rejecting netns delete with unsafe name");
598        return;
599    }
600    let netns_path = format!("/var/run/netns/{}", name);
601    if let Ok(path_c) = std::ffi::CString::new(netns_path.as_str()) {
602        unsafe { libc::umount2(path_c.as_ptr(), libc::MNT_DETACH) };
603    }
604    // J4: Log unexpected remove_file errors instead of silently discarding
605    if let Err(e) = std::fs::remove_file(&netns_path) {
606        if e.kind() != std::io::ErrorKind::NotFound {
607            tracing::warn!(
608                name,
609                path = %netns_path,
610                error = %e,
611                "J4: unexpected error removing netns file"
612            );
613        }
614    }
615}

691        let branch_id = "a1b2c3d4-e5f6-7890-abcd-ef1234567890";
692        let short_id = format!("{:08x}", crc32fast::hash(branch_id.as_bytes()));
693        let host_veth = format!("vh{}", &short_id);

705            agent_veth
706        );
707
708        // Deterministic: same input produces same output

710        assert_eq!(short_id, short_id2, "CRC32 hash should be deterministic");

712        // Different inputs produce different outputs
713        let other_id = "different-branch-id";
714        let other_short = format!("{:08x}", crc32fast::hash(other_id.as_bytes()));
