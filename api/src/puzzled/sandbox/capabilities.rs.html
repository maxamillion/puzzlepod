2//! Capability dropping and credential switching for agent sandboxes.
3//!
4//! After clone3(), Landlock, and seccomp are applied, the child process
5//! must drop all unnecessary capabilities and switch to the agent's
6//! non-root UID/GID. Without this, the agent runs as root with full
7//! capabilities, which undermines all other sandbox layers.
8
9use crate::error::{PuzzledError, Result};
10
11/// Drop all capabilities from the bounding set except those in `keep`.
12///
13/// Uses `prctl(PR_CAPBSET_DROP)` to remove each capability from the
14/// bounding set. This is irreversible — once dropped, the capability
15/// cannot be re-acquired by the process or its children.
16///
17/// Typical `keep` for restricted agents: empty (no capabilities retained).
18/// Standard agents may retain `CAP_NET_BIND_SERVICE` if needed.
19#[cfg(target_os = "linux")]
20pub fn drop_capabilities(keep: &[u32]) -> Result<()> {
21    // CAP_LAST_CAP is typically 40-41 on modern kernels; iterate to 63 for safety
22    const MAX_CAP: u32 = 63;
23
24    // Step 1: Drop capabilities from the bounding set.
25    // CRITICAL: PR_CAPBSET_DROP requires CAP_SETPCAP (cap 8) in the effective
26    // set. This function must be called BEFORE setuid() (which clears
27    // effective/permitted) — see drop_bounding_set() for the pre-setuid path.
28    // If called after setuid, bounding set drops will silently fail.
29    // This is irreversible — once dropped, the capability cannot be
30    // re-acquired by the process or its children.
31    let mut dropped = 0u32;
32    for cap in 0..=MAX_CAP {
33        if keep.contains(&cap) {
34            continue;
35        }
36
37        let ret = unsafe { libc::prctl(libc::PR_CAPBSET_DROP, cap, 0, 0, 0) };
38        if ret == 0 {
39            dropped += 1;
40        }
41        // EINVAL means the capability doesn't exist on this kernel — skip
42    }
43
44    // Step 2: Clear the effective and permitted capability sets.
45    clear_effective_and_permitted_sets(keep)?;
46
47    // Step 3: Clear the ambient capability set (prevents re-acquisition via execve)
48    let ret = unsafe {
49        libc::prctl(
50            libc::PR_CAP_AMBIENT,
51            libc::PR_CAP_AMBIENT_CLEAR_ALL,
52            0,
53            0,
54            0,
55        )
56    };
57    if ret < 0 {
58        let err = std::io::Error::last_os_error();
59        // EINVAL on older kernels that don't support ambient caps — non-fatal
60        if err.raw_os_error() != Some(libc::EINVAL) {
61            tracing::warn!(error = %err, "failed to clear ambient capabilities");
62        }
63    }
64
65    // Step 4: Set NO_NEW_PRIVS to prevent privilege escalation via setuid binaries
66    let ret = unsafe { libc::prctl(libc::PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0) };
67    if ret < 0 {
68        return Err(PuzzledError::Sandbox(format!(
69            "prctl(PR_SET_NO_NEW_PRIVS) failed: {}",
70            std::io::Error::last_os_error()
71        )));
72    }
73
74    // Verify the bounding set is empty (except for kept caps) by checking
75    // each capability with PR_CAPBSET_READ. This belt-and-suspenders check
76    // catches cases where a capability was not successfully dropped.
77    let mut remaining = Vec::new();
78    for cap in 0..=MAX_CAP {
79        if keep.contains(&cap) {
80            continue;
81        }
82        let ret = unsafe { libc::prctl(libc::PR_CAPBSET_READ, cap, 0, 0, 0) };
83        if ret > 0 {
84            remaining.push(cap);
85        }
86    }
87    if !remaining.is_empty() {
88        tracing::warn!(
89            remaining = ?remaining,
90            "capabilities still present in bounding set after drop"
91        );
92    }
93
94    tracing::info!(
95        dropped,
96        kept = keep.len(),
97        remaining_caps = remaining.len(),
98        "capabilities dropped — effective, permitted, and bounding sets cleared"
99    );
100
101    Ok(())
102}

104/// Drop all capabilities from the bounding set except those in `keep`.
105///
106/// MUST be called BEFORE setuid() / switch_credentials(), because
107/// PR_CAPBSET_DROP requires CAP_SETPCAP (cap 8) in the effective set.
108/// setuid() to a non-root user clears effective/permitted, making
109/// subsequent PR_CAPBSET_DROP calls fail with EPERM.
110#[cfg(target_os = "linux")]
111pub fn drop_bounding_set(keep: &[u32]) -> Result<()> {
112    const MAX_CAP: u32 = 63;
113
114    let mut dropped = 0u32;
115    for cap in 0..=MAX_CAP {
116        if keep.contains(&cap) {
117            continue;
118        }
119
120        let ret = unsafe { libc::prctl(libc::PR_CAPBSET_DROP, cap, 0, 0, 0) };
121        if ret == 0 {
122            dropped += 1;
123        }
124        // EINVAL means the capability doesn't exist on this kernel — skip
125    }
126
127    // Verify the bounding set is cleared
128    let mut remaining = Vec::new();
129    for cap in 0..=MAX_CAP {
130        if keep.contains(&cap) {
131            continue;
132        }
133        let ret = unsafe { libc::prctl(libc::PR_CAPBSET_READ, cap, 0, 0, 0) };
134        if ret > 0 {
135            remaining.push(cap);
136        }
137    }
138    if !remaining.is_empty() {
139        return Err(PuzzledError::Sandbox(format!(
140            "failed to drop {} capabilities from bounding set: {:?}",
141            remaining.len(),
142            remaining
143        )));
144    }
145
146    tracing::info!(
147        dropped,
148        kept = keep.len(),
149        "bounding set capabilities dropped"
150    );
151    Ok(())
152}

161/// Clear the effective and permitted capability sets using capget/capset.
162///
163/// The correct order for full capability removal is:
164///   1. Clear effective set (so no capabilities are active)
165///   2. Clear permitted set (so no capabilities can be raised to effective)
166///   3. Drop bounding set entries (handled by caller)
167///
168/// Capabilities in `keep` are retained in both effective and permitted sets
169/// (needed if the agent profile requires them, e.g., CAP_NET_BIND_SERVICE).
170#[cfg(target_os = "linux")]
171fn clear_effective_and_permitted_sets(keep: &[u32]) -> Result<()> {
172    // Linux capability data structures for capget/capset (v3, 64-bit)
173    #[repr(C)]
174    struct CapUserHeader {
175        version: u32,
176        pid: i32,
177    }
178
179    #[repr(C)]
180    #[derive(Clone, Copy, Default)]
181    struct CapUserData {
182        effective: u32,
183        permitted: u32,
184        inheritable: u32,
185    }
186
187    // _LINUX_CAPABILITY_VERSION_3 (0x20080522) — supports caps 0..63 via two data structs
188    const LINUX_CAPABILITY_VERSION_3: u32 = 0x20080522;
189
190    let mut header = CapUserHeader {
191        version: LINUX_CAPABILITY_VERSION_3,
192        pid: 0, // current process
193    };
194
195    // capget/capset with v3 uses two CapUserData structs (caps 0-31 and 32-63)
196    let mut data = [CapUserData::default(); 2];
197
198    // Get current capabilities so we know what the kernel reports
199    let ret = unsafe {
200        libc::syscall(
201            libc::SYS_capget,
202            &mut header as *mut CapUserHeader,
203            data.as_mut_ptr(),
204        )
205    };
206    if ret < 0 {
207        return Err(PuzzledError::Sandbox(format!(
208            "capget failed: {}",
209            std::io::Error::last_os_error()
210        )));
211    }
212
213    tracing::debug!(
214        effective_lo = format!("{:#010x}", data[0].effective),
215        effective_hi = format!("{:#010x}", data[1].effective),
216        permitted_lo = format!("{:#010x}", data[0].permitted),
217        permitted_hi = format!("{:#010x}", data[1].permitted),
218        "current capability sets before clearing"
219    );
220
221    // Build bitmasks for capabilities to keep
222    let mut keep_lo: u32 = 0;
223    let mut keep_hi: u32 = 0;
224    for &cap in keep {
225        if cap < 32 {
226            keep_lo |= 1 << cap;
227        } else if cap < 64 {
228            keep_hi |= 1 << (cap - 32);
229        }
230    }
231
232    // Clear effective set (only retain kept caps that were already permitted)
233    data[0].effective = keep_lo & data[0].permitted;
234    data[1].effective = keep_hi & data[1].permitted;
235
236    // Clear permitted set (only retain kept caps)
237    data[0].permitted &= keep_lo;
238    data[1].permitted &= keep_hi;
239
240    // Clear inheritable set entirely (agents should not pass caps to children)
241    data[0].inheritable = 0;
242    data[1].inheritable = 0;
243
244    // Apply the cleared capability sets
245    header.version = LINUX_CAPABILITY_VERSION_3;
246    header.pid = 0;
247
248    let ret = unsafe {
249        libc::syscall(
250            libc::SYS_capset,
251            &header as *const CapUserHeader,
252            data.as_ptr(),
253        )
254    };
255    if ret < 0 {
256        return Err(PuzzledError::Sandbox(format!(
257            "capset failed (clearing effective/permitted): {}",
258            std::io::Error::last_os_error()
259        )));
260    }
261
262    tracing::debug!(
263        effective_lo = format!("{:#010x}", data[0].effective),
264        effective_hi = format!("{:#010x}", data[1].effective),
265        permitted_lo = format!("{:#010x}", data[0].permitted),
266        permitted_hi = format!("{:#010x}", data[1].permitted),
267        "capability sets after clearing effective and permitted"
268    );
269
270    Ok(())
271}

273/// Switch the process to a non-root UID/GID.
274///
275/// Calls setgroups(0, NULL) -> setgid(gid) -> setuid(uid) in the
276/// correct order (gid first, since setuid drops the ability to setgid).
277///
278/// After this call, the process runs as the specified non-root user
279/// and cannot regain root privileges (due to NO_NEW_PRIVS + dropped caps).
280///
281/// If `reject_root` is true and uid=0, returns an error (for FailClosed profiles).
282/// Otherwise, uid=0 is allowed with a warning.
283#[cfg(target_os = "linux")]
284pub fn switch_credentials(uid: u32, gid: u32, reject_root: bool) -> Result<()> {
285    if uid == 0 {
286        if reject_root {
287            return Err(PuzzledError::Sandbox(
288                "refusing to run agent as root (FailClosed profile requires non-root UID)"
289                    .to_string(),
290            ));
291        }
292        tracing::warn!("switch_credentials called with uid=0, agent will run as root");
293        return Ok(());
294    }
295
296    // Clear supplementary groups
297    let ret = unsafe { libc::setgroups(0, std::ptr::null()) };
298    if ret < 0 {
299        return Err(PuzzledError::Sandbox(format!(
300            "setgroups(0) failed: {}",
301            std::io::Error::last_os_error()
302        )));
303    }
304
305    // Set GID first (setuid drops ability to change GID)
306    let ret = unsafe { libc::setgid(gid) };
307    if ret < 0 {
308        return Err(PuzzledError::Sandbox(format!(
309            "setgid({}) failed: {}",
310            gid,
311            std::io::Error::last_os_error()
312        )));
313    }
314
315    // Set UID last
316    let ret = unsafe { libc::setuid(uid) };
317    if ret < 0 {
318        return Err(PuzzledError::Sandbox(format!(
319            "setuid({}) failed: {}",
320            uid,
321            std::io::Error::last_os_error()
322        )));
323    }
324
325    tracing::info!(uid, gid, "credentials switched to non-root");
326    Ok(())
327}

329/// M8: Map a capability name string to its numeric value.
330///
331/// Accepts both "CAP_NET_BIND_SERVICE" and "NET_BIND_SERVICE" forms.
332/// Returns None for unrecognized capability names.
333pub fn cap_name_to_number(name: &str) -> Option<u32> {
334    // Strip optional "CAP_" prefix for convenience
335    let name = name.strip_prefix("CAP_").unwrap_or(name);
336    match name {
337        "CHOWN" => Some(0),
338        "DAC_OVERRIDE" => Some(1),
339        "DAC_READ_SEARCH" => Some(2),
340        "FOWNER" => Some(3),
341        "FSETID" => Some(4),
342        "KILL" => Some(5),
343        "SETGID" => Some(6),
344        "SETUID" => Some(7),
345        "SETPCAP" => Some(8),
346        "LINUX_IMMUTABLE" => Some(9),
347        "NET_BIND_SERVICE" => Some(10),
348        "NET_BROADCAST" => Some(11),
349        "NET_ADMIN" => Some(12),
350        "NET_RAW" => Some(13),
351        "IPC_LOCK" => Some(14),
352        "IPC_OWNER" => Some(15),
353        "SYS_MODULE" => Some(16),
354        "SYS_RAWIO" => Some(17),
355        "SYS_CHROOT" => Some(18),
356        "SYS_PTRACE" => Some(19),
357        "SYS_PACCT" => Some(20),
358        "SYS_ADMIN" => Some(21),
359        "SYS_BOOT" => Some(22),
360        "SYS_NICE" => Some(23),
361        "SYS_RESOURCE" => Some(24),
362        "SYS_TIME" => Some(25),
363        "SYS_TTY_CONFIG" => Some(26),
364        "MKNOD" => Some(27),
365        "LEASE" => Some(28),
366        "AUDIT_WRITE" => Some(29),
367        "AUDIT_CONTROL" => Some(30),
368        "SETFCAP" => Some(31),
369        "MAC_OVERRIDE" => Some(32),
370        "MAC_ADMIN" => Some(33),
371        "SYSLOG" => Some(34),
372        "WAKE_ALARM" => Some(35),
373        "BLOCK_SUSPEND" => Some(36),
374        "AUDIT_READ" => Some(37),
375        "PERFMON" => Some(38),
376        "BPF" => Some(39),
377        "CHECKPOINT_RESTORE" => Some(40),
378        _ => {
379            tracing::warn!(capability = name, "unrecognized capability name");
380            None
381        }
382    }
383}
