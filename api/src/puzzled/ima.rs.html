2//! IMA changeset signing — generates and signs commit manifests.
3//!
4//! Each commit produces a manifest (YAML) listing all files in the changeset
5//! with their paths, change kinds, sizes, and checksums. The manifest is signed
6//! with Ed25519 to provide integrity and non-repudiation.
7//!
8//! On Linux, the module also extends the IMA measurement log (best-effort).
9//! On non-Linux platforms, only manifest generation and signing are available.
10
11use std::path::{Path, PathBuf};
12
13use ed25519_dalek::{Signer, SigningKey, Verifier, VerifyingKey};
14use puzzled_types::{BranchId, FileChange, FileChangeKind};
15use serde::{Deserialize, Serialize};
16
17use crate::error::{PuzzledError, Result};

19/// A signed commit manifest.
20#[derive(Debug, Clone, Serialize, Deserialize)]
21pub struct CommitManifest {
22    /// Branch that was committed.
23    pub branch_id: String,
24    /// ISO 8601 timestamp of the commit.
25    pub timestamp: String,
26    /// C5: ID of the agent that performed the commit.
27    #[serde(default)]
28    pub agent_id: String,
29    /// C5: Profile under which the agent was running.
30    #[serde(default)]
31    pub agent_profile: String,
32    /// C5: Checksum of the branch state before the commit (for auditability).
33    #[serde(default)]
34    pub checksum_before: String,
35    /// List of files in the changeset.
36    pub files: Vec<ManifestEntry>,
37    /// Ed25519 signature over the canonical manifest content (hex-encoded).
38    pub signature: String,
39}
40
41/// A single file entry in the manifest.
42#[derive(Debug, Clone, Serialize, Deserialize)]
43pub struct ManifestEntry {
44    pub path: String,
45    pub kind: String,
46    pub size: u64,
47    pub checksum: String,
48}

50/// Default key rotation threshold in days.
51/// Keys older than this will be rotated when `check_key_rotation()` is called.
52const DEFAULT_KEY_ROTATION_DAYS: u64 = 90;

54/// IMA integration for changeset signing and verification.
55pub struct ImaIntegration {
56    /// Ed25519 signing key.
57    signing_key: SigningKey,
58    /// Ed25519 verifying key (derived from signing key).
59    verifying_key: VerifyingKey,
60    /// Directory where manifests are stored.
61    manifest_dir: PathBuf,
62    /// Path to the signing key file (needed for rotation).
63    key_path: PathBuf,
64}
65
66impl ImaIntegration {
67    /// Create a new IMA integration instance.
68    ///
69    /// If `key_path` exists, loads the key from it. Otherwise, generates a new
70    /// keypair and saves it.
71    ///
72    /// # L8: IMA failure is configurable
73    ///
74    /// Callers should check `config.require_ima` (in `DaemonConfig`) before
75    /// calling this method:
76    ///
77    /// - When `require_ima` is **true**: a failure from `new()` should be
78    ///   treated as fatal (propagate the error, abort daemon startup). This is
79    ///   the setting for deployments that require signed commit manifests for
80    ///   compliance or audit trail integrity.
81    ///
82    /// - When `require_ima` is **false** (the default): a failure from `new()`
83    ///   should be logged as a warning and IMA signing should be skipped for
84    ///   the lifetime of the daemon. Commits proceed without manifest
85    ///   signatures. This is appropriate for development, testing, and
86    ///   environments where the IMA key infrastructure is not yet deployed.
87    pub fn new(manifest_dir: PathBuf, key_path: &Path) -> Result<Self> {
88        let signing_key = if key_path.exists() {
89            let key_bytes = std::fs::read(key_path)
90                .map_err(|e| PuzzledError::Ima(format!("reading signing key: {}", e)))?;
91            if key_bytes.len() != 32 {
92                return Err(PuzzledError::Ima(
93                    "signing key must be 32 bytes".to_string(),
94                ));
95            }
96            let mut arr = [0u8; 32];
97            arr.copy_from_slice(&key_bytes);
98            SigningKey::from_bytes(&arr)
99        } else {
100            // Generate a new keypair
101            let mut rng_bytes = [0u8; 32];
102            csprng_fill(&mut rng_bytes)?;
103            let key = SigningKey::from_bytes(&rng_bytes);
104
105            // Save the key with restricted permissions set atomically at creation
106            // time (no TOCTOU race window where the key is world-readable).
107            if let Some(parent) = key_path.parent() {
108                std::fs::create_dir_all(parent)
109                    .map_err(|e| PuzzledError::Ima(format!("creating key directory: {}", e)))?;
110            }
111            {
112                use std::io::Write;
113                #[cfg(unix)]
114                use std::os::unix::fs::OpenOptionsExt;
115
116                let mut opts = std::fs::OpenOptions::new();
117                opts.write(true).create_new(true);
118                #[cfg(unix)]
119                opts.mode(0o600);
120
121                let mut file = opts
122                    .open(key_path)
123                    .map_err(|e| PuzzledError::Ima(format!("saving signing key: {}", e)))?;
124                file.write_all(&key.to_bytes())
125                    .map_err(|e| PuzzledError::Ima(format!("writing signing key: {}", e)))?;
126            }
127
128            key
129        };
130
131        let verifying_key = signing_key.verifying_key();
132
133        std::fs::create_dir_all(&manifest_dir)
134            .map_err(|e| PuzzledError::Ima(format!("creating manifest directory: {}", e)))?;
135
136        Ok(Self {
137            signing_key,
138            verifying_key,
139            manifest_dir,
140            key_path: key_path.to_path_buf(),
141        })
142    }

144    /// Generate and sign a commit manifest from the changeset.
145    ///
146    /// The manifest is saved to `<manifest_dir>/<branch_id>.manifest.yaml`.
147    ///
148    /// H-8: `agent_id`, `agent_profile`, and `checksum_before` are included in
149    /// the canonical signed payload so they cannot be tampered with after signing.
150    pub fn sign_commit(
151        &self,
152        branch_id: &BranchId,
153        changes: &[FileChange],
154    ) -> Result<CommitManifest> {
155        self.sign_commit_with_metadata(branch_id, changes, "", "", "")
156    }
157
158    /// Generate and sign a commit manifest with full metadata.
159    ///
160    /// H-8: All metadata fields (`agent_id`, `agent_profile`, `checksum_before`)
161    /// are included in the canonical signed payload to prevent post-signature tampering.
162    pub fn sign_commit_with_metadata(
163        &self,
164        branch_id: &BranchId,
165        changes: &[FileChange],
166        agent_id: &str,
167        agent_profile: &str,
168        checksum_before: &str,
169    ) -> Result<CommitManifest> {
170        let files: Vec<ManifestEntry> = changes
171            .iter()
172            .map(|c| ManifestEntry {
173                path: c.path.display().to_string(),
174                kind: match c.kind {
175                    FileChangeKind::Added => "added".to_string(),
176                    FileChangeKind::Modified => "modified".to_string(),
177                    FileChangeKind::Deleted => "deleted".to_string(),
178                    FileChangeKind::MetadataChanged => "metadata_changed".to_string(),
179                    FileChangeKind::Renamed => "renamed".to_string(),
180                    FileChangeKind::Symlink => "symlink".to_string(),
181                    // Q6: New special file type variants
182                    FileChangeKind::Hardlink => "hardlink".to_string(),
183                    FileChangeKind::BlockDevice => "block_device".to_string(),
184                    FileChangeKind::CharDevice => "char_device".to_string(),
185                    FileChangeKind::Fifo => "fifo".to_string(),
186                },
187                size: c.size,
188                checksum: c.checksum.clone(),
189            })
190            .collect();
191
192        let timestamp = chrono::Utc::now().to_rfc3339();
193
194        // H-8: Build canonical content including agent_id, agent_profile,
195        // checksum_before so they are covered by the Ed25519 signature.
196        let canonical = build_canonical(
197            branch_id.as_str(),
198            &timestamp,
199            &files,
200            agent_id,
201            agent_profile,
202            checksum_before,
203        )
204        .map_err(|e| PuzzledError::Ima(format!("canonical form failed: {e}")))?;
205
206        // Sign the canonical content
207        let signature = self.signing_key.sign(canonical.as_bytes());
208        let sig_hex = hex_encode(signature.to_bytes().as_slice());
209
210        let manifest = CommitManifest {
211            branch_id: branch_id.as_str().to_string(),
212            timestamp,
213            agent_id: agent_id.to_string(),
214            agent_profile: agent_profile.to_string(),
215            checksum_before: checksum_before.to_string(),
216            files,
217            signature: sig_hex,
218        };
219
220        // C4: Atomic manifest write — write to temp file, fsync, then rename.
221        // This prevents a crash from leaving a partially-written manifest that
222        // could pass signature verification with truncated content.
223        let manifest_path = self
224            .manifest_dir
225            .join(format!("{}.manifest.yaml", branch_id.as_str()));
226        let yaml = serde_yaml::to_string(&manifest)
227            .map_err(|e| PuzzledError::Ima(format!("serializing manifest: {}", e)))?;
228        {
229            use std::io::Write;
230            let mut tmp = tempfile::NamedTempFile::new_in(&self.manifest_dir)
231                .map_err(|e| PuzzledError::Ima(format!("creating temp manifest file: {}", e)))?;
232            tmp.write_all(yaml.as_bytes())
233                .map_err(|e| PuzzledError::Ima(format!("writing temp manifest: {}", e)))?;
234            tmp.as_file()
235                .sync_all()
236                .map_err(|e| PuzzledError::Ima(format!("fsync temp manifest: {}", e)))?;
237            tmp.persist(&manifest_path)
238                .map_err(|e| PuzzledError::Ima(format!("persisting manifest: {}", e)))?;
239        }
240
241        // On Linux, extend IMA measurement log (best-effort)
242        #[cfg(target_os = "linux")]
243        self.extend_ima_log(&manifest_path);
244
245        tracing::info!(
246            branch = %branch_id,
247            manifest = %manifest_path.display(),
248            files = manifest.files.len(),
249            "commit manifest signed"
250        );
251
252        self.verify_manifest(&manifest)?;
253
254        Ok(manifest)
255    }

257    /// Verify a manifest's signature.
258    ///
259    /// Returns `Ok(())` if the signature is valid, or an error if verification
260    /// fails. This ensures callers cannot silently ignore a bad signature by
261    /// forgetting to check a boolean return value.
262    pub fn verify_manifest(&self, manifest: &CommitManifest) -> Result<()> {
263        let canonical = build_canonical(
264            &manifest.branch_id,
265            &manifest.timestamp,
266            &manifest.files,
267            &manifest.agent_id,
268            &manifest.agent_profile,
269            &manifest.checksum_before,
270        )
271        .map_err(|e| PuzzledError::Ima(format!("canonical form failed: {e}")))?;
272
273        let sig_bytes = hex_decode(&manifest.signature)
274            .map_err(|e| PuzzledError::Ima(format!("decoding signature: {}", e)))?;
275
276        if sig_bytes.len() != 64 {
277            return Err(PuzzledError::Ima("signature must be 64 bytes".to_string()));
278        }
279
280        let mut arr = [0u8; 64];
281        arr.copy_from_slice(&sig_bytes);
282        let signature = ed25519_dalek::Signature::from_bytes(&arr);
283
284        self.verifying_key
285            .verify(canonical.as_bytes(), &signature)
286            .map_err(|e| PuzzledError::Ima(format!("signature verification failed: {}", e)))
287    }

311    /// Check the age of the signing key and rotate if older than the threshold.
312    ///
313    /// When rotation occurs:
314    /// 1. The old public key is saved to `<key_path>.pub.<timestamp>` for
315    ///    verification of existing manifests signed with the old key.
316    /// 2. A new Ed25519 keypair is generated and written to the key file.
317    /// 3. The new key file permissions are set to 0600 (owner-read-write only).
318    ///
319    /// Returns `Ok(true)` if the key was rotated, `Ok(false)` if no rotation
320    /// was needed.
321    pub fn check_key_rotation(&mut self) -> Result<bool> {
322        self.check_key_rotation_with_threshold(DEFAULT_KEY_ROTATION_DAYS)
323    }

325    /// Check key rotation with a custom threshold in days.
326    pub fn check_key_rotation_with_threshold(&mut self, max_age_days: u64) -> Result<bool> {
327        if !self.key_path.exists() {
328            return Ok(false);
329        }
330
331        let metadata = std::fs::metadata(&self.key_path)
332            .map_err(|e| PuzzledError::Ima(format!("reading key file metadata: {}", e)))?;
333
334        let modified = metadata
335            .modified()
336            .map_err(|e| PuzzledError::Ima(format!("reading key file mtime: {}", e)))?;
337
338        // S11: Detect clock skew — if the key file's mtime is in the future,
339        // `duration_since()` fails. Previously `unwrap_or_default()` silently
340        // returned Duration::ZERO, making future-dated keys appear 0 days old
341        // and preventing rotation. Now we return an error so the caller can
342        // investigate the clock skew.
343        let age = std::time::SystemTime::now()
344            .duration_since(modified)
345            .map_err(|_| {
346                PuzzledError::Ima(
347                    "S11: signing key file mtime is in the future — possible clock skew. \
348                 Cannot determine key age reliably. Check system clock and key file timestamp."
349                        .to_string(),
350                )
351            })?;
352
353        // L3: Use saturating_mul to avoid panic/wrap on large max_age_days.
354        // Saturating at u64::MAX means an absurdly large threshold simply
355        // never triggers rotation, which is the correct semantic.
356        let max_age_secs = max_age_days
357            .saturating_mul(24)
358            .saturating_mul(60)
359            .saturating_mul(60);
360        let max_age = std::time::Duration::from_secs(max_age_secs);
361
362        if age < max_age {
363            tracing::debug!(
364                key_age_days = age.as_secs() / 86400,
365                max_age_days,
366                "signing key is within rotation threshold"
367            );
368            return Ok(false);
369        }
370
371        tracing::warn!(
372            key_age_days = age.as_secs() / 86400,
373            max_age_days,
374            "signing key exceeds rotation threshold, rotating"
375        );
376
377        // Preserve old public key for verification of existing manifests
378        let timestamp = chrono::Utc::now().format("%Y%m%dT%H%M%SZ").to_string();
379        let old_pubkey_path = self.key_path.with_extension(format!("pub.{}", timestamp));
380        let old_pubkey_bytes = self.verifying_key.to_bytes();
381        std::fs::write(&old_pubkey_path, old_pubkey_bytes)
382            .map_err(|e| PuzzledError::Ima(format!("saving old public key: {}", e)))?;
383
384        tracing::info!(
385            old_pubkey = %old_pubkey_path.display(),
386            "preserved old public key for verification of existing manifests"
387        );
388
389        // Generate new keypair
390        let mut rng_bytes = [0u8; 32];
391        csprng_fill(&mut rng_bytes)?;
392        let new_key = SigningKey::from_bytes(&rng_bytes);
393        let new_verifying_key = new_key.verifying_key();
394
395        // Write new key to disk atomically: write to a temp file with
396        // restricted permissions, then rename over the old key file.
397        // This avoids a TOCTOU race window where the key is world-readable.
398        {
399            use std::io::Write;
400            #[cfg(unix)]
401            use std::os::unix::fs::OpenOptionsExt;
402
403            let tmp_path = self.key_path.with_extension("tmp");
404            // R22: Remove stale temp file before creating with O_EXCL to prevent symlink attacks
405            let _ = std::fs::remove_file(&tmp_path);
406
407            let mut opts = std::fs::OpenOptions::new();
408            opts.write(true).create_new(true);
409            #[cfg(unix)]
410            opts.mode(0o600);
411
412            let mut file = opts
413                .open(&tmp_path)
414                .map_err(|e| PuzzledError::Ima(format!("saving rotated signing key: {}", e)))?;
415            file.write_all(&new_key.to_bytes())
416                .map_err(|e| PuzzledError::Ima(format!("writing rotated signing key: {}", e)))?;
417
418            // C6: Fsync the new key file before renaming to ensure the key
419            // bytes are durable on disk. Without this, a crash after rename
420            // but before the OS flushes the key data could leave a zero-length
421            // or corrupted key file.
422            file.sync_all()
423                .map_err(|e| PuzzledError::Ima(format!("C6: fsync rotated signing key: {}", e)))?;
424
425            std::fs::rename(&tmp_path, &self.key_path)
426                .map_err(|e| PuzzledError::Ima(format!("replacing signing key file: {}", e)))?;
427        }
428
429        // Update in-memory state
430        self.signing_key = new_key;
431        self.verifying_key = new_verifying_key;
432
433        tracing::info!(
434            new_pubkey = %self.public_key_hex(),
435            "signing key rotated successfully"
436        );
437
438        Ok(true)
439    }

441    /// Verify a manifest using a specific public key (hex-encoded).
442    ///
443    /// Useful for verifying manifests signed with a previous key after rotation.
444    pub fn verify_manifest_with_pubkey(manifest: &CommitManifest, pubkey_hex: &str) -> Result<()> {
445        let pubkey_bytes = hex_decode(pubkey_hex)
446            .map_err(|e| PuzzledError::Ima(format!("decoding public key: {}", e)))?;
447
448        if pubkey_bytes.len() != 32 {
449            return Err(PuzzledError::Ima("public key must be 32 bytes".to_string()));
450        }
451
452        let mut arr = [0u8; 32];
453        arr.copy_from_slice(&pubkey_bytes);
454        let verifying_key = VerifyingKey::from_bytes(&arr)
455            .map_err(|e| PuzzledError::Ima(format!("invalid public key: {}", e)))?;
456
457        let canonical = build_canonical(
458            &manifest.branch_id,
459            &manifest.timestamp,
460            &manifest.files,
461            &manifest.agent_id,
462            &manifest.agent_profile,
463            &manifest.checksum_before,
464        )
465        .map_err(|e| PuzzledError::Ima(format!("canonical form failed: {e}")))?;
466
467        let sig_bytes = hex_decode(&manifest.signature)
468            .map_err(|e| PuzzledError::Ima(format!("decoding signature: {}", e)))?;
469
470        if sig_bytes.len() != 64 {
471            return Err(PuzzledError::Ima("signature must be 64 bytes".to_string()));
472        }
473
474        let mut sig_arr = [0u8; 64];
475        sig_arr.copy_from_slice(&sig_bytes);
476        let signature = ed25519_dalek::Signature::from_bytes(&sig_arr);
477
478        verifying_key
479            .verify(canonical.as_bytes(), &signature)
480            .map_err(|e| PuzzledError::Ima(format!("signature verification failed: {}", e)))
481    }

485    fn extend_ima_log(&self, manifest_path: &Path) {
486        // Write to the IMA measurement log via /sys/kernel/security/ima/ascii_runtime_measurements
487        // This requires CAP_SYS_ADMIN. If unavailable, we skip silently.
488        let ima_path = Path::new("/sys/kernel/security/ima/policy");

492                "IMA measurement log available (measurement recorded by kernel on file access)"

494        } else {
495            tracing::debug!("IMA measurement log not available (continuing without)");
496        }
497    }

500/// M21: Build canonical content for signing using JSON canonical form.
501///
502/// Uses `BTreeMap` for deterministic key ordering and `serde_json::to_string`
503/// for consistent serialization. Each file entry is a JSON object with sorted
504/// keys: `{"checksum":"...","kind":"...","path":"...","size":N}`. Files are
505/// sorted by path for deterministic ordering.
506///
507/// H-8: Includes `agent_id`, `agent_profile`, and `checksum_before` fields in
508/// the canonical output so they are covered by the Ed25519 signature.
509///
510/// Output format:
511/// ```json
512/// {"agent_id":"...","agent_profile":"...","branch_id":"...","checksum_before":"...","files":[...],"timestamp":"..."}
513/// ```
514fn build_canonical(
515    branch_id: &str,
516    timestamp: &str,
517    files: &[ManifestEntry],
518    agent_id: &str,
519    agent_profile: &str,
520    checksum_before: &str,
521) -> anyhow::Result<String> {
522    use std::collections::BTreeMap;
523
524    // Sort files by path for deterministic ordering
525    let mut sorted_files: Vec<&ManifestEntry> = files.iter().collect();
526    sorted_files.sort_by(|a, b| a.path.cmp(&b.path));
527
528    let file_entries: Vec<serde_json::Value> = sorted_files
529        .iter()
530        .map(|f| {
531            let mut entry = BTreeMap::new();
532            entry.insert("checksum", serde_json::Value::String(f.checksum.clone()));
533            entry.insert("kind", serde_json::Value::String(f.kind.clone()));
534            entry.insert("path", serde_json::Value::String(f.path.clone()));
535            entry.insert("size", serde_json::json!(f.size));
536            // N11: Propagate serialization errors instead of panicking
537            serde_json::to_value(entry)
538                .map_err(|e| anyhow::anyhow!("canonical serialization failed: {e}"))
539        })
540        .collect::<anyhow::Result<Vec<_>>>()?;
541
542    let mut canonical = BTreeMap::new();
543    // H-8: Include agent_id, agent_profile, and checksum_before in signed payload
544    canonical.insert("agent_id", serde_json::Value::String(agent_id.to_string()));
545    canonical.insert(
546        "agent_profile",
547        serde_json::Value::String(agent_profile.to_string()),
548    );
549    canonical.insert(
550        "branch_id",
551        serde_json::Value::String(branch_id.to_string()),
552    );
553    canonical.insert(
554        "checksum_before",
555        serde_json::Value::String(checksum_before.to_string()),
556    );
557    canonical.insert("files", serde_json::Value::Array(file_entries));
558    canonical.insert(
559        "timestamp",
560        serde_json::Value::String(timestamp.to_string()),
561    );
562
563    // N11: Propagate serialization errors instead of panicking
564    serde_json::to_string(&canonical)
565        .map_err(|e| anyhow::anyhow!("canonical serialization failed: {e}"))
566}

570/// Fill buffer with cryptographically secure random bytes.
571///
572/// Uses the `getrandom` crate which calls the OS CSPRNG (`getrandom(2)` syscall
573/// on Linux, `/dev/urandom` as fallback). Returns an error if the OS cannot
574/// provide entropy — generating an Ed25519 signing key with insufficient
575/// entropy is a critical security failure (CVE-2025-2814).
576fn csprng_fill(buf: &mut [u8]) -> std::result::Result<(), PuzzledError> {
577    getrandom::getrandom(buf).map_err(|e| {
578        PuzzledError::Ima(format!(
579            "CSPRNG entropy source unavailable — refusing to generate signing key with \
580             insufficient randomness. Ensure /dev/urandom is accessible or the getrandom(2) \
581             syscall is permitted by seccomp policy: {}",
582            e
583        ))
584    })
585}

848    // S11: Clock skew on key file should return error, not silently default to 0
849    //
850    // When a key file has an mtime in the future, `duration_since()` fails and
851    // `unwrap_or_default()` returns Duration::ZERO. This makes a future-dated
852    // key appear 0 days old, preventing rotation even when the key should be
853    // rotated (e.g., 90-day threshold). The fix should detect clock skew and
854    // return an error.
855    #[test]
856    fn test_s11_future_dated_key_returns_error() {
857        let dir = tempfile::tempdir().unwrap();
858        let manifest_dir = dir.path().join("manifests");
859        let key_path = dir.path().join("key");
860
861        let mut ima = ImaIntegration::new(manifest_dir, &key_path).unwrap();
862
863        // Set the key file's mtime far into the future (year 2100)
864        let future_time =
865            std::time::SystemTime::now() + std::time::Duration::from_secs(365 * 24 * 60 * 60 * 75); // ~75 years ahead
866
867        let file = std::fs::File::open(&key_path).unwrap();
868        file.set_times(std::fs::FileTimes::new().set_modified(future_time))
869            .unwrap();
870
871        // With the default 90-day threshold, a future-dated key with age
872        // silently defaulted to 0 would appear "fresh" and skip rotation.
873        // This should return an error about clock skew instead.
874        let result = ima.check_key_rotation_with_threshold(DEFAULT_KEY_ROTATION_DAYS);
875        assert!(
876            result.is_err(),
877            "S11: future-dated key should return an error about clock skew, \
878             got {:?} — unwrap_or_default() makes the key appear 0 days old, \
879             preventing rotation of a key that may actually be very old",
880            result
881        );
882        let err_msg = format!("{}", result.unwrap_err());
883        assert!(
884            err_msg.contains("clock") || err_msg.contains("future"),
885            "S11: error message should mention clock skew, got: {}",
886            err_msg
887        );
888    }
