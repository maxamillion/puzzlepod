10/// A provenance record linking cause to effect.
11#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
12pub struct ProvenanceRecord {
13    pub id: String,
14    pub record_type: ProvenanceType,
15    pub branch_id: String,
16    pub timestamp: String,
17}
18
19/// Provenance record type -- each variant captures a different stage of the chain.
20#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
21#[serde(tag = "type", rename_all = "snake_case")]
22pub enum ProvenanceType {
23    /// Reported by agent framework SDK.
24    Request {
25        request_id: String,
26        user_uid: u32,
27        prompt_hash: String,
28    },
29    /// Reported by agent framework SDK.
30    Inference {
31        inference_id: String,
32        request_id: String,
33        model: String,
34        token_count: u32,
35        tool_calls: Vec<String>,
36    },
37    /// From seccomp USER_NOTIF handler + optional SDK enrichment.
38    ToolInvocation {
39        invocation_id: String,
40        #[serde(default, skip_serializing_if = "Option::is_none")]
41        inference_id: Option<String>,
42        tool_path: String,
43        #[serde(default, skip_serializing_if = "Option::is_none")]
44        arguments_hash: Option<String>,
45        pid: u32,
46        #[serde(default, skip_serializing_if = "Option::is_none")]
47        exit_code: Option<i32>,
48        /// Timestamp when the tool invocation started (RFC 3339).
49        #[serde(default, skip_serializing_if = "Option::is_none")]
50        started_at: Option<String>,
51        /// Timestamp when the tool invocation exited (RFC 3339).
52        #[serde(default, skip_serializing_if = "Option::is_none")]
53        exited_at: Option<String>,
54    },
55    /// From DiffEngine + fanotify correlation.
56    FileChange {
57        change_id: String,
58        #[serde(default, skip_serializing_if = "Option::is_none")]
59        invocation_id: Option<String>,
60        path: String,
61        kind: FileChangeKind,
62        size: u64,
63        checksum: String,
64    },
65    /// From policy evaluation + CommitManifest.
66    Governance {
67        decision_id: String,
68        change_ids: Vec<String>,
69        policy_version: String,
70        result: String,
71        violations: Vec<String>,
72        #[serde(default, skip_serializing_if = "Option::is_none")]
73        manifest_hash: Option<String>,
74    },
75}
