2//! Unified audit record types for NDJSON deserialization.
3//!
4//! V49: Previously duplicated in `puzzlectl/src/main.rs` (extended version with
5//! attestation fields) and `puzzlectl/src/compliance.rs` (minimal 3-field version).
6//! Unified here as the superset: compliance consumers simply ignore the optional
7//! attestation fields, while attestation verification has access to all fields.
8//! All optional fields use `#[serde(default, skip_serializing_if = "Option::is_none")]`
9//! to maintain wire compatibility with both the minimal and extended NDJSON formats.
10
11use crate::{AgentIdentity, GovernanceDecision};
12
13/// Deserialization struct for audit events read from NDJSON.
14///
15/// Mirrors `StoredAuditEvent` in puzzled. The three core fields (`seq`,
16/// `timestamp`, `event`) are always present; the remaining fields are
17/// populated only for governance-significant records that carry attestation
18/// metadata.
19#[derive(Debug, Clone, Default, PartialEq, serde::Serialize, serde::Deserialize)]
20pub struct AuditRecord {
21    pub seq: u64,
22    pub timestamp: String,
23    pub event: AuditRecordEvent,
24    #[serde(default, skip_serializing_if = "Option::is_none")]
25    pub record_id: Option<String>,
26    #[serde(default, skip_serializing_if = "Option::is_none")]
27    pub agent_identity: Option<AgentIdentity>,
28    #[serde(default, skip_serializing_if = "Option::is_none")]
29    pub policy_version: Option<String>,
30    #[serde(default, skip_serializing_if = "Option::is_none")]
31    pub changeset_hash: Option<String>,
32    #[serde(default, skip_serializing_if = "Option::is_none")]
33    pub governance_decision: Option<GovernanceDecision>,
34    #[serde(default, skip_serializing_if = "Option::is_none")]
35    pub parent_record_id: Option<String>,
36    #[serde(default, skip_serializing_if = "Option::is_none")]
37    pub signature: Option<String>,
38    /// HMAC chain signature (puzzled stores this as "hmac" in NDJSON).
39    #[serde(default, skip_serializing_if = "Option::is_none")]
40    pub hmac: Option<String>,
41    #[serde(default, skip_serializing_if = "Option::is_none")]
42    pub merkle_leaf_index: Option<u64>,
43}
44
45/// The event payload within an [`AuditRecord`].
46#[derive(Debug, Clone, Default, PartialEq, serde::Serialize, serde::Deserialize)]
47pub struct AuditRecordEvent {
48    pub event_type: String,
49    #[serde(default, skip_serializing_if = "Option::is_none")]
50    pub branch_id: Option<String>,
51    pub details: serde_json::Value,
52}
