10/// Kind of change detected in the OverlayFS upper layer.
11#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Type)]
12pub enum FileChangeKind {
13    /// m4: PRD canonical name is "Created"; "Added" kept as primary for backward compat.
14    #[serde(alias = "Created")]
15    Added,
16    Modified,
17    Deleted,
18    /// m5: PRD canonical name is "PermissionChanged"; "MetadataChanged" kept as primary.
19    #[serde(alias = "PermissionChanged")]
20    MetadataChanged,
21    /// File was renamed (OverlayFS redirect xattr).
22    Renamed,
23    /// H9: Symbolic link detected in changeset. Rejected by default unless
24    /// the agent profile sets `allow_symlinks: true`.
25    Symlink,
26    /// Q6: Hard link (nlink > 1) detected in changeset.
27    Hardlink,
28    /// Q6: Block device special file detected in changeset.
29    BlockDevice,
30    /// Q6: Character device special file detected in changeset.
31    CharDevice,
32    /// Q6: Named pipe (FIFO) detected in changeset.
33    Fifo,
34}
35
36/// A single file change in a branch diff.
37#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
38pub struct FileChange {
39    /// Path relative to the branch root.
40    pub path: PathBuf,
41    pub kind: FileChangeKind,
42    /// Size in bytes (0 for deletions).
43    pub size: u64,
44    /// Size of the file in the base layer (for Modified changes).
45    #[serde(default, skip_serializing_if = "Option::is_none")]
46    pub old_size: Option<u64>,
47    /// File mode in the base layer (for Modified/MetadataChanged).
48    #[serde(default, skip_serializing_if = "Option::is_none")]
49    pub old_mode: Option<u32>,
50    /// File mode in the upper layer.
51    #[serde(default, skip_serializing_if = "Option::is_none")]
52    pub new_mode: Option<u32>,
53    /// SHA-256 checksum of file contents (empty for deletions).
54    pub checksum: String,
55    /// RFC 3339 timestamp of the file modification.
56    #[serde(default, skip_serializing_if = "Option::is_none")]
57    pub timestamp: Option<String>,
58    /// K60: Symlink target path (only populated for Symlink changes).
59    /// Included in Rego input so policies can validate symlink destinations.
60    #[serde(default, skip_serializing_if = "Option::is_none")]
61    pub target: Option<String>,
62    /// Shannon entropy of file content (0.0-8.0). Only set for Added/Modified text files.
63    #[serde(default, skip_serializing_if = "Option::is_none")]
64    pub entropy: Option<f64>,
65    /// Whether the file contains base64 blocks longer than 64 chars.
66    #[serde(default, skip_serializing_if = "Option::is_none")]
67    pub has_base64_blocks: Option<bool>,
68}
69
70impl Default for FileChange {
71    fn default() -> Self {
72        Self {
73            path: PathBuf::new(),
74            kind: FileChangeKind::Added,
75            size: 0,
76            old_size: None,
77            old_mode: None,
78            new_mode: None,
79            checksum: String::new(),
80            timestamp: None,
81            target: None,
82            entropy: None,
83            has_base64_blocks: None,
84        }
85    }
86}
87
88impl FileChange {
89    /// Create a `FileChange` with the given path, kind, and size.
90    ///
91    /// All optional fields default to `None` and checksum defaults to empty.
92    /// Use struct update syntax to override additional fields:
93    /// ```
94    /// # use puzzled_types::{FileChange, FileChangeKind};
95    /// let change = FileChange::new("/app/main.py", FileChangeKind::Modified, 1024);
96    /// ```
97    pub fn new(path: impl Into<PathBuf>, kind: FileChangeKind, size: u64) -> Self {
98        Self {
99            path: path.into(),
100            kind,
101            size,
102            ..Default::default()
103        }
104    }
105}
