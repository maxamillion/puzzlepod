42// Merkle tree crypto utilities (A-M1: deduplicated from attestation.rs + puzzlectl)

44
45/// Shared Merkle tree cryptographic functions used by both `puzzled` (attestation)
46/// and `puzzlectl` (verification). Domain-separated hashing per RFC 6962.
47pub mod merkle {
48    use sha2::{Digest, Sha256};
49
50    /// Domain separation prefix for leaf nodes (RFC 6962 §2.1).
51    const LEAF_PREFIX: u8 = 0x00;
52    /// Domain separation prefix for internal nodes (RFC 6962 §2.1).
53    const NODE_PREFIX: u8 = 0x01;
54
55    /// Compute domain-separated leaf hash: SHA-256(0x00 || data).
56    pub fn hash_leaf(data: &[u8]) -> [u8; 32] {
57        let mut hasher = Sha256::new();
58        hasher.update([LEAF_PREFIX]);
59        hasher.update(data);
60        hasher.finalize().into()
61    }
62
63    /// Compute domain-separated internal node hash: SHA-256(0x01 || left || right).
64    pub fn hash_node(left: &[u8; 32], right: &[u8; 32]) -> [u8; 32] {
65        let mut hasher = Sha256::new();
66        hasher.update([NODE_PREFIX]);
67        hasher.update(left);
68        hasher.update(right);
69        hasher.finalize().into()
70    }
71
72    /// Largest power of 2 strictly less than `n`.
73    pub fn largest_power_of_2_less_than(n: u64) -> u64 {
74        if n <= 1 {
75            return 0;
76        }
77        1u64 << (63 - (n - 1).leading_zeros())
78    }
79
80    /// Encode a byte slice as a lowercase hex string.
81    pub fn hex_encode(bytes: &[u8]) -> String {
82        const HEX_CHARS: &[u8; 16] = b"0123456789abcdef";
83        let mut s = String::with_capacity(bytes.len() * 2);
84        for &b in bytes {
85            s.push(HEX_CHARS[(b >> 4) as usize] as char);
86            s.push(HEX_CHARS[(b & 0x0f) as usize] as char);
87        }
88        s
89    }
90
91    /// Decode a hex string to bytes.
92    ///
93    /// Returns an error for odd-length strings, non-ASCII input, or invalid hex digits.
94    pub fn hex_decode(s: &str) -> Result<Vec<u8>, String> {
95        // A-M3: Guard against multi-byte UTF-8 input which would cause panics
96        // in the byte-offset indexing below (&s[i..i + 2]).
97        if !s.is_ascii() {
98            return Err("non-ASCII characters in hex string".to_string());
99        }
100        if !s.len().is_multiple_of(2) {
101            return Err("odd-length hex string".to_string());
102        }
103        (0..s.len())
104            .step_by(2)
105            .map(|i| {
106                u8::from_str_radix(&s[i..i + 2], 16)
107                    .map_err(|e| format!("invalid hex at position {}: {}", i, e))
108            })
109            .collect()
110    }
111
112    /// Recompute root hash from an inclusion proof (bottom-up traversal).
113    ///
114    /// Proof elements are consumed bottom-up via `pos` index, matching the
115    /// order they were generated (leaf-level sibling first, root-level last).
116    pub fn compute_root_from_inclusion(
117        leaf_index: u64,
118        tree_size: u64,
119        leaf_hash: &[u8; 32],
120        proof: &[[u8; 32]],
121        pos: &mut usize,
122    ) -> Option<[u8; 32]> {
123        if tree_size <= 1 {
124            return Some(*leaf_hash);
125        }
126        let k = largest_power_of_2_less_than(tree_size);
127        if leaf_index < k {
128            let left = compute_root_from_inclusion(leaf_index, k, leaf_hash, proof, pos)?;
129            let right = *proof.get(*pos)?;
130            *pos += 1;
131            Some(hash_node(&left, &right))
132        } else {
133            let right =
134                compute_root_from_inclusion(leaf_index - k, tree_size - k, leaf_hash, proof, pos)?;
135            let left = *proof.get(*pos)?;
136            *pos += 1;
137            Some(hash_node(&left, &right))
138        }
139    }
140
141    /// Verify a Merkle inclusion proof against an expected root hash.
142    ///
143    /// Returns `Ok(true)` if the proof is valid, `Ok(false)` if the computed root
144    /// doesn't match, or `Err` if the proof is malformed.
145    pub fn verify_merkle_inclusion(
146        leaf_hash: &[u8; 32],
147        proof: &super::InclusionProof,
148        expected_root: &[u8; 32],
149    ) -> Result<bool, String> {
150        let proof_hashes: Vec<[u8; 32]> = proof
151            .proof_hashes
152            .iter()
153            .map(|h| {
154                let bytes = hex_decode(h)?;
155                if bytes.len() != 32 {
156                    return Err(format!("proof hash must be 32 bytes, got {}", bytes.len()));
157                }
158                let mut arr = [0u8; 32];
159                arr.copy_from_slice(&bytes);
160                Ok(arr)
161            })
162            .collect::<Result<Vec<_>, _>>()?;
163
164        let mut pos = 0;
165        let computed = compute_root_from_inclusion(
166            proof.leaf_index,
167            proof.tree_size,
168            leaf_hash,
169            &proof_hashes,
170            &mut pos,
171        )
172        .ok_or_else(|| "malformed inclusion proof: insufficient proof hashes".to_string())?;
173
174        // RFC 6962: all proof hashes must be consumed
175        if pos != proof_hashes.len() {
176            return Err(format!(
177                "malformed inclusion proof: {} extra hashes",
178                proof_hashes.len() - pos
179            ));
180        }
181
182        Ok(computed == *expected_root)
183    }
184}
