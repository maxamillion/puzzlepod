10/// Output a JSON action result or a plain text message.
11/// L-ctl2: Uses serde_json::to_string_pretty for JSON output consistency.
12pub fn output_action(format: OutputFormat, status: &str, id: &str, reason: &str, text_msg: &str) {
13    match format {
14        OutputFormat::Json => {
15            let mut obj = serde_json::json!({"status": status, "branch_id": id});
16            if !reason.is_empty() {
17                obj["reason"] = serde_json::Value::String(reason.to_string());
18            }
19            println!(
20                "{}",
21                serde_json::to_string_pretty(&obj).unwrap_or_else(|_| obj.to_string())
22            );
23        }
24        OutputFormat::Text => println!("{text_msg}"),
25    }
26}

28/// Output a D-Bus result string as pretty JSON or plain text.
29/// Handles the common pattern where the daemon returns a JSON string that needs
30/// pretty-printing in JSON mode, or direct printing in text mode.
31pub fn output_json_or_text(format: OutputFormat, json_str: &str, text_fn: impl FnOnce(&str)) {
32    match format {
33        OutputFormat::Json => {
34            // Try to parse as JSON for pretty-printing; fall back to raw string
35            let pretty = serde_json::from_str::<serde_json::Value>(json_str)
36                .ok()
37                .and_then(|v| serde_json::to_string_pretty(&v).ok())
38                .unwrap_or_else(|| json_str.to_string());
39            println!("{pretty}");
40        }
41        OutputFormat::Text => text_fn(json_str),
42    }
43}

45pub fn format_bytes(bytes: u64) -> String {
46    if bytes < 1024 {
47        format!("{} B", bytes)
48    } else if bytes < 1024 * 1024 {
49        format!("{:.1} KiB", bytes as f64 / 1024.0)
50    } else if bytes < 1024 * 1024 * 1024 {
51        format!("{:.1} MiB", bytes as f64 / (1024.0 * 1024.0))
52    } else {
53        format!("{:.1} GiB", bytes as f64 / (1024.0 * 1024.0 * 1024.0))
54    }
55}

57/// L11: UTF-8 safe truncation. Uses char_indices to find a valid boundary
58/// instead of byte slicing, which could panic on multi-byte characters.
59pub fn truncate(s: &str, max_len: usize) -> &str {
60    if s.len() <= max_len {
61        return s;
62    }
63    match s.char_indices().nth(max_len) {
64        Some((idx, _)) => &s[..idx],
65        None => s,
66    }
67}

69/// L-ctl1: Filter branches JSON array by state.
70/// Supported state filters: "active", "reviewing" (governance_review), "degraded", "all".
71pub fn filter_branches_by_state(branches_json: &str, state_filter: &str) -> String {
72    if state_filter == "all" {
73        return branches_json.to_string();
74    }
75
76    let target_state = match state_filter {
77        "active" => "active",
78        "reviewing" => "governance_review",
79        "degraded" => "degraded",
80        other => other,
81    };
82
83    match serde_json::from_str::<Vec<serde_json::Value>>(branches_json) {
84        Ok(branches) => {
85            let filtered: Vec<&serde_json::Value> = branches
86                .iter()
87                .filter(|b| {
88                    b.get("state")
89                        .and_then(|v| v.as_str())
90                        .map(|s| s == target_state)
91                        .unwrap_or(false)
92                })
93                .collect();
94            serde_json::to_string(&filtered).unwrap_or_else(|_| branches_json.to_string())
95        }
96        Err(_) => branches_json.to_string(),
97    }
98}

100/// M-ctl2: Validate a branch ID before sending it over D-Bus.
101/// Returns the validated BranchId, or a clear CLI error instead of a cryptic D-Bus error on invalid input.
102pub fn validate_branch_id(id: &str) -> Result<BranchId> {
103    BranchId::validated(id.to_string())
104        .map_err(|e| anyhow::anyhow!("invalid branch ID '{}': {}", id, e))
105}

107/// Read a credential value from stdin with bounded size.
108/// H62: Uses Zeroizing<String> so credential material is zeroized on drop.
109/// J42: Limits stdin read to MAX_CREDENTIAL_SIZE to prevent unbounded allocation.
110pub fn read_credential_stdin(context_msg: &str) -> Result<zeroize::Zeroizing<String>> {
111    let mut value = zeroize::Zeroizing::new(String::new());
112    std::io::Read::read_to_string(
113        &mut std::io::Read::take(std::io::stdin(), MAX_CREDENTIAL_SIZE),
114        &mut value,
115    )
116    .context(context_msg.to_string())?;
117    Ok(zeroize::Zeroizing::new(value.trim_end().to_string()))
118}
