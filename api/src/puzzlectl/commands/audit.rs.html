6/// Handle audit commands that require D-Bus (List, Export).
7pub async fn handle_dbus(
8    action: crate::cli::AuditAction,
9    client: &client::PuzzledClient,
10    output: OutputFormat,
11) -> Result<()> {
12    match action {
13        crate::cli::AuditAction::List {
14            branch_id,
15            event_type,
16            since,
17            limit,
18        } => {
19            let filter = serde_json::json!({
20                "branch_id": branch_id,
21                "event_type": event_type,
22                "since": since,
23                "limit": limit,
24            });
25            let result = client.query_audit_events(&filter.to_string()).await?;
26            println!("{result}");
27        }
28        crate::cli::AuditAction::Export { format, file } => {
29            let result = client.export_audit_events(&format).await?;
30            if let Some(path) = file {
31                std::fs::write(&path, &result).with_context(|| format!("writing to {}", path))?;
32                match output {
33                    OutputFormat::Json => {
34                        println!(
35                            "{}",
36                            serde_json::json!({"status": "exported", "file": path})
37                        );
38                    }
39                    OutputFormat::Text => println!("Exported to {}", path),
40                }
41            } else {
42                println!("{result}");
43            }
44        }
45        crate::cli::AuditAction::Verify { .. } => unreachable!(),
46    }
47    Ok(())
48}

50/// Verify an IMA manifest signature.
51pub fn cmd_audit_verify(hash_or_path: &str) -> Result<()> {
52    let path = if std::path::Path::new(hash_or_path).exists() {
53        std::path::PathBuf::from(hash_or_path)
54    } else {
55        // Try to find the manifest by branch ID hash
56        let manifest_dir = std::path::PathBuf::from("/var/lib/puzzled/branches/manifests");
57        let candidate = manifest_dir.join(format!("{}.manifest.yaml", hash_or_path));
58        if candidate.exists() {
59            candidate
60        } else {
61            anyhow::bail!(
62                "manifest not found: {} (tried path and /var/lib/puzzled/branches/manifests/)",
63                hash_or_path
64            );
65        }
66    };
67
68    let contents =
69        std::fs::read_to_string(&path).with_context(|| format!("reading {}", path.display()))?;
70
71    println!("Manifest: {}", path.display());
72    println!("{}", contents);
73
74    // T23: Signature verification is not yet implemented — fail explicitly
75    // rather than silently succeeding and misleading the operator.
76    eprintln!("WARNING: signature verification is not yet implemented.");
77    eprintln!(
78        "The manifest contents are displayed above but have NOT been cryptographically verified."
79    );
80    eprintln!(
81        "Use puzzled's attestation bundle (puzzlectl attestation verify-bundle) for verified audit."
82    );
83
84    Ok(())
85}
