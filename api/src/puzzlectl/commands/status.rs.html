2use crate::cli::OutputFormat;
3use crate::client;
4use crate::commands::branch::print_branches_text;
5use crate::output::{output_json_or_text, validate_branch_id};
6use anyhow::Result;
7
8pub async fn handle(
9    id: Option<String>,
10    client: &client::PuzzledClient,
11    output: OutputFormat,
12) -> Result<()> {
13    if let Some(branch_id) = id {
14        validate_branch_id(&branch_id)?;
15        let info = client.agent_info(&branch_id).await?;
16        output_json_or_text(output, &info, |s| {
17            println!("Branch status for {}:", branch_id);
18            println!("{s}");
19        });
20    } else {
21        let branches = client.list_branches().await?;
22        match output {
23            OutputFormat::Json => {
24                let parsed: serde_json::Value = serde_json::from_str(&branches)
25                    .unwrap_or(serde_json::Value::String(branches.clone()));
26                let count = parsed.as_array().map(|a| a.len()).unwrap_or(0);
27                let status = serde_json::json!({
28                    "daemon": "connected",
29                    "branch_count": count,
30                    "branches": parsed,
31                });
32                println!(
33                    "{}",
34                    serde_json::to_string_pretty(&status).unwrap_or(branches)
35                );
36            }
37            OutputFormat::Text => {
38                println!("Daemon: connected");
39                let parsed: Result<Vec<serde_json::Value>, _> = serde_json::from_str(&branches);
40                match parsed {
41                    Ok(list) => {
42                        println!("Active branches: {}", list.len());
43                        if !list.is_empty() {
44                            println!();
45                            print_branches_text(&branches);
46                        }
47                    }
48                    Err(_) => println!("{branches}"),
49                }
50            }
51        }
52    }
53    Ok(())
54}
