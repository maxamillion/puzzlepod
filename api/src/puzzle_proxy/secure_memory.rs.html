2//! §3.4 G3/G4: Secure memory region for credential storage.
3//!
4//! Provides `SecureRegion` — an mmap-backed memory region with:
5//! - `mlock()` to prevent swapping to disk
6//! - `MADV_DONTDUMP` to exclude from core dumps
7//! - Guard pages (`PROT_NONE`) for overflow detection
8//! - Volatile zeroize on `Drop` with compiler fence
9//!
10//! Also provides `SecureCredentialStore` (slot-based credential storage) and
11//! `CredentialManager` (thread-safe async wrapper with zero-copy access).
12
13use std::collections::HashMap;
14use std::sync::atomic::{compiler_fence, Ordering};
15
16/// Errors from secure memory operations.
17#[derive(Debug, thiserror::Error)]
18pub enum SecureMemoryError {
19    #[error(
20        "mlock failed: {0} (mlock_required=true, cannot store credentials in swappable memory)"
21    )]
22    MlockFailed(std::io::Error),
23    #[error("mmap failed: {0}")]
24    MmapFailed(std::io::Error),
25    #[error("store is full: {used}/{capacity} slots used")]
26    StoreFull { used: usize, capacity: usize },
27    #[error("credential too large: {size} bytes exceeds slot capacity {capacity}")]
28    CredentialTooLarge { size: usize, capacity: usize },
29    #[error("credential not found: {0}")]
30    NotFound(String),
31    #[error("rotate failed: new value ({new_size} bytes) exceeds slot capacity ({capacity})")]
32    RotateExceedsCapacity { new_size: usize, capacity: usize },
33}
34
35// ---------------------------------------------------------------------------
36// SecureRegion — mmap-backed secure memory
37// ---------------------------------------------------------------------------
38
39/// A memory region backed by `mmap(MAP_ANONYMOUS | MAP_PRIVATE)` with
40/// security hardening: `mlock`, `MADV_DONTDUMP`, and `PROT_NONE` guard pages.
41pub(crate) struct SecureRegion {
42    /// Pointer to the start of the full allocation (including guard pages).
43    base_ptr: *mut u8,
44    /// Total allocation size including guard pages.
45    total_size: usize,
46    /// Offset from base_ptr to the usable region (after the leading guard page).
47    usable_offset: usize,
48    /// Size of the usable region (excluding guard pages).
49    usable_size: usize,
50    /// Page size on this system.
51    #[allow(dead_code)]
52    page_size: usize,
53}
54
55// SAFETY: SecureRegion manages raw memory that is not shared across threads
56// by itself. Thread safety is provided by the CredentialManager's RwLock.
57unsafe impl Send for SecureRegion {}
58unsafe impl Sync for SecureRegion {}
59
60impl SecureRegion {
61    /// Create a new secure memory region.
62    ///
63    /// - `size`: desired usable size (will be rounded up to page boundary)
64    /// - `mlock_required`: if true, `mlock()` failure is fatal
65    ///
66    /// Layout: [guard page] [usable region] [guard page]
67    pub fn new(size: usize, mlock_required: bool) -> Result<Self, SecureMemoryError> {
68        let page_size = page_size();
69        // Round up to page boundary
70        let usable_size = (size + page_size - 1) & !(page_size - 1);
71        // Total = leading guard + usable + trailing guard
72        let total_size = usable_size + 2 * page_size;
73
74        // SAFETY: mmap with MAP_ANONYMOUS|MAP_PRIVATE allocates zeroed private memory.
75        // NULL base lets the kernel choose the address. fd=-1, offset=0 for anonymous.
76        // Return value is checked for MAP_FAILED before use.
77        let base_ptr = unsafe {
78            libc::mmap(
79                std::ptr::null_mut(),
80                total_size,
81                libc::PROT_READ | libc::PROT_WRITE,
82                libc::MAP_ANONYMOUS | libc::MAP_PRIVATE,
83                -1,
84                0,
85            )
86        };
87        if base_ptr == libc::MAP_FAILED {
88            return Err(SecureMemoryError::MmapFailed(
89                std::io::Error::last_os_error(),
90            ));
91        }
92        let base_ptr = base_ptr as *mut u8;
93
94        // SAFETY: mprotect on regions within our mmap allocation. Pointers are valid
95        // and within bounds (leading guard = base_ptr, trailing = base_ptr + page_size + usable_size).
96        // On failure, we munmap the entire allocation before returning.
97        unsafe {
98            // Leading guard page
99            if libc::mprotect(base_ptr as *mut libc::c_void, page_size, libc::PROT_NONE) != 0 {
100                libc::munmap(base_ptr as *mut libc::c_void, total_size);
101                return Err(SecureMemoryError::MmapFailed(
102                    std::io::Error::last_os_error(),
103                ));
104            }
105            // Trailing guard page
106            let trailing = base_ptr.add(page_size + usable_size);
107            if libc::mprotect(trailing as *mut libc::c_void, page_size, libc::PROT_NONE) != 0 {
108                libc::munmap(base_ptr as *mut libc::c_void, total_size);
109                return Err(SecureMemoryError::MmapFailed(
110                    std::io::Error::last_os_error(),
111                ));
112            }
113        }
114
115        // SAFETY: base_ptr.add(page_size) is within the mmap'd region (total_size > page_size).
116        let usable_ptr = unsafe { base_ptr.add(page_size) };
117
118        // SAFETY: usable_ptr points to page-aligned memory within our allocation.
119        // Lock usable region in RAM (prevent swapping).
120        let mlock_result = unsafe { libc::mlock(usable_ptr as *const libc::c_void, usable_size) };
121        if mlock_result != 0 {
122            let err = std::io::Error::last_os_error();
123            if mlock_required {
124                unsafe {
125                    libc::munmap(base_ptr as *mut libc::c_void, total_size);
126                }
127                return Err(SecureMemoryError::MlockFailed(err));
128            }
129            // §3.4 T2.4: PRD requires Critical audit event when mlock fails with
130            // mlock_required=false. Use tracing::error! with structured audit fields
131            // so the audit subsystem can capture this as a Critical event.
132            tracing::error!(
133                error = %err,
134                audit_type = "credential.mlock_failed",
135                severity = "critical",
136                "§3.4 G3: mlock() failed — credentials may be swapped to disk. \
137                 Set LimitMEMLOCK=infinity in puzzled.service."
138            );
139        }
140
141        // SAFETY: Advisory only — MADV_DONTDUMP tells the kernel to exclude this
142        // region from core dumps. usable_ptr and usable_size are valid.
143        unsafe {
144            libc::madvise(
145                usable_ptr as *mut libc::c_void,
146                usable_size,
147                libc::MADV_DONTDUMP,
148            );
149        }
150
151        Ok(Self {
152            base_ptr,
153            total_size,
154            usable_offset: page_size,
155            usable_size,
156            page_size,
157        })
158    }
159
160    /// Write data to the region at the given offset.
161    ///
162    /// # Panics
163    /// Panics if `offset + data.len()` exceeds the usable region.
164    pub fn write(&self, offset: usize, data: &[u8]) {
165        assert!(
166            offset + data.len() <= self.usable_size,
167            "SecureRegion::write out of bounds: offset={} len={} usable={}",
168            offset,
169            data.len(),
170            self.usable_size
171        );
172        // SAFETY: Bounds checked by assert above. dst is within the usable region.
173        unsafe {
174            let dst = self.base_ptr.add(self.usable_offset + offset);
175            std::ptr::copy_nonoverlapping(data.as_ptr(), dst, data.len());
176        }
177    }
178
179    /// Read data from the region at the given offset.
180    ///
181    /// # Panics
182    /// Panics if `offset + len` exceeds the usable region.
183    pub fn read(&self, offset: usize, len: usize) -> &[u8] {
184        assert!(
185            offset + len <= self.usable_size,
186            "SecureRegion::read out of bounds: offset={} len={} usable={}",
187            offset,
188            len,
189            self.usable_size
190        );
191        // SAFETY: Bounds checked by assert above. src is within the usable region.
192        // The returned slice borrows &self, so the region cannot be dropped while in use.
193        unsafe {
194            let src = self.base_ptr.add(self.usable_offset + offset);
195            std::slice::from_raw_parts(src, len)
196        }
197    }
198
199    /// Volatile-zeroize a specific slot in the region.
200    pub fn zeroize_slot(&self, offset: usize, len: usize) {
201        assert!(
202            offset + len <= self.usable_size,
203            "SecureRegion::zeroize_slot out of bounds"
204        );
205        // SAFETY: Bounds checked by assert above. write_volatile ensures the compiler
206        // cannot elide the zeroing. compiler_fence after the block prevents reordering.
207        unsafe {
208            let ptr = self.base_ptr.add(self.usable_offset + offset);
209            for i in 0..len {
210                std::ptr::write_volatile(ptr.add(i), 0u8);
211            }
212        }
213        compiler_fence(Ordering::SeqCst);
214    }
215
216    /// Total usable size of the region.
217    #[cfg(test)]
218    pub fn usable_size(&self) -> usize {
219        self.usable_size
220    }
221}
222
223impl Drop for SecureRegion {
224    fn drop(&mut self) {
225        // SAFETY: Zeroize the entire usable region using write_volatile to prevent
226        // the compiler from eliding the write. compiler_fence ensures ordering.
227        unsafe {
228            let ptr = self.base_ptr.add(self.usable_offset);
229            for i in 0..self.usable_size {
230                std::ptr::write_volatile(ptr.add(i), 0u8);
231            }
232        }
233        compiler_fence(Ordering::SeqCst);
234
235        // SAFETY: munlock + munmap on the original allocation. base_ptr and sizes
236        // are unchanged since construction. Must occur unconditionally in drop.
237        unsafe {
238            let usable_ptr = self.base_ptr.add(self.usable_offset);
239            libc::munlock(usable_ptr as *const libc::c_void, self.usable_size);
240            libc::munmap(self.base_ptr as *mut libc::c_void, self.total_size);
241        }
242    }
243}

245/// Get the system page size.
246fn page_size() -> usize {
247    // SAFETY: sysconf(_SC_PAGESIZE) is a read-only query with no side effects.
248    unsafe { libc::sysconf(libc::_SC_PAGESIZE) as usize }
249}

255/// A single credential slot within the SecureRegion.
256#[derive(Debug, Clone)]
257struct CredentialSlot {
258    /// Offset into the SecureRegion.
259    offset: usize,
260    /// Current length of the credential value.
261    len: usize,
262    /// Maximum capacity of this slot (fixed at allocation time).
263    capacity: usize,
264}
265
266/// Slot-based credential store backed by a `SecureRegion`.
267///
268/// Credentials are stored in fixed-size slots within the secure region.
269/// Each slot has a maximum capacity set at allocation time.
270pub struct SecureCredentialStore {
271    region: SecureRegion,
272    /// Maps phantom token → slot metadata.
273    index: HashMap<String, CredentialSlot>,
274    /// Next available offset for allocation.
275    next_offset: usize,
276    /// Maximum credential value size per slot.
277    slot_capacity: usize,
278    /// Maximum number of credentials.
279    max_credentials: usize,
280}
281
282impl SecureCredentialStore {
283    /// Create a new credential store.
284    ///
285    /// - `max_credentials`: maximum number of credentials
286    /// - `slot_capacity`: maximum bytes per credential value
287    /// - `mlock_required`: if true, fail if mlock() is unavailable
288    pub fn new(
289        max_credentials: usize,
290        slot_capacity: usize,
291        mlock_required: bool,
292    ) -> Result<Self, SecureMemoryError> {
293        let total_size = max_credentials * slot_capacity;
294        let region = SecureRegion::new(total_size, mlock_required)?;
295        Ok(Self {
296            region,
297            index: HashMap::with_capacity(max_credentials),
298            next_offset: 0,
299            slot_capacity,
300            max_credentials,
301        })
302    }
303
304    /// Store a credential value under the given phantom token key.
305    pub fn store(&mut self, phantom_token: &str, value: &[u8]) -> Result<(), SecureMemoryError> {
306        if value.len() > self.slot_capacity {
307            return Err(SecureMemoryError::CredentialTooLarge {
308                size: value.len(),
309                capacity: self.slot_capacity,
310            });
311        }
312        if self.index.len() >= self.max_credentials {
313            return Err(SecureMemoryError::StoreFull {
314                used: self.index.len(),
315                capacity: self.max_credentials,
316            });
317        }
318
319        // Allocate a new slot
320        let offset = self.next_offset;
321        self.next_offset += self.slot_capacity;
322
323        // Write the credential value
324        self.region.write(offset, value);
325
326        self.index.insert(
327            phantom_token.to_string(),
328            CredentialSlot {
329                offset,
330                len: value.len(),
331                capacity: self.slot_capacity,
332            },
333        );
334
335        Ok(())
336    }
337
338    /// Resolve a phantom token to the credential value bytes.
339    pub fn resolve(&self, phantom_token: &str) -> Option<&[u8]> {
340        let slot = self.index.get(phantom_token)?;
341        Some(self.region.read(slot.offset, slot.len))
342    }
343
344    /// Rotate a credential: zeroize old value, write new value in the same slot.
345    pub fn rotate(
346        &mut self,
347        phantom_token: &str,
348        new_value: &[u8],
349    ) -> Result<(), SecureMemoryError> {
350        let slot = self
351            .index
352            .get_mut(phantom_token)
353            .ok_or_else(|| SecureMemoryError::NotFound(phantom_token.to_string()))?;
354
355        if new_value.len() > slot.capacity {
356            return Err(SecureMemoryError::RotateExceedsCapacity {
357                new_size: new_value.len(),
358                capacity: slot.capacity,
359            });
360        }
361
362        // Zeroize old value — zeroize full capacity (not just len) to clear tail
363        // bytes from prior credentials that may have been longer than the current one.
364        self.region.zeroize_slot(slot.offset, slot.capacity);
365        // Write new value
366        self.region.write(slot.offset, new_value);
367        slot.len = new_value.len();
368
369        Ok(())
370    }
371
372    /// Remove a credential: zeroize the slot and remove from index.
373    pub fn remove(&mut self, phantom_token: &str) -> Result<(), SecureMemoryError> {
374        let slot = self
375            .index
376            .remove(phantom_token)
377            .ok_or_else(|| SecureMemoryError::NotFound(phantom_token.to_string()))?;
378        // Zeroize full slot capacity to clear all residual credential bytes.
379        self.region.zeroize_slot(slot.offset, slot.capacity);
380        Ok(())
381    }
382
383    /// Scan a byte buffer for any credential values (direct byte comparison).
384    ///
385    /// Returns a list of (phantom_token, byte_offset) for each match found.
386    /// Uses direct byte comparison — NO Aho-Corasick or other algorithms that
387    /// would copy credential values to heap memory.
388    pub fn scan_bytes(&self, haystack: &[u8]) -> Vec<(String, usize)> {
389        let mut matches = Vec::new();
390        for (token, slot) in &self.index {
391            let needle = self.region.read(slot.offset, slot.len);
392            if needle.is_empty() {
393                continue;
394            }
395            // Simple byte search — O(n*m) but credentials are short and
396            // this avoids any heap allocation of credential bytes.
397            for i in 0..=(haystack.len().saturating_sub(needle.len())) {
398                if &haystack[i..i + needle.len()] == needle {
399                    matches.push((token.clone(), i));
400                }
401            }
402        }
403        matches
404    }
405
406    /// Number of credentials currently stored.
407    pub fn credential_count(&self) -> usize {
408        self.index.len()
409    }
410
411    /// Check if a phantom token exists in the store.
412    pub fn contains(&self, phantom_token: &str) -> bool {
413        self.index.contains_key(phantom_token)
414    }
415}

421use std::ops::Deref;
422use std::sync::Arc;
423use tokio::sync::{OwnedRwLockReadGuard, RwLock};
424
425/// Thread-safe async credential manager wrapping `SecureCredentialStore`.
426///
427/// All credential access goes through `tokio::sync::RwLock` to ensure
428/// safe concurrent reads and exclusive writes.
429pub struct CredentialManager {
430    store: Arc<RwLock<SecureCredentialStore>>,
431}
432
433impl CredentialManager {
434    /// Create a new credential manager.
435    pub fn new(
436        max_credentials: usize,
437        slot_capacity: usize,
438        mlock_required: bool,
439    ) -> Result<Self, SecureMemoryError> {
440        let store = SecureCredentialStore::new(max_credentials, slot_capacity, mlock_required)?;
441        Ok(Self {
442            store: Arc::new(RwLock::new(store)),
443        })
444    }
445
446    /// Resolve a phantom token to a `SecureRef` that borrows credential bytes
447    /// from the secure region.
448    ///
449    /// The returned `SecureRef` holds the read lock — the credential bytes remain
450    /// valid and accessible as long as the `SecureRef` is alive. No credential
451    /// data is copied to heap memory.
452    pub async fn resolve(&self, phantom_token: &str) -> Option<SecureRef> {
453        let guard = self.store.clone().read_owned().await;
454        let slot = guard.index.get(phantom_token)?;
455        let offset = slot.offset;
456        let len = slot.len;
457        Some(SecureRef {
458            _guard: guard,
459            // SAFETY: The guard keeps the RwLock held, and the SecureRegion's
460            // memory is valid for the lifetime of the guard. We store offset/len
461            // to reconstruct the slice when dereferenced.
462            offset,
463            len,
464        })
465    }
466
467    /// Store a credential value under the given phantom token.
468    pub async fn store(&self, phantom_token: &str, value: &[u8]) -> Result<(), SecureMemoryError> {
469        let mut guard = self.store.write().await;
470        guard.store(phantom_token, value)
471    }
472
473    /// Rotate a credential: zeroize old value, write new value (same slot).
474    ///
475    /// M-12: Caller responsibility — the `new_value` parameter borrows from the caller.
476    /// Callers MUST ensure the source buffer is wrapped in `Zeroizing<Vec<u8>>` (or
477    /// equivalent) so that the caller's copy is zeroized after this call returns.
478    /// The mlock'd store will hold its own copy in the secure region; the caller's
479    /// heap copy must not persist.
480    pub async fn rotate(
481        &self,
482        phantom_token: &str,
483        new_value: &[u8],
484    ) -> Result<(), SecureMemoryError> {
485        let mut guard = self.store.write().await;
486        guard.rotate(phantom_token, new_value)
487    }
488
489    /// Remove a credential: zeroize the slot and remove from index.
490    pub async fn remove(&self, phantom_token: &str) -> Result<(), SecureMemoryError> {
491        let mut guard = self.store.write().await;
492        guard.remove(phantom_token)
493    }
494
495    /// Scan a byte buffer for any stored credential values.
496    ///
497    /// Returns (phantom_token, byte_offset) pairs for each match.
498    pub async fn scan_bytes(&self, haystack: &[u8]) -> Vec<(String, usize)> {
499        let guard = self.store.read().await;
500        guard.scan_bytes(haystack)
501    }
502
503    /// Number of credentials currently stored.
504    pub async fn credential_count(&self) -> usize {
505        let guard = self.store.read().await;
506        guard.credential_count()
507    }
508
509    /// Check if a phantom token exists.
510    pub async fn contains(&self, phantom_token: &str) -> bool {
511        let guard = self.store.read().await;
512        guard.contains(phantom_token)
513    }
514}
515
516/// A zero-copy reference to credential bytes in the secure region.
517///
518/// Holds an `OwnedRwLockReadGuard` to keep the lock alive while the caller
519/// accesses the credential. Does NOT implement `Clone` or `Copy` —
520/// credential bytes cannot be duplicated.
521pub struct SecureRef {
522    _guard: OwnedRwLockReadGuard<SecureCredentialStore>,
523    offset: usize,
524    len: usize,
525}
526
527impl Deref for SecureRef {
528    type Target = [u8];
529
530    fn deref(&self) -> &[u8] {
531        // SAFETY: The guard keeps the SecureCredentialStore (and its SecureRegion)
532        // alive. The offset/len were validated when the SecureRef was created.
533        self._guard.region.read(self.offset, self.len)
534    }
535}
536
537impl std::fmt::Debug for SecureRef {
538    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
539        f.write_str("[REDACTED credential ref]")
540    }
541}

784        let mgr_b = CredentialManager::new(16, 4096, false).unwrap();
785
786        mgr_a.store("pt_branch_a_token", b"secret-A").await.unwrap();

790        let result = mgr_b.resolve("pt_branch_a_token").await;

810    // §4.3: Concurrent rotate + resolve race condition test
811    #[tokio::test]
812    async fn test_concurrent_rotate_and_resolve() {
813        let mgr = std::sync::Arc::new(CredentialManager::new(16, 4096, false).unwrap());
814        mgr.store("pt_race", b"initial-value").await.unwrap();
815
816        let mgr_reader = mgr.clone();
817        let mgr_writer = mgr.clone();
818
819        // Spawn concurrent readers and a writer
820        let reader_handle = tokio::spawn(async move {
821            for _ in 0..50 {
822                let r = mgr_reader.resolve("pt_race").await.unwrap();
823                let val = std::str::from_utf8(&r).unwrap().to_string();
824                // Value must be either "initial-value" or "rotated-value" — never corrupt
825                assert!(
826                    val == "initial-value" || val == "rotated-value",
827                    "§4.3: concurrent read during rotate must see consistent value, got: {}",
828                    val
829                );
830                tokio::task::yield_now().await;
831            }
832        });
833
834        let writer_handle = tokio::spawn(async move {
835            for _ in 0..10 {
836                mgr_writer
837                    .rotate("pt_race", b"rotated-value")
838                    .await
839                    .unwrap();
840                tokio::task::yield_now().await;
841            }
842        });
843
844        reader_handle.await.unwrap();
845        writer_handle.await.unwrap();
846
847        // After all rotations, value should be "rotated-value"
848        let final_val = mgr.resolve("pt_race").await.unwrap();
849        assert_eq!(&*final_val, b"rotated-value");
850    }
