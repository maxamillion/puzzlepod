4use maxminddb::{geoip2, Reader};
5use std::net::IpAddr;
6use std::path::Path;
7
8/// GeoIP database reader wrapping MaxMind GeoLite2-Country.
9pub struct GeoIpDatabase {
10    reader: Reader<Vec<u8>>,
11}
12
13impl std::fmt::Debug for GeoIpDatabase {
14    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
15        f.debug_struct("GeoIpDatabase").finish()
16    }
17}
18
19impl GeoIpDatabase {
20    /// Load a MaxMind .mmdb database from disk.
21    pub fn open(path: &Path) -> Result<Self, maxminddb::MaxMindDbError> {
22        let reader = Reader::open_readfile(path)?;
23        Ok(Self { reader })
24    }
25
26    /// Lookup the ISO country code for an IP address.
27    /// Returns None if the IP is not found in the database.
28    pub fn lookup_country(&self, ip: IpAddr) -> Option<String> {
29        let result = self.reader.lookup(ip).ok()?;
30        let country: geoip2::Country = result.decode().ok()??;
31        country.country.iso_code.map(|s: &str| s.to_string())
32    }
33
34    /// Check whether an IP's country is in the allowed regions list.
35    /// Supports ISO 3166-1 alpha-2 codes and region aliases:
36    /// "EU" (all EU member states), "EEA" (EU + IS/LI/NO), "US", "APAC".
37    pub fn is_region_allowed(&self, ip: IpAddr, allowed_regions: &[String]) -> Option<bool> {
38        let country = self.lookup_country(ip)?;
39        Some(allowed_regions.iter().any(|region| {
40            // Always try region alias expansion first (handles "EU", "EEA", "US", "APAC").
41            // Fall back to direct ISO country code comparison for unrecognized aliases.
42            is_country_in_region(&country, region) || region.eq_ignore_ascii_case(&country)
43        }))
44    }
45}
46
47/// EU member state ISO codes.
48const EU_COUNTRIES: &[&str] = &[
49    "AT", "BE", "BG", "HR", "CY", "CZ", "DK", "EE", "FI", "FR", "DE", "GR", "HU", "IE", "IT", "LV",
50    "LT", "LU", "MT", "NL", "PL", "PT", "RO", "SK", "SI", "ES", "SE",
51];
52
53/// EEA = EU + Iceland, Liechtenstein, Norway.
54const EEA_EXTRA: &[&str] = &["IS", "LI", "NO"];
55
56/// APAC region codes (major markets).
57const APAC_COUNTRIES: &[&str] = &[
58    "AU", "BD", "BN", "KH", "CN", "FJ", "HK", "IN", "ID", "JP", "KR", "LA", "MO", "MY", "MV", "MN",
59    "MM", "NP", "NZ", "PK", "PH", "SG", "LK", "TW", "TH", "VN",
60];
61
62/// Check if a country code belongs to a named region alias.
63fn is_country_in_region(country: &str, region: &str) -> bool {
64    let upper = country.to_uppercase();
65    match region.to_uppercase().as_str() {
66        "EU" => EU_COUNTRIES.contains(&upper.as_str()),
67        "EEA" => EU_COUNTRIES.contains(&upper.as_str()) || EEA_EXTRA.contains(&upper.as_str()),
68        "US" => upper == "US",
69        "APAC" => APAC_COUNTRIES.contains(&upper.as_str()),
70        _ => false,
71    }
72}
73
74/// Check if a domain is in the geo exception list.
75pub fn is_geo_exception(domain: &str, exceptions: &[puzzled_types::GeoException]) -> bool {
76    exceptions.iter().any(|exc| {
77        // Check expiry
78        if let Some(ref expires) = exc.expires {
79            if let Ok(exp) = chrono::DateTime::parse_from_rfc3339(expires) {
80                if chrono::Utc::now() > exp {
81                    return false; // Exception expired
82                }
83            }
84        }
85        crate::credentials::domain_matches(domain, &exc.domain)
86    })
87}

89#[cfg(test)]

93    #[test]
94    fn test_eu_region_membership() {
95        assert!(is_country_in_region("DE", "EU"));
96        assert!(is_country_in_region("FR", "EU"));
97        assert!(is_country_in_region("de", "eu")); // case insensitive
98        assert!(!is_country_in_region("US", "EU"));
99        assert!(!is_country_in_region("NO", "EU")); // Norway is EEA, not EU
100    }
101
102    #[test]
103    fn test_eea_region_membership() {
104        assert!(is_country_in_region("DE", "EEA"));
105        assert!(is_country_in_region("NO", "EEA")); // EEA includes Norway
106        assert!(is_country_in_region("IS", "EEA")); // Iceland
107        assert!(!is_country_in_region("US", "EEA"));
108    }
109
110    #[test]
111    fn test_us_region() {
112        assert!(is_country_in_region("US", "US"));
113        assert!(!is_country_in_region("CA", "US"));
114    }

116    #[test]
117    fn test_apac_region() {
118        assert!(is_country_in_region("JP", "APAC"));
119        assert!(is_country_in_region("AU", "APAC"));
120        assert!(!is_country_in_region("US", "APAC"));
121    }

123    #[test]
124    fn test_direct_country_code() {

129    #[test]
130    fn test_geo_exception_matching() {
131        let exceptions = vec![
132            puzzled_types::GeoException {
133                domain: "api.anthropic.com".to_string(),
134                reason: "LLM API".to_string(),
135                approved_by: None,
136                expires: None,
137            },
138            puzzled_types::GeoException {
139                domain: "*.openai.com".to_string(),
140                reason: "LLM API".to_string(),

142                expires: Some("2099-01-01T00:00:00Z".to_string()),
143            },
144        ];
145        assert!(is_geo_exception("api.anthropic.com", &exceptions));
146        assert!(is_geo_exception("api.openai.com", &exceptions));
147        assert!(!is_geo_exception("google.com", &exceptions));
148    }
