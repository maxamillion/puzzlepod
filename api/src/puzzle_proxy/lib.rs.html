2//! HTTP proxy for PuzzlePod network gating.
3//!
4//! Provides application-layer network control for agent sandboxes in Gated mode:
5//! - GET/HEAD requests: forwarded immediately if domain is in allowlist
6//! - POST/PUT/DELETE/PATCH requests: serialized to network journal for replay at commit
7//! - CONNECT tunneling: allowed with domain check (for HTTPS)
8//!
9//! Runs in-process within puzzled (not a separate binary).

33/// Audit events emitted from the proxy handler back to puzzled.
34///
35/// The proxy crate cannot depend on puzzled's AuditEvent type, so this
36/// lightweight enum carries the information needed for puzzled to emit
37/// the corresponding structured audit event.
38#[derive(Debug, Clone)]
39pub enum ProxyAuditEvent {
40    /// §3.3: DLP blocked a request or response (event 2616).
41    DlpBlocked {
42        branch_id: BranchId,
43        rule_name: String,
44        domain: String,
45        match_hash: String,
46    },
47    /// §3.3: DLP detected but allowed (LogAndAllow) (event 2617).
48    DlpDetected {
49        branch_id: BranchId,
50        rule_name: String,
51        domain: String,
52        match_hash: String,
53    },
54    /// §3.3: DLP redacted content (event 2618).
55    DlpRedacted {
56        branch_id: BranchId,
57        rule_name: String,
58        domain: String,
59    },
60    /// §3.3: DLP quarantined branch (event 2619).
61    DlpQuarantine {
62        branch_id: BranchId,
63        rule_name: String,
64        domain: String,
65    },
66    /// §3.4: Credential injected (event 2626).
67    CredentialInjected {
68        branch_id: BranchId,
69        credential_name: String,
70        domain: String,
71    },
72    /// §3.4: Credential denied (event 2627).
73    CredentialDenied {
74        branch_id: BranchId,
75        credential_name: String,
76        domain: String,
77        reason: String,
78    },
79    /// §3.4 G29: Credential resolve failed.
80    CredentialResolveFailed {
81        branch_id: BranchId,
82        credential_name: String,
83        reason: String,
84    },
85    /// §3.4 G29: Real credential value detected in response body.
86    CredentialResponseLeak { branch_id: BranchId, domain: String },
87    /// §3.4 G29: Phantom token stripped from request headers.
88    PhantomTokenStripped {
89        branch_id: BranchId,
90        header_name: String,
91    },
92    /// §3.4 T2.1: Credential resolved — D-Bus signal data for CredentialResolved.
93    CredentialResolved {
94        branch_id: BranchId,
95        credential_name: String,
96        domain: String,
97    },
98    /// §3.4 T2.2: Credential proxy error — D-Bus signal data for CredentialProxyError.
99    CredentialProxyError {
100        branch_id: BranchId,
101        error: String,
102        domain: String,
103    },
104}

106/// Proxy operating mode per PRD network gating specification.
107#[derive(Debug, Clone, Default, PartialEq, Eq)]
108pub enum ProxyMode {
109    /// Block all outbound network requests (403 for everything).
110    Blocked,
111    /// Gated mode: check allow/deny domain lists (default, current behavior).
112    #[default]
113    Gated,
114    /// Monitored mode: log all requests but forward everything.
115    Monitored,
116    /// Unrestricted mode: forward all requests without logging.
117    Unrestricted,
118    /// §3.4 G7: Transparent mode — connections arrive via DNAT, hostname
119    /// extracted from TLS SNI (not CONNECT). Used for credential isolation
120    /// where agents should not know they are going through a proxy.
121    Transparent,
122}
123
124/// Configuration for the proxy server.
125#[derive(Debug, Clone)]
126pub struct ProxyConfig {
127    /// Listen address.
128    pub listen_addr: SocketAddr,
129    /// Allowed domains for read operations (GET/HEAD/OPTIONS).
130    pub read_allowed_domains: Vec<String>,
131    /// Allowed domains for write operations (POST/PUT/DELETE/PATCH).
132    pub write_allowed_domains: Vec<String>,
133    /// Denied domains — checked BEFORE allow lists. Deny overrides allow.
134    pub denied_domains: Vec<String>,
135    /// Proxy operating mode.
136    pub mode: ProxyMode,
137    /// Branch directory for network journal storage.
138    pub branch_dir: PathBuf,
139    /// Branch ID for this proxy instance.
140    pub branch_id: BranchId,
141    /// Optional per-agent CA for TLS MITM interception (C4).
142    /// When `Some`, CONNECT requests use TLS interception instead of opaque tunneling.
143    pub ca: Option<Arc<AgentCa>>,
144    /// §3.3: Optional DLP engine for content inspection on request/response bodies.
145    pub dlp_engine: Option<Arc<DlpEngine>>,
146    /// §3.3: Maximum request body size to inspect (bytes). Bodies larger than this
147    /// are handled per `oversized_body_action`. Default: 10MB.
148    pub max_inspection_body_size: usize,
149    /// §3.3: Action for oversized bodies exceeding `max_inspection_body_size`.
150    pub oversized_body_action: crate::dlp::OversizedAction,
151    /// §3.3/§3.4: Optional channel to signal quarantine (cgroup.freeze) for a branch.
152    /// Sends the branch ID when a Quarantine DLP action is triggered.
153    pub quarantine_sender: Option<tokio::sync::mpsc::Sender<BranchId>>,
154    /// §3.4: Optional phantom token manager for credential injection.
155    pub phantom_token_manager: Option<Arc<RwLock<PhantomTokenManager>>>,
156    /// Agent profile name for credential lookup scoping.
157    pub agent_profile: Option<String>,
158    /// §3.3: Optional GeoIP database for data residency enforcement.
159    pub geo_database: Option<Arc<crate::geo::GeoIpDatabase>>,
160    /// §3.3: Data residency configuration (allowed regions, enforcement mode, exceptions).
161    pub data_residency: Option<puzzled_types::DataResidencyConfig>,
162    /// §3.3/§3.4: Audit event channel for DLP and credential events.
163    /// Handler sends ProxyAuditEvents; puzzled receiver converts to AuditEvents.
164    pub audit_sender: Option<tokio::sync::mpsc::Sender<ProxyAuditEvent>>,
165    /// §3.4: Credential injection mode (Phantom/Passthrough/Blocked).
166    pub credential_mode: puzzled_types::CredentialMode,
167    /// §3.4 G7: Enable transparent proxy mode (DNAT + SNI-based routing).
168    /// When true, the proxy accepts raw TCP connections and extracts the
169    /// upstream hostname from TLS SNI instead of HTTP CONNECT.
170    pub transparent_mode: bool,
171}

173/// Proxy server state.
174pub struct ProxyServer {
175    config: ProxyConfig,
176    journal: Arc<Mutex<replay::NetworkJournal>>,
177    /// H-15: Semaphore limiting concurrent connections to prevent resource exhaustion.
178    connection_semaphore: Arc<tokio::sync::Semaphore>,
179}
180
181impl ProxyServer {
182    /// Create a new proxy server.
183    pub fn new(config: ProxyConfig) -> Self {
184        let journal_dir = config.branch_dir.join("network_journal");
185        let journal = replay::NetworkJournal::new(journal_dir, config.branch_id.clone());
186
187        Self {
188            config,
189            journal: Arc::new(Mutex::new(journal)),
190            connection_semaphore: Arc::new(tokio::sync::Semaphore::new(
191                handler::MAX_CONCURRENT_CONNECTIONS,
192            )),
193        }
194    }
195
196    /// Start the proxy server.
197    ///
198    /// Returns when the server is shut down.
199    pub async fn run(&self) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
200        use hyper::server::conn::http1;
201        use hyper::service::service_fn;
202        use hyper_util::rt::TokioIo;
203
204        let listener = tokio::net::TcpListener::bind(&self.config.listen_addr).await?;
205
206        tracing::info!(
207            addr = %self.config.listen_addr,
208            read_domains = self.config.read_allowed_domains.len(),
209            write_domains = self.config.write_allowed_domains.len(),
210            denied_domains = self.config.denied_domains.len(),
211            mode = ?self.config.mode,
212            "HTTP proxy started"
213        );
214
215        let transparent_mode = self.config.transparent_mode;
216
217        // Construct shared context once — cloned per connection.
218        let base_ctx = handler::ProxyRequestContext {
219            journal: self.journal.clone(),
220            branch_id: self.config.branch_id.clone(),
221            connection_semaphore: self.connection_semaphore.clone(),
222            dlp_engine: self.config.dlp_engine.clone(),
223            max_inspection_body_size: self.config.max_inspection_body_size,
224            oversized_body_action: self.config.oversized_body_action,
225            quarantine_sender: self.config.quarantine_sender.clone(),
226            phantom_token_manager: self.config.phantom_token_manager.clone(),
227            agent_profile: self.config.agent_profile.clone(),
228            audit_sender: self.config.audit_sender.clone(),
229            credential_mode: self.config.credential_mode,
230        };
231
232        loop {
233            let (stream, peer_addr) = listener.accept().await?;
234
235            let ctx = base_ctx.clone();
236            let ca = self.config.ca.clone();
237
238            if transparent_mode {
239                // §3.4 G7: Transparent mode — handle DNAT'd TLS connections.
240                // Connections arrive as raw TCP; SNI is extracted from ClientHello.
241                let ca = match ca {
242                    Some(ca) => ca,
243                    None => {
244                        tracing::error!(
245                            peer = %peer_addr,
246                            "§3.4 G7: transparent mode requires a CA — dropping connection"
247                        );
248                        continue;
249                    }
250                };
251
252                tokio::spawn(async move {
253                    handler::handle_transparent_connection(stream, ca, ctx).await;
254                });
255            } else {
256                // Explicit proxy mode — standard HTTP/1.1 with CONNECT tunneling.
257                let io = TokioIo::new(stream);
258                let read_allowed_domains = self.config.read_allowed_domains.clone();
259                let write_allowed_domains = self.config.write_allowed_domains.clone();
260                let denied_domains = self.config.denied_domains.clone();
261                let mode = self.config.mode.clone();
262                let geo_database = self.config.geo_database.clone();
263                let data_residency = self.config.data_residency.clone();
264
265                tokio::spawn(async move {
266                    let service = service_fn(move |req| {
267                        let read_domains = read_allowed_domains.clone();
268                        let write_domains = write_allowed_domains.clone();
269                        let denied = denied_domains.clone();
270                        let mode = mode.clone();
271                        let ctx = ctx.clone();
272                        let ca = ca.clone();
273                        let geo_db = geo_database.clone();
274                        let data_res = data_residency.clone();
275                        async move {
276                            handler::handle_request(
277                                req,
278                                &read_domains,
279                                &write_domains,
280                                &denied,
281                                &mode,
282                                &ctx,
283                                ca.as_deref(),
284                                geo_db,
285                                data_res,
286                            )
287                            .await
288                        }
289                    });
290
291                    if let Err(e) = http1::Builder::new()
292                        .preserve_header_case(true)
293                        .serve_connection(io, service)
294                        .with_upgrades()
295                        .await
296                    {
297                        tracing::debug!(
298                            peer = %peer_addr,
299                            error = %e,
300                            "proxy connection error"
301                        );
302                    }
303                });
304            }
305        }
306    }
307
308    /// Get the network journal for replay at commit time.
309    pub fn journal(&self) -> Arc<Mutex<replay::NetworkJournal>> {
310        self.journal.clone()
311    }

318    #[test]
319    fn test_proxy_mode_default() {
320        let mode = ProxyMode::default();
321        assert_eq!(mode, ProxyMode::Gated);

324    #[test]
325    fn test_proxy_mode_equality() {
326        assert_eq!(ProxyMode::Blocked, ProxyMode::Blocked);
327        assert_eq!(ProxyMode::Gated, ProxyMode::Gated);
328        assert_eq!(ProxyMode::Monitored, ProxyMode::Monitored);
329        assert_eq!(ProxyMode::Unrestricted, ProxyMode::Unrestricted);
330
331        assert_ne!(ProxyMode::Blocked, ProxyMode::Gated);
332        assert_ne!(ProxyMode::Gated, ProxyMode::Monitored);
333        assert_ne!(ProxyMode::Monitored, ProxyMode::Unrestricted);
334        assert_ne!(ProxyMode::Unrestricted, ProxyMode::Blocked);
335    }

340        let debug_gated = format!("{:?}", ProxyMode::Gated);
341        let debug_monitored = format!("{:?}", ProxyMode::Monitored);
342        let debug_unrestricted = format!("{:?}", ProxyMode::Unrestricted);

345        assert!(debug_gated.contains("Gated"));
