2//! Network request journal and replay engine.
3//!
4//! Side-effect requests (POST/PUT/DELETE/PATCH) from agents in Gated mode
5//! are serialized to disk as a journal. At commit time, after OPA policy
6//! approval, the journal entries are replayed against the upstream servers.
7//!
8//! On rollback, the journal directory is simply discarded (zero residue).
9//!
10//! C11: Crash recovery is supported via a `.replay_progress` state file that
11//! tracks the index of the last successfully replayed entry. On restart,
12//! replay resumes from where it left off.
13//!
14//! M18: Each journal entry write is followed by `sync_data()` to ensure
15//! durability in case of crash.
16
17use std::path::{Path, PathBuf};
18
19use puzzled_types::BranchId;
20use serde::{Deserialize, Serialize};

22/// A journaled network request entry.
23#[derive(Debug, Clone, Serialize, Deserialize)]
24pub struct JournalEntry {
25    /// HTTP method.
26    pub method: String,
27    /// Full URI.
28    pub uri: String,
29    /// HTTP headers.
30    pub headers: Vec<(String, String)>,
31    /// Request body (serialized as base64 in JSON).
32    #[serde(with = "base64_bytes")]
33    pub body: Vec<u8>,
34    /// Timestamp of the original request.
35    pub timestamp: String,
36    /// H-28: When true, this entry is explicitly marked safe for replay even if
37    /// the method is non-idempotent (POST, PUT, DELETE, PATCH). Defaults to false.
38    #[serde(default)]
39    pub safe_replay: bool,
40}
41
42/// PXH3: Token bucket rate limiter for journal writes.
43/// Limits to MAX_ENTRIES_PER_SECOND per branch. When exceeded, the request
44/// is still proxied but the journal entry is skipped (logged as warning).
45struct JournalRateLimiter {
46    /// Timestamps of recent journal writes within the current window.
47    timestamps: Vec<std::time::Instant>,
48}
49
50impl JournalRateLimiter {
51    /// Maximum journal entries per second per branch.
52    const MAX_ENTRIES_PER_SECOND: usize = 100;
53
54    fn new() -> Self {
55        Self {
56            timestamps: Vec::new(),
57        }
58    }
59
60    /// Returns true if the write is allowed, false if rate-limited.
61    fn check(&mut self) -> bool {
62        let now = std::time::Instant::now();
63        let cutoff = now - std::time::Duration::from_secs(1);
64        self.timestamps.retain(|t| *t > cutoff);
65        if self.timestamps.len() >= Self::MAX_ENTRIES_PER_SECOND {
66            return false;
67        }
68        self.timestamps.push(now);
69        true
70    }
71}
72
73/// M-px3: Maximum number of journal entries per branch.
74const MAX_JOURNAL_ENTRIES: usize = 10_000;
75
76/// M-px3: Maximum total journal size in bytes per branch (100 MB).
77const MAX_JOURNAL_BYTES: u64 = 100 * 1024 * 1024;
78
79/// H-28: HTTP methods that are safe to replay (idempotent or read-only).
80/// Non-idempotent methods (POST, DELETE, PUT, PATCH) are skipped during
81/// replay unless explicitly marked safe via the `safe_replay` field.
82// V18: Non-idempotent method replay is intentionally disabled (safe_replay always false).
83// Journal records are retained for audit/forensics; replay only executes GET/HEAD.
84const SAFE_REPLAY_METHODS: &[&str] = &["GET", "HEAD"];

86/// Network request journal for a single branch.
87pub struct NetworkJournal {
88    journal_dir: PathBuf,
89    branch_id: BranchId,
90    entry_count: u64,
91    /// M-px3: Approximate total bytes written to the journal file.
92    journal_bytes: u64,
93    /// PXH3: Rate limiter — max 100 entries/second per branch.
94    rate_limiter: JournalRateLimiter,
95}
96
97impl NetworkJournal {
98    /// Create a new network journal.
99    ///
100    /// If the journal file already exists on disk (e.g., after a daemon restart),
101    /// `entry_count` is recovered by counting lines in the journal file and
102    /// `journal_bytes` from the file size.
103    pub fn new(journal_dir: PathBuf, branch_id: BranchId) -> Self {
104        let entry_count = Self::count_existing_entries(&journal_dir);
105        let journal_bytes = Self::measure_existing_bytes(&journal_dir);
106        Self {
107            journal_dir,
108            branch_id,
109            entry_count,
110            journal_bytes,
111            rate_limiter: JournalRateLimiter::new(),
112        }
113    }
114
115    /// Count existing journal entries by reading lines from the journal file.
116    fn count_existing_entries(journal_dir: &Path) -> u64 {
117        use std::io::BufRead;
118        let journal_file = journal_dir.join("requests.ndjson");
119        match std::fs::File::open(&journal_file) {
120            Ok(file) => {
121                let reader = std::io::BufReader::new(file);
122                reader
123                    .lines()
124                    .filter(|line| line.as_ref().map(|l| !l.trim().is_empty()).unwrap_or(false))
125                    // N6: Use try_from instead of bare `as u64` cast
126                    .count()
127                    .try_into()
128                    .unwrap_or(u64::MAX)
129            }
130            Err(_) => 0,
131        }
132    }
133
134    /// M-px3: Measure the size of the existing journal file in bytes.
135    fn measure_existing_bytes(journal_dir: &Path) -> u64 {
136        let journal_file = journal_dir.join("requests.ndjson");
137        match std::fs::metadata(&journal_file) {
138            Ok(meta) => meta.len(),
139            Err(_) => 0,
140        }
141    }
142
143    /// Append an entry to the journal.
144    ///
145    /// M18: Calls `sync_data()` after writing each entry to ensure durability.
146    /// PXH3: Rate-limited to 100 entries/second per branch. When exceeded,
147    /// the entry is skipped (request is still proxied) and a warning is logged.
148    ///
149    /// CQ-1: The blocking filesystem I/O (file open, write, sync_data) is
150    /// offloaded to `tokio::task::spawn_blocking` to avoid blocking the
151    /// async runtime's executor threads.
152    pub async fn append(&mut self, entry: JournalEntry) -> Result<(), String> {
153        // PXH3: Check rate limit before writing
154        if !self.rate_limiter.check() {
155            tracing::warn!(
156                branch = %self.branch_id,
157                method = %entry.method,
158                uri = %entry.uri,
159                "journal rate limit exceeded (>{}/s) — skipping journal entry (request still proxied)",
160                JournalRateLimiter::MAX_ENTRIES_PER_SECOND,
161            );
162            return Ok(());
163        }
164
165        // M-px3: Check journal entry count limit
166        // Q2: Safe cast — avoid bare `as usize` which silently truncates on 16-bit platforms
167        if usize::try_from(self.entry_count).unwrap_or(usize::MAX) >= MAX_JOURNAL_ENTRIES {
168            tracing::warn!(
169                branch = %self.branch_id,
170                entry_count = self.entry_count,
171                limit = MAX_JOURNAL_ENTRIES,
172                method = %entry.method,
173                uri = %entry.uri,
174                "M-px3: journal entry count limit reached — rejecting new entry"
175            );
176            return Err(format!(
177                "journal entry count limit exceeded ({}/{})",
178                self.entry_count, MAX_JOURNAL_ENTRIES
179            ));
180        }
181
182        // Serialize before spawning blocking task to keep JournalEntry out of Send bounds issues
183        let json = serde_json::to_string(&entry)
184            .map_err(|e| format!("serializing journal entry: {}", e))?;
185
186        // M-px3: Check journal byte size limit (use serialized size as approximation)
187        // Q3: Safe cast — avoid bare `as u64` which could silently truncate on exotic platforms
188        let entry_size = u64::try_from(json.len())
189            .unwrap_or(u64::MAX)
190            .saturating_add(1); // +1 for newline
191        if self.journal_bytes + entry_size > MAX_JOURNAL_BYTES {
192            tracing::warn!(
193                branch = %self.branch_id,
194                journal_bytes = self.journal_bytes,
195                entry_size = entry_size,
196                limit = MAX_JOURNAL_BYTES,
197                method = %entry.method,
198                uri = %entry.uri,
199                "M-px3: journal byte size limit reached — rejecting new entry"
200            );
201            return Err(format!(
202                "journal byte size limit exceeded ({} + {} > {} bytes)",
203                self.journal_bytes, entry_size, MAX_JOURNAL_BYTES
204            ));
205        }
206
207        let journal_dir = self.journal_dir.clone();
208        let journal_file = journal_dir.join("requests.ndjson");
209
210        // CQ-1: Offload blocking filesystem I/O to a blocking thread
211        tokio::task::spawn_blocking(move || {
212            // Create journal directory (idempotent — succeeds if already exists)
213            std::fs::create_dir_all(&journal_dir)
214                .map_err(|e| format!("creating journal dir: {}", e))?;
215
216            use std::io::Write;
217            let mut file = std::fs::OpenOptions::new()
218                .create(true)
219                .append(true)
220                .open(&journal_file)
221                .map_err(|e| format!("opening journal file: {}", e))?;
222
223            writeln!(file, "{}", json).map_err(|e| format!("writing journal entry: {}", e))?;
224
225            // M18: fsync after each entry write to ensure durability
226            file.sync_data()
227                .map_err(|e| format!("syncing journal entry to disk: {}", e))?;
228
229            Ok::<(), String>(())
230        })
231        .await
232        .map_err(|e| format!("journal write task panicked: {}", e))??;
233
234        self.entry_count += 1;
235        self.journal_bytes += entry_size;
236
237        tracing::debug!(
238            branch = %self.branch_id,
239            entry = self.entry_count,
240            journal_bytes = self.journal_bytes,
241            method = %entry.method,
242            uri = %entry.uri,
243            "journal entry written and synced"
244        );
245
246        Ok(())
247    }

249    /// Read all journal entries.
250    pub fn read_all(&self) -> Result<Vec<JournalEntry>, String> {
251        let journal_file = self.journal_dir.join("requests.ndjson");
252        if !journal_file.exists() {
253            return Ok(Vec::new());
254        }
255
256        let contents = std::fs::read_to_string(&journal_file)
257            .map_err(|e| format!("reading journal: {}", e))?;
258
259        let mut entries = Vec::new();
260        for line in contents.lines() {
261            if line.trim().is_empty() {
262                continue;
263            }
264            let entry: JournalEntry =
265                serde_json::from_str(line).map_err(|e| format!("parsing journal entry: {}", e))?;
266            entries.push(entry);
267        }
268
269        Ok(entries)
270    }

272    /// Path to the replay progress state file.
273    fn progress_file_path(&self) -> PathBuf {
274        self.journal_dir.join(".replay_progress")
275    }
276
277    /// Read the last completed replay index from the progress file.
278    ///
279    /// Returns 0 if no progress file exists (start from beginning).
280    fn read_replay_progress(&self) -> u64 {
281        let path = self.progress_file_path();
282        if !path.exists() {
283            return 0;
284        }
285        match std::fs::read_to_string(&path) {
286            Ok(contents) => contents.trim().parse::<u64>().unwrap_or(0),
287            Err(_) => 0,
288        }
289    }
290
291    /// Write the replay progress (index of last successfully replayed entry + 1).
292    ///
293    /// M18: Calls `sync_data()` to ensure the progress is durable.
294    ///
295    /// CQ-1: Blocking filesystem I/O is offloaded to `tokio::task::spawn_blocking`.
296    async fn write_replay_progress(&self, completed_index: u64) -> Result<(), String> {
297        let path = self.progress_file_path();
298        let journal_dir = self.journal_dir.clone();
299
300        tokio::task::spawn_blocking(move || {
301            // Create journal directory (idempotent — succeeds if already exists)
302            std::fs::create_dir_all(&journal_dir)
303                .map_err(|e| format!("creating journal dir for progress: {}", e))?;
304
305            use std::io::Write;
306            let mut file = std::fs::File::create(&path)
307                .map_err(|e| format!("creating progress file: {}", e))?;
308            write!(file, "{}", completed_index).map_err(|e| format!("writing progress: {}", e))?;
309            file.sync_data()
310                .map_err(|e| format!("syncing progress to disk: {}", e))?;
311
312            Ok::<(), String>(())
313        })
314        .await
315        .map_err(|e| format!("progress write task panicked: {}", e))?
316    }

318    /// Replay all journaled requests.
319    ///
320    /// Called at commit time after OPA policy approval.
321    /// Returns the number of entries that were replayed (i.e., attempted replay).
322    /// This count includes entries where the upstream request succeeded as well as
323    /// entries that were already replayed in a previous run (crash recovery).
324    /// It does NOT include entries that were skipped due to domain validation
325    /// failures or entries that failed to replay.
326    ///
327    /// C11: Resumes from the last completed entry on crash recovery.
328    /// Progress is written to `.replay_progress` after each successful replay.
329    ///
330    /// M18: Re-validates each entry's domain against the current allowed_domains list
331    /// and checks for DNS rebinding before replaying. Entries for disallowed domains
332    /// are skipped with a warning.
333    ///
334    // U19: Replay timeout and port validation are deferred to Phase 2 — current replay is for development/debugging only
335    pub async fn replay(&self, allowed_domains: &[String]) -> Result<u64, String> {
336        let entries = self.read_all()?;
337        if entries.is_empty() {
338            return Ok(0);
339        }
340
341        // C11: Read replay progress to resume from where we left off
342        let start_index = self.read_replay_progress();
343
344        // Q4: Safe cast — avoid bare `as u64`
345        if start_index >= u64::try_from(entries.len()).unwrap_or(u64::MAX) {
346            tracing::info!(
347                branch = %self.branch_id,
348                total = entries.len(),
349                already_completed = start_index,
350                "all journal entries already replayed (crash recovery: nothing to do)"
351            );
352            return Ok(start_index);
353        }
354
355        if start_index > 0 {
356            tracing::info!(
357                branch = %self.branch_id,
358                total = entries.len(),
359                resuming_from = start_index,
360                "resuming replay from crash recovery checkpoint"
361            );
362        }
363
364        tracing::info!(
365            branch = %self.branch_id,
366            count = entries.len(),
367            start_index = start_index,
368            "replaying journaled network requests"
369        );
370
371        let mut replayed_count = start_index;
372        // L6: Use try_from to safely handle u64→usize on 32-bit targets
373        for (i, entry) in entries
374            .iter()
375            .enumerate()
376            .skip(usize::try_from(start_index).unwrap_or(usize::MAX))
377        {
378            // H-28: Skip non-idempotent methods unless explicitly marked safe_replay.
379            // POST, DELETE, PUT, PATCH are not safe to blindly replay (they may
380            // duplicate side effects). Only GET and HEAD are replayed by default.
381            let method_upper = entry.method.to_uppercase();
382            if !SAFE_REPLAY_METHODS.contains(&method_upper.as_str()) && !entry.safe_replay {
383                tracing::warn!(
384                    branch = %self.branch_id,
385                    method = %entry.method,
386                    uri = %entry.uri,
387                    index = i,
388                    "H-28: skipping non-idempotent method during replay (not marked safe_replay)"
389                );
390                replayed_count += 1;
391                // Write progress to skip this entry on crash recovery too
392                if let Err(e) = self.write_replay_progress(replayed_count).await {
393                    tracing::warn!(
394                        branch = %self.branch_id,
395                        error = %e,
396                        index = i,
397                        "failed to write replay progress for skipped entry (non-fatal)"
398                    );
399                }
400                continue;
401            }
402
403            match replay_entry(entry, allowed_domains, &self.branch_id).await {
404                Ok(status) => {
405                    tracing::info!(
406                        branch = %self.branch_id,
407                        method = %entry.method,
408                        uri = %entry.uri,
409                        status,
410                        index = i,
411                        "replayed request"
412                    );
413                    replayed_count += 1;
414
415                    // C11: Write progress after each successful replay
416                    if let Err(e) = self.write_replay_progress(replayed_count).await {
417                        tracing::warn!(
418                            branch = %self.branch_id,
419                            error = %e,
420                            index = i,
421                            "failed to write replay progress (non-fatal)"
422                        );
423                    }
424                }
425                Err(e) => {
426                    tracing::error!(
427                        branch = %self.branch_id,
428                        method = %entry.method,
429                        uri = %entry.uri,
430                        error = %e,
431                        index = i,
432                        "failed to replay request"
433                    );
434                }
435            }
436        }
437
438        Ok(replayed_count)
439    }

441    /// Discard the journal (on rollback).
442    ///
443    /// Also removes the replay progress file.
444    pub fn discard(&self) {
445        if self.journal_dir.exists() {
446            // L43: Log error instead of silently ignoring
447            if let Err(e) = std::fs::remove_dir_all(&self.journal_dir) {
448                tracing::warn!(dir = %self.journal_dir.display(), error = %e, "L43: failed to discard journal directory");
449            }
450        }
451    }

459/// Replay a single journal entry by making the actual HTTP request.
460///
461/// M18: Re-validates the entry's domain against the current allowed_domains list
462/// and checks for DNS rebinding before making the request. Returns an error
463/// (skipping the entry) if the domain is no longer allowed.
464async fn replay_entry(
465    entry: &JournalEntry,
466    allowed_domains: &[String],
467    branch_id: &BranchId,
468) -> Result<u16, String> {
469    use bytes::Bytes;
470    use http_body_util::Full;
471    use hyper::Request;
472    use hyper_util::client::legacy::Client;
473    use hyper_util::rt::TokioExecutor;
474
475    // M18: Re-validate domain before replaying
476    // Parse the host from the stored URI
477    let uri: hyper::Uri = entry
478        .uri
479        .parse()
480        .map_err(|e| format!("invalid URI '{}': {}", entry.uri, e))?;
481
482    let host = uri
483        .host()
484        .ok_or_else(|| format!("no host in URI '{}'", entry.uri))?;
485
486    // Check domain allowlist
487    if !crate::handler::is_domain_allowed(host, allowed_domains) {
488        tracing::warn!(
489            branch = %branch_id,
490            uri = %entry.uri,
491            host = %host,
492            "replay: skipping entry — domain no longer in allowlist"
493        );
494        return Err(format!(
495            "domain '{}' no longer in allowlist, skipping replay",
496            host
497        ));
498    }
499
500    // Check for DNS rebinding (SSRF protection)
501    if let Err(_resp) = crate::handler::check_dns_rebinding(host, branch_id).await {
502        tracing::warn!(
503            branch = %branch_id,
504            uri = %entry.uri,
505            host = %host,
506            "replay: skipping entry — DNS rebinding check failed"
507        );
508        return Err(format!(
509            "DNS rebinding check failed for '{}', skipping replay",
510            host
511        ));
512    }
513
514    tracing::debug!(
515        method = %entry.method,
516        uri = %entry.uri,
517        body_len = entry.body.len(),
518        "replaying request"
519    );
520
521    let scheme = uri.scheme_str().unwrap_or("http");
522
523    // Build an appropriate client based on the URI scheme.
524    // HTTPS entries require a TLS-capable connector; HTTP entries use plain TCP.
525    let is_https = scheme.eq_ignore_ascii_case("https");
526
527    let method: hyper::Method = entry
528        .method
529        .parse()
530        .map_err(|e| format!("invalid method '{}': {}", entry.method, e))?;
531
532    let mut req_builder = Request::builder().method(method).uri(&entry.uri);
533
534    // Q9: No credential stripping needed here — journal entries have credentials redacted
535    // at write time (handler.rs N10/D-I8 redaction), so the replay path inherits that
536    // protection. Headers stored in the journal never contain real credential values.
537    for (name, value) in &entry.headers {
538        req_builder = req_builder.header(name.as_str(), value.as_str());
539    }
540
541    let req = req_builder
542        .body(Full::new(Bytes::from(entry.body.clone())))
543        .map_err(|e| format!("building request: {}", e))?;
544
545    if is_https {
546        // Build an HTTPS-capable client using hyper-rustls with webpki root certs
547        let https_connector = hyper_rustls::HttpsConnectorBuilder::new()
548            .with_webpki_roots()
549            .https_only()
550            .enable_http1()
551            .build();
552        let client: Client<_, Full<Bytes>> =
553            Client::builder(TokioExecutor::new()).build(https_connector);
554        match client.request(req).await {
555            Ok(resp) => Ok(resp.status().as_u16()),
556            Err(e) => Err(format!("request to {} failed: {}", entry.uri, e)),
557        }
558    } else {
559        let client: Client<_, Full<Bytes>> = Client::builder(TokioExecutor::new()).build_http();
560        match client.request(req).await {
561            Ok(resp) => Ok(resp.status().as_u16()),
562            Err(e) => Err(format!("request to {} failed: {}", entry.uri, e)),
563        }
564    }
565}

567/// Base64 serialization for binary body data in JSON.
568mod base64_bytes {
569    use base64::Engine;
570    use serde::{Deserialize, Deserializer, Serializer};
571
572    pub fn serialize<S>(bytes: &[u8], serializer: S) -> Result<S::Ok, S::Error>
573    where
574        S: Serializer,
575    {
576        let encoded = base64::engine::general_purpose::STANDARD.encode(bytes);
577        serializer.serialize_str(&encoded)
578    }
579
580    pub fn deserialize<'de, D>(deserializer: D) -> Result<Vec<u8>, D::Error>
581    where
582        D: Deserializer<'de>,
583    {
584        let s = String::deserialize(deserializer)?;
585        base64::engine::general_purpose::STANDARD
586            .decode(&s)
587            .map_err(serde::de::Error::custom)
588    }
589}
