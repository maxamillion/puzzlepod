21/// Handle the intercepted (decrypted) HTTP stream between agent and upstream.
22///
23/// Reads a single HTTP/1.1 request from the agent side, routes it based on method,
24/// and relays the response back through the TLS streams.
25pub(super) async fn handle_intercepted_stream<A, U>(
26    mut agent_tls: A,
27    mut upstream_tls: U,
28    domain: &str,
29    _target_with_port: &str,
30    ctx: &ProxyRequestContext,
31) -> Result<(), String>
32where
33    A: tokio::io::AsyncRead + tokio::io::AsyncWrite + Unpin,
34    U: tokio::io::AsyncRead + tokio::io::AsyncWrite + Unpin,
35{
36    let branch_id = &ctx.branch_id;
37    let journal = ctx.journal.clone();
38    let dlp_engine = &ctx.dlp_engine;
39    let max_inspection_body_size = ctx.max_inspection_body_size;
40    let oversized_body_action = ctx.oversized_body_action;
41    let quarantine_sender = &ctx.quarantine_sender;
42    let audit_sender = &ctx.audit_sender;
43    let phantom_token_manager = &ctx.phantom_token_manager;
44    let agent_profile = &ctx.agent_profile;
45    let credential_mode = ctx.credential_mode;
46
47    // V4: 4KB buffer is a tradeoff — larger buffers waste stack space per connection,
48    // smaller buffers increase syscall overhead. Slowloris is mitigated by TLS_SESSION_TIMEOUT.
49    let mut tmp_buf = [0u8; 4096];
50    // Track leftover bytes read beyond the current request (pipelining / keep-alive)
51    let mut leftover: Vec<u8> = Vec::new();
52
53    // D-I6: Session-level timeout to prevent indefinite keep-alive loops.
54    let session_start = std::time::Instant::now();
55
56    // HTTP/1.1 keep-alive loop: process multiple requests on the same TLS connection.
57    // Break on: connection close (read returns 0), Connection: close header, or I/O error.
58    loop {
59        // D-I6: Check session timeout at the start of each iteration.
60        if session_start.elapsed() > TLS_SESSION_TIMEOUT {
61            tracing::info!(
62                branch = %branch_id,
63                domain = %domain,
64                elapsed_secs = session_start.elapsed().as_secs(),
65                "D-I6: TLS session timeout exceeded, closing keep-alive connection"
66            );
67            break;
68        }
69        // Read the HTTP request from the agent (decrypted)
70        // We read into a buffer and parse the HTTP/1.1 request manually
71        let mut request_buf = Vec::with_capacity(8192);
72
73        // Prepend any leftover bytes from the previous iteration
74        if !leftover.is_empty() {
75            request_buf.extend_from_slice(&leftover);
76            leftover.clear();
77        }
78
79        // Read until we have a complete HTTP request (headers end with \r\n\r\n)
80        let header_end;
81        loop {
82            // Check if we already have a complete header set (from leftover data)
83            if let Some(pos) = find_header_end(&request_buf) {
84                header_end = pos;
85                break;
86            }
87
88            let n = match agent_tls.read(&mut tmp_buf).await {
89                Ok(0) => {
90                    // Connection closed cleanly by the agent — not an error.
91                    tracing::debug!(
92                        branch = %branch_id,
93                        domain = %domain,
94                        "C4: agent closed TLS connection (keep-alive end)"
95                    );
96                    return Ok(());
97                }
98                Ok(n) => n,
99                Err(e) => {
100                    tracing::debug!(
101                        branch = %branch_id,
102                        error = %e,
103                        "C4: I/O error reading from agent, closing intercepted connection"
104                    );
105                    return Err(format!("reading from agent: {}", e));
106                }
107            };
108            request_buf.extend_from_slice(&tmp_buf[..n]);
109
110            // D-M1: Use MAX_HEADER_SIZE (64KB) instead of MAX_BODY_SIZE (100MB)
111            // to prevent excessive memory consumption from oversized headers.
112            if request_buf.len() > MAX_HEADER_SIZE {
113                return Err("request headers too large".to_string());
114            }
115        }
116
117        // Parse the request line and headers
118        let header_bytes = &request_buf[..header_end];
119        let header_str = std::str::from_utf8(header_bytes)
120            .map_err(|e| format!("invalid UTF-8 in headers: {}", e))?;
121
122        let mut lines = header_str.split("\r\n");
123        let request_line = lines.next().ok_or("empty request")?;
124        let mut parts = request_line.split_whitespace();
125        let method_str = parts.next().ok_or("no method in request line")?;
126        // D-C1: path must be mutable for QueryParameter credential injection.
127        let mut path = parts.next().ok_or("no path in request line")?.to_string();
128
129        // Parse headers
130        let mut headers: Vec<(String, String)> = Vec::new();
131        let mut content_length: usize = 0;
132        let mut content_length_seen = false;
133        let mut connection_close = false;
134        let mut has_transfer_encoding = false;
135        let mut has_content_encoding = false;
136        for line in lines {
137            if line.is_empty() {
138                break;
139            }
140            if let Some((name, value)) = line.split_once(':') {
141                let name = name.trim().to_string();
142                let value = value.trim().to_string();
143
144                // D-C3: Validate header name and value per RFC 9110.
145                // Skip headers with invalid characters to prevent injection attacks.
146                if !is_valid_http_header_name(&name) {
147                    tracing::warn!(
148                        branch = %branch_id,
149                        header_name = %name,
150                        "D-C3: skipping header with invalid name characters"
151                    );
152                    continue;
153                }
154                if !is_valid_http_header_value(&value) {
155                    tracing::warn!(
156                        branch = %branch_id,
157                        header_name = %name,
158                        "D-C3: skipping header with invalid value characters (null/control bytes)"
159                    );
160                    continue;
161                }
162
163                if name.eq_ignore_ascii_case("content-length") {
164                    // V5: Reject malformed Content-Length — unwrap_or(0) could enable request smuggling
165                    let new_cl: usize = match value.parse() {
166                        Ok(cl) => cl,
167                        Err(_) => {
168                            tracing::warn!(
169                                branch = %branch_id,
170                                raw_value = %value,
171                                "V5: rejecting request with malformed Content-Length"
172                            );
173                            let error_body = "Malformed Content-Length header\n";
174                            let error_response = format!(
175                                "HTTP/1.1 400 Bad Request\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
176                                error_body.len(), error_body
177                            );
178                            if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
179                                tracing::debug!(error = %e, "V5: failed to write TLS error response to agent");
180                            }
181                            if let Err(e) = agent_tls.flush().await {
182                                tracing::debug!(error = %e, "V5: failed to flush TLS error response to agent");
183                            }
184                            return Ok(());
185                        }
186                    };
187                    // D-C2: RFC 9112 §8.6 — reject requests with conflicting Content-Length
188                    // values to prevent request smuggling via CL desync.
189                    if content_length_seen && new_cl != content_length {
190                        tracing::warn!(
191                            branch = %branch_id,
192                            first_cl = content_length,
193                            second_cl = new_cl,
194                            "D-C2: rejecting request with conflicting Content-Length headers"
195                        );
196                        let error_body = "Conflicting Content-Length headers\n";
197                        let error_response = format!(
198                            "HTTP/1.1 400 Bad Request\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
199                            error_body.len(), error_body
200                        );
201                        if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
202                            tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
203                        }
204                        if let Err(e) = agent_tls.flush().await {
205                            tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
206                        }
207                        return Ok(());
208                    }
209                    content_length = new_cl;
210                    content_length_seen = true;
211                }
212                if name.eq_ignore_ascii_case("connection") && value.eq_ignore_ascii_case("close") {
213                    connection_close = true;
214                }
215                // C7: Detect Transfer-Encoding header (especially chunked)
216                if name.eq_ignore_ascii_case("transfer-encoding") {
217                    has_transfer_encoding = true;
218                }
219                // D-C1: Detect Content-Encoding header (gzip/deflate/br/zstd)
220                if name.eq_ignore_ascii_case("content-encoding") {
221                    has_content_encoding = true;
222                }
223                headers.push((name, value));
224            }
225        }
226
227        // D-C1: Reject requests with Content-Encoding when DLP is active.
228        // Compressed bodies bypass DLP pattern matching.
229        if has_content_encoding && dlp_engine.is_some() {
230            tracing::warn!(
231                branch = %branch_id,
232                domain = %domain,
233                method = %method_str,
234                "D-C1: rejecting request with Content-Encoding in TLS intercept (DLP bypass prevention)"
235            );
236            let body = "Content-Encoding blocked: DLP inspection is active\n";
237            let error_response = format!(
238                "HTTP/1.1 403 Forbidden\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
239                body.len(),
240                body
241            );
242            if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
243                tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
244            }
245            if let Err(e) = agent_tls.flush().await {
246                tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
247            }
248            return Ok(());
249        }
250
251        // C7: Reject requests with Transfer-Encoding header to prevent request
252        // smuggling. The TLS MITM parser does not implement chunked decoding,
253        // so allowing it would let an attacker smuggle additional requests.
254        if has_transfer_encoding {
255            tracing::warn!(
256                branch = %branch_id,
257                domain = %domain,
258                method = %method_str,
259                "C7: rejecting request with Transfer-Encoding (request smuggling prevention)"
260            );
261            let body = "Transfer-Encoding is not supported by this proxy\n";
262            let error_response = format!(
263                "HTTP/1.1 501 Not Implemented\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
264                body.len(),
265                body
266            );
267            if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
268                tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
269            }
270            if let Err(e) = agent_tls.flush().await {
271                tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
272            }
273            return Ok(());
274        }
275
276        // Read the body if Content-Length indicates one
277        // We may have already read some body bytes beyond the header end
278        let body_start = header_end + 4; // skip \r\n\r\n
279        let mut body = Vec::new();
280        if content_length > 0 {
281            if content_length > MAX_BODY_SIZE {
282                return Err(format!(
283                    "request body too large ({} bytes, max {})",
284                    content_length, MAX_BODY_SIZE
285                ));
286            }
287            // Copy any body bytes already in the buffer
288            if body_start < request_buf.len() {
289                body.extend_from_slice(&request_buf[body_start..]);
290            }
291            // Read remaining body bytes
292            while body.len() < content_length {
293                let n = agent_tls
294                    .read(&mut tmp_buf)
295                    .await
296                    .map_err(|e| format!("reading request body from agent: {}", e))?;
297                if n == 0 {
298                    break;
299                }
300                body.extend_from_slice(&tmp_buf[..n]);
301            }
302            // If we read more than content_length, stash the overflow as leftover
303            if body.len() > content_length {
304                leftover.extend_from_slice(&body[content_length..]);
305                body.truncate(content_length);
306            }
307        } else if body_start < request_buf.len() {
308            // No body expected but there are bytes after headers — leftover for next request
309            leftover.extend_from_slice(&request_buf[body_start..]);
310        }
311
312        // T18: full_uri is intentionally constructed BEFORE credential injection.
313        // This ensures credentials injected via QueryParameter do not leak into
314        // logs or journal entries. Do not move this after injection.
315        let full_uri = format!("https://{}{}", domain, &path);
316
317        // U18: Capture the original path BEFORE credential injection for the T14
318        // exfiltration check below. QueryParameter injection modifies `path` by
319        // appending credentials to the query string, so checking `path` after
320        // injection would false-positive on the proxy's own injected credential.
321        let original_path = path.clone();
322
323        tracing::info!(
324            branch = %branch_id,
325            method = %method_str,
326            uri = %full_uri,
327            body_len = body.len(),
328            "C4: intercepted HTTP request"
329        );
330
331        // §3.4/C4: Credential injection on TLS-intercepted requests.
332        // D-I3: Enforce Blocked credential mode — strip all auth headers.
333        let mut injected_credential_value: Option<zeroize::Zeroizing<String>> = None;
334        // N10: Track custom header name from credential injection for redaction in journal.
335        let mut injected_custom_header_name: Option<String> = None;
336        // N5: Track whether credential was injected as query parameter for URI redaction.
337        let mut injected_via_query_param = false;
338        if credential_mode == puzzled_types::CredentialMode::Blocked {
339            let before_len = headers.len();
340            headers.retain(|(name, _)| {
341                let lower = name.to_lowercase();
342                lower != "authorization" && lower != "proxy-authorization" && lower != "x-api-key"
343            });
344            if headers.len() < before_len {
345                tracing::warn!(
346                    branch = %branch_id,
347                    stripped = before_len - headers.len(),
348                    "D-I3/§3.4: credential mode=Blocked, stripped auth headers in TLS intercept path"
349                );
350            }
351        } else if credential_mode == puzzled_types::CredentialMode::Phantom {
352            if let Some(ref ptm) = phantom_token_manager {
353                let mgr = ptm.read().await;
354                // Check each header for phantom tokens, stripping auth scheme prefixes
355                let mut inject_header_idx: Option<usize> = None;
356                let mut inject_token: Option<String> = None;
357                // Q5: Removed dead `_inject_auth_prefix` variable (was unused after D-C4 refactor)
358                for (idx, (_name, value)) in headers.iter().enumerate() {
359                    // D-C4: Strip "Bearer " or "Basic " prefix before checking,
360                    // matching the non-TLS path in inject_credentials().
361                    // H61: Use case-insensitive matching for auth scheme prefixes
362                    let (_prefix, token_part) =
363                        if value.len() >= 7 && value[..7].eq_ignore_ascii_case("Bearer ") {
364                            (Some("Bearer "), &value[7..])
365                        } else if value.len() >= 6 && value[..6].eq_ignore_ascii_case("Basic ") {
366                            (Some("Basic "), &value[6..])
367                        } else {
368                            (None, value.as_str())
369                        };
370                    if mgr.is_phantom_token(token_part) {
371                        inject_header_idx = Some(idx);
372                        inject_token = Some(token_part.to_string());
373                        break;
374                    }
375                }
376                if let Some((idx, token)) = inject_header_idx.zip(inject_token) {
377                    if let Some(resolved) = mgr.resolve(&token, Some(branch_id)).await {
378                        // Domain scope check
379                        let domain_ok = resolved
380                            .target_domains
381                            .iter()
382                            .any(|d| crate::credentials::domain_matches(domain, d));
383                        if domain_ok {
384                            // Profile check
385                            let profile_ok = match &agent_profile {
386                                Some(prof) => resolved
387                                    .allowed_profiles
388                                    .iter()
389                                    .any(|p| p == "*" || p == prof),
390                                None => true,
391                            };
392                            if profile_ok {
393                                // D-C1: Use helper that dispatches on InjectionMethod
394                                injected_credential_value = inject_credential_into_header_vec(
395                                    &mut headers,
396                                    &mut path,
397                                    &resolved,
398                                    idx,
399                                );
400                                // N10: Capture custom header name for journal redaction.
401                                if let InjectionMethod::CustomHeader { ref header_name } =
402                                    resolved.injection
403                                {
404                                    injected_custom_header_name = Some(header_name.clone());
405                                }
406                                // N5: Track query parameter injection for URI redaction.
407                                if matches!(
408                                    resolved.injection,
409                                    InjectionMethod::QueryParameter { .. }
410                                ) {
411                                    injected_via_query_param = true;
412                                }
413                                tracing::info!(
414                                    branch = %branch_id,
415                                    credential = %resolved.credential_name,
416                                    "§3.4/C4: credential injected in TLS intercept path"
417                                );
418                                if let Some(ref sender) = audit_sender {
419                                    send_audit(
420                                        sender,
421                                        crate::ProxyAuditEvent::CredentialInjected {
422                                            branch_id: branch_id.clone(),
423                                            credential_name: resolved.credential_name.clone(),
424                                            domain: domain.to_string(),
425                                        },
426                                    );
427                                    // §3.4 T2.1: Emit CredentialResolved for D-Bus signal
428                                    send_audit(
429                                        sender,
430                                        crate::ProxyAuditEvent::CredentialResolved {
431                                            branch_id: branch_id.clone(),
432                                            credential_name: resolved.credential_name.clone(),
433                                            domain: domain.to_string(),
434                                        },
435                                    );
436                                }
437                            } else {
438                                tracing::warn!(
439                                    branch = %branch_id,
440                                    credential = %resolved.credential_name,
441                                    "§3.4/C4: credential not authorized for this profile"
442                                );
443                                let error_body = "Credential not authorized for this profile\n";
444                                let error_response = format!(
445                                    "HTTP/1.1 403 Forbidden\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
446                                    error_body.len(), error_body
447                                );
448                                if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await
449                                {
450                                    tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
451                                }
452                                if let Err(e) = agent_tls.flush().await {
453                                    tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
454                                }
455                                return Ok(());
456                            }
457                        } else {
458                            tracing::warn!(
459                                branch = %branch_id,
460                                credential = %resolved.credential_name,
461                                "§3.4/C4: credential not authorized for domain '{}'", domain
462                            );
463                            let error_body = "Credential not authorized for this domain\n";
464                            let error_response = format!(
465                                "HTTP/1.1 403 Forbidden\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
466                                error_body.len(), error_body
467                            );
468                            if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
469                                tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
470                            }
471                            if let Err(e) = agent_tls.flush().await {
472                                tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
473                            }
474                            return Ok(());
475                        }
476                    } else {
477                        tracing::warn!(
478                            branch = %branch_id,
479                            "§3.4/C4: invalid or expired phantom token in TLS intercept"
480                        );
481                        let error_body = "Invalid or expired phantom token\n";
482                        let error_response = format!(
483                            "HTTP/1.1 401 Unauthorized\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
484                            error_body.len(), error_body
485                        );
486                        if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
487                            tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
488                        }
489                        if let Err(e) = agent_tls.flush().await {
490                            tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
491                        }
492                        return Ok(());
493                    }
494                }
495
496                // D-C2: Strip any remaining phantom tokens from other headers.
497                // If the agent sent phantom tokens in multiple headers (e.g., both
498                // Authorization and X-Api-Key), only one was resolved above. Any
499                // unresolved phantom tokens must be removed to prevent leaking the
500                // phantom token prefix pattern to upstream.
501                let resolved_idx = inject_header_idx; // the index we already resolved
502                let mut indices_to_remove: Vec<usize> = Vec::new();
503                for (idx, (_name, value)) in headers.iter().enumerate() {
504                    if Some(idx) == resolved_idx {
505                        continue; // already handled
506                    }
507                    // J43: Case-insensitive prefix matching per RFC 7235 §2.1
508                    let token_part =
509                        if value.len() >= 7 && value[..7].eq_ignore_ascii_case("bearer ") {
510                            &value[7..]
511                        } else if value.len() >= 6 && value[..6].eq_ignore_ascii_case("basic ") {
512                            &value[6..]
513                        } else {
514                            value.as_str()
515                        };
516                    if mgr.is_phantom_token(token_part) {
517                        tracing::warn!(
518                            branch = %branch_id,
519                            header_name = %_name,
520                            "D-C2: stripping additional phantom token from header (not resolved, would leak to upstream)"
521                        );
522                        indices_to_remove.push(idx);
523                    }
524                }
525                // Remove in reverse order to preserve indices
526                for idx in indices_to_remove.into_iter().rev() {
527                    headers.remove(idx);
528                }
529
530                // D-I2/M-4: If no phantom token was found, strip all credential-bearing
531                // headers (defense-in-depth). The agent should only use phantom tokens;
532                // any other auth header is fabricated or leaked. PRD §3.4.8 Step 5.
533                if inject_header_idx.is_none() {
534                    let before_len = headers.len();
535                    headers.retain(|(name, _)| {
536                        !name.eq_ignore_ascii_case("authorization")
537                            && !name.eq_ignore_ascii_case("x-api-key")
538                            && !name.eq_ignore_ascii_case("proxy-authorization")
539                    });
540                    if headers.len() < before_len {
541                        tracing::warn!(
542                            branch = %branch_id,
543                            "D-I2/M-4/§3.4: stripped non-phantom credential headers in TLS intercept path"
544                        );
545                    }
546                }
547            }
548        }
549
550        // T14: URI credential exfiltration check in TLS intercept path.
551        // Mirrors the G23 check in the non-TLS path — blocks agents from
552        // embedding credentials in request path or query parameters.
553        // U18: Check original_path (captured before injection) to avoid false-positive
554        // when QueryParameter injection appended the credential to the query string.
555        if let Some(ref cred_val) = injected_credential_value {
556            if original_path.contains(cred_val.as_str()) {
557                tracing::warn!(
558                    branch = %branch_id,
559                    "T14/§3.4: real credential value found in request URI after injection — blocking"
560                );
561                if let Some(ref sender) = audit_sender {
562                    send_audit(
563                        sender,
564                        crate::ProxyAuditEvent::CredentialDenied {
565                            branch_id: branch_id.clone(),
566                            credential_name: "unknown".to_string(),
567                            domain: domain.to_string(),
568                            reason: "credential_exfiltration_in_uri".to_string(),
569                        },
570                    );
571                }
572                let error_body = "Credential value detected in request URI — blocked\n";
573                let error_response = format!(
574                    "HTTP/1.1 403 Forbidden\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
575                    error_body.len(),
576                    error_body
577                );
578                if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
579                    tracing::debug!(error = %e, "failed to write TLS error response");
580                }
581                if let Err(e) = agent_tls.flush().await {
582                    tracing::debug!(error = %e, "failed to flush TLS error response");
583                }
584                return Ok(());
585            }
586        }
587
588        // D-I4: Credential body exfiltration check in TLS intercept path.
589        // If a credential was injected, verify the request body does NOT contain it.
590        // D-I2: Uses byte-string search instead of from_utf8 to detect credentials
591        // in non-UTF-8 bodies (binary payloads with embedded credential bytes).
592        if let Some(ref cred_val) = injected_credential_value {
593            let cred_bytes = cred_val.as_bytes();
594            if !body.is_empty()
595                && !cred_bytes.is_empty()
596                && body.windows(cred_bytes.len()).any(|w| w == cred_bytes)
597            {
598                tracing::warn!(
599                    branch = %branch_id,
600                    "D-I4/§3.4: real credential value found in request body after injection — blocking"
601                );
602                if let Some(ref sender) = audit_sender {
603                    send_audit(
604                        sender,
605                        crate::ProxyAuditEvent::CredentialDenied {
606                            branch_id: branch_id.clone(),
607                            credential_name: "unknown".to_string(),
608                            domain: domain.to_string(),
609                            reason: "credential_exfiltration_in_body".to_string(),
610                        },
611                    );
612                }
613                let error_body = "Credential value detected in request body — blocked\n";
614                let error_response = format!(
615                    "HTTP/1.1 403 Forbidden\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
616                    error_body.len(),
617                    error_body
618                );
619                if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
620                    tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
621                }
622                if let Err(e) = agent_tls.flush().await {
623                    tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
624                }
625                return Ok(());
626            }
627        }
628
629        // Route based on method
630        let method_upper = method_str.to_uppercase();
631        let is_side_effect = matches!(method_upper.as_str(), "POST" | "PUT" | "DELETE" | "PATCH");
632
633        // §3.3: DLP inspection on intercepted side-effect request bodies
634        // D-C3: Compressed request bodies are blocked by D-C1 (Content-Encoding
635        // rejection) when DLP is active, so only uncompressed bodies reach here.
636        // Compressed *response* bodies are decompressed before DLP/credential
637        // scanning below (see resp_content_encoding handling).
638        if is_side_effect {
639            if let Some(ref dlp) = dlp_engine {
640                if body.len() > max_inspection_body_size {
641                    match oversized_body_action {
642                        crate::dlp::OversizedAction::BlockAndAlert => {
643                            tracing::warn!(
644                                branch = %branch_id,
645                                body_size = body.len(),
646                                limit = max_inspection_body_size,
647                                "§3.3/C4: oversized body blocked in TLS intercept (fail closed)"
648                            );
649                            let error_body = format!(
650                                "Request body too large for DLP inspection (max {} bytes)\n",
651                                max_inspection_body_size
652                            );
653                            let error_response = format!(
654                                "HTTP/1.1 413 Payload Too Large\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
655                                error_body.len(),
656                                error_body
657                            );
658                            if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
659                                tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
660                            }
661                            if let Err(e) = agent_tls.flush().await {
662                                tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
663                            }
664                            return Ok(());
665                        }
666                        crate::dlp::OversizedAction::AllowAndLog => {
667                            tracing::warn!(
668                                branch = %branch_id,
669                                body_size = body.len(),
670                                limit = max_inspection_body_size,
671                                "§3.3/C4: oversized body allowed without DLP inspection in TLS intercept (fail open)"
672                            );
673                            // Skip DLP inspection, continue to journal
674                        }
675                    }
676                } else {
677                    match inspect_dlp_body(
678                        &body,
679                        dlp,
680                        branch_id,
681                        quarantine_sender.as_ref(),
682                        audit_sender.as_ref(),
683                        domain,
684                    ) {
685                        Ok(Some(redacted)) => {
686                            body = redacted;
687                        }
688                        Ok(None) => {
689                            // No redaction needed, continue with original body
690                        }
691                        Err(resp) => {
692                            // L21: DLP blocked the request — use a fixed-size error
693                            // message instead of collecting the (potentially large)
694                            // response body unboundedly.
695                            let status = resp.status();
696                            let dlp_err_body = "DLP blocked request\n";
697                            let error_response = format!(
698                                "HTTP/1.1 {} {}\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
699                                status.as_u16(),
700                                status.canonical_reason().unwrap_or("Forbidden"),
701                                dlp_err_body.len(),
702                                dlp_err_body,
703                            );
704                            if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
705                                tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
706                            }
707                            if let Err(e) = agent_tls.flush().await {
708                                tracing::debug!(error = %e, "L40: failed to flush TLS error response to agent");
709                            }
710                            return Ok(());
711                        }
712                    }
713                }
714            }
715        }
716
717        if is_side_effect {
718            // D-I8: Redact credential headers before journaling to avoid writing
719            // real credentials to disk in plaintext. Create a redacted copy of headers
720            // for the journal entry while preserving original headers for the upstream request.
721            // N10: Also redact custom header names from credential injection.
722            let redacted_headers: Vec<(String, String)> = headers
723                .iter()
724                .map(|(name, value)| {
725                    let lower_name = name.to_lowercase();
726                    if lower_name == "authorization"
727                        || lower_name == "x-api-key"
728                        || lower_name == "proxy-authorization"
729                        || injected_custom_header_name
730                            .as_ref()
731                            .is_some_and(|h| h.to_lowercase() == lower_name)
732                    {
733                        (name.clone(), "[REDACTED]".to_string())
734                    } else {
735                        (name.clone(), value.clone())
736                    }
737                })
738                .collect();
739
740            // N5: Redact query string from journal URI when credential was injected
741            // as a query parameter, to avoid writing the credential to disk.
742            let journal_uri = if injected_via_query_param {
743                if let Some(qpos) = full_uri.find('?') {
744                    format!("{}?REDACTED", &full_uri[..qpos])
745                } else {
746                    full_uri.clone()
747                }
748            } else {
749                full_uri.clone()
750            };
751
752            // Journal the request for replay at commit time
753            let entry = crate::replay::JournalEntry {
754                method: method_str.to_string(),
755                uri: journal_uri,
756                headers: redacted_headers,
757                body: body.clone(),
758                timestamp: chrono_now(),
759                safe_replay: false,
760            };
761
762            let mut journal_guard = journal.lock().await;
763            if let Err(e) = journal_guard.append(entry).await {
764                tracing::error!(
765                    branch = %branch_id,
766                    error = %e,
767                    "C4: failed to journal intercepted request"
768                );
769                // C8: Send a 502 response back to the agent with correct Content-Length
770                let error_body = "Journal write failed\r\n";
771                let error_response = format!(
772                    "HTTP/1.1 502 Bad Gateway\r\nContent-Length: {}\r\n\r\n{}",
773                    error_body.len(),
774                    error_body
775                );
776                if let Err(e) = agent_tls.write_all(error_response.as_bytes()).await {
777                    tracing::debug!(error = %e, "L40: failed to write TLS error response to agent");
778                }
779                return Err(format!("journal write failed: {}", e));
780            }
781            drop(journal_guard);
782
783            tracing::info!(
784                branch = %branch_id,
785                method = %method_str,
786                uri = %full_uri,
787                "C4: side-effect request journaled"
788            );
789
790            // Return synthetic 200 OK — do NOT forward side-effect to upstream.
791            // Side-effects are replayed at commit time, not executed immediately.
792            // C8: Content-Length must match actual body size exactly.
793            let synthetic_body = "Request journaled; will be replayed at commit\n";
794            let synthetic_response = format!(
795                "HTTP/1.1 200 OK\r\nContent-Length: {}\r\nX-PuzzlePod-Journaled: true\r\n\r\n{}",
796                synthetic_body.len(),
797                synthetic_body
798            );
799            agent_tls
800                .write_all(synthetic_response.as_bytes())
801                .await
802                .map_err(|e| format!("writing synthetic response to agent: {}", e))?;
803            agent_tls
804                .flush()
805                .await
806                .map_err(|e| format!("flushing synthetic response: {}", e))?;
807
808            tracing::info!(
809                branch = %branch_id,
810                method = %method_str,
811                uri = %full_uri,
812                "C4: side-effect request journaled — synthetic 200 OK returned (not forwarded to upstream)"
813            );
814        } else {
815            // Forward the raw request bytes to upstream
816            // H65: Validate path and method contain no CR/LF to prevent header injection
817            // V17: Header value CR/LF validation is handled by D-C3 (is_valid_http_header_value) — defense is split but complete
818            if method_str.contains('\r')
819                || method_str.contains('\n')
820                || path.contains('\r')
821                || path.contains('\n')
822            {
823                return Err(
824                    "H65: request line contains CR/LF characters — possible header injection"
825                        .to_string(),
826                );
827            }
828            // H64: Validate domain contains no CR/LF to prevent header injection
829            if domain.contains('\r') || domain.contains('\n') {
830                return Err(
831                    "H64: domain contains CR/LF characters — possible header injection".to_string(),
832                );
833            }
834            // Reconstruct the HTTP request to send to upstream
835            let mut upstream_request = format!("{} {} HTTP/1.1\r\n", method_str, path);
836            // Ensure Host header is present
837            let has_host = headers.iter().any(|(n, _)| n.eq_ignore_ascii_case("host"));
838            if !has_host {
839                upstream_request.push_str(&format!("Host: {}\r\n", domain));
840            }
841            for (name, value) in &headers {
842                // Skip hop-by-hop headers
843                let lower = name.to_lowercase();
844                if HOP_BY_HOP_HEADERS.contains(&lower.as_str()) {
845                    continue;
846                }
847                // §3.4.8 step 7: Strip Accept-Encoding to force uncompressed responses,
848                // ensuring credential scanning can inspect response bodies reliably.
849                if lower == "accept-encoding" {
850                    continue;
851                }
852                upstream_request.push_str(&format!("{}: {}\r\n", name, value));
853            }
854            upstream_request.push_str("\r\n");
855
856            // §3.4 G26: Detect Expect: 100-continue header
857            let has_expect_continue = headers.iter().any(|(n, v)| {
858                n.eq_ignore_ascii_case("expect") && v.trim().eq_ignore_ascii_case("100-continue")
859            });
860
861            upstream_tls
862                .write_all(upstream_request.as_bytes())
863                .await
864                .map_err(|e| format!("writing request to upstream: {}", e))?;
865
866            upstream_tls
867                .flush()
868                .await
869                .map_err(|e| format!("flushing upstream: {}", e))?;
870
871            // §3.4 G26: If Expect: 100-continue, wait for upstream's 100 Continue
872            // response and relay it back to the agent before sending the body.
873            if has_expect_continue && !body.is_empty() {
874                use tokio::io::AsyncReadExt;
875                let mut continue_buf = [0u8; 256];
876                match tokio::time::timeout(
877                    Duration::from_secs(5),
878                    upstream_tls.read(&mut continue_buf),
879                )
880                .await
881                {
882                    Ok(Ok(n)) if n > 0 => {
883                        // Relay the 100 Continue (or other response) to agent
884                        agent_tls
885                            .write_all(&continue_buf[..n])
886                            .await
887                            .map_err(|e| format!("relaying 100-continue to agent: {}", e))?;
888                        agent_tls
889                            .flush()
890                            .await
891                            .map_err(|e| format!("flushing 100-continue: {}", e))?;
892                    }
893                    Ok(Ok(_)) => {
894                        tracing::debug!(
895                            branch = %branch_id,
896                            "§3.4 G26: upstream closed connection during 100-continue"
897                        );
898                    }
899                    Ok(Err(e)) => {
900                        tracing::debug!(
901                            branch = %branch_id,
902                            error = %e,
903                            "§3.4 G26: error reading 100-continue from upstream"
904                        );
905                    }
906                    Err(_) => {
907                        tracing::debug!(
908                            branch = %branch_id,
909                            "§3.4 G26: 100-continue timeout — sending body anyway"
910                        );
911                    }
912                }
913            }
914
915            if !body.is_empty() {
916                upstream_tls
917                    .write_all(&body)
918                    .await
919                    .map_err(|e| format!("writing request body to upstream: {}", e))?;
920            }
921
922            upstream_tls
923                .flush()
924                .await
925                .map_err(|e| format!("flushing upstream: {}", e))?;
926
927            // D-I5: Read the response from upstream using Content-Length or EOF.
928            // Previously this read until EOF, which blocks on HTTP/1.1 keep-alive
929            // connections where the server does not close the connection after each response.
930            let mut response_buf = Vec::new();
931
932            // U21: Response header size is bounded by hyper's default limits (currently ~64KB)
933            // Step 1: Read until response headers are complete (\r\n\r\n)
934            let resp_header_end;
935            loop {
936                if let Some(pos) = find_header_end(&response_buf) {
937                    resp_header_end = pos;
938                    break;
939                }
940                let n = upstream_tls
941                    .read(&mut tmp_buf)
942                    .await
943                    .map_err(|e| format!("reading response headers from upstream: {}", e))?;
944                if n == 0 {
945                    // Server closed before sending complete headers
946                    if response_buf.is_empty() {
947                        return Err("upstream closed connection without response".to_string());
948                    }
949                    // Treat whatever we have as the full response (incomplete headers)
950                    resp_header_end = response_buf.len().saturating_sub(4);
951                    break;
952                }
953                response_buf.extend_from_slice(&tmp_buf[..n]);
954                if response_buf.len() > MAX_RESPONSE_BODY_BYTES {
955                    tracing::warn!(
956                        branch = %branch_id,
957                        bytes = response_buf.len(),
958                        "C4: response headers exceeded size limit, closing"
959                    );
960                    resp_header_end = response_buf.len().saturating_sub(4);
961                    break;
962                }
963            }
964
965            // Step 2: Parse response headers to extract Content-Length and Transfer-Encoding
966            let resp_body_start = resp_header_end + 4;
967            let mut resp_content_length: Option<usize> = None;
968            let mut resp_is_chunked = false;
969            let mut resp_content_encoding: Option<String> = None;
970            if let Ok(header_str) = std::str::from_utf8(&response_buf[..resp_header_end]) {
971                for line in header_str.split("\r\n") {
972                    if let Some((name, value)) = line.split_once(':') {
973                        if name.trim().eq_ignore_ascii_case("content-length") {
974                            resp_content_length = value.trim().parse().ok();
975                        }
976                        // D-C1: Detect chunked Transfer-Encoding in upstream responses
977                        if name.trim().eq_ignore_ascii_case("transfer-encoding")
978                            && value.trim().eq_ignore_ascii_case("chunked")
979                        {
980                            resp_is_chunked = true;
981                        }
982                        // D-C3: Detect Content-Encoding for response decompression
983                        if name.trim().eq_ignore_ascii_case("content-encoding") {
984                            resp_content_encoding = Some(value.trim().to_lowercase());
985                        }
986                    }
987                }
988            }
989
990            // Step 3: Read body bytes based on Content-Length, chunked encoding, or EOF
991            if resp_is_chunked {
992                // D-M2: Read chunked body with per-read timeout instead of fragile
993                // terminal marker detection. The `decode_chunked_body()` parser after
994                // this loop handles proper chunk boundary parsing.
995                loop {
996                    match tokio::time::timeout(
997                        Duration::from_secs(30),
998                        upstream_tls.read(&mut tmp_buf),
999                    )
1000                    .await
1001                    {
1002                        Ok(Ok(0)) => break, // EOF — server closed
1003                        Ok(Ok(n)) => {
1004                            response_buf.extend_from_slice(&tmp_buf[..n]);
1005                            if response_buf.len() > MAX_RESPONSE_BODY_BYTES {
1006                                tracing::warn!(
1007                                    branch = %branch_id,
1008                                    bytes = response_buf.len(),
1009                                    "D-C1: chunked response exceeded size limit, closing"
1010                                );
1011                                break;
1012                            }
1013                            // Probe: try to decode what we have so far. If it succeeds,
1014                            // the chunked body is complete.
1015                            let body_so_far = &response_buf[resp_body_start..];
1016                            if decode_chunked_body(body_so_far).is_ok() {
1017                                break;
1018                            }
1019                        }
1020                        Ok(Err(e)) => {
1021                            return Err(format!("reading chunked response from upstream: {}", e));
1022                        }
1023                        Err(_) => {
1024                            tracing::warn!(
1025                                branch = %branch_id,
1026                                bytes = response_buf.len(),
1027                                "D-M2: per-read timeout (30s) while reading chunked response, treating as complete"
1028                            );
1029                            break;
1030                        }
1031                    }
1032                }
1033
1034                // Decode the chunked body
1035                let raw_chunked_body = &response_buf[resp_body_start..];
1036                match decode_chunked_body(raw_chunked_body) {
1037                    Ok(decoded_body) => {
1038                        // Rebuild response: replace Transfer-Encoding: chunked with
1039                        // Content-Length: <decoded_length> and use decoded body.
1040                        let header_bytes = &response_buf[..resp_header_end];
1041                        if let Ok(header_str) = std::str::from_utf8(header_bytes) {
1042                            let mut new_response = Vec::new();
1043                            for line in header_str.split("\r\n") {
1044                                if let Some((name, _)) = line.split_once(':') {
1045                                    if name.trim().eq_ignore_ascii_case("transfer-encoding") {
1046                                        // Replace with Content-Length
1047                                        new_response.extend_from_slice(
1048                                            format!("Content-Length: {}", decoded_body.len())
1049                                                .as_bytes(),
1050                                        );
1051                                        new_response.extend_from_slice(b"\r\n");
1052                                        continue;
1053                                    }
1054                                }
1055                                new_response.extend_from_slice(line.as_bytes());
1056                                new_response.extend_from_slice(b"\r\n");
1057                            }
1058                            new_response.extend_from_slice(b"\r\n");
1059                            new_response.extend_from_slice(&decoded_body);
1060                            response_buf = new_response;
1061                        } else {
1062                            // Cannot re-parse headers — just truncate and append decoded body
1063                            response_buf.truncate(resp_body_start);
1064                            response_buf.extend_from_slice(&decoded_body);
1065                        }
1066                    }
1067                    Err(e) => {
1068                        tracing::warn!(
1069                            branch = %branch_id,
1070                            error = %e,
1071                            "D-C1: failed to decode chunked response body, passing raw"
1072                        );
1073                        // Fall through with the raw response — DLP will inspect chunk framing
1074                        // but this is a best-effort fallback for malformed responses.
1075                    }
1076                }
1077            } else if let Some(cl) = resp_content_length {
1078                // Read exactly `cl` body bytes
1079                let already_read = response_buf.len().saturating_sub(resp_body_start);
1080                let remaining = cl.saturating_sub(already_read);
1081                if remaining > 0 && (already_read + remaining) <= MAX_RESPONSE_BODY_BYTES {
1082                    let mut bytes_left = remaining;
1083                    while bytes_left > 0 {
1084                        let n = upstream_tls
1085                            .read(&mut tmp_buf)
1086                            .await
1087                            .map_err(|e| format!("reading response body from upstream: {}", e))?;
1088                        if n == 0 {
1089                            break; // Server closed early
1090                        }
1091                        response_buf.extend_from_slice(&tmp_buf[..n]);
1092                        bytes_left = bytes_left.saturating_sub(n);
1093                    }
1094                }
1095            } else {
1096                // D-C2: No Content-Length or chunked encoding — fall back to reading
1097                // until EOF (HTTP/1.0 style) with a per-read timeout to prevent hangs
1098                // on keep-alive connections that never send EOF.
1099                tracing::warn!(
1100                    branch = %branch_id,
1101                    "D-C2: upstream response has no Content-Length or chunked encoding, reading until EOF with 30s timeout"
1102                );
1103                loop {
1104                    match tokio::time::timeout(
1105                        Duration::from_secs(30),
1106                        upstream_tls.read(&mut tmp_buf),
1107                    )
1108                    .await
1109                    {
1110                        Ok(Ok(0)) => break, // EOF
1111                        Ok(Ok(n)) => {
1112                            response_buf.extend_from_slice(&tmp_buf[..n]);
1113                            if response_buf.len() > MAX_RESPONSE_BODY_BYTES {
1114                                tracing::warn!(
1115                                    branch = %branch_id,
1116                                    bytes = response_buf.len(),
1117                                    "C4: response exceeded size limit, closing"
1118                                );
1119                                break;
1120                            }
1121                        }
1122                        Ok(Err(e)) => {
1123                            return Err(format!("reading from upstream: {}", e));
1124                        }
1125                        Err(_) => {
1126                            tracing::warn!(
1127                                branch = %branch_id,
1128                                bytes = response_buf.len(),
1129                                "D-C2: per-read timeout (30s) in EOF fallback path, treating response as complete"
1130                            );
1131                            break;
1132                        }
1133                    }
1134                }
1135            }
1136
1137            // D-C3: Decompress response body for DLP + credential leak scanning.
1138            // We decompress into a separate buffer for inspection only; the original
1139            // compressed response is forwarded to the agent unchanged (the agent
1140            // expects the Content-Encoding it negotiated with the upstream).
1141            let decompressed_body: Option<Vec<u8>> = if resp_content_encoding
1142                .as_deref()
1143                .is_some_and(|e| !e.is_empty() && e != "identity")
1144            {
1145                if let Some(hdr_end) = find_header_end(&response_buf) {
1146                    let body_off = hdr_end + 4;
1147                    if body_off < response_buf.len() {
1148                        let raw_body = &response_buf[body_off..];
1149                        match decompress_for_scanning(
1150                            raw_body,
1151                            resp_content_encoding.as_deref().unwrap_or(""),
1152                        ) {
1153                            Ok(decompressed) => {
1154                                tracing::debug!(
1155                                    branch = %branch_id,
1156                                    encoding = resp_content_encoding.as_deref().unwrap_or(""),
1157                                    compressed_size = raw_body.len(),
1158                                    decompressed_size = decompressed.len(),
1159                                    "D-C3: decompressed response body for DLP/credential scanning"
1160                                );
1161                                Some(decompressed)
1162                            }
1163                            Err(reason) => {
1164                                // Fail-closed: unsupported or corrupt encoding blocks the response.
1165                                tracing::warn!(
1166                                    branch = %branch_id,
1167                                    reason = %reason,
1168                                    "D-C3: blocking response — cannot decompress for scanning"
