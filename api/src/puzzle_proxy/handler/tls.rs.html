20/// C4: TLS MITM interception handler.
21///
22/// Intercepts CONNECT requests by:
23/// 1. Issuing a leaf cert for the target domain via `AgentCa::issue_leaf_cert(domain)`
24/// 2. Accepting TLS from the agent using the leaf cert (the agent trusts the CA)
25/// 3. Establishing TLS to the upstream server
26/// 4. Reading the decrypted HTTP/1.1 request from the agent
27/// 5. Routing based on method:
28///    - GET/HEAD/OPTIONS: forward to upstream and relay response back
29///    - POST/PUT/DELETE/PATCH: journal the request, then forward to upstream
30///
31/// The `AgentCa` cert PEM must be injected into the agent's trust store before
32/// this handler is used, so the agent accepts the proxy's leaf certificates.
33pub(super) async fn handle_tls_intercept(
34    req: Request<hyper::body::Incoming>,
35    ca: &AgentCa,
36    resolved_addrs: Option<&[std::net::SocketAddr]>,
37    ctx: &ProxyRequestContext,
38) -> Result<Response<BoxBody>, hyper::Error> {
39    let branch_id = &ctx.branch_id;
40
41    // 1. Extract target domain and port from the CONNECT request URI
42    let target_addr = req
43        .uri()
44        .authority()
45        .map(|a| a.to_string())
46        .unwrap_or_else(|| req.uri().to_string());
47
48    let domain = target_addr
49        .split(':')
50        .next()
51        .unwrap_or(&target_addr)
52        .to_string();
53
54    let target_with_port = if target_addr.contains(':') {
55        target_addr.clone()
56    } else {
57        format!("{}:443", target_addr)
58    };
59
60    tracing::info!(
61        branch = %branch_id,
62        target = %target_addr,
63        domain = %domain,
64        "C4: TLS MITM intercept requested"
65    );
66
67    // 2. Issue a leaf cert for the target domain
68    let (leaf_cert_der, leaf_key_der) = match ca.issue_leaf_cert(&domain) {
69        Ok(pair) => pair,
70        Err(e) => {
71            tracing::error!(
72                branch = %branch_id,
73                domain = %domain,
74                error = %e,
75                "C4: failed to issue leaf cert"
76            );
77            // R2: generic error — do not leak domain or crypto error details
78            return Ok(Response::builder()
79                .status(StatusCode::INTERNAL_SERVER_ERROR)
80                .body(Full::new(Bytes::from("TLS certificate error\n")))
81                .unwrap());
82        }
83    };
84
85    // 3. Connect to upstream with timeout (do this before upgrade so we can
86    //    return an error response if upstream is unreachable)
87    let upstream_tcp = match connect_upstream(&target_with_port, branch_id, resolved_addrs).await {
88        Ok(stream) => stream,
89        Err(resp) => return Ok(resp),
90    };
91
92    // Clone context for the spawned task
93    let ctx = ctx.clone();
94    let domain_clone = domain.clone();
95    let target_port_clone = target_with_port.clone();
96
97    // Spawn the TLS interception pipeline after sending 200 to the client
98    tokio::spawn(async move {
99        // 4. Upgrade the connection to get the raw TCP stream from the client
100        let upgraded = match hyper::upgrade::on(req).await {
101            Ok(upgraded) => upgraded,
102            Err(e) => {
103                tracing::error!(error = %e, "C4: upgrade failed");
104                return;
105            }
106        };
107        let client_io = hyper_util::rt::TokioIo::new(upgraded);
108
109        // 5. Create TLS acceptor with the leaf cert to accept TLS from the agent
110        let tls_acceptor = match build_tls_acceptor(leaf_cert_der, leaf_key_der) {
111            Ok(acceptor) => acceptor,
112            Err(e) => {
113                tracing::error!(error = %e, "C4: failed to build TLS acceptor");
114                return;
115            }
116        };
117
118        // Accept TLS from the agent
119        let agent_tls = match tls_acceptor.accept(client_io).await {
120            Ok(stream) => stream,
121            Err(e) => {
122                tracing::error!(
123                    domain = %domain_clone,
124                    error = %e,
125                    "C4: TLS handshake with agent failed"
126                );
127                return;
128            }
129        };
130
131        // 6. Establish TLS to the upstream server
132        let upstream_tls = match connect_upstream_tls(upstream_tcp, &domain_clone).await {
133            Ok(stream) => stream,
134            Err(e) => {
135                tracing::error!(
136                    domain = %domain_clone,
137                    error = %e,
138                    "C4: TLS handshake with upstream failed"
139                );
140                return;
141            }
142        };
143
144        // 7. Read the decrypted HTTP/1.1 request from the agent, then route it
145        if let Err(e) = handle_intercepted_stream(
146            agent_tls,
147            upstream_tls,
148            &domain_clone,
149            &target_port_clone,
150            &ctx,
151        )
152        .await
153        {
154            tracing::error!(
155                branch = %ctx.branch_id,
156                domain = %domain_clone,
157                error = %e,
158                "C4: intercepted stream handling failed"
159            );
160        }
161    });
162
163    // Send 200 Connection Established to the client, triggering the upgrade
164    Ok(Response::builder()
165        .status(StatusCode::OK)
166        .body(Full::new(Bytes::new()))
167        .unwrap())
168}

170/// Build a `TlsAcceptor` from a leaf cert and private key for agent-side TLS.
171pub(super) fn build_tls_acceptor(
172    cert_der: rustls::pki_types::CertificateDer<'static>,
173    key_der: rustls::pki_types::PrivateKeyDer<'static>,
174) -> Result<TlsAcceptor, String> {
175    let mut server_config = ServerConfig::builder()
176        .with_no_client_auth()
177        .with_single_cert(vec![cert_der], key_der)
178        .map_err(|e| format!("building TLS server config: {}", e))?;
179
180    // §3.4 G15: Force HTTP/1.1 negotiation — the proxy handler only supports
181    // HTTP/1.1. Without this, clients may negotiate HTTP/2 via ALPN, which
182    // would break the HTTP/1.1-only request/response parsing pipeline.
183    server_config.alpn_protocols = vec![b"http/1.1".to_vec()];
184
185    Ok(TlsAcceptor::from(Arc::new(server_config)))
186}

188/// Connect to an upstream server with TLS, using the system root CA store.
189pub(super) async fn connect_upstream_tls(
190    tcp_stream: tokio::net::TcpStream,
191    domain: &str,
192) -> Result<tokio_rustls::client::TlsStream<tokio::net::TcpStream>, String> {
193    let mut root_store = rustls::RootCertStore::empty();
194    // Add the webpki/mozilla root certificates
195    root_store.extend(webpki_roots::TLS_SERVER_ROOTS.iter().cloned());
196
197    let mut client_config = rustls::ClientConfig::builder()
198        .with_root_certificates(root_store)
199        .with_no_client_auth();
200    // M-3: Explicitly negotiate HTTP/1.1 on upstream connections, matching the
201    // server-side ALPN restriction (line 3274). Defense-in-depth: prevents silent
202    // HTTP/2 upgrade that would break the credential injection pipeline.
203    client_config.alpn_protocols = vec![b"http/1.1".to_vec()];
204
205    let connector = tokio_rustls::TlsConnector::from(Arc::new(client_config));
206    let server_name = rustls::pki_types::ServerName::try_from(domain.to_string())
207        .map_err(|e| format!("invalid server name '{}': {}", domain, e))?;
208
209    connector
210        .connect(server_name, tcp_stream)
211        .await
212        .map_err(|e| format!("upstream TLS handshake failed: {}", e))
213}
