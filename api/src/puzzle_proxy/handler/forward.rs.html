4use bytes::Bytes;
5use http_body_util::Full;
6use hyper::{Request, Response, StatusCode};
7
8use super::routing::strip_hop_by_hop;
9use super::{BoxBody, CONNECT_TIMEOUT, MAX_RESPONSE_BODY_BYTES, REQUEST_TIMEOUT};

11/// Forward a request to the upstream server using hyper client.
12///
13/// H7: Applies connect timeout and total request timeout.
14/// H-3: When `resolved_addrs` is provided, uses the pre-resolved IP addresses
15/// to connect instead of re-resolving DNS (prevents DNS rebinding TOCTOU).
16/// M17: Streams the request body instead of buffering entirely in memory.
17pub(super) async fn forward_request(
18    req: Request<hyper::body::Incoming>,
19    resolved_addrs: Option<&[std::net::SocketAddr]>,
20) -> Result<Response<BoxBody>, hyper::Error> {
21    use hyper_util::client::legacy::Client;
22    use hyper_util::rt::TokioExecutor;
23
24    let uri = req.uri().clone();
25    tracing::debug!(uri = %uri, "forwarding request");
26
27    // H-3: If we have pre-resolved addresses, connect directly to the pinned IP
28    // instead of letting the HTTP client re-resolve DNS (TOCTOU protection).
29    if let Some(addrs) = resolved_addrs {
30        if let Some(addr) = addrs.first() {
31            let port = uri.port_u16().unwrap_or(match uri.scheme_str() {
32                Some("https") => 443,
33                _ => 80,
34            });
35            let target_addr = std::net::SocketAddr::new(addr.ip(), port);
36
37            return match tokio::time::timeout(REQUEST_TIMEOUT, async {
38                // Connect to the pinned IP address directly
39                let tcp_stream = tokio::net::TcpStream::connect(target_addr).await
40                    .map_err(|e| {
41                        tracing::error!(uri = %uri, addr = %target_addr, error = %e, "failed to connect to pinned IP");
42                        e
43                    });
44
45                let tcp_stream = match tcp_stream {
46                    Ok(s) => s,
47                    Err(_e) => {
48                        return Ok(Ok(Response::builder()
49                            .status(StatusCode::BAD_GATEWAY)
50                            // K41: Omit internal IP and error details from response
51                            .body(Full::new(Bytes::from("Failed to connect to upstream\n")))
52                            .unwrap()));
53                    }
54                };
55
56                // Use the TCP stream with hyper's handshake for HTTP/1.1
57                let io = hyper_util::rt::TokioIo::new(tcp_stream);
58                let (mut sender, conn) = match hyper::client::conn::http1::handshake(io).await {
59                    Ok(parts) => parts,
60                    Err(e) => {
61                        tracing::warn!(error = %e, "R1: HTTP handshake failed");
62                        return Ok(Ok(Response::builder()
63                            .status(StatusCode::BAD_GATEWAY)
64                            // R1: generic error — do not leak internal handshake error
65                            .body(Full::new(Bytes::from("HTTP handshake failed\n")))
66                            .unwrap()));
67                    }
68                };
69
70                // Spawn the connection driver
71                tokio::spawn(async move {
72                    if let Err(e) = conn.await {
73                        tracing::debug!(error = %e, "pinned connection driver error");
74                    }
75                });
76
77                // Build the outgoing request — rewrite URI to path-only for the direct connection
78                let method = req.method().clone();
79                let mut headers = req.headers().clone();
80                strip_hop_by_hop(&mut headers);
81
82                // §3.4.8 step 7: Strip Accept-Encoding to force uncompressed responses,
83                // ensuring credential scanning can inspect response bodies reliably.
84                headers.remove("accept-encoding");
85
86                // Ensure Host header is set (required for HTTP/1.1)
87                if !headers.contains_key("host") {
88                    if let Some(authority) = uri.authority() {
89                        if let Ok(val) = hyper::header::HeaderValue::from_str(authority.as_ref()) {
90                            headers.insert("host", val);
91                        }
92                    }
93                }
94
95                // Stream body (G25: with size limit to prevent memory exhaustion)
96                let mut body_parts = Vec::new();
97                let mut total_body_size: usize = 0;
98                let mut body_stream = req.into_body();
99                use http_body_util::BodyExt;
100                while let Some(frame_result) = body_stream.frame().await {
101                    match frame_result {
102                        Ok(frame) => {
103                            if let Ok(data) = frame.into_data() {
104                                total_body_size += data.len();
105                                if total_body_size > MAX_RESPONSE_BODY_BYTES {
106                                    tracing::warn!(
107                                        total_body_size,
108                                        limit = MAX_RESPONSE_BODY_BYTES,
109                                        "G25: request body too large in pinned-IP path"
110                                    );
111                                    return Ok(Ok(Response::builder()
112                                        .status(StatusCode::PAYLOAD_TOO_LARGE)
113                                        .body(Full::new(Bytes::from(
114                                            "G25: request body too large\n",
115                                        )))
116                                        .unwrap()));
117                                }
118                                body_parts.push(data);
119                            }
120                        }
121                        Err(e) => return Err(e),
122                    }
123                }
124                let body_bytes: Bytes = if body_parts.is_empty() {
125                    Bytes::new()
126                } else if body_parts.len() == 1 {
127                    body_parts.into_iter().next().expect("V24: guarded by body_parts len check")
128                } else {
129                    let total_len: usize = body_parts.iter().map(|b| b.len()).sum();
130                    let mut combined = bytes::BytesMut::with_capacity(total_len);
131                    for part in body_parts {
132                        combined.extend_from_slice(&part);
133                    }
134                    combined.freeze()
135                };
136
137                // Build request with path-only URI for direct connection
138                let path_and_query = uri.path_and_query().map(|pq| pq.to_string()).unwrap_or_else(|| "/".to_string());
139                let mut outgoing = Request::builder().method(method).uri(path_and_query);
140                for (name, value) in headers.iter() {
141                    outgoing = outgoing.header(name, value);
142                }
143                let outgoing = outgoing.body(Full::new(body_bytes)).unwrap();
144
145                match sender.send_request(outgoing).await {
146                    Ok(resp) => {
147                        let status = resp.status();
148                        let resp_headers = resp.headers().clone();
149                        let body = resp.into_body();
150
151                        let mut body_parts = Vec::new();
152                        let mut total_resp_size: usize = 0;
153                        let mut body_stream = body;
154                        while let Some(frame_result) = body_stream.frame().await {
155                            match frame_result {
156                                Ok(frame) => {
157                                    if let Ok(data) = frame.into_data() {
158                                        total_resp_size += data.len();
159                                        if total_resp_size > MAX_RESPONSE_BODY_BYTES {
160                                            tracing::warn!(
161                                                uri = %uri,
162                                                bytes = total_resp_size,
163                                                limit = MAX_RESPONSE_BODY_BYTES,
164                                                "response body exceeded size limit"
165                                            );
166                                            return Ok(Ok(Response::builder()
167                                                .status(StatusCode::BAD_GATEWAY)
168                                                .body(Full::new(Bytes::from(format!(
169                                                    "Response body too large (limit {} bytes)\n",
170                                                    MAX_RESPONSE_BODY_BYTES
171                                                ))))
172                                                .unwrap()));
173                                        }
174                                        body_parts.push(data);
175                                    }
176                                }
177                                Err(e) => return Err(e),
178                            }
179                        }
180                        let body_bytes: Bytes = if body_parts.is_empty() {
181                            Bytes::new()
182                        } else if body_parts.len() == 1 {
183                            body_parts.into_iter().next().expect("V24: guarded by body_parts len check")
184                        } else {
185                            let mut combined = bytes::BytesMut::with_capacity(total_resp_size);
186                            for part in body_parts {
187                                combined.extend_from_slice(&part);
188                            }
189                            combined.freeze()
190                        };
191
192                        let mut filtered_resp_headers = resp_headers;
193                        strip_hop_by_hop(&mut filtered_resp_headers);
194
195                        let mut response = Response::builder().status(status);
196                        for (name, value) in filtered_resp_headers.iter() {
197                            response = response.header(name, value);
198                        }
199                        Ok(Ok(response.body(Full::new(body_bytes)).unwrap()))
200                    }
201                    Err(e) => {
202                        tracing::error!(uri = %uri, error = %e, "upstream request failed (pinned IP)");
203                        // K45: Omit internal error details from response
204                        Ok(Ok(Response::builder()
205                            .status(StatusCode::BAD_GATEWAY)
206                            .body(Full::new(Bytes::from("Upstream request failed\n")))
207                            .unwrap()))
208                    }
209                }
210            })
211            .await
212            {
213                Ok(result) => result?,
214                Err(_elapsed) => {
215                    tracing::error!(uri = %uri, "request timed out (pinned IP)");
216                    Ok(Response::builder()
217                        .status(StatusCode::GATEWAY_TIMEOUT)
218                        .body(Full::new(Bytes::from(
219                            "Request timed out\n"
220                        )))
221                        .unwrap())
222                }
223            };
224        }
225    }
226
227    // Fallback: no pre-resolved addresses, use standard HTTP client (re-resolves DNS)
228    // D-C1: Redirect safety — hyper_util::client::legacy::Client does NOT follow
229    // redirects automatically. It returns 3xx responses as-is to the caller.
230    // The pinned-IP path above uses raw hyper::client::conn::http1::handshake which
231    // also does not follow redirects. Therefore injected credentials in Authorization
232    // headers are never carried to a redirect target domain by this code.
233    // V1: Refuse to forward https:// URIs over plaintext — prevents credential leakage
234    if uri.scheme_str() == Some("https") {
235        tracing::error!(uri = %uri, "V1: refusing to forward https:// URI over plaintext HTTP client — TLS downgrade prevented");
236        return Ok(Response::builder()
237            .status(StatusCode::BAD_GATEWAY)
238            .body(Full::new(Bytes::from("TLS required for https:// URIs\n")))
239            .unwrap());
240    }
241    // H7: Wrap the entire request in a total timeout
242    match tokio::time::timeout(REQUEST_TIMEOUT, async {
243        let client: Client<_, http_body_util::Full<Bytes>> =
244            Client::builder(TokioExecutor::new()).build_http();
245
246        // M17: Stream the body — collect it but forward as a single chunk.
247        // For hyper 1.x with the legacy client, we build the outgoing request
248        // and stream body data through.
249        let method = req.method().clone();
250        let mut headers = req.headers().clone();
251
252        // M15: Strip hop-by-hop headers before forwarding
253        strip_hop_by_hop(&mut headers);
254
255        // §3.4.8 step 7: Strip Accept-Encoding to force uncompressed responses,
256        // ensuring credential scanning can inspect response bodies reliably.
257        headers.remove("accept-encoding");
258
259        // Stream body: read chunks as they arrive instead of buffering all at once
260        // H60: Track total body size and enforce MAX_BODY_SIZE limit
261        let mut body_parts = Vec::new();
262        let mut total_body_size: usize = 0; // H60: running size counter
263        let mut body_stream = req.into_body();
264        use http_body_util::BodyExt;
265        while let Some(frame_result) = body_stream.frame().await {
266            match frame_result {
267                Ok(frame) => {
268                    if let Ok(data) = frame.into_data() {
269                        // H60: Check accumulated size before accepting chunk
270                        total_body_size = total_body_size.saturating_add(data.len());
271                        if total_body_size > MAX_BODY_SIZE {
272                            tracing::warn!(
273                                total_body_size,
274                                max = MAX_BODY_SIZE,
275                                "H60: fallback request body exceeds MAX_BODY_SIZE"
276                            );
277                            return Ok(Ok(Response::builder()
278                                .status(StatusCode::PAYLOAD_TOO_LARGE)
279                                .body(Full::new(Bytes::from(format!(
280                                    "H60: request body ({} bytes) exceeds limit ({} bytes)\n",
281                                    total_body_size, MAX_BODY_SIZE
282                                ))))
283                                .unwrap()));
284                        }
285                        body_parts.push(data);
286                    }
287                }
288                Err(e) => return Err(e),
289            }
290        }
291        let body_bytes: Bytes = if body_parts.is_empty() {
292            Bytes::new()
293        } else if body_parts.len() == 1 {
294            body_parts
295                .into_iter()
296                .next()
297                .expect("V24: guarded by body_parts len check")
298        } else {
299            let total_len: usize = body_parts.iter().map(|b| b.len()).sum();
300            let mut combined = bytes::BytesMut::with_capacity(total_len);
301            for part in body_parts {
302                combined.extend_from_slice(&part);
303            }
304            combined.freeze()
305        };
306
307        let mut outgoing = Request::builder().method(method).uri(uri.clone());
308        for (name, value) in headers.iter() {
309            outgoing = outgoing.header(name, value);
310        }
311        let outgoing = outgoing.body(Full::new(body_bytes)).unwrap();
312
313        // H7: Connect timeout is handled by the overall request timeout
314        match client.request(outgoing).await {
315            Ok(resp) => {
316                let status = resp.status();
317                let resp_headers = resp.headers().clone();
318                let body = resp.into_body();
319
320                // M16: Read response body with size limit enforcement
321                let mut body_parts = Vec::new();
322                let mut total_resp_size: usize = 0;
323                let mut body_stream = body;
324                while let Some(frame_result) = body_stream.frame().await {
325                    match frame_result {
326                        Ok(frame) => {
327                            if let Ok(data) = frame.into_data() {
328                                total_resp_size += data.len();
329                                if total_resp_size > MAX_RESPONSE_BODY_BYTES {
330                                    tracing::warn!(
331                                        uri = %uri,
332                                        bytes = total_resp_size,
333                                        limit = MAX_RESPONSE_BODY_BYTES,
334                                        "response body exceeded size limit"
335                                    );
336                                    return Ok(Ok(Response::builder()
337                                        .status(StatusCode::BAD_GATEWAY)
338                                        .body(Full::new(Bytes::from(format!(
339                                            "Response body too large (limit {} bytes)\n",
340                                            MAX_RESPONSE_BODY_BYTES
341                                        ))))
342                                        .unwrap()));
343                                }
344                                body_parts.push(data);
345                            }
346                        }
347                        Err(e) => return Err(e),
348                    }
349                }
350                let body_bytes: Bytes = if body_parts.is_empty() {
351                    Bytes::new()
352                } else if body_parts.len() == 1 {
353                    body_parts
354                        .into_iter()
355                        .next()
356                        .expect("V24: guarded by body_parts len check")
357                } else {
358                    let mut combined = bytes::BytesMut::with_capacity(total_resp_size);
359                    for part in body_parts {
360                        combined.extend_from_slice(&part);
361                    }
362                    combined.freeze()
363                };
364
365                // PM2: Strip hop-by-hop headers from the response before
366                // returning to the client, mirroring the request-side stripping.
367                let mut filtered_resp_headers = resp_headers;
368                strip_hop_by_hop(&mut filtered_resp_headers);
369
370                let mut response = Response::builder().status(status);
371                for (name, value) in filtered_resp_headers.iter() {
372                    response = response.header(name, value);
373                }
374                Ok(Ok(response.body(Full::new(body_bytes)).unwrap()))
375            }
376            Err(e) => {
377                tracing::error!(uri = %uri, error = %e, "upstream request failed");
378                // K45: Omit internal error details from response
379                Ok(Ok(Response::builder()
380                    .status(StatusCode::BAD_GATEWAY)
381                    .body(Full::new(Bytes::from("Upstream request failed\n")))
382                    .unwrap()))
383            }
384        }
385    })
386    .await
387    {
388        Ok(result) => result?,
389        Err(_elapsed) => {
390            tracing::error!(uri = %uri, "request timed out");
391            Ok(Response::builder()
392                .status(StatusCode::GATEWAY_TIMEOUT)
393                .body(Full::new(Bytes::from("Request timed out\n")))
394                .unwrap())
395        }
396    }
397}

399/// Maximum request body size (100 MB). Requests exceeding this are rejected with HTTP 413.
400pub(super) const MAX_BODY_SIZE: usize = 100 * 1024 * 1024;
401
402/// D-M1: Maximum HTTP header size (64 KB). Headers should be much smaller than bodies.
403/// Using MAX_BODY_SIZE (100 MB) for headers allowed excessive memory consumption.
404pub(super) const MAX_HEADER_SIZE: usize = 64 * 1024;

406/// Connect to an upstream server with timeout, using pre-resolved addresses if available.
407pub(super) async fn connect_upstream(
408    target_with_port: &str,
409    branch_id: &puzzled_types::BranchId,
410    resolved_addrs: Option<&[std::net::SocketAddr]>,
411) -> Result<tokio::net::TcpStream, Response<BoxBody>> {
412    match tokio::time::timeout(CONNECT_TIMEOUT, async {
413        if let Some(addrs) = resolved_addrs {
414            let addrs_with_port: Vec<std::net::SocketAddr> = addrs
415                .iter()
416                .map(|a| {
417                    let port = target_with_port
418                        .rsplit(':')
419                        .next()
420                        .and_then(|p| p.parse::<u16>().ok())
421                        .unwrap_or(443);
422                    std::net::SocketAddr::new(a.ip(), port)
423                })
424                .collect();
425            tokio::net::TcpStream::connect(addrs_with_port.as_slice()).await
426        } else {
427            tokio::net::TcpStream::connect(target_with_port).await
428        }
429    })
430    .await
431    {
432        Ok(Ok(stream)) => Ok(stream),
433        Ok(Err(e)) => {
434            tracing::error!(
435                branch = %branch_id,
436                target = %target_with_port,
437                error = %e,
438                "failed to connect to upstream"
439            );
440            Err(Response::builder()
441                .status(StatusCode::BAD_GATEWAY)
442                // K40: Omit internal IP/port and error details from response
443                .body(Full::new(Bytes::from("Failed to connect to upstream\n")))
444                .unwrap())
445        }
446        Err(_elapsed) => {
447            tracing::error!(
448                branch = %branch_id,
449                target = %target_with_port,
450                "connect timeout to upstream"
451            );
452            // N2: Do not leak target in timeout error response
453            Err(Response::builder()
454                .status(StatusCode::GATEWAY_TIMEOUT)
455                .body(Full::new(Bytes::from("Connect timeout\n")))
456                .unwrap())
457        }
458    }
